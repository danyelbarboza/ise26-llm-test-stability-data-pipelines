import pandas as pd
import pytest
from ise26.targets import deduplicate_events


def test_deduplicate_keeps_most_recent():
    df = pd.DataFrame({
        "event_id": [1, 1, 2],
        "updated_at": ["2020-01-01", "2020-01-02", "2020-01-03"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "updated_at": ["2020-01-02", "2020-01-03"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_preserves_null_ids():
    df = pd.DataFrame({
        "event_id": [None, 1, 1, None],
        "updated_at": ["2020-01-01", "2020-01-01", "2020-01-02", "2020-01-03"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [None, 1, None],
        "updated_at": ["2020-01-01", "2020-01-02", "2020-01-03"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_treats_invalid_timestamps_as_older():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["invalid", "2020-01-02"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2020-01-02"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_timestamp_tie_keeps_last_row():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2020-01-01", "2020-01-01"]
    })
    result = deduplicate_events(df)
    # second row should be kept (index 1)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2020-01-01"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_all_invalid_timestamps_keeps_last_row():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["bad", "worse"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["worse"]  # second row kept
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_mixed_ids_and_nulls():
    df = pd.DataFrame({
        "event_id": [1, None, 1, None, 2],
        "updated_at": ["2020-01-01", "2020-01-05", "2020-01-02", "2020-01-04", "2020-01-03"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1, None, None, 2],
        "updated_at": ["2020-01-02", "2020-01-05", "2020-01-04", "2020-01-03"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_all_null_ids():
    df = pd.DataFrame({
        "event_id": [None, None],
        "updated_at": ["2020-01-01", "2020-01-02"]
    })
    result = deduplicate_events(df)
    # all rows preserved
    pd.testing.assert_frame_equal(result, df)


def test_deduplicate_empty_df():
    df = pd.DataFrame({"event_id": pd.Series(dtype='object'), "updated_at": pd.Series(dtype='object')})
    result = deduplicate_events(df)
    pd.testing.assert_frame_equal(result, df)


def test_deduplicate_preserves_input_df():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2020-01-01", "2020-01-02"]
    })
    df_before = df.copy()
    _ = deduplicate_events(df)
    pd.testing.assert_frame_equal(df, df_before)


def test_deduplicate_returns_new_dataframe():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2020-01-01", "2020-01-02"]
    })
    result = deduplicate_events(df)
    assert result is not df
    assert isinstance(result, pd.DataFrame)


def test_deduplicate_custom_columns():
    df = pd.DataFrame({
        "custom_id": [1, 1, 2],
        "custom_ts": ["2020-01-02", "2020-01-01", "2020-01-03"]
    })
    result = deduplicate_events(df, id_col="custom_id", timestamp_col="custom_ts")
    expected = pd.DataFrame({
        "custom_id": [1, 2],
        "custom_ts": ["2020-01-02", "2020-01-03"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_deduplicate_maintains_original_order():
    df = pd.DataFrame({
        "event_id": [1, 2, 1, 2, 3, None],
        "updated_at": ["2020-02-01", "2020-01-01", "2020-01-01", "2020-02-01", "2020-03-01", "2020-04-01"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1, 2, 3, None],
        "updated_at": ["2020-02-01", "2020-02-01", "2020-03-01", "2020-04-01"]
    })
    pd.testing.assert_frame_equal(result, expected)
