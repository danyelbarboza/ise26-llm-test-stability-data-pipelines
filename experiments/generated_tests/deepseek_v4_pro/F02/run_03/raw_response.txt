import pandas as pd
import pytest
from pandas.testing import assert_frame_equal

from ise26.targets import deduplicate_events


class TestDeduplicateEvents:
    def test_returns_new_dataframe_and_does_not_mutate_input(self):
        df = pd.DataFrame({
            "event_id": ["a", "a"],
            "updated_at": ["2023-01-01 10:00:00", "2023-01-01 11:00:00"],
        })
        df_original = df.copy()
        result = deduplicate_events(df)
        assert result is not df
        assert_frame_equal(df, df_original)

    def test_basic_deduplication_most_recent_kept(self):
        df = pd.DataFrame({
            "event_id": ["a", "a", "b"],
            "updated_at": ["2023-01-01 10:00:00", "2023-01-01 11:00:00", "2023-01-01 12:00:00"],
        })
        expected = pd.DataFrame({
            "event_id": ["a", "b"],
            "updated_at": ["2023-01-01 11:00:00", "2023-01-01 12:00:00"],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_null_identifiers_preserved(self):
        df = pd.DataFrame({
            "event_id": [None, None, "a"],
            "updated_at": ["2023-01-01 10:00:00", "2023-01-01 09:00:00", "2023-01-01 08:00:00"],
        })
        expected = df.copy().reset_index(drop=True)
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_invalid_timestamp_treated_as_older(self):
        df = pd.DataFrame({
            "event_id": ["a", "a"],
            "updated_at": ["invalid_date", "2023-01-01 10:00:00"],
        })
        expected = pd.DataFrame({
            "event_id": ["a"],
            "updated_at": ["2023-01-01 10:00:00"],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_timestamp_tie_keeps_last_original_order(self):
        df = pd.DataFrame({
            "event_id": ["a", "a", "a"],
            "updated_at": ["2023-01-01 10:00:00"] * 3,
            "value": [1, 2, 3],
        })
        expected = pd.DataFrame({
            "event_id": ["a"],
            "updated_at": ["2023-01-01 10:00:00"],
            "value": [3],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_mixed_scenarios(self):
        df = pd.DataFrame({
            "event_id": [None, "a", "a", "b", "b", "c"],
            "updated_at": [
                "2023-01-01",
                "2023-01-01 10:00:00",
                "2023-01-01 09:00:00",
                "2023-01-01 12:00:00",
                "invalid",
                "2023-01-01 08:00:00",
            ],
            "extra": [1, 2, 3, 4, 5, 6],
        })
        # For a: two rows, most recent is 10:00 (index 1)
        # For b: valid 12:00 (index 3) vs invalid (index 4) → keep valid (index 3)
        # For c: single (index 5)
        # Null id: index 0 preserved
        # Order after sorting by original index: 0 (None), 1 (a), 3 (b), 5 (c)
        expected = pd.DataFrame({
            "event_id": [None, "a", "b", "c"],
            "updated_at": ["2023-01-01", "2023-01-01 10:00:00", "2023-01-01 12:00:00", "2023-01-01 08:00:00"],
            "extra": [1, 2, 4, 6],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "custom_id": ["x", "x", "y"],
            "custom_ts": ["2023-01-02", "2023-01-01", "2023-01-03"],
        })
        expected = pd.DataFrame({
            "custom_id": ["x", "y"],
            "custom_ts": ["2023-01-02", "2023-01-03"],
        })
        result = deduplicate_events(df, id_col="custom_id", timestamp_col="custom_ts")
        assert_frame_equal(result, expected)

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["event_id", "updated_at"])
        result = deduplicate_events(df)
        assert_frame_equal(result, df)

    def test_all_null_identifiers(self):
        df = pd.DataFrame({
            "event_id": [None, None],
            "updated_at": ["2023-01-01", "2023-01-02"],
        })
        expected = df.copy().reset_index(drop=True)
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_all_invalid_timestamps_tie_break(self):
        df = pd.DataFrame({
            "event_id": ["a", "a", "a"],
            "updated_at": ["abc", "xyz", "def"],
            "value": [10, 20, 30],
        })
        # All timestamps invalid (NaT), tie, keep last original order → index 2 (value 30)
        expected = pd.DataFrame({
            "event_id": ["a"],
            "updated_at": ["def"],
            "value": [30],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)

    def test_single_row(self):
        df = pd.DataFrame({
            "event_id": ["single"],
            "updated_at": ["2023-01-01"],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, df)

    def test_output_order_preserves_original_order_of_kept_rows(self):
        df = pd.DataFrame({
            "event_id": ["b", "a", "a"],
            "updated_at": ["2023-01-01", "2023-01-01 09:00:00", "2023-01-01 10:00:00"],
        })
        # b (index 0) kept as is, a duplicate: keep index 2 (most recent).
        # Order in result should be b (original 0) then a (original 2).
        expected = pd.DataFrame({
            "event_id": ["b", "a"],
            "updated_at": ["2023-01-01", "2023-01-01 10:00:00"],
        })
        result = deduplicate_events(df)
        assert_frame_equal(result, expected)
