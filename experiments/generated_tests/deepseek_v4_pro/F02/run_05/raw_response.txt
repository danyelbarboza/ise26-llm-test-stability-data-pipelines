import pandas as pd
import pytest
from ise26.targets import deduplicate_events


def test_keeps_most_recent_per_event_id():
    df = pd.DataFrame({
        "event_id": ["a", "a", "b", "b", "b"],
        "updated_at": [
            "2020-01-01",
            "2020-01-02",
            "2020-02-01",
            "2020-02-03",
            "2020-02-02",
        ],
        "data": [1, 2, 3, 4, 5],
    })
    result = deduplicate_events(df)
    # For "a", keep last row (2020-01-02) with data=2.
    # For "b", keep row with 2020-02-03, data=4.
    expected = pd.DataFrame({
        "event_id": ["a", "b"],
        "updated_at": ["2020-01-02", "2020-02-03"],
        "data": [2, 4],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )


def test_preserves_null_event_ids():
    df = pd.DataFrame({
        "event_id": [None, "a", "a", None, None],
        "updated_at": [
            "2020-01-01",
            "2020-01-01",
            "2020-01-02",
            "2020-02-01",
            None,
        ],
        "data": [1, 2, 3, 4, 5],
    })
    result = deduplicate_events(df)
    # Null ids are kept as-is (all three rows: index 0,3,4)
    # Identified "a" keeps latest => row with updated_at "2020-01-02", data=3.
    # The result order must be original order of kept rows: index 0 (null), 2 (a), 3 (null), 4 (null)
    expected_rows = [
        {"event_id": None, "updated_at": "2020-01-01", "data": 1},
        {"event_id": "a", "updated_at": "2020-01-02", "data": 3},
        {"event_id": None, "updated_at": "2020-02-01", "data": 4},
        {"event_id": None, "updated_at": None, "data": 5},
    ]
    expected = pd.DataFrame(expected_rows)
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )


def test_invalid_timestamps_are_treated_as_older():
    df = pd.DataFrame({
        "event_id": [1, 1, 2, 2, 2],
        "updated_at": [
            "invalid_date",
            "2020-01-02",
            "invalid",
            "another_invalid",
            "2020-03-04",
        ],
        "value": [10, 20, 30, 40, 50],
    })
    result = deduplicate_events(df)
    # For id=1: invalid and valid -> keep valid "2020-01-02", value=20
    # For id=2: two invalid and one valid -> keep valid "2020-03-04", value=50
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "updated_at": ["2020-01-02", "2020-03-04"],
        "value": [20, 50],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )


def test_id_with_only_invalid_timestamps_keeps_last_original_row():
    df = pd.DataFrame({
        "event_id": ["x", "x", "x"],
        "updated_at": ["bad", "also_bad", "not_a_date"],
        "info": ["first", "second", "third"],

    expected = pd.DataFrame({
        "event_id": ["x"],
        "updated_at": ["not_a_date"],
        "info": ["third"],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )


def test_timestamp_tie_keeps_last_original_order():
    df = pd.DataFrame({

            "2020-05-05",
            "2020-05-05",

        "payload": [3],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )


def test_original_order_of_rows_is_preserved_after_dedup():
    df = pd.DataFrame({
        "event_id": ["c", "a", "b", "a", "b", "c"],
        "updated_at": [
            "2020-01-01",
            "2020-01-02",
            "2020-01-03",
            "2020-01-05",
            "2020-01-04",
            "2020-01-06",
        ],
        "data": [1, 2, 3, 4, 5, 6],
    })
    result = deduplicate_events(df)
    # Original order: indices 0(c),1(a),2(b),3(a),4(b),5(c).
    # Kept: for "a" keep index 3 (latest timestamp), for "b" keep index 4 (latest among 2 and 4), for "c" keep index 5.
    # Result order: 0 (c from idx0? No, idx0 is not kept, we keep idx5 for c). Actually we keep idx3 for a, idx4 for b, idx5 for c.
    # The order of kept rows in the final output is the original order: idx3, idx4, idx5.
    # So final rows: data=4,5,6.
    expected = pd.DataFrame({
        "event_id": ["a", "b", "c"],
        "updated_at": ["2020-01-05", "2020-01-04", "2020-01-06"],
        "data": [4, 5, 6],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )


def test_input_dataframe_is_not_mutated():
    df = pd.DataFrame({
        "event_id": ["p", "p", "q"],
        "updated_at": [
            "2020-01-01",
            "2020-01-02",
            "2020-03-04",
        ],
        "score": [10, 20, 30],
    })
    original = df.copy()

    pd.testing.assert_frame_equal(df, original)


def test_with_custom_column_names():
    df = pd.DataFrame({
        "id": ["E1", "E1", "E2"],
        "ts": [
            "2020-06-01",
            "2020-06-02",
            "2020-06-03",
        ],
        "val": [100, 200, 300],
    })
    result = deduplicate_events(df, id_col="id", timestamp_col="ts")
    expected = pd.DataFrame({
        "id": ["E1", "E2"],
        "ts": ["2020-06-02", "2020-06-03"],
        "val": [200, 300],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )

def test_mixed_null_and_identified_with_invalid_dates():
    df = pd.DataFrame({
        "event_id": [None, "id1", "id1", None, "id1"],
        "updated_at": [
            "ok",
            "bad",
            "2020-06-01",
            None,
            "2020-05-30",
        ],
        "n": [1, 2, 3, 4, 5],
    })
    result = deduplicate_events(df)
    # null ids: rows with n=1 and n=4 preserved.
    # id1: rows with timestamps "bad" (NaT), "2020-06-01", "2020-05-30". Sorted NaT first, then ascending dates: "2020-05-30", "2020-06-01".
    # keep='last' picks "2020-06-01" (n=3). So keep n=3.
    # order: n=1 (null), n=3 (id1), n=4 (null).
    expected = pd.DataFrame({
        "event_id": [None, "id1", None],
        "updated_at": ["ok", "2020-06-01", None],
        "n": [1, 3, 4],
    })
    pd.testing.assert_frame_equal(
        result.reset_index(drop=True), expected.reset_index(drop=True),
        check_like=True,
    )
