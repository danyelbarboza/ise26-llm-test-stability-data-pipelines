import pandas as pd
import pytest
from ise26.targets import deduplicate_events


class TestDeduplicateEvents:
    def test_basic_deduplication_keeps_most_recent(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 2],
            "updated_at": [
                "2023-01-01 10:00:00",
                "2023-01-02 10:00:00",
                "2023-01-01 10:00:00",
            ],
            "value": [10, 20, 30],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # For id=1, most recent is the second row (value=20)
        expected_ids = [1, 2]
        assert list(result["event_id"]) == expected_ids
        assert list(result["value"]) == [20, 30]

    def test_null_ids_preserved(self):
        df = pd.DataFrame({
            "event_id": [None, None, 1],
            "updated_at": [
                "2023-01-01",
                "2022-01-01",
                "2023-01-01",
            ],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # All null-id rows must be kept, only the identified one is kept
        assert len(result) == 3
        assert result["event_id"].isna().sum() == 2

    def test_invalid_timestamps_treated_as_oldest(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": [
                "not_a_date",
                "2023-01-01",
                "2022-02-02",
            ],
            "marker": ["bad", "recent", "old"],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # The row with valid most recent timestamp is "recent" (2023-01-01)
        assert len(result) == 1
        assert result.iloc[0]["marker"] == "recent"

    def test_timestamp_ties_keep_last_in_order(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": [
                "2023-01-01",
                "2023-01-01",
                "2023-01-01",
            ],
            "payload": ["a", "b", "c"],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # Ties: keep the last row in original order => payload "c"
        assert len(result) == 1
        assert result.iloc[0]["payload"] == "c"

    def test_all_timestamps_invalid_for_specific_id(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 2],
            "updated_at": ["bad", "worse", "2023-01-01"],
            "content": ["x", "y", "z"],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # For id=1, both timestamps invalid. The last row in original order is kept (content="y").
        # For id=2, single row kept.
        assert len(result) == 2
        kept_1 = result[result["event_id"] == 1].iloc[0]
        assert kept_1["content"] == "y"
        kept_2 = result[result["event_id"] == 2].iloc[0]
        assert kept_2["content"] == "z"

    def test_mixed_valid_and_invalid_multiple_ids(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 2, 2, None],
            "updated_at": [
                "invalid",
                "2023-01-01",
                "2023-01-02",
                "2023-01-01",
                "2023-01-03",
            ],
            "info": ["a", "b", "c", "d", "e"],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # id=1: invalid vs valid, keep valid (b)
        # id=2: keep most recent (c)
        # null: keep row
        assert len(result) == 3
        result_ids = result.set_index("event_id")
        assert result_ids.loc[1, "info"] == "b"
        assert result_ids.loc[2, "info"] == "c"
        null_row = result[result["event_id"].isna()].iloc[0]
        assert null_row["info"] == "e"

    def test_input_dataframe_not_mutated(self):
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2020-01-01", "2021-01-01"],
        })
        df_copy = df.copy()
        _ = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        pd.testing.assert_frame_equal(df, df_copy)

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["event_id", "updated_at"])
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 0

    def test_all_null_ids(self):
        df = pd.DataFrame({
            "event_id": [None, None, None],
            "updated_at": ["2023-01-01", "2022-01-01", "2021-01-01"],
            "data": [1, 2, 3],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # All rows must be preserved
        assert len(result) == 3
        assert list(result["data"]) == [1, 2, 3]

    def test_columns_other_than_id_and_timestamp_preserved(self):
        df = pd.DataFrame({
            "event_id": [10, 10],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "name": ["old", "new"],
            "score": [1, 2],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 1
        assert result.iloc[0]["name"] == "new"
        assert result.iloc[0]["score"] == 2

    def test_no_duplicates_no_change_in_rows(self):
        df = pd.DataFrame({
            "event_id": [1, 2, 3],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 3
        assert list(result["event_id"]) == [1, 2, 3]

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "my_id": ["A", "A", "B"],
            "ts": ["2022-05-05", "2022-05-06", "2022-05-05"],
            "val": [1, 2, 3],
        })
        result = deduplicate_events(df, id_col="my_id", timestamp_col="ts")
        assert len(result) == 2
        kept_a = result[result["my_id"] == "A"].iloc[0]
        assert kept_a["val"] == 2
