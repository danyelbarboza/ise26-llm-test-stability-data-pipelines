from ise26.targets import deduplicate_events

import pandas as pd
import pytest
from pandas.testing import assert_frame_equal


def test_deduplicate_events_keeps_most_recent_timestamp():
    df = pd.DataFrame({
        "event_id": [1, 1, 2, 2],
        "value": ["a", "b", "c", "d"],
        "updated_at": pd.to_datetime([
            "2023-01-01 10:00",
            "2023-01-01 12:00",
            "2023-01-02 09:00",
            "2023-01-02 08:00"
        ])
    })
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "value": ["b", "c"],
        "updated_at": pd.to_datetime(["2023-01-01 12:00", "2023-01-02 09:00"])
    })
    result = deduplicate_events(df)
    assert_frame_equal(result, expected)

def test_deduplicate_events_null_ids_preserved():
    df = pd.DataFrame({
        "event_id": [1, pd.NA, 1, pd.NA, 2],
        "value": ["a", "b", "c", "d", "e"],
        "updated_at": pd.to_datetime([
            "2023-01-01", "2023-01-02", "2023-01-03",
            "2023-01-04", "2023-01-05"
        ])
    })
    expected = pd.DataFrame({
        "event_id": [1, pd.NA, pd.NA, 2],
        "value": ["c", "b", "d", "e"],
        "updated_at": pd.to_datetime([
            "2023-01-03", "2023-01-02", "2023-01-04", "2023-01-05"
        ])
    })
    result = deduplicate_events(df)
    # Null id handling: both null-id rows preserved
    assert_frame_equal(result, expected, check_index_type=False)


def test_deduplicate_events_does_not_mutate_input():

def test_deduplicate_events_result_retains_original_order():
    df = pd.DataFrame({
        "event_id": [3, 1, 1, 2, 3],
        "value": ["a", "b", "c", "d", "e"],
        "updated_at": pd.to_datetime([
            "2023-01-01", "2023-01-01", "2023-01-02",
            "2023-01-03", "2023-01-04"
        ])
    })
    # For id=3: two rows, latest is e (2023-01-04), original pos 4
    # For id=1: tie, keep last -> c (original pos 2)
    # For id=2: only one -> d (original pos 3)
    # Final order should be: rows at original positions of 0(a not kept, replaced by c?), Wait: dedup keeps c at pos2? Let's think algorithm: unidentified_rows are empty. deduplicated_identified sorted: 1,2,3. For each id:
    # id=1: rows b (pos1), c (pos2) sorted by timestamp tie and original_order ascending: b then c, keep last -> c.
    # id=2: row d (pos3) -> kept.
    # id=3: rows a (pos0), e (pos4) sorted by timestamp: a (2023-01-01) then e (2023-01-04), keep last -> e.
    # Then concat and sort by _original_order: c (pos2), d (pos3), e (pos4). Order: pos2, pos3, pos4 -> c, d, e.
    expected = pd.DataFrame({
        "event_id": [1, 2, 3],
        "value": ["c", "d", "e"],
        "updated_at": pd.to_datetime(["2023-01-02", "2023-01-03", "2023-01-04"])
    })
    result = deduplicate_events(df)
    assert_frame_equal(result, expected)


def test_deduplicate_events_handles_datetime_edge():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "value": ["old", "new"],
        "updated_at": [
            pd.Timestamp("2020-01-01"),
            pd.Timestamp("2023-12-31 23:59:59.999999")
        ]
    })
    expected = pd.DataFrame({
        "event_id": [1],
        "value": ["new"],
        "updated_at": [pd.Timestamp("2023-12-31 23:59:59.999999")]
    })
    result = deduplicate_events(df)
    assert_frame_equal(result, expected)
