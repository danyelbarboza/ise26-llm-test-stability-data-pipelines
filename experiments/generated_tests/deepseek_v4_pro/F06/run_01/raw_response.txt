import pandas as pd
import pytest
from ise26.targets import classify_payment_status


class TestInvalidReferenceDate:
    def test_reference_date_invalid_raises_error(self):
        df = pd.DataFrame({"due_date": ["2023-01-01"], "paid_date": [None], "amount": [100]})
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, reference_date="not-a-date")


class TestOriginalDataFrameUntouched:
    def test_original_df_not_modified(self):
        original = pd.DataFrame({
            "due_date": ["2023-01-01", "2023-02-01"],
            "paid_date": ["2023-01-01", None],
            "amount": [100, 200],
        })
        original_before = original.copy(deep=True)
        classify_payment_status(original, reference_date="2023-02-15")
        pd.testing.assert_frame_equal(original, original_before)


@pytest.mark.parametrize(
    "row,reason",
    [
        ({"due_date": "abc", "paid_date": None, "amount": 100}, "invalid due_date"),
        ({"due_date": "2023-01-01", "paid_date": None, "amount": None}, "invalid amount (NaN)"),
        ({"due_date": "2023-01-01", "paid_date": None, "amount": 0}, "invalid amount (zero)"),
        ({"due_date": "2023-01-01", "paid_date": None, "amount": -5}, "invalid amount (negative)"),
        ({"due_date": "2023-01-01", "paid_date": "abc", "amount": 100}, "provided invalid paid_date"),
        ({"due_date": "2023-01-01", "paid_date": " ", "amount": 100}, "paid_date whitespace -> provided invalid"),
    ],
)
def test_invalid_rows(row, reason):
    """Rows that should be classified as 'invalid' immediately."""
    df = pd.DataFrame([row])
    result = classify_payment_status(df, reference_date="2023-02-01")
    assert result.loc[0, "payment_status"] == "invalid", f"Failed for {reason}"


@pytest.mark.parametrize(
    "due_date,paid_date,amount,reference_date,expected",
    [
        # paid_on_time: paid_date <= due_date
        ("2023-01-15", "2023-01-15", 150, "2023-02-01", "paid_on_time"),
        ("2023-01-15", "2023-01-10", 150, "2023-02-01", "paid_on_time"),
        # paid_late: paid_date > due_date
        ("2023-01-15", "2023-01-20", 150, "2023-02-01", "paid_late"),
        # overdue: unpaid and reference_date > due_date
        ("2023-01-15", None, 100, "2023-02-01", "overdue"),
        ("2023-01-15", pd.NaT, 100, "2023-02-01", "overdue"),
        # pending: unpaid and reference_date <= due_date
        ("2023-02-15", None, 200, "2023-02-01", "pending"),
        ("2023-02-15", "", 200, "2023-02-01", "pending"),  # empty string -> unpaid
    ],
)
def test_standard_classifications(due_date, paid_date, amount, reference_date, expected):
    df = pd.DataFrame([{"due_date": due_date, "paid_date": paid_date, "amount": amount}])
    result = classify_payment_status(df, reference_date=reference_date)
    assert result.loc[0, "payment_status"] == expected


def test_amount_string_numeric():
    df = pd.DataFrame({"due_date": ["2023-01-01"], "paid_date": [None], "amount": ["150.50"]})
    result = classify_payment_status(df, reference_date="2023-02-01")
    assert result.loc[0, "payment_status"] == "overdue"  # unpaid, ref > due


def test_amount_invalid_string():
    df = pd.DataFrame({"due_date": ["2023-01-01"], "paid_date": [None], "amount": ["foo"]})
    result = classify_payment_status(df, reference_date="2023-02-01")
    assert result.loc[0, "payment_status"] == "invalid"


def test_mixed_statuses():
    df = pd.DataFrame({
        "due_date": ["2023-01-01", "2023-05-10", "2023-03-15", "2023-07-01"],
        "paid_date": ["2023-01-01", None, "2023-03-20", "not-a-date"],
        "amount": [100, 200, 150, 50],
    })
    reference_date = "2023-06-01"
    result = classify_payment_status(df, reference_date=reference_date)
    expected = ["paid_on_time", "overdue", "paid_late", "invalid"]
    assert result["payment_status"].tolist() == expected


def test_overwrite_existing_output_column():
    df = pd.DataFrame({
        "due_date": ["2023-01-01"],
        "paid_date": [None],
        "amount": [100],
        "payment_status": ["old_value"],
    })
    result = classify_payment_status(df, reference_date="2023-02-01")
    assert result.loc[0, "payment_status"] == "overdue"
    assert "payment_status" in result.columns


def test_custom_column_names():
    df = pd.DataFrame({
        "date_due": ["2023-01-01"],
        "date_paid": [None],
        "valor": [200],
    })
    reference_date = "2023-02-01"
    result = classify_payment_status(
        df,
        reference_date=reference_date,
        due_date_col="date_due",
        paid_date_col="date_paid",
        amount_col="valor",
        output_col="status",
    )
    assert result.loc[0, "status"] == "overdue"
    assert "status" in result.columns
    assert "payment_status" not in result.columns


def test_reference_date_as_timestamp():
    df = pd.DataFrame({"due_date": ["2023-01-01"], "paid_date": [None], "amount": [100]})
    ref = pd.Timestamp("2023-02-01")
    result = classify_payment_status(df, reference_date=ref)
    assert result.loc[0, "payment_status"] == "overdue"


def test_paid_date_empty_string_is_unpaid():
    """Empty string should be treated as not provided -> unpaid."""
    df = pd.DataFrame({
        "due_date": ["2023-01-01"],
        "paid_date": [""],
        "amount": [100],
    })
    reference_date = "2023-02-01"
    result = classify_payment_status(df, reference_date=reference_date)
    # reference_date > due_date, so overdue
    assert result.loc[0, "payment_status"] == "overdue"
