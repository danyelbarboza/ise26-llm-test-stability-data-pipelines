class TestInvalidProvidedPaidDate:
    def test_provided_unparseable_paid_date(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01"],
            "paid_date": ["no date"],
            "amount": [100],
        })
        result = classify_payment_status(df, valid_ref_date)
        assert result["payment_status"].iloc[0] == "invalid"

    def test_provided_paid_date_nat(self, valid_ref_date):
        # NaT is considered provided? notna() is False for NaT, so it's missing.
        # Strictly speaking, NaT is a missing value, not an invalid provided date.
        # We'll test that NaT is treated as missing (unpaid).
        df = pd.DataFrame({
            "due_date": ["2023-06-01"],
            "paid_date": [pd.NaT],
            "amount": [100],
        })
        result = classify_payment_status(df, valid_ref_date)
        # ref_date(2023-06-15) > due_date -> overdue
        assert result["payment_status"].iloc[0] == "overdue"

    def test_empty_string_paid_date_treated_as_missing(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01"],
            "paid_date": [""],
            "amount": [100],
        })
        result = classify_payment_status(df, valid_ref_date)
        assert result["payment_status"].iloc[0] == "overdue"

        assert result["payment_status"].iloc[0] == "overdue"


class TestDataFrameIntegrity:
    def test_original_df_not_mutated(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01"],
            "paid_date": [None],
            "amount": [100],
        })
        original_columns = df.columns.tolist()
        original_data = df.iloc[0].to_dict()
        result = classify_payment_status(df, valid_ref_date)
        # Original should remain unchanged (no extra column, same values)
        assert df.columns.tolist() == original_columns
        assert df.iloc[0].to_dict() == original_data

    def test_returned_df_has_extra_column(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01"],
            "paid_date": [None],
            "amount": [100],
        })
        result = classify_payment_status(df, valid_ref_date)
        assert len(result.columns) == len(df.columns) + 1

    def test_duplicate_rows_handled(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01", "2023-06-01"],
            "paid_date": [None, "2023-06-01"],
            "amount": [100, 100],
        })
        result = classify_payment_status(df, valid_ref_date)
        expected = ["overdue", "paid_on_time"]
        assert result["payment_status"].tolist() == expected

    def test_empty_dataframe(self, valid_ref_date):
        df = pd.DataFrame({"due_date": [], "paid_date": [], "amount": []})
        result = classify_payment_status(df, valid_ref_date)
        assert result.empty
        assert "payment_status" in result.columns


class TestComprehensiveScenarios:
    def test_mixed_valid_invalid(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01", "invalid", "2023-06-01", "2023-07-01", "2023-06-01"],
            "paid_date": ["2023-06-01", "2023-06-01", None, "2023-07-02", None],
            "amount": [100, 100, 100, 100, -5],
        })
        result = classify_payment_status(df, valid_ref_date)
        # row0: due valid, paid on time -> paid_on_time
        # row1: due invalid -> invalid
        # row2: due valid, unpaid, due < ref -> overdue
        # row3: due valid, paid late (2023-07-02 > 2023-07-01) -> paid_late
        # row4: amount invalid -> invalid
        expected = ["paid_on_time", "invalid", "overdue", "paid_late", "invalid"]
        assert result["payment_status"].tolist() == expected

    def test_nat_in_paid_date_treated_as_unpaid(self, valid_ref_date):
        df = pd.DataFrame({
            "due_date": ["2023-06-01"],
            "paid_date": [pd.NaT],
            "amount": [100],
        })
        result = classify_payment_status(df, valid_ref_date)
        assert result["payment_status"].iloc[0] == "overdue"
