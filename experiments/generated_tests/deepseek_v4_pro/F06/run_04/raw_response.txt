import pandas as pd
import pytest

from ise26.targets import classify_payment_status


# ---------------------------------------------------------------
# Fixtures / helpers
# ---------------------------------------------------------------

@pytest.fixture
def valid_reference_date():
    return pd.Timestamp("2024-01-31")


# ---------------------------------------------------------------
# Positive cases: paid on time, paid late, overdue, pending
# ---------------------------------------------------------------

def test_paid_on_time(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "paid_on_time"


def test_paid_late(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-20"],
        "amount": [200.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "paid_late"


def test_overdue_unpaid(valid_reference_date):
    # reference_date = 2024-01-31, due_date earlier -> overdue
    df = pd.DataFrame({
        "due_date": ["2024-01-20"],
        "paid_date": [pd.NaT],
        "amount": [300.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "overdue"


def test_pending_unpaid(valid_reference_date):
    # reference_date = 2024-01-31, due_date later -> pending
    df = pd.DataFrame({
        "due_date": ["2024-02-10"],
        "paid_date": [None],
        "amount": [400.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "pending"


# ---------------------------------------------------------------
# Invalid amount cases
# ---------------------------------------------------------------

@pytest.mark.parametrize("amount", [
    None,           # missing
    0,              # zero
    -50.0,          # negative
    "abc",          # non-numeric
])
def test_invalid_amount(amount, valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [amount],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "invalid"


# ---------------------------------------------------------------
# Invalid due_date cases
# ---------------------------------------------------------------

@pytest.mark.parametrize("due", [
    None,
    "2024-02-30",   # impossible date -> NaT after coerce
    "invalid",
])
def test_invalid_due_date(due, valid_reference_date):
    df = pd.DataFrame({
        "due_date": [due],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "invalid"


# ---------------------------------------------------------------
# Provided but invalid paid_date -> invalid
# ---------------------------------------------------------------

def test_invalid_paid_date_provided(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["not-a-date"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "invalid"


# ---------------------------------------------------------------
# Genuinely missing paid_date (NaT, None, empty string) -> overdue/pending,
# not invalid.
# ---------------------------------------------------------------

@pytest.mark.parametrize("paid", [
    pd.NaT,
    None,
    "",             # empty string is considered missing
])
def test_missing_paid_date_treated_as_unpaid(paid, valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-02-10"],
        "paid_date": [paid],
        "amount": [100.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    # reference_date (2024-01-31) <= due_date -> pending
    assert result["payment_status"].iloc[0] == "pending"


# ---------------------------------------------------------------
# Reference date validation
# ---------------------------------------------------------------

def test_invalid_reference_date():
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
        classify_payment_status(df, "garbage-date")


# ---------------------------------------------------------------
# Multiple rows / mixed statuses
# ---------------------------------------------------------------

def test_mixed_statuses(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15", "2024-01-15", "2024-02-10"],
        "paid_date": ["2024-01-10", "2024-01-20", None],
        "amount": [100, 200, 300],
    })
    result = classify_payment_status(df, valid_reference_date)
    expected = ["paid_on_time", "paid_late", "pending"]
    assert list(result["payment_status"]) == expected


def test_mixed_invalids_and_valids(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15", "2024-01-15", "2024-01-15"],
        "paid_date": ["not-a-date", pd.NaT, "2024-01-20"],
        "amount": [100, 100, 100],
    })
    result = classify_payment_status(df, valid_reference_date)
    expected = ["invalid", "overdue", "paid_late"]  # unpaid + overdue
    assert list(result["payment_status"]) == expected


# ---------------------------------------------------------------
# Schema and immutability checks
# ---------------------------------------------------------------

def test_original_dataframe_unchanged(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    original = df.copy()
    classify_payment_status(df, valid_reference_date)
    pd.testing.assert_frame_equal(df, original)


def test_return_type_is_copy(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result is not df
    assert "payment_status" in result.columns
    assert "payment_status" not in df.columns


def test_custom_output_column_name(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, valid_reference_date, output_col="status")
    assert "status" in result.columns
    assert result["status"].iloc[0] == "paid_on_time"


def test_output_column_overwrites_existing(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
        "payment_status": ["existing"],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["payment_status"].iloc[0] == "paid_on_time"


def test_extra_columns_are_preserved(valid_reference_date):
    df = pd.DataFrame({
        "due_date": ["2024-01-15"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
        "extra": ["value"],
    })
    result = classify_payment_status(df, valid_reference_date)
    assert result["extra"].iloc[0] == "value"
    assert "payment_status" in result.columns


def test_missing_required_column_raises():
    df = pd.DataFrame({
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    with pytest.raises(KeyError):
        classify_payment_status(df, "2024-01-31")
