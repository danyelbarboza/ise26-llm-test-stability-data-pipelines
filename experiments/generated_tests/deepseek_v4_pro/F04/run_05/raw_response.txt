import pandas as pd
import pytest
from pandas.testing import assert_frame_equal
from ise26.targets import join_customers_orders


def test_basic_full_outer_join():
    customers = pd.DataFrame({"customer_id": [1, 2, 3], "name": ["Alice", "Bob", "Charlie"]})
    orders = pd.DataFrame(
        {"customer_id": [1, 2, 4], "amount": [100, 200, 300], "order_id": [10, 11, 12]}
    )

    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame(
        {
            "customer_id": [1, 2, 3, 4],
            "name": ["Alice", "Bob", "Charlie", pd.NA],
            "amount": [100, 200, pd.NA, 300],
            "order_id": [10, 11, pd.NA, 12],
            "record_status": [
                "matched",
                "matched",
                "customer_without_order",
                "order_without_customer",
            ],
        }
    )
    assert_frame_equal(result, expected)


def test_null_keys_never_matched():
    customers = pd.DataFrame({"customer_id": [1.0, None], "name": ["Alice", "Bob"]})
    orders = pd.DataFrame(
        {"customer_id": [1.0, None], "amount": [100, 200], "order_id": [10, 11]}
    )

    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame(
        {
            "customer_id": [1.0, None, None],
            "name": ["Alice", "Bob", pd.NA],
            "amount": [100, pd.NA, 200],
            "order_id": [10, pd.NA, 11],
            "record_status": [
                "matched",
                "customer_without_order",
                "order_without_customer",
            ],
        }
    )
    assert_frame_equal(result, expected)
