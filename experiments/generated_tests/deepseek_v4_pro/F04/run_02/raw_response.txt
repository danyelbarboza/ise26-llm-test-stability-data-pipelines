import pandas as pd
import pytest
from pandas.testing import assert_frame_equal

from ise26.targets import join_customers_orders


def test_basic_matched_join():
    """All keys present in both dataframes produce 'matched' status."""
    customers = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "name": ["Alice", "Bob", "Charlie"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "order_id": [101, 102, 103],
    })

    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "name": ["Alice", "Bob", "Charlie"],
        "order_id": [101, 102, 103],
        "record_status": ["matched", "matched", "matched"],
    })
    # Deterministic order based on source order
    assert_frame_equal(result, expected)


def test_unmatched_customers_and_orders():
    """Left-only and right-only rows get correct statuses."""
    customers = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "name": ["Alice", "Bob", "Charlie"],
    })
    orders = pd.DataFrame({
        "customer_id": [2, 3, 4],
        "order_id": [102, 103, 104],
    })

    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame({
        "customer_id": [1, 2, 3, 4],
        "name": ["Alice", "Bob", "Charlie", pd.NA],
        "order_id": [pd.NA, 102, 103, 104],
        "record_status": [
            "customer_without_order",
            "matched",
            "matched",
            "order_without_customer",
        ],
    })
    assert_frame_equal(result, expected, check_dtype=False)


def test_null_keys_not_matched():
    """Rows with null join keys are never classified as 'matched'."""
    customers = pd.DataFrame({
        "customer_id": [1, None, 3],
        "name": ["Alice", "Bob", "Charlie"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, None, 3],
        "order_id": [101, 102, 103],
    })

    result = join_customers_orders(customers, orders)

    # Explicit null-key rows are added separately, never joined together
    expected = pd.DataFrame({
        "customer_id": [1, 3, None, None],
        "name": ["Alice", "Charlie", "Bob", pd.NA],
        "order_id": [101, 103, pd.NA, 102],
        "record_status": [
            "matched",
            "matched",
            "customer_without_order",
            "order_without_customer",
        ],
    })
    assert_frame_equal(result, expected, check_dtype=False)


def test_duplicates_handled():
    """Duplicated keys produce cross product rows with 'matched' status."""
    customers = pd.DataFrame({
        "customer_id": [1, 1, 2],
        "name": ["Alice", "Alice2", "Bob"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 1, 2],
        "order_id": [101, 102, 103],
    })

    result = join_customers_orders(customers, orders)

    # Factor combinations; order deterministic by source positions
    expected = pd.DataFrame({
        "customer_id": [1, 1, 1, 1, 2],
        "name": ["Alice", "Alice2", "Alice", "Alice2", "Bob"],
        "order_id": [101, 101, 102, 102, 103],
        "record_status": ["matched"] * 5,
    })
    assert_frame_equal(result, expected)


def test_empty_inputs():
    """Empty dataframes produce empty output with expected columns."""
    customers = pd.DataFrame(columns=["customer_id", "name"])
    orders = pd.DataFrame(columns=["customer_id", "order_id"])

    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame({
        "customer_id": pd.Series(dtype="object"),
        "name": pd.Series(dtype="object"),
        "order_id": pd.Series(dtype="object"),
        "record_status": pd.Series(dtype="object"),
    })
    assert_frame_equal(result, expected, check_index_type=False)


def test_original_data_unchanged():
    """The input DataFrames must not be mutated."""
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", "Bob"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 2],
        "order_id": [101, 102],
    })
    customers_before = customers.copy()
    orders_before = orders.copy()

    _ = join_customers_orders(customers, orders)

    assert_frame_equal(customers, customers_before)
    assert_frame_equal(orders, orders_before)


def test_index_reset():
    """Output must have a clean default integer index."""
    customers = pd.DataFrame({
        "customer_id": [10, 20],
        "name": ["A", "B"],
    })
    orders = pd.DataFrame({
        "customer_id": [20, 30],
        "order_id": [200, 300],
    })
    result = join_customers_orders(customers, orders)
    expected_index = pd.RangeIndex(start=0, stop=3, step=1)
    assert result.index.equals(expected_index)


def test_custom_key_name():
    """Works when a non-default join key is supplied."""
    customers = pd.DataFrame({
        "cust_id": [1, 2],
        "name": ["X", "Y"],
    })
    orders = pd.DataFrame({
        "cust_id": [2, 3],
        "order_id": [10, 20],
    })
    result = join_customers_orders(customers, orders, customer_key="cust_id")

    expected = pd.DataFrame({
        "cust_id": [1, 2, 3],
        "name": ["X", "Y", pd.NA],
        "order_id": [pd.NA, 10, 20],
        "record_status": [
            "customer_without_order",
            "matched",
            "order_without_customer",
        ],
    })
    assert_frame_equal(result, expected, check_dtype=False)


def test_different_column_sets():
    """Columns from both sides are preserved, missing ones become NA."""
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", "Bob"],
        "since": [2020, 2021],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 3],
        "order_id": [101, 103],
        "amount": [50, 75],
    })
    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "name": ["Alice", "Bob", pd.NA],
        "since": [2020.0, 2021.0, pd.NA],   # float due to NA
        "order_id": [101, pd.NA, 103],
        "amount": [50.0, pd.NA, 75.0],
        "record_status": [
            "matched",
            "customer_without_order",
            "order_without_customer",
        ],
    })
    # Expect float columns for integer columns with NA (pandas inference)
    assert_frame_equal(result, expected, check_dtype=False)


def test_deterministic_order():
    """Result order is stable and respects original row order within groups."""
    customers = pd.DataFrame({
        "customer_id": [2, 1, 3],
        "name": ["Second", "First", "Third"],
    })
    orders = pd.DataFrame({
        "customer_id": [3, 2, 1],
        "order_id": [300, 200, 100],
    })
    result = join_customers_orders(customers, orders)
    # Expected order: original customer rows [2,1,3] matched with corresponding orders;
    # joins should produce rows in that order with each customer matched to its order.
    expected = pd.DataFrame({
        "customer_id": [2, 1, 3],
        "name": ["Second", "First", "Third"],
        "order_id": [200, 100, 300],
        "record_status": ["matched"] * 3,
    })
    assert_frame_equal(result, expected)


def test_missing_key_column_raises():
    """If customer_key not in both DataFrames, a KeyError is expected."""
    customers = pd.DataFrame({"customer_id": [1]})
    orders = pd.DataFrame({"order_id": [100]})
    with pytest.raises(KeyError):
        join_customers_orders(customers, orders)


def test_mixed_null_and_nonnull_keys():
    """Null keys from both sides stay separate; non-null keys match."""
    customers = pd.DataFrame({
        "customer_id": [1, None],
        "name": ["Alice", "Unknown"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, None],
        "order_id": [101, 999],
    })
    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame({
        "customer_id": [1, None, None],
        "name": ["Alice", "Unknown", pd.NA],
        "order_id": [101, pd.NA, 999],
        "record_status": [
            "matched",
            "customer_without_order",
            "order_without_customer",
        ],
    })
    assert_frame_equal(result, expected, check_dtype=False)
