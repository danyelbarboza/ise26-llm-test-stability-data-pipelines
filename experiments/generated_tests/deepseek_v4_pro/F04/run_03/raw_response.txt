def test_full_outer_join_basic(customers_basic, orders_basic):
    """All three statuses appear: matched, customer_only, order_only."""
    result = join_customers_orders(customers_basic, orders_basic)

    # Expected order: [customer_id, name, city, order_id, amount, record_status]
    expected = pd.DataFrame({
        "customer_id": [1, 2, 3, 4],
        "name": ["Alice", "Bob", "Charlie", pd.NA],
        "city": ["NY", "LA", "CH", pd.NA],
        "order_id": [pd.NA, 100, 101, 102],
        "amount": [pd.NA, 200.0, 150.0, 99.0],
        "record_status": [
            "customer_without_order",
            "matched",
            "matched",
            "order_without_customer",
        ],
    })

    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_output_columns_union(customers_basic, orders_basic):
    """Result contains all columns from both frames plus record_status."""
    result = join_customers_orders(customers_basic, orders_basic)
    columns = set(result.columns)
    expected_cols = set(customers_basic.columns) | set(orders_basic.columns) | {"record_status"}
    assert columns == expected_cols


def test_no_internal_columns():
    """The return columns must not leak helper columns like _sort_order."""
    customers = pd.DataFrame({"customer_id": [1]})
    orders = pd.DataFrame({"customer_id": [2]})
    result = join_customers_orders(customers, orders)
    assert "_sort_order" not in result.columns
    assert "_customer_source_order" not in result.columns
    assert "_order_source_order" not in result.columns
    assert "_merge" not in result.columns


def test_default_customer_key(customers_basic, orders_basic):
    """Function works with the default customer_key='customer_id'."""
    result = join_customers_orders(customers_basic, orders_basic)
    assert "customer_id" in result.columns
    # Quick sanity: presence of all statuses
    statuses = set(result["record_status"])
    assert statuses == {"matched", "customer_without_order", "order_without_customer"}

# Null key handling
# ------------------------------------------------------------------------------

def test_null_keys_never_matched():
    """Rows with null join keys are never classified as 'matched'."""
    customers = pd.DataFrame({
        "customer_id": [1, None],
        "name": ["Alice", "Bob"],
    })
    orders = pd.DataFrame({
        "customer_id": [None, 1],
        "order_id": [100, 101],
    })

    result = join_customers_orders(customers, orders)

    # Row with null customer_id on either side must be left_only / right_only
    mask_null = result["customer_id"].isna()
    assert (result.loc[mask_null, "record_status"] != "matched").all()
    # Specifically: first join row should be matched (id=1), then two null rows
    expected_status = pd.array(["matched", "customer_without_order", "order_without_customer"])
    pd.testing.assert_series_equal(
        result["record_status"], pd.Series(expected_status), check_names=False
    )


def test_null_keys_on_one_side_only():
    """Left-only null keys appear as customer_without_order, right-only as order_without_customer."""
    customers = pd.DataFrame({"customer_id": [None], "name": ["Ghost"]})
    orders = pd.DataFrame({"customer_id": [None], "order_id": [99]})

    result = join_customers_orders(customers, orders)
    assert len(result) == 2
    assert result["record_status"].tolist() == [
        "customer_without_order",
        "order_without_customer",
    ]
    # Key is NA for both
    assert result["customer_id"].isna().all()


def test_mixed_null_and_valid_keys():
    """A null key on one side should not match a non-null key on the other side."""
    customers = pd.DataFrame({"customer_id": [1, None], "x": ["a", "b"]})
    orders = pd.DataFrame({"customer_id": [2, None], "y": [10, 20]})

    result = join_customers_orders(customers, orders)

    # Expected rows: customer 1 (no match) -> customer_without_order
    #               customer null (no match) -> customer_without_order
    #               order 2 (no match) -> order_without_customer
    #               order null (no match) -> order_without_customer
    assert len(result) == 4
    statuses = result["record_status"].values
    assert list(statuses) == [
        "customer_without_order",
        "customer_without_order",
        "order_without_customer",
        "order_without_customer",
    ]

def test_empty_customers():
    """Empty customers yields only order rows (with their data)."""
    customers = pd.DataFrame(columns=["customer_id", "name"])
    orders = pd.DataFrame({"customer_id": [1], "order_id": [100]})
    result = join_customers_orders(customers, orders)
    assert len(result) == 1
    assert result["record_status"].iloc[0] == "order_without_customer"
    pd.testing.assert_series_equal(
        result["customer_id"], pd.Series([1], name="customer_id")
    )


def test_empty_orders():
    """Empty orders yields only customer rows (with their data)."""
    customers = pd.DataFrame({"customer_id": [1], "name": ["Alice"]})
    orders = pd.DataFrame(columns=["customer_id", "order_id"])
    result = join_customers_orders(customers, orders)
    assert len(result) == 1
    assert result["record_status"].iloc[0] == "customer_without_order"


def test_both_empty():
    """Both empty returns an empty frame with correct schema."""
    customers = pd.DataFrame(columns=["customer_id"])
    orders = pd.DataFrame(columns=["customer_id"])
    result = join_customers_orders(customers, orders)
    assert len(result) == 0
    assert set(result.columns) == {"customer_id", "record_status"}


# ------------------------------------------------------------------------------
# Duplicates
# ------------------------------------------------------------------------------

def test_duplicate_keys():
    """Natural outer join behaviour: duplicates produce Cartesian product of matches."""
    customers = pd.DataFrame({
        "customer_id": [1, 1],
        "name": ["A", "B"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 1],
        "order_id": [10, 20],
    })

    result = join_customers_orders(customers, orders)
    assert len(result) == 4  # 2x2
    assert (result["record_status"] == "matched").all()


def test_duplicates_on_one_side():
    """Duplicates only on one side still produce outer join behaviour."""
    customers = pd.DataFrame({"customer_id": [1, 1], "x": [1, 2]})
    orders = pd.DataFrame({"customer_id": [2], "y": [10]})

    result = join_customers_orders(customers, orders)

    # Left duplicates -> two customer_without_order rows, one order_without_customer
    assert len(result) == 3
    status = result["record_status"].tolist()
    assert status.count("customer_without_order") == 2
    assert status.count("order_without_customer") == 1

def test_extra_columns():
    """Columns present only in one side are filled with pd.NA on the other."""
    customers = pd.DataFrame({
        "customer_id": [1],
        "name": ["Alice"],
        "segment": ["VIP"],
    })
    orders = pd.DataFrame({
        "customer_id": [2],
        "order_id": [100],
        "channel": ["web"],
    })

    result = join_customers_orders(customers, orders)

    # First row: customer 1 (left only) -> segment present, order_id NA, channel NA
    # Second row: order 2 (right only) -> name NA, order_id present, channel present
    expected = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", pd.NA],
        "segment": ["VIP", pd.NA],
        "order_id": [pd.NA, 100],
        "channel": [pd.NA, "web"],
        "record_status": ["customer_without_order", "order_without_customer"],
    })

    pd.testing.assert_frame_equal(result, expected.reindex(columns=result.columns), check_dtype=False)


def test_record_status_values_only_expected():
    """record_status column contains only the three defined labels."""
    customers = pd.DataFrame({"customer_id": range(5)})
    orders = pd.DataFrame({"customer_id": range(3, 8)})
    result = join_customers_orders(customers, orders)
    allowed = {"matched", "customer_without_order", "order_without_customer"}
    assert set(result["record_status"].unique()) <= allowed
