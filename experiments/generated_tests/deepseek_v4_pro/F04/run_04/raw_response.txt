from ise26.targets import join_customers_orders

import pandas as pd
import pytest

def test_basic_matched_join():
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", "Bob"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 2],
        "amount": [100, 200],
    })

    result = join_customers_orders(customers, orders)

    assert isinstance(result, pd.DataFrame)
    assert list(result.columns) == ["customer_id", "name", "amount", "record_status"]
    assert len(result) == 2
    assert result["record_status"].tolist() == ["matched", "matched"]
    assert result["customer_id"].tolist() == [1, 2]
    assert result["name"].tolist() == ["Alice", "Bob"]
    assert result["amount"].tolist() == [100, 200]
    assert result.index.tolist() == [0, 1]  # reset index

def test_unmatched_rows():
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", "Bob"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 3],
        "amount": [100, 300],
    })

    result = join_customers_orders(customers, orders)

    # Expect: matched id=1, customer_without_order id=2, order_without_customer id=3
    assert len(result) == 3
    statuses = result["record_status"].tolist()
    assert statuses == ["matched", "customer_without_order", "order_without_customer"]

    # Check values
    row_id1 = result[result["customer_id"] == 1].iloc[0]
    assert row_id1["name"] == "Alice"
    assert row_id1["amount"] == 100

    row_id2 = result[result["customer_id"] == 2].iloc[0]
    assert row_id2["name"] == "Bob"
    assert pd.isna(row_id2["amount"])

    row_id3 = result[result["customer_id"] == 3].iloc[0]
    assert pd.isna(row_id3["name"])
    assert row_id3["amount"] == 300
