import pandas as pd
import pytest
from ise26.targets import join_customers_orders


def test_basic_full_outer_join():
    """Matched, unmatched customer, and unmatched order rows appear with correct statuses."""
    customers = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "name": ["Alice", "Bob", "Carol"],
    })
    orders = pd.DataFrame({
        "order_id": [101, 102, 103],
        "customer_id": [1, 3, 4],
        "product": ["apple", "banana", "cherry"],
    })

    result = join_customers_orders(customers, orders)

    # Check presence of all rows
    assert len(result) == 4

    # Check record_status distribution
    assert set(result["record_status"]) == {
        "matched",
        "customer_without_order",
        "order_without_customer",
    }

    matched = result[result["record_status"] == "matched"]
    assert len(matched) == 2
    assert set(matched["customer_id"]) == {1, 3}

    cust_no_order = result[result["record_status"] == "customer_without_order"]
    assert len(cust_no_order) == 1
    assert cust_no_order.iloc[0]["customer_id"] == 2
    assert pd.isna(cust_no_order.iloc[0]["order_id"])

    order_no_cust = result[result["record_status"] == "order_without_customer"]
    assert len(order_no_cust) == 1
    assert order_no_cust.iloc[0]["customer_id"] == 4
    assert pd.isna(order_no_cust.iloc[0]["name"])

    # Deterministic order: id1 matched, id2 unmatched, id3 matched, id4 unmatched
    expected_order_customer_ids = [1, 2, 3, 4]
    actual = result["customer_id"].tolist()
    assert actual == expected_order_customer_ids

    # Index is reset
    assert result.index.equals(pd.RangeIndex(len(result)))


def test_original_dataframes_unchanged():
    """Calling the function must not modify the input DataFrames."""
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", "Bob"],
    })
    orders = pd.DataFrame({
        "order_id": [101],
        "customer_id": [1],
        "product": ["apple"],
    })

    cust_original = customers.copy()
    ord_original = orders.copy()

    _ = join_customers_orders(customers, orders)

    pd.testing.assert_frame_equal(customers, cust_original)
    pd.testing.assert_frame_equal(orders, ord_original)


def test_null_keys_never_matched():
    """Rows with null join keys are never classified as 'matched'."""
    customers = pd.DataFrame({
        "customer_id": [1, None, None],
        "name": ["Alice", "Bob (null)", "Carol (null)"],
    })
    orders = pd.DataFrame({
        "order_id": [101, 102],
        "customer_id": [1, None],
        "product": ["apple", "unknown"],
    })

    result = join_customers_orders(customers, orders)

    # No row should be 'matched' if it originated from a null key
    matched = result[result["record_status"] == "matched"]
    assert len(matched) == 1  # only the row with customer_id=1
    assert matched.iloc[0]["customer_id"] == 1

    # Customer rows with null key should be "customer_without_order"
    cust_null = result[
        (result["customer_id"].isna()) & (result["record_status"] == "customer_without_order")
    ]
    assert len(cust_null) == 2  # Bob and Carol

    # Order rows with null key should be "order_without_customer"
    ord_null = result[
        (result["customer_id"].isna()) & (result["record_status"] == "order_without_customer")
    ]
    assert len(ord_null) == 1  # the "unknown" order


def test_duplicate_keys_product():
    """Multiple rows with the same join key produce a full outer product."""
    customers = pd.DataFrame({
        "customer_id": [1, 1],
        "name": ["Alice", "Alice duplicate"],
    })
    orders = pd.DataFrame({
        "order_id": [101, 102],
        "customer_id": [1, 1],
        "product": ["apple", "banana"],
    })

    result = join_customers_orders(customers, orders)

    # 2 customers × 2 orders = 4 matched rows
    matched = result[result["record_status"] == "matched"]
    assert len(matched) == 4

    # The output should contain all combinations
    expected_names = ["Alice", "Alice", "Alice duplicate", "Alice duplicate"]
    expected_order_ids = [101, 102, 101, 102]
    assert matched["name"].tolist() == expected_names
    assert matched["order_id"].tolist() == expected_order_ids


def test_empty_inputs():
    """Handle empty customers or orders gracefully."""
    customers_empty = pd.DataFrame(columns=["customer_id", "name"])
    orders = pd.DataFrame({
        "order_id": [1],
        "customer_id": [10],
        "product": ["x"],
    })

    result = join_customers_orders(customers_empty, orders)
    assert len(result) == 1
    assert result.iloc[0]["record_status"] == "order_without_customer"
    assert result.iloc[0]["customer_id"] == 10

    customers = pd.DataFrame({
        "customer_id": [1],
        "name": ["Alice"],
    })
    orders_empty = pd.DataFrame(columns=["order_id", "customer_id", "product"])

    result2 = join_customers_orders(customers, orders_empty)
    assert len(result2) == 1
    assert result2.iloc[0]["record_status"] == "customer_without_order"
    assert result2.iloc[0]["customer_id"] == 1

    # Both empty
    result3 = join_customers_orders(customers_empty, orders_empty)
    assert len(result3) == 0
    assert list(result3.columns) == ["customer_id", "name", "order_id", "product", "record_status"]


def test_deterministic_output():
    """Multiple calls with identical inputs produce identical output (same row order)."""
    customers = pd.DataFrame({
        "customer_id": [3, 2, 1],
        "name": ["a", "b", "c"],
    })
    orders = pd.DataFrame({
        "order_id": [10, 20],
        "customer_id": [1, 3],
        "product": ["x", "y"],
    })

    result1 = join_customers_orders(customers, orders)
    result2 = join_customers_orders(customers, orders)
    pd.testing.assert_frame_equal(result1, result2)

    # Check a specific deterministic order:
    # Original order: customers [3,2,1] -> source_order 0,1,2; orders [10->1,20->3] -> source_order 0,1
    # Merged: customer_id=1 matched (_sort_order=2, _customer=2, _order=0), =3 matched (_sort_order=0, _cust=0, _order=1)
    # Left only: customer_id=2 (_sort_order=1, _cust=1, _order=NaN)
    # Sorted: first _sort_order=0 (id3 matched), then _sort_order=1 (id2 unmatched), then _sort_order=2 (id1 matched)
    # Inside _sort_order=2, _customer=2, _order=0, record_status='matched'. Tied resolved by record_status stable.
    expected_customer_ids = [3, 2, 1]
    assert result1["customer_id"].tolist() == expected_customer_ids


def test_column_overlap_beyond_join_key():
    """If DataFrames share extra columns, Pandas merge suffixes apply."""
    customers = pd.DataFrame({
        "customer_id": [1],
        "info": ["cust_info"],
        "extra": ["cust_extra"],
    })
    orders = pd.DataFrame({
        "order_id": [101],
        "customer_id": [1],
        "info": ["ord_info"],
        "extra": ["ord_extra"],
    })

    result = join_customers_orders(customers, orders)

    # Overlapping columns 'info' and 'extra' should appear with suffixes
    assert "info_x" in result.columns or "info_y" in result.columns
    assert "extra_x" in result.columns or "extra_y" in result.columns

    # The key column 'customer_id' appears only once
    assert list(result.columns).count("customer_id") == 1
