import pytest
import pandas as pd
import numpy as np
from ise26.targets import calculate_monthly_revenue


def test_basic_aggregation():
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-10", "2023-02-01", "2023-02-02"],
        "amount": [100, 200, 150, 250],
        "status": ["completed", "completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [300, 400]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_canceled_statuses_ignored():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04", "2023-01-05"],
        "amount": [100, 200, 300, 400, 500],
        "status": ["cancelled", "CANCELED", "  cancelled  ", "Cancelado", "CANCELADO"]
    })
    result = calculate_monthly_revenue(df)
    assert result.empty


def test_invalid_dates_ignored():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "not-a-date", None, "2023-01-10", ""],
        "amount": [100, 200, 300, 400, 500],
        "status": ["completed"] * 5
    })
    result = calculate_monthly_revenue(df)
    # only the two valid dates should count
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [500]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_missing_amounts_treated_as_zero():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "amount": [100, np.nan, "invalid"],
        "status": ["completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [100]  # only the first amount counted, others become 0
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert list(result.columns) == ["month", "revenue"]
    assert result.empty


def test_all_canceled():

        "amount": [100, 200],
        "status": ["cancelled", "cancelado"]

def test_single_month():
    df = pd.DataFrame({
        "order_date": ["2023-05-10", "2023-05-15"],
        "amount": [50, 150],
        "status": ["completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-05"],
        "revenue": [200]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_multiple_years_sorting():
    df = pd.DataFrame({
        "order_date": ["2021-12-01", "2022-01-01", "2022-02-01", "2023-01-01"],
        "amount": [10, 20, 30, 40],
        "status": ["completed"] * 4
    })
    result = calculate_monthly_revenue(df)
    expected_months = ["2021-12", "2022-01", "2022-02", "2023-01"]
    assert list(result["month"]) == expected_months


def test_duplicate_months():
    df = pd.DataFrame({
        "order_date": ["2023-03-01", "2023-03-05", "2023-04-01"],
        "amount": [100, 200, 300],
        "status": ["completed"] * 3
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-03", "2023-04"],
        "revenue": [300, 300]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_mixed_scenario():
    df = pd.DataFrame({
        "order_date": [
            "2023-01-01", "2023-01-05", "2023-02-10", "2023-02-11",
            "2023-02-12", "2023-03-15", "2023-03-16", "invalid-date",
            "2023-04-01"
        ],
        "amount": [100, 50.5, 200, np.nan, "n/a", 75, 25, 10, 0],
        "status": [
            "completed", "cancelled", "completed", "completed",
            "   cancelado   ", "completed", "completed", "completed", "completed"
        ]
    })
    result = calculate_monthly_revenue(df)
    # valid months:
    # 2023-01: 100 (50.5 from cancelled row ignored) -> 100
    # 2023-02: 200 (NaN treated as 0, cancelado row ignored) -> 200
    # 2023-03: 75+25 = 100
    # 2023-04: 0
    # invalid date row ignored
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02", "2023-03", "2023-04"],
        "revenue": [100.0, 200.0, 100.0, 0.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)
