import pytest
import pandas as pd
from ise26.targets import calculate_monthly_revenue


def test_valid_orders_aggregation():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-05", "2023-01-10", "2023-02-03", "2023-02-20"],
            "amount": [100, 200, 50, 75],
            "status": ["completed", "shipped", "completed", "shipped"],
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {"month": ["2023-01", "2023-02"], "revenue": [300, 125]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_canceled_orders_ignored():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01", "2023-01-02", "2023-02-01", "2023-02-02"],
            "amount": [10, 20, 30, 40],
            "status": ["completed", "cancelled", "completed", "CANCELED"],
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {"month": ["2023-01", "2023-02"], "revenue": [10, 30]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_canceled_status_variations():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04"],
            "amount": [1, 2, 3, 4],
            "status": ["  cancelado ", "Cancelled", "Canceled", "CANCELADO"],
        }
    )
    # All should be ignored
    result = calculate_monthly_revenue(df)
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]


def test_invalid_dates_ignored():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01", "not-a-date", "2023-02-01", ""],
            "amount": [100, 200, 300, 400],
            "status": ["completed"] * 4,
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {"month": ["2023-01", "2023-02"], "revenue": [100, 300]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_missing_amounts_treated_as_zero():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01", "2023-01-02", "2023-02-01"],
            "amount": [100, None, 200],
            "status": "completed",
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {"month": ["2023-01", "2023-02"], "revenue": [100, 200]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_nan_amount_column():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01"],
            "amount": [float("nan")],
            "status": "completed",
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {"month": ["2023-01"], "revenue": [0.0]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_amounts_treated_as_zero():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01", "2023-01-01", "2023-01-01"],
            "amount": [100, "abc", "   "],
            "status": "completed",
        }
    )
    result = calculate_monthly_revenue(df)
    # "abc" and whitespace become 0.0 after coerce+fillna
    expected = pd.DataFrame(
        {"month": ["2023-01"], "revenue": [100.0]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_empty_dataframe():
    df = pd.DataFrame(
        {"order_date": [], "amount": [], "status": []}
    )
    result = calculate_monthly_revenue(df)
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]


def test_all_canceled():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-01", "2023-01-02"],
            "amount": [10, 20],
            "status": ["cancelled", "cancelado"],
        }
    )
    result = calculate_monthly_revenue(df)
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]


def test_single_order_month_format():
    df = pd.DataFrame(
        {
            "order_date": ["2023-12-31"],
            "amount": [999],
            "status": "completed",
        }
    )
    result = calculate_monthly_revenue(df)
    assert result.iloc[0]["month"] == "2023-12"
    assert result.iloc[0]["revenue"] == 999


def test_output_sorted_by_month():
    df = pd.DataFrame(
        {
            "order_date": ["2023-03-01", "2023-01-01", "2023-02-01", "2022-12-15"],
            "amount": [1, 2, 3, 4],
            "status": "completed",
        }
    )
    result = calculate_monthly_revenue(df)
    months = result["month"].tolist()
    assert months == sorted(months)


def test_custom_column_names():
    df = pd.DataFrame(
        {
            "data": ["2023-01-05", "2023-01-10"],
            "valor": [10, 20],
            "situacao": ["completed", "completed"],
        }
    )
    result = calculate_monthly_revenue(
        df, date_col="data", amount_col="valor", status_col="situacao"
    )
    expected = pd.DataFrame(
        {"month": ["2023-01"], "revenue": [30]}
    )
    pd.testing.assert_frame_equal(result, expected)


def test_mixed_scenarios():
    df = pd.DataFrame(
        {
            "order_date": [
                "2023-01-01",   # valid, completed
                "2023-01-02",   # valid, canceled -> ignored
                "bad-date",     # invalid -> ignored
                "2023-02-01",   # valid, None amount -> zero
                "2023-02-02",   # valid, string amount -> zero
                "2023-02-03",   # valid, cancelado -> ignored
            ],
            "amount": [100, 200, 300, None, "xyz", 400],
            "status": [
                "completed",
                " cancelled ",
                "completed",
                "completed",
                "completed",
                "Cancelado",
            ],
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {"month": ["2023-01", "2023-02"], "revenue": [100.0, 0.0]}
    )
    pd.testing.assert_frame_equal(result, expected)
