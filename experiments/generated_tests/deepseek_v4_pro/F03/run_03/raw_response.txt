import pytest
import pandas as pd
from ise26.targets import calculate_monthly_revenue


def test_basic_grouping_two_months():
    """Valid orders are aggregated and sorted by month."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-20", "2023-02-18"],
        "amount": [100, 200, 300],
        "status": ["completed", "completed", "completed"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [300, 300],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_canceled_statuses_ignored():
    """Rows with cancelled, canceled or cancelado in any case are excluded."""
    df = pd.DataFrame({
        "order_date": ["2023-03-01", "2023-03-02", "2023-03-03"],
        "amount": [10, 20, 30],
        "status": ["CANCELLED", "canceled ", "CANCELADO"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(columns=["month", "revenue"]).astype({"month": object, "revenue": float})
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_invalid_dates_ignored():
    """Rows with unparseable dates are excluded."""
    df = pd.DataFrame({
        "order_date": ["not-a-date", None, "2023-04-01"],
        "amount": [100, 200, 300],
        "status": ["completed", "completed", "completed"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-04"],
        "revenue": [300],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_invalid_amounts_treated_as_zero():
    """Non‑numeric amounts become 0 and contribute nothing to revenue."""
    df = pd.DataFrame({
        "order_date": ["2023-05-01", "2023-05-02", "2023-05-03"],
        "amount": ["abc", None, 50],
        "status": ["completed", "completed", "completed"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-05"],
        "revenue": [50],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_empty_input_returns_empty_result():
    """An empty DataFrame yields an empty result with the expected columns."""
    df = pd.DataFrame({"order_date": [], "amount": [], "status": []})
    result = calculate_monthly_revenue(df)
    assert list(result.columns) == ["month", "revenue"]
    assert len(result) == 0


def test_all_rows_filtered_out():
    """When every row is cancelled (or invalid), result is empty."""
    df = pd.DataFrame({
        "order_date": ["2023-06-01", "2023-06-02"],
        "amount": [100, 200],
        "status": ["cancelled", "cancelado"],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0
    assert list(result.columns) == ["month", "revenue"]


def test_status_normalization_handles_spaces_and_case():
    """Leading/trailing spaces and case variations still identify cancelled orders."""
    df = pd.DataFrame({
        "order_date": ["2023-07-01"],
        "amount": [10],
        "status": ["  Cancelled  "],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0


def test_portuguese_canceled():
    """'Cancelado' in any case is treated as cancelled."""
    df = pd.DataFrame({
        "order_date": ["2023-08-01"],
        "amount": [15],
        "status": ["cAnCeLaDo"],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0


def test_sort_by_month_ascending():
    """Output is sorted lexicographically by 'month' (YYYY‑MM)."""
    df = pd.DataFrame({
        "order_date": ["2023-12-01", "2023-01-01", "2023-06-01"],
        "amount": [1, 2, 3],
        "status": ["completed"] * 3,
    })
    result = calculate_monthly_revenue(df)
    months = result["month"].tolist()
    assert months == sorted(months)


def test_input_not_mutated():
    """The original DataFrame is left unchanged after the call."""
    df = pd.DataFrame({
        "order_date": ["2023-09-01"],
        "amount": [42],
        "status": ["completed"],
        "extra_col": [99],
    })
    original = df.copy()
    _ = calculate_monthly_revenue(df)
    pd.testing.assert_frame_equal(df, original)


def test_month_format_is_yyyy_mm():
    """The 'month' column contains strings in 'YYYY-MM' format."""
    df = pd.DataFrame({
        "order_date": ["2023-11-15"],
        "amount": [10],
        "status": ["completed"],
    })
    result = calculate_monthly_revenue(df)
    assert result["month"].iloc[0] == "2023-11"


def test_zero_amount_orders_yield_zero_revenue():
    """All amounts zero still produce a row for the month with revenue 0."""
    df = pd.DataFrame({
        "order_date": ["2024-01-05", "2024-01-10"],
        "amount": [0, 0.0],
        "status": ["completed", "completed"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2024-01"],
        "revenue": [0],
    })
    # Allow float/int difference
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_mixed_scenario():
    """Combination of valid, cancelled, invalid date, and zero amount rows."""
    df = pd.DataFrame({
        "order_date": [
            "2024-02-01",  # valid, completed, 100
            "2024-02-10",  # valid, CANCELLED, 50  -> ignored
            "bad-date",    # invalid date, completed, 200 -> ignored
            "2024-02-15",  # valid, completed, "garbage" -> 0
            "2024-03-01",  # valid, completed, 300
        ],
        "amount": [100, 50, 200, "garbage", 300],
        "status": ["completed", "CANCELLED", "completed", "completed", "completed"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2024-02", "2024-03"],
        "revenue": [100, 300],  # 100 + 0 for Feb, 300 for Mar
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)
