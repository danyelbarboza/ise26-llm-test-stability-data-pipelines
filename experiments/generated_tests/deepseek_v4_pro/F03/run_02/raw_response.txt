import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue


def test_basic_aggregation():
    df = pd.DataFrame({
        "order_date": ["2023-01-15", "2023-01-20", "2023-02-10"],
        "amount": [100, 200, 300],
        "status": ["completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [300.0, 300.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert list(result.columns) == ["month", "revenue"]
    assert len(result) == 0


def test_all_canceled():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02"],
        "amount": [100, 200],
        "status": ["cancelled", "canceled"]
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0


def test_all_invalid_dates():
    df = pd.DataFrame({
        "order_date": ["not-a-date", "2023-13-01", ""],
        "amount": [100, 200, 300],
        "status": ["completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0


def test_invalid_amounts_treated_as_zero():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "amount": ["invalid", None, 50],
        "status": ["completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [50.0]
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected, check_dtype=False)


def test_case_insensitive_and_portuguese_status():

def test_status_with_whitespace():

        "order_date": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "amount": [100, 200, 300],
        "status": ["  cancelled  ", " CANCELADO ", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [300.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_mixed_scenarios():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "bad-date", "2023-02-01", "2023-02-14"],
        "amount": [100, "NA", 999, 200, None],
        "status": ["completed", "cancelled", "completed", "  Cancelado ", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [100.0, 0.0]  # 2023-02 valid row amount None -> 0, so revenue 0
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_return_type_and_columns():
    df = pd.DataFrame({
        "order_date": ["2023-01-01"],
        "amount": [100],
        "status": ["completed"]
    })
    result = calculate_monthly_revenue(df)
    assert isinstance(result, pd.DataFrame)
    assert list(result.columns) == ["month", "revenue"]


def test_sorting_order():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2023-12-31", "2023-01-01"],
        "amount": [100, 200, 300],
        "status": ["completed"] * 3
    })
    result = calculate_monthly_revenue(df)
    months = result["month"].tolist()
    assert months == sorted(months)  # lexicographic ascending order is chronological


def test_zero_amount_valid_rows():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02"],
        "amount": [0, 0],
        "status": ["completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [0.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_duplicate_months_aggregation():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-15", "2023-02-01", "2023-01-20"],
        "amount": [10, 20, 30, 40],
        "status": ["completed"] * 4
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [70.0, 30.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_multiple_years_sorting():
    df = pd.DataFrame({
        "order_date": ["2023-12-01", "2024-01-15", "2024-02-01", "2023-01-01"],
        "amount": [5, 10, 20, 2],
        "status": ["completed"] * 4
    })
    result = calculate_monthly_revenue(df)
    expected_months = ["2023-01", "2023-12", "2024-01", "2024-02"]
    assert result["month"].tolist() == expected_months
