import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue


def test_basic_aggregation():
    df = pd.DataFrame({
        "order_date": ["2021-01-15", "2021-01-20", "2021-02-10"],
        "amount": [100.0, 200.0, 300.0],
        "status": ["shipped", "delivered", "shipped"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2021-01", "2021-02"],
        "revenue": [300.0, 300.0],
    })
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_canceled_statuses_excluded():
    df = pd.DataFrame({
        "order_date": ["2021-01-01"] * 6,
        "amount": [100] * 6,
        "status": [
            "cancelled",
            "canceled",
            "cancelado",
            "CANCELED",
            " Cancelado ",
            "shipped",
        ],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2021-01"], "revenue": [100]})
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_invalid_dates_excluded():
    df = pd.DataFrame({
        "order_date": ["2021-01-10", "not_a_date", None, "2021-01-11"],
        "amount": [100, 200, 300, 400],
        "status": ["ok"] * 4,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2021-01"], "revenue": [500]})
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_invalid_amounts_treated_as_zero():
    df = pd.DataFrame({
        "order_date": ["2021-01-01", "2021-01-02"],
        "amount": ["invalid", 50.0],
        "status": ["ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2021-01"], "revenue": [50.0]})
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_missing_amounts():
    df = pd.DataFrame({
        "order_date": ["2021-01-01", "2021-01-02"],
        "amount": [None, pd.NA],
        "status": ["ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2021-01"], "revenue": [0.0]})
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_default_column_names():
    df = pd.DataFrame({
        "order_date": ["2021-01-01"],
        "amount": [1.0],
        "status": ["ok"],
    })
    result = calculate_monthly_revenue(df)
    assert list(result.columns) == ["month", "revenue"]
    assert result["month"].iloc[0] == "2021-01"
    assert result["revenue"].iloc[0] == 1.0


def test_custom_column_names():
    df = pd.DataFrame({
        "dt": ["2021-02-01"],
        "val": [42.0],
        "st": ["ok"],
    })
    result = calculate_monthly_revenue(df, date_col="dt", amount_col="val", status_col="st")
    expected = pd.DataFrame({"month": ["2021-02"], "revenue": [42.0]})
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert list(result.columns) == ["month", "revenue"]
    assert result.empty


def test_all_rows_invalid():
    df = pd.DataFrame({
        "order_date": ["invalid"] * 3 + ["2021-01-01"],
        "amount": [1, 2, 3, 4],
        "status": ["cancelled"] * 3 + ["ok"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2021-01"], "revenue": [4]})
    pd.testing.assert_frame_equal(result.astype(expected.dtypes), expected)


def test_sorting():
    df = pd.DataFrame({
        "order_date": ["2022-12-01", "2021-01-01", "2022-01-15"],
        "amount": [1, 2, 3],
        "status": ["ok"] * 3,
    })
    result = calculate_monthly_revenue(df)
    expected_months = ["2021-01", "2022-01", "2022-12"]
    assert result["month"].tolist() == expected_months


def test_original_df_unchanged():
    df = pd.DataFrame({
        "order_date": ["2021-01-01"],
        "amount": [1.0],
        "status": ["ok"],
    })
    original = df.copy()
    calculate_monthly_revenue(df)
    pd.testing.assert_frame_equal(df, original)
