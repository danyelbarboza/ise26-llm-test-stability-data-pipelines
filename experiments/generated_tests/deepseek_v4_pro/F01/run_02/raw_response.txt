import pandas as pd
import pytest
from ise26.targets import clean_customer_names


def assert_series_with_na(actual_series, expected_list):
    """Compare a pandas Series containing possibly pd.NA against a list."""
    for i, (a, e) in enumerate(zip(actual_series, expected_list)):
        if pd.isna(e):
            assert pd.isna(a), f"Index {i}: expected NA, got {a}"
        else:
            assert a == e, f"Index {i}: expected {e}, got {a}"


def test_original_dataframe_immutable():
    df = pd.DataFrame({"customer_name": ["John DOE", "María"]})
    original_columns = df.columns.tolist()
    original_values = df["customer_name"].tolist()
    result = clean_customer_names(df)
    # Original DataFrame should be unchanged
    assert df.columns.tolist() == original_columns
    assert df["customer_name"].tolist() == original_values
    # Result is a new DataFrame, not the same object
    assert result is not df


def test_basic_cleaning():
    df = pd.DataFrame({"customer_name": ["  John   DOE  ", "María José", "  ", None, "JOSÉ"]})
    result = clean_customer_names(df)
    expected_clean = ["john doe", "maria jose", pd.NA, pd.NA, "jose"]
    assert "customer_name_clean" in result.columns
    assert_series_with_na(result["customer_name_clean"], expected_clean)


def test_lowercases_strings():
    df = pd.DataFrame({"customer_name": ["UPPER", "MiXeD", "lower"]})
    result = clean_customer_names(df)
    expected = ["upper", "mixed", "lower"]
    assert_series_with_na(result["customer_name_clean"], expected)


def test_collapses_internal_whitespace():
    df = pd.DataFrame({"customer_name": ["a   b", "c  d   e", " single "]})
    result = clean_customer_names(df)
    expected = ["a b", "c d e", "single"]
    assert_series_with_na(result["customer_name_clean"], expected)


def test_removes_accents():
    df = pd.DataFrame({"customer_name": ["José", "Máiréad", "Łukasz", "上海"]})
    result = clean_customer_names(df)
    # accented characters should be stripped to base form; non-ASCII that is
    # an accent mark -> removed; "Ł" might become "L", "上海" stays as is
    # (lowercased)
    # Trust the implementation; just verify it runs and produces lowercased
    for val in result["customer_name_clean"]:
        assert isinstance(val, str) or pd.isna(val)
    # Check specific known conversions (assuming unicodedata.normalize is used)
    if "josé" in result["customer_name_clean"].tolist():
        pass  # some implementations keep é? but spec says remove accent marks
    # We'll just trust the function; test that "José" becomes "jose"
    assert result.loc[0, "customer_name_clean"] == "jose"


def test_blank_strings_and_nulls_become_pdna():
    df = pd.DataFrame({"customer_name": ["", "   ", None, pd.NA, pd.NaT, float("nan")]})
    result = clean_customer_names(df)
    for val in result["customer_name_clean"]:
        assert pd.isna(val), f"Expected NA but got {val}"


def test_custom_column_names():
    df = pd.DataFrame({"raw": [" John ", "MARY"]})
    result = clean_customer_names(df, name_col="raw", output_col="clean")
    assert "clean" in result.columns
    assert_series_with_na(result["clean"], ["john", "mary"])


def test_output_column_overwrites_existing_if_present():
    df = pd.DataFrame({"customer_name": ["Foo"], "customer_name_clean": [123]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].iloc[0] == "foo"
    # Original df should not be affected
    assert df["customer_name_clean"].iloc[0] == 123


def test_empty_dataframe():
    df = pd.DataFrame({"customer_name": pd.Series([], dtype=str)})
    result = clean_customer_names(df)
    assert len(result) == 0
    assert "customer_name_clean" in result.columns
    assert result["customer_name_clean"].dtype == object  # or string?


def test_missing_name_col_raises():
    df = pd.DataFrame({"other": ["a"]})
    with pytest.raises(KeyError):
        clean_customer_names(df, name_col="nonexistent")
