import pandas as pd
import numpy as np
import pytest
from ise26.targets import clean_customer_names


def test_basic_cleaning():
    df = pd.DataFrame({"customer_name": [" John   Doe ", "Évelyne", "Ñoño"]})
    result = clean_customer_names(df)
    expected = ["john doe", "evelyne", "nono"]
    assert result["customer_name_clean"].tolist() == expected
    assert "customer_name" in result.columns and "customer_name_clean" in result.columns
    # original unchanged
    assert "customer_name_clean" not in df.columns
    assert df["customer_name"].tolist() == [" John   Doe ", "Évelyne", "Ñoño"]


def test_null_values_become_na():
    df = pd.DataFrame({"customer_name": [None, np.nan, pd.NA, "valid"]})
    result = clean_customer_names(df)
    clean = result["customer_name_clean"]
    assert clean.iloc[0] is pd.NA
    assert clean.iloc[1] is pd.NA
    assert clean.iloc[2] is pd.NA
    assert clean.iloc[3] == "valid"
    # original values unchanged
    assert df["customer_name"].isna().sum() == 3
    assert df["customer_name"].iloc[3] == "valid"


def test_blank_strings_become_na():
    df = pd.DataFrame({"customer_name": ["", "   ", "\t\n", "keep"]})
    result = clean_customer_names(df)
    clean = result["customer_name_clean"]
    assert clean.iloc[0] is pd.NA
    assert clean.iloc[1] is pd.NA
    assert clean.iloc[2] is pd.NA
    assert clean.iloc[3] == "keep"


def test_whitespace_collapse_and_strip():
    df = pd.DataFrame({"customer_name": ["  hello   world  ", "a\tb", "x\ny"]})
    result = clean_customer_names(df)
    expected = ["hello world", "a b", "x y"]
    assert result["customer_name_clean"].tolist() == expected
