import pytest
import pandas as pd
from ise26.targets import clean_customer_names


class TestCleanCustomerNames:
    """Test suite for clean_customer_names function."""

    # --- normalization tests ---
    @pytest.mark.parametrize(
        "raw, expected",
        [
            ("John Doe", "john doe"),
            ("  John   Doe  ", "john doe"),
            ("JOSÉ ÁVILA", "jose avila"),
            ("ÁÉÍÓÚáéíóúñÑ", "aeiouaeiounn"),
            ("Ana", "ana"),
            ("already clean", "already clean"),
            ("\t\nLeading and trailing\n", "leading and trailing"),
            ("Multiple   spaces   everywhere   ", "multiple spaces everywhere"),
            ("", pd.NA),
            ("   ", pd.NA),
            (None, pd.NA),
            (pd.NA, pd.NA),
        ],
    )
    def test_normalization(self, raw, expected):
        """Individual name normalization including blanks and nulls."""
        df = pd.DataFrame({"customer_name": [raw]})
        result = clean_customer_names(df)

        cleaned = result.loc[0, "customer_name_clean"]
        if expected is pd.NA:
            assert pd.isna(cleaned)
        else:
            assert cleaned == expected

    def test_mixed_values_column(self):
        """Normalization handles multiple rows with mixed values."""
        raw_values = ["  Maria ", None, "JÜRGEN", "", "  normal  "]
        expected = ["maria", pd.NA, "jurgen", pd.NA, "normal"]
        df = pd.DataFrame({"customer_name": raw_values})
        result = clean_customer_names(df)
        actual = result["customer_name_clean"].tolist()

        for a, e in zip(actual, expected):
            if e is pd.NA:
                assert pd.isna(a)
            else:
                assert a == e

    # --- preservation of original DataFrame ---
    def test_original_dataframe_unchanged_columns(self):
        """Original DataFrame columns must not be modified."""
        original = pd.DataFrame({"customer_name": ["Alice"]})
        result = clean_customer_names(original)

        assert list(original.columns) == ["customer_name"]
        assert "customer_name_clean" not in original.columns

    def test_original_dataframe_unchanged_values(self):
        """Original DataFrame values must remain identical."""
        original = pd.DataFrame({"customer_name": ["Alice", "Bob"]})
        result = clean_customer_names(original)

        # values untouched
        pd.testing.assert_series_equal(
            original["customer_name"],
            pd.Series(["Alice", "Bob"], name="customer_name"),
        )

    def test_original_not_affected_by_result_mutation(self):
        """Mutating the returned DataFrame must not alter the original."""
        original = pd.DataFrame({"customer_name": ["Alice"]})
        result = clean_customer_names(original)

        # add a column to result; original must stay the same
        result["extra"] = "test"
        assert "extra" not in original.columns

        # modify an original column in result
        result.loc[0, "customer_name"] = "changed"
        assert original.loc[0, "customer_name"] == "Alice"

    # --- custom column names ---
    def test_custom_name_and_output_columns(self):
        """Function respects user-supplied column names."""
        df = pd.DataFrame({"raw": ["  CHLOÉ  "]})
        result = clean_customer_names(df, name_col="raw", output_col="clean")
        assert "clean" in result.columns
        assert result.loc[0, "clean"] == "chloe"

    def test_default_column_names(self):
        """Default column names are applied when not specified."""
        df = pd.DataFrame({"customer_name": ["John"]})
        result = clean_customer_names(df)
        assert "customer_name_clean" in result.columns
        assert result.loc[0, "customer_name_clean"] == "john"

    # --- edge cases ---
    def test_empty_dataframe(self):
        """Works with an empty DataFrame (no rows)."""
        df = pd.DataFrame({"customer_name": pd.Series([], dtype=object)})
        result = clean_customer_names(df)
        assert len(result) == 0
        assert "customer_name_clean" in result.columns

    def test_missing_input_column_raises(self):
        """Raises KeyError when the specified name column does not exist."""
        df = pd.DataFrame({"wrong_name": ["Alice"]})
        with pytest.raises(KeyError):
            clean_customer_names(df, name_col="customer_name")

    def test_non_string_columns_preserved(self):
        """Additional columns are retained unchanged in the result."""
        df = pd.DataFrame({
            "customer_name": ["John"],
            "id": [1],
            "score": [99.9],
        })
        result = clean_customer_names(df)
        assert result.loc[0, "id"] == 1
        assert result.loc[0, "score"] == 99.9
