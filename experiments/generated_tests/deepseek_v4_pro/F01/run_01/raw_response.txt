import pandas as pd
import pytest
from ise26.targets import clean_customer_names


def test_basic_normalization():
    """Lowercase and strip simple names."""
    df = pd.DataFrame({"customer_name": ["John Doe", "Jane Smith"]})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    assert result["customer_name_clean"].tolist() == ["john doe", "jane smith"]
    # original DataFrame is intact
    assert "customer_name_clean" not in df.columns


def test_collapses_repeated_whitespace():
    """Multiple spaces and tabs are collapsed to a single space."""
    df = pd.DataFrame({"customer_name": ["  John   Doe  ", "  Jane\tSmith"]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].tolist() == ["john doe", "jane smith"]


def test_null_and_blank_become_pd_na():
    """None, pd.NA, empty string and whitespace-only become pd.NA."""
    df = pd.DataFrame(
        {
            "customer_name": [
                None,
                pd.NA,
                "",
                "   ",
                "valid",
            ]
        }
    )
    result = clean_customer_names(df)
    # Use pd.isna to handle pd.NA comparison
    na_flags = result["customer_name_clean"].isna()
    assert na_flags.tolist() == [True, True, True, True, False]
    assert result.loc[4, "customer_name_clean"] == "valid"


def test_accent_removal():
    """Accented characters are normalised to ASCII equivalents."""
    df = pd.DataFrame({"customer_name": ["José", "Renée", "François", "Åsa"]})
    result = clean_customer_names(df)
    expected = ["jose", "renee", "francois", "asa"]
    assert result["customer_name_clean"].tolist() == expected


def test_custom_column_names():
    """Arbitrary input and output column names can be used."""
    df = pd.DataFrame({"name": ["Alice", "Bob"]})
    result = clean_customer_names(df, name_col="name", output_col="clean_name")
    assert "clean_name" in result.columns
    assert result["clean_name"].tolist() == ["alice", "bob"]


def test_output_column_default_name():
    """Default output column is 'customer_name_clean'."""
    df = pd.DataFrame({"customer_name": ["Alice"]})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns


def test_original_dataframe_not_mutated():
    """The input DataFrame is not modified in any way."""
    original = pd.DataFrame({"customer_name": ["Original Value"]})
    original_before = original.copy()
    _ = clean_customer_names(original)
    pd.testing.assert_frame_equal(original, original_before)
    assert list(original.columns) == ["customer_name"]


def test_missing_name_col_raises_keyerror():
    """If name_col does not exist, a KeyError is raised."""
    df = pd.DataFrame({"other_col": ["value"]})
    with pytest.raises(KeyError):
        clean_customer_names(df, name_col="customer_name")


def test_index_preserved():
    """The returned DataFrame retains the original index."""
    df = pd.DataFrame(
        {"customer_name": ["a", "b", "c"]}, index=[10, 20, 30]
    )
    result = clean_customer_names(df)
    assert result.index.tolist() == [10, 20, 30]


def test_mixed_case_and_extra_blanks():
    """Uppercase, leading/trailing blanks, and interior repeated spaces."""
    df = pd.DataFrame({"customer_name": ["  HELLO   WORLD  ", "Python"]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].tolist() == ["hello world", "python"]
