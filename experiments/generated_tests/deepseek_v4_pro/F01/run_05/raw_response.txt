import pandas as pd
import pytest
import numpy as np
from ise26.targets import clean_customer_names


def test_basic_normalization():
    df = pd.DataFrame({
        "customer_name": ["  María  ", "JOSÉ", "  Ana  ", "John   Doe", "already clean"]
    })
    result = clean_customer_names(df)
    expected = ["maria", "jose", "ana", "john doe", "already clean"]
    assert result["customer_name_clean"].tolist() == expected
    # Original column unchanged
    assert result["customer_name"].tolist() == ["  María  ", "JOSÉ", "  Ana  ", "John   Doe", "already clean"]


def test_null_and_blank_to_na():
    df = pd.DataFrame({
        "customer_name": [None, pd.NA, "", "   ", "valid"]
    })
    result = clean_customer_names(df)
    out = result["customer_name_clean"]
    # Verify conversion to pd.NA
    assert out.iloc[0] is pd.NA
    assert out.iloc[1] is pd.NA
    assert out.iloc[2] is pd.NA
    assert out.iloc[3] is pd.NA
    assert out.iloc[4] == "valid"


def test_nan_input():
    df = pd.DataFrame({"customer_name": [np.nan]})
    result = clean_customer_names(df)
    assert pd.isna(result["customer_name_clean"].iloc[0])


def test_original_unchanged():
    df = pd.DataFrame({"customer_name": ["  Test  "]})
    df_original = df.copy()
    clean_customer_names(df)
    pd.testing.assert_frame_equal(df, df_original)


def test_custom_column_names():
    df = pd.DataFrame({"raw_name": ["  HELLO  "]})
    result = clean_customer_names(df, name_col="raw_name", output_col="clean")
    assert "clean" in result.columns
    assert result["clean"].iloc[0] == "hello"
    # original columns preserved
    assert list(result.columns) == ["raw_name", "clean"]
    assert result["raw_name"].iloc[0] == "  HELLO  "


def test_empty_dataframe():
    df = pd.DataFrame({"customer_name": []}, dtype=str)
    result = clean_customer_names(df)
    assert len(result) == 0
    assert "customer_name_clean" in result.columns
    # dtype of empty output should be object (because of pd.NA)
    assert result["customer_name_clean"].dtype == object


def test_missing_name_column_raises():
    df = pd.DataFrame({"other_col": ["val"]})
    with pytest.raises(KeyError):
        clean_customer_names(df, name_col="nonexistent")


def test_preserves_other_columns():
    df = pd.DataFrame({
        "customer_name": ["  Pedro  "],
        "age": [30],
        "city": ["Madrid"]
    })
    result = clean_customer_names(df)
    expected_cols = ["customer_name", "age", "city", "customer_name_clean"]
    assert list(result.columns) == expected_cols
    # Verify data preserved in other columns
    assert result.loc[0, "age"] == 30
    assert result.loc[0, "city"] == "Madrid"
    assert result.loc[0, "customer_name"] == "  Pedro  "


def test_accents_removed():
    df = pd.DataFrame({"customer_name": ["àéîõùç"]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].iloc[0] == "aeiouc"


def test_multiple_spaces_collapsed():
    df = pd.DataFrame({"customer_name": ["  hello   world  "]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].iloc[0] == "hello world"


def test_no_mutation_of_input_dataframe():
    df = pd.DataFrame({"customer_name": ["  Example  "]})
    original = df.copy()
    result = clean_customer_names(df)
    # Original not changed
    pd.testing.assert_frame_equal(df, original)
    # Result is a different DataFrame
    assert result is not df
