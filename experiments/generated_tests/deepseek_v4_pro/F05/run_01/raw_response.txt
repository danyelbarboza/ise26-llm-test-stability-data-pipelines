import pandas as pd
import pytest
from ise26.targets import validate_schema


def test_valid_schema_all_supported_types():
    df = pd.DataFrame({
        "int_col": pd.array([1, 2], dtype="Int64"),
        "float_col": pd.array([1.0, 2.0], dtype="Float64"),
        "num_int": pd.array([10, 20], dtype="Int64"),
        "num_float": pd.array([1.1, 2.2], dtype="Float64"),
        "str_col": pd.array(["a", "b"], dtype="string"),
        "datetime_col": pd.Series(pd.to_datetime(["2023-01-01", "2023-06-15"])),
        "bool_col": pd.array([True, False], dtype="boolean"),
    })
    schema = {
        "int_col": "int",
        "float_col": "float",
        "num_int": "number",
        "num_float": "number",
        "str_col": "string",
        "datetime_col": "datetime",
        "bool_col": "bool",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_columns_reported():
    df = pd.DataFrame({"a": [1, 2]})
    schema = {"a": "int", "b": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["b"]
    assert result["type_errors"] == []


def test_type_mismatch_reported():
    df = pd.DataFrame({
        "col": pd.array(["x", "y"], dtype="string")
    })
    schema = {"col": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "col"
    assert err["expected_type"] == "int"
    assert err["actual_type"] == "string"

def test_empty_dataframe_missing_all_columns():
    df = pd.DataFrame()
    schema = {"col": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["col"]
    assert result["type_errors"] == []


def test_normalization_trims_whitespace_and_lowercases():
    df = pd.DataFrame({"a": pd.array([1], dtype="Int64")})
    schema = {"a": "   Int   "}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["type_errors"] == []

    schema = {"a": "  DatetimE "}
    df_date = pd.DataFrame({"a": pd.Series(pd.to_datetime(["2023-01-01"]))})
    result = validate_schema(df_date, schema)
    assert result["valid"] is True


def test_unsupported_expected_type_raises_valueerror():

def test_unsupported_type_after_normalization_raises():

def test_number_rejects_string_column():
    df = pd.DataFrame({"x": ["a", "b"]})
    schema = {"x": "number"}
    result = validate_schema(df, schema)
    assert not result["valid"]
    err = result["type_errors"][0]
    assert err["expected_type"] == "number"
    assert err["actual_type"] == "string"
    assert err["column"] == "x"

def test_bool_rejects_object_bool():
    # plain Python bool in object column may be inferred differently
    df = pd.DataFrame({"b": [True, False]})  # dtype object
    schema = {"b": "bool"}
    result = validate_schema(df, schema)
    # likely fails because series type is object -> string
    assert not result["valid"]
    err = result["type_errors"][0]
    assert err["actual_type"] == "string"
    assert err["expected_type"] == "bool"
