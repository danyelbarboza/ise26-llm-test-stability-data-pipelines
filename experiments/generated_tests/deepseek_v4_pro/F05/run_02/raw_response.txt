import pandas as pd
import pytest
from ise26.targets import validate_schema


# --- helpers ----------------------------------------------------------------

def _make_df(columns_and_dtypes):
    """Create a DataFrame with one row, given a dict of column name -> pandas dtype."""
    data = {}
    for col, dtype in columns_and_dtypes.items():
        if dtype == "int":
            data[col] = pd.Series([1], dtype="int64")
        elif dtype == "float":
            data[col] = pd.Series([1.0], dtype="float64")
        elif dtype == "string":
            data[col] = pd.Series(["a"], dtype="object")
        elif dtype == "datetime":
            data[col] = pd.Series([pd.Timestamp("2023-01-01")], dtype="datetime64[ns]")
        elif dtype == "bool":
            data[col] = pd.Series([True], dtype="bool")
        else:
            raise ValueError(f"Unknown dtype shortcut: {dtype}")
    return pd.DataFrame(data)


# --- valid schemas ----------------------------------------------------------

def test_valid_schema_all_supported_types():
    df = _make_df({
        "col_int": "int",
        "col_float": "float",
        "col_str": "string",
        "col_dt": "datetime",
        "col_bool": "bool",
    })
    schema = {
        "col_int": "int",
        "col_float": "float",
        "col_str": "string",
        "col_dt": "datetime",
        "col_bool": "bool",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []

# --- missing columns --------------------------------------------------------

def test_missing_column():
    df = _make_df({"a": "int"})
    schema = {"a": "int", "b": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["b"]
    assert result["type_errors"] == []


def test_all_columns_missing():
    df = pd.DataFrame()
    schema = {"a": "int", "b": "float"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert sorted(result["missing_columns"]) == ["a", "b"]
    assert result["type_errors"] == []

# --- type mismatches --------------------------------------------------------

def test_type_mismatch_int_expected_float_actual():
    df = _make_df({"a": "float"})
    schema = {"a": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "a"
    assert err["expected_type"] == "int"
    # actual_type inferred from float64 series should be "float"
    assert err["actual_type"] == "float"


def test_type_mismatch_string_expected_datetime():
    df = _make_df({"a": "string"})
    schema = {"a": "datetime"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "a"
    assert err["expected_type"] == "datetime"
    assert err["actual_type"] == "string"


def test_type_mismatch_bool_expected_int():
    df = _make_df({"a": "bool"})

    result = validate_schema(df, schema)
    assert not result["valid"]
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["expected_type"] == "int"
    assert err["actual_type"] == "bool"

# --- combined errors --------------------------------------------------------

def test_mixed_missing_and_type_mismatch():
    df = _make_df({"a": "float", "c": "string"})
    schema = {
        "a": "int",           # type mismatch: float vs int
        "b": "int",           # missing
        "c": "datetime",      # type mismatch: string vs datetime
    }
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert sorted(result["missing_columns"]) == ["b"]
    assert len(result["type_errors"]) == 2
    columns = {e["column"] for e in result["type_errors"]}
    assert columns == {"a", "c"}


# --- ValueError for unsupported types ---------------------------------------

def test_unsupported_schema_type_raises():
    df = _make_df({"a": "int"})
    with pytest.raises(ValueError, match="unsupported"):
        validate_schema(df, {"a": "category"})


def test_unsupported_schema_type_after_normalization():
    df = _make_df({"a": "int"})
    # "  Int  " is fine, but "  Blob  " is not
    with pytest.raises(ValueError, match="unsupported"):
        validate_schema(df, {"a": "  Blob  "})


# --- normalization of expected types ----------------------------------------

def test_expected_type_with_whitespace():
    df = _make_df({"a": "int"})
    schema = {"a": "  INT  "}
    result = validate_schema(df, schema)
    assert result["valid"] is True


def test_expected_type_mixed_case_normalized():
    df = _make_df({"a": "float"})
    schema = {"a": "fLoAt"}
    result = validate_schema(df, schema)
    assert result["valid"] is True


def test_expected_type_normalization_reported():
    # When there is a mismatch, the expected_type in the error dict is the normalized version.
    df = _make_df({"a": "string"})
    schema = {"a": "  Int  "}
    result = validate_schema(df, schema)
    assert not result["valid"]
    err = result["type_errors"][0]
    assert err["expected_type"] == "int"
    assert err["actual_type"] == "string"
