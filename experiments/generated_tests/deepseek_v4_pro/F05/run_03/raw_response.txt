import pytest
import pandas as pd
from ise26.targets import validate_schema


def test_valid_schema_all_types():
    df = pd.DataFrame(
        {
            "col_int": [1, 2],
            "col_float": [1.1, 2.2],
            "col_str": ["a", "b"],
            "col_dt": pd.to_datetime(["2020-01-01", "2020-01-02"]),
            "col_bool": [True, False],
        }
    )
    schema = {
        "col_int": "int",
        "col_float": "float",
        "col_str": "string",
        "col_dt": "datetime",
        "col_bool": "bool",
    }
    result = validate_schema(df, schema)
    assert result == {"valid": True, "missing_columns": [], "type_errors": []}


def test_valid_normalization_strip_and_lower():
    df = pd.DataFrame({"col_int": [1, 2], "col_float": [1.1, 2.2]})
    schema = {"   col_int  ": "  INt ", "col_float": "Float  "}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_columns_reported():
    df = pd.DataFrame({"a": [1], "b": [2]})
    schema = {"a": "int", "c": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["c"]
    assert result["type_errors"] == []


def test_type_mismatch_int_vs_float():
    df = pd.DataFrame({"num": [1.1, 2.2]})
    schema = {"num": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "num"
    assert err["expected_type"] == "int"
    assert err["actual_type"] == "float"

def test_unsupported_expected_type_raises_valueerror():
    df = pd.DataFrame({"col": [1]})
    schema = {"col": "unknown_type"}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_unsupported_after_normalization_raises_valueerror():
    df = pd.DataFrame({"col": [1]})
    schema = {"col": "   BLAH  "}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_empty_df_and_empty_schema_valid():
    df = pd.DataFrame()
    schema = {}
    result = validate_schema(df, schema)
    assert result == {"valid": True, "missing_columns": [], "type_errors": []}


def test_multiple_errors_combined():
    df = pd.DataFrame({"a": [1], "c": ["x"]})
    schema = {"a": "string", "b": "int", "c": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert set(result["missing_columns"]) == {"b"}
    assert len(result["type_errors"]) == 2
    columns_with_error = {e["column"] for e in result["type_errors"]}
    assert columns_with_error == {"a", "c"}
