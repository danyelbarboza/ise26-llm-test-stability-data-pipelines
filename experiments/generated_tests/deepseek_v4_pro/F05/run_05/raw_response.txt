import pandas as pd
import pytest

from ise26.targets import validate_schema


class TestValidateSchema:
    def test_valid_schema_all_types_match(self):
        df = pd.DataFrame(
            {
                "int_col": [1, 2],
                "float_col": [1.0, 2.0],
                "string_col": ["a", "b"],
                "bool_col": [True, False],
                "datetime_col": pd.to_datetime(["2020-01-01", "2020-01-02"]),
            }
        )
        schema = {
            "int_col": "int",
            "float_col": "float",
            "string_col": "string",
            "bool_col": "bool",
            "datetime_col": "datetime",
        }
        result = validate_schema(df, schema)
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_normalization_strip_and_lowercase(self):
        df = pd.DataFrame({"col": [1, 2]})
        schema = {"col": "  INT  "}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_number_matches_int(self):
        df = pd.DataFrame({"col": [1, 2]})
        schema = {"col": "number"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_number_matches_float(self):
        df = pd.DataFrame({"col": [1.0, 2.5]})
        schema = {"col": "number"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_missing_columns(self):
        df = pd.DataFrame({"existing": [1, 2]})
        schema = {"existing": "int", "missing_col": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["missing_col"]
        assert len(result["type_errors"]) == 0

    def test_type_mismatch_int_for_string(self):
        df = pd.DataFrame({"col": [1, 2]})
        schema = {"col": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert len(result["type_errors"]) == 1
        err = result["type_errors"][0]
        assert err["column"] == "col"
        assert err["expected_type"] == "string"
        assert err["actual_type"] == "int"

    def test_type_mismatch_string_for_int(self):
        df = pd.DataFrame({"col": ["a", "b"]})
        schema = {"col": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        err = result["type_errors"][0]
        assert err["expected_type"] == "int"
        assert err["actual_type"] == "string"

    def test_type_mismatch_float_for_datetime(self):
        df = pd.DataFrame({"col": [1.0, 2.0]})
        schema = {"col": "datetime"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        err = result["type_errors"][0]
        assert err["expected_type"] == "datetime"
        assert err["actual_type"] == "float"

    def test_type_mismatch_datetime_for_bool(self):
        df = pd.DataFrame({"col": pd.to_datetime(["2020-01-01", "2020-01-02"])})
        schema = {"col": "bool"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        err = result["type_errors"][0]
        assert err["expected_type"] == "bool"
        assert err["actual_type"] == "datetime"

    def test_extra_columns_allowed(self):
        df = pd.DataFrame({"a": [1], "b": [2.0], "extra": ["x"]})
        schema = {"a": "int", "b": "float"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_unsupported_type_raises_value_error(self):
        df = pd.DataFrame({"col": [1]})
        schema = {"col": "list"}
        with pytest.raises(ValueError, match="Unsupported expected type"):
            validate_schema(df, schema)

    def test_unsupported_type_after_normalization_raises(self):
        df = pd.DataFrame({"col": [1]})
        schema = {"col": "  LiST "}
        with pytest.raises(ValueError):
            validate_schema(df, schema)

    def test_empty_schema(self):
        df = pd.DataFrame({"anycol": [1, 2]})
        schema = {}
        result = validate_schema(df, schema)
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_empty_dataframe_with_schema(self):
        df = pd.DataFrame()
        schema = {"col1": "int", "col2": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert set(result["missing_columns"]) == {"col1", "col2"}
        assert result["type_errors"] == []

    def test_null_values_preserve_string_type(self):
        df = pd.DataFrame({"col": ["a", None, "b"]})
        schema = {"col": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_multiple_missing_and_type_errors(self):
        df = pd.DataFrame({"int_col": [1], "str_col": ["ok"]})
        schema = {
            "int_col": "string",  # mismatch
            "str_col": "int",     # mismatch
            "missing": "float",
        }
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["missing"]
        assert len(result["type_errors"]) == 2
        # order not guaranteed but check presence
        cols = {e["column"] for e in result["type_errors"]}
        assert cols == {"int_col", "str_col"}

    def test_boolean_type_recognized(self):
        df = pd.DataFrame({"col": [True, False, True]})
        schema = {"col": "bool"}
        result = validate_schema(df, schema)
        assert result["valid"] is True
