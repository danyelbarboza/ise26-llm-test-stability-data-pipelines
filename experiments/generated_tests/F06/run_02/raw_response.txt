import pytest
import pandas as pd
from ise26.targets import classify_payment_status


def test_classify_payment_status_basic_paid_on_time():
    """A payment made on or before the due date is 'paid_on_time'."""
    df = pd.DataFrame({
        'due_date': ['2025-01-15'],
        'paid_date': ['2025-01-10'],
        'amount': [100.0]
    })
    result = classify_payment_status(df, reference_date='2025-01-20')
    assert result['payment_status'].iloc[0] == 'paid_on_time'


def test_classify_payment_status_basic_paid_late():
    """A payment made after the due date is 'paid_late'."""
    df = pd.DataFrame({
        'due_date': ['2025-01-15'],
        'paid_date': ['2025-01-20'],
        'amount': [100.0]
    })
    result = classify_payment_status(df, reference_date='2025-01-20')
    assert result['payment_status'].iloc[0] == 'paid_late'


def test_classify_payment_status_basic_pending():
    """Unpaid row with reference_date <= due_date is 'pending'."""
    df = pd.DataFrame({
        'due_date': ['2025-01-25'],
        'paid_date': [None],
        'amount': [100.0]
    })
    result = classify_payment_status(df, reference_date='2025-01-20')
    assert result['payment_status'].iloc[0] == 'pending'


def test_classify_payment_status_basic_overdue():

    """An empty string paid_date should be treated as unpaid (pending/overdue), not invalid."""
    df = pd.DataFrame({

        'paid_date': [''],

    })

def test_classify_payment_status_mixed_scenarios():
    """Test multiple rows with different statuses in one call."""
    df = pd.DataFrame({
        'due_date': [
            '2025-01-15',  # paid on time
            '2025-01-15',  # paid late
            '2025-01-25',  # pending
            '2025-01-15',  # overdue
            None,          # invalid due date
            '2025-01-15',  # invalid amount
            '2025-01-15',  # invalid paid date
            '2025-01-15',  # valid, paid on time (edge: exactly due)
            '2025-01-15',  # valid, unpaid, ref == due => pending
        ],
        'paid_date': [
            '2025-01-10',  # on time
            '2025-01-20',  # late
            None,          # pending
            None,          # overdue
            '2025-01-10',
            '2025-01-10',
            'bad_date',
            '2025-01-15',  # exactly due
            None,
        ],
        'amount': [
            100.0,   # ok
            100.0,   # ok
            100.0,   # ok
            100.0,   # ok
            100.0,   # ok
            -1.0,    # invalid
            100.0,   # ok
            100.0,   # ok
            100.0,   # ok
        ]
    })
    result = classify_payment_status(df, reference_date='2025-01-20')
    expected = [
        'paid_on_time',
        'paid_late',
        'pending',
        'overdue',
        'invalid',
        'invalid',
        'invalid',
        'paid_on_time',
        'pending'
    ]
    assert result['payment_status'].tolist() == expected


def test_classify_payment_status_edge_equal_dates():
    """When paid_date exactly equals due_date, it is paid_on_time."""
    df = pd.DataFrame({
        'due_date': ['2025-01-15'],
        'paid_date': ['2025-01-15'],
        'amount': [100.0]
    })
    result = classify_payment_status(df, reference_date='2025-01-20')
    assert result['payment_status'].iloc[0] == 'paid_on_time'


def test_classify_payment_status_does_not_mutate_original():
    """The original DataFrame should not be modified."""
    original = pd.DataFrame({
        'due_date': ['2025-01-15'],
        'paid_date': ['2025-01-10'],
        'amount': [100.0]
    })
    original_copy = original.copy()
    classify_payment_status(original, reference_date='2025-01-20')
    pd.testing.assert_frame_equal(original, original_copy)
