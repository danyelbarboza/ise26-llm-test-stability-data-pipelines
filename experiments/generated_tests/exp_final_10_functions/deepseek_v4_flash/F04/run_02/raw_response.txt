import pandas as pd
import pytest

from ise26.targets import join_customers_orders


class TestJoinCustomersOrders:
    """Test suite for join_customers_orders."""

    def test_happy_path(self):
        """Normal join with some matches, some left-only, some right-only."""
        customers = pd.DataFrame({
            "customer_id": [1, 2, 3],
            "name": ["Alice", "Bob", "Charlie"],
            "region": ["US", "EU", "ASIA"],
        })
        orders = pd.DataFrame({
            "customer_id": [1, 3, 4],
            "product": ["Widget", "Gadget", "Doohickey"],
            "amount": [100, 200, 150],
        })
        result = join_customers_orders(customers, orders)

        # Expected rows:
        # matched: (1, Alice, US, Widget, 100) and (3, Charlie, ASIA, Gadget, 200)
        # customer_without_order: (2, Bob, EU, NaN, NaN)
        # order_without_customer: (4, NaN, NaN, Doohickey, 150)
        # No null keys involved.
        assert len(result) == 4
        assert list(result.columns) == [
            "customer_id", "name", "region", "product", "amount", "record_status"
        ]
        assert result["record_status"].tolist() == [
            "matched",
            "customer_without_order",
            "matched",
            "order_without_customer",
        ]
        # Check specific rows by index position (deterministic sort based on source order)
        # Source order for customers: [1,2,3]; for orders: [1,3,4]
        # The join preserves intra-group ordering.
        # For matched rows: customer 1 first, then customer 3.
        # For unmatched: customer 2 after matched 1? Actually sorting by _sort_order which is
        # combine_first of customer and order source indices.
        # The expected order after sorting: row for c1 (indices 0,0), c2 (1,NaN), c3 (2,1), o4 (NaN,2).
        row0 = result.iloc[0]
        assert row0["customer_id"] == 1
        assert row0["name"] == "Alice"
        assert row0["region"] == "US"
        assert row0["product"] == "Widget"
        assert row0["record_status"] == "matched"

        row1 = result.iloc[1]
        assert row1["customer_id"] == 2
        assert row1["name"] == "Bob"
        assert row1["region"] == "EU"
        assert pd.isna(row1["product"])
        assert row1["record_status"] == "customer_without_order"

        row2 = result.iloc[2]
        assert row2["customer_id"] == 3
        assert row2["name"] == "Charlie"
        assert row2["region"] == "ASIA"
        assert row2["product"] == "Gadget"
        assert row2["record_status"] == "matched"

        row3 = result.iloc[3]
        assert row3["customer_id"] == 4
        assert pd.isna(row3["name"])
        assert pd.isna(row3["region"])
        assert row3["product"] == "Doohickey"
        assert row3["record_status"] == "order_without_customer"

    def test_null_keys_customers(self):
        """Customers with null keys are never matched and appear as customer_without_order."""
        customers = pd.DataFrame({
            "customer_id": [1, None, 2],
            "name": ["Alice", "Bob", "Charlie"],
        })
        orders = pd.DataFrame({
            "customer_id": [1, 2],
            "product": ["Widget", "Gadget"],
        })
        result = join_customers_orders(customers, orders)
        # Expected: customer 1 matched, customer null -> customer_without_order, customer 2 matched
        # No order_without_customer
        assert len(result) == 3
        assert result["record_status"].tolist() == [
            "matched",
            "customer_without_order",
            "matched",
        ]
        # Check null key row
        row_null = result.iloc[1]
        assert pd.isna(row_null["customer_id"])
        assert row_null["name"] == "Bob"
        assert pd.isna(row_null["product"])

    def test_null_keys_orders(self):
        """Orders with null keys are never matched and appear as order_without_customer."""
        customers = pd.DataFrame({
            "customer_id": [1],
            "name": ["Alice"],
        })
        orders = pd.DataFrame({
            "customer_id": [1, None, 2],
            "product": ["Widget", "Gadget", "Doohickey"],
        })
        result = join_customers_orders(customers, orders)
        # Expected: matched (1, Widget), order_without_customer (null, Gadget), order_without_customer (2, Doohickey)
        assert len(result) == 3
        statuses = result["record_status"].tolist()
        # Deterministic sort: matched first (source index 0), then null order (source index 1), then order 2 (source index 2)
        assert statuses == ["matched", "order_without_customer", "order_without_customer"]
        # Check null order row
        row_null = result.iloc[1]
        assert pd.isna(row_null["customer_id"])
        assert pd.isna(row_null["name"])
        assert row_null["product"] == "Gadget"

    def test_both_sides_null_keys(self):
        """Both sides have null keys; they remain unmatched."""
        customers = pd.DataFrame({
            "customer_id": [None, None],
            "name": ["Alice", "Bob"],
        })
        orders = pd.DataFrame({
            "customer_id": [None, None],
            "product": ["Widget", "Gadget"],
        })
        result = join_customers_orders(customers, orders)
        # All rows are unmatched: two customer_without_order, two order_without_customer
        assert len(result) == 4
        # Order: first customer null rows (source indices 0,1), then order null rows (source indices 0,1)
        assert result["record_status"].tolist() == [
            "customer_without_order",
            "customer_without_order",
            "order_without_customer",
            "order_without_customer",
        ]

    def test_empty_dataframes(self):
        """Both DataFrames empty -> empty result."""
        customers = pd.DataFrame({"customer_id": pd.Series(dtype=int)})
        orders = pd.DataFrame({"customer_id": pd.Series(dtype=int)})
        result = join_customers_orders(customers, orders)
        assert result.empty
        # Column order: note that overlapping columns are not present, but join preserves non-key?
        # The function will still produce columns: customer_id (if exists) plus record_status.
        # Since both have only customer_id, after merge (no columns), record_status is added.
        # But there are no rows. Let's just check empty.
        assert len(result) == 0
        # Columns? At least record_status should be present.
        assert "record_status" in result.columns

    def test_no_matches(self):
        """No shared keys between customers and orders."""
        customers = pd.DataFrame({
            "customer_id": [1, 2],
            "name": ["Alice", "Bob"],
        })
        orders = pd.DataFrame({
            "customer_id": [3, 4],
            "product": ["Widget", "Gadget"],
        })
        result = join_customers_orders(customers, orders)
        # Should yield two customer_without_order and two order_without_customer
        assert len(result) == 4
        statuses = result["record_status"].tolist()
        # Order: customer rows first (1,2), then order rows (3,4)
        assert statuses == [
            "customer_without_order",
            "customer_without_order",
            "order_without_customer",
            "order_without_customer",
        ]

    def test_overlapping_non_key_columns(self):
        """Non-key columns with same name get suffixes."""
        customers = pd.DataFrame({
            "customer_id": [1],
            "code": ["C1"],
        })
        orders = pd.DataFrame({
            "customer_id": [1],
            "code": ["O1"],
        })
        result = join_customers_orders(customers, orders)
        assert "code_customer" in result.columns
        assert "code_order" in result.columns
        assert result.iloc[0]["code_customer"] == "C1"
        assert result.iloc[0]["code_order"] == "O1"
        # record_status should be matched
        assert result.iloc[0]["record_status"] == "matched"

    def test_custom_key(self):
        """Use a custom join key."""
        customers = pd.DataFrame({
            "key": [1, 2],

        })
        orders = pd.DataFrame({
            "key": [1, 3],

            "order_without_customer",
        ]

    def test_customers_unmodified(self):
        """Original customers DataFrame is not mutated."""
        customers_orig = pd.DataFrame({
            "customer_id": [1, 2],
            "name": ["Alice", "Bob"],
        })
        customers_copy = customers_orig.copy()
        orders = pd.DataFrame({
            "customer_id": [1, 3],
            "product": ["Widget", "Gadget"],
        })
        _ = join_customers_orders(customers_orig, orders)
        pd.testing.assert_frame_equal(customers_orig, customers_copy)

    def test_duplicate_keys(self):
        """Both sides can have multiple rows with same key; full outer join handles duplicates."""
        customers = pd.DataFrame({
            "customer_id": [1, 1],
            "name": ["Alice", "Bob"],
        })
        orders = pd.DataFrame({
            "customer_id": [1, 1, 2],
            "product": ["Widget", "Gadget", "Doohickey"],
        })
        result = join_customers_orders(customers, orders)
        # Should give cartesian product for matches: each customer row with each order row.
        # 2 customers * 2 orders = 4 matched rows, plus order 2 unmatched.
        assert len(result) == 5
        matched = result[result["record_status"] == "matched"]
        assert len(matched) == 4
        unmatched = result[result["record_status"] == "order_without_customer"]
        assert len(unmatched) == 1

    def test_no_non_key_columns(self):

        assert list(result.columns) == ["customer_id", "record_status"]
