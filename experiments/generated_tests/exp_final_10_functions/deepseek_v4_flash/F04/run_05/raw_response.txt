# Basic full outer join scenarios
# -------------------------------------------------------------------

class TestBasicJoin:
    def test_matched_and_unmatched(self, sample_customers, sample_orders):
        result = join_customers_orders(sample_customers, sample_orders)
        # Expected record_status per row:
        #  customer_id 1 -> matched
        #  customer_id 2 -> customer_without_order
        #  customer_id 3 -> matched
        #  customer_id None -> customer_without_order (null key -> not matched)
        #  customer_id 5 -> order_without_customer
        #  customer_id None (order side) -> order_without_customer
        # Total rows: 5 + 1 unmatched order null key? Wait, orders None also appears.
        # Actually orders has None customer_id as well. That will be kept as order_without_customer.
        # So total rows: from customers_with_key: 1,2,3 => 3 rows (but after merge 1 and 3 matched, 2 left_only)
        # customers_without_key: None => 1 row
        # orders_with_key: 1,3,5 => after merge 1,3 matched, 5 right_only
        # orders_without_key: None => 1 row
        # So total rows: 3 (merged) + 1 (cust null) + 1 (order null) = 5? Actually merged with keys has 3 rows (both, left_only, right_only). Then add 2 null rows = 5.
        # Let's verify:
        # merged_with_keys: customers_with_key (1,2,3) outer join orders_with_key (1,3,5) -> rows: (1,both), (2,left_only), (3,both), (5,right_only) = 4 rows? Wait no, outer join on keys [1,2,3] and [1,3,5] gives keys 1,2,3,5 -> 4 rows.
        # Actually because customers_with_key has 3 rows, orders_with_key has 3 rows, outer join results in 4 rows: 1,2,3,5.
        # So merged_with_keys yields 4 rows (statuses: matched, customer_without_order, matched, order_without_customer).
        # Then customers_without_key: 1 row -> customer_without_order.
        # orders_without_key: 1 row -> order_without_customer.
        # Total: 6 rows.
        assert len(result) == 6
        statuses = result["record_status"].tolist()
        assert "matched" in statuses
        assert "customer_without_order" in statuses
        assert "order_without_customer" in statuses

    def test_no_unmatched(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "item": ["x", "y"]})
        result = join_customers_orders(customers, orders)
        assert len(result) == 2
        assert result["record_status"].tolist() == ["matched", "matched"]

    def test_only_customers(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [10], "item": ["z"]})
        result = join_customers_orders(customers, orders)
        assert len(result) == 3
        statuses = result["record_status"].tolist()
        assert statuses.count("customer_without_order") == 2
        assert statuses.count("order_without_customer") == 1

    def test_only_orders(self):
        customers = pd.DataFrame({"customer_id": [10], "name": ["X"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "item": ["a", "b"]})
        result = join_customers_orders(customers, orders)
        assert len(result) == 3
        statuses = result["record_status"].tolist()
        assert statuses.count("customer_without_order") == 1
        assert statuses.count("order_without_customer") == 2


# -------------------------------------------------------------------
# Null key handling
# -------------------------------------------------------------------

class TestNullKey:
    def test_null_in_customers(self):
        customers = pd.DataFrame({"customer_id": [1, None], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "item": ["x", "y"]})
        result = join_customers_orders(customers, orders)
        assert len(result) == 4  # 1 matched, 1 customer_without_order (null), 2 order_without_customer (2)
        null_cust = result[result["customer_id"].isna() & (result["record_status"] == "customer_without_order")]
        assert len(null_cust) == 1

    def test_null_in_orders(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, None], "item": ["x", "y"]})
        result = join_customers_orders(customers, orders)
        assert len(result) == 4
        null_order = result[result["customer_id"].isna() & (result["record_status"] == "order_without_customer")]
        assert len(null_order) == 1

    def test_both_null(self):
        customers = pd.DataFrame({"customer_id": [None], "name": ["A"]})
        orders = pd.DataFrame({"customer_id": [None], "item": ["x"]})
        result = join_customers_orders(customers, orders)
        # Two null rows, each from one side, never matched
        assert len(result) == 2
        statuses = result["record_status"].tolist()
        assert "customer_without_order" in statuses
        assert "order_without_customer" in statuses
        assert "matched" not in statuses

# Overlapping non-key columns
# -------------------------------------------------------------------

class TestOverlappingColumns:
    def test_suffixes_applied(self):
        customers = pd.DataFrame({"customer_id": [1], "value": [10]})
        orders = pd.DataFrame({"customer_id": [1], "value": [20]})
        result = join_customers_orders(customers, orders)
        # The overlapping column 'value' should be renamed to value_customer and value_order
        assert "value_customer" in result.columns
        assert "value_order" in result.columns
        assert "value" not in result.columns
        assert result["value_customer"].iloc[0] == 10
        assert result["value_order"].iloc[0] == 20

    def test_suffixes_with_unmatched(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "value": [10, 30]})
        orders = pd.DataFrame({"customer_id": [1], "value": [20]})
        result = join_customers_orders(customers, orders)
        # Row with customer_id=2 has no order, so value_order should be NA
        row = result[result["customer_id"] == 2]
        assert row["value_customer"].iloc[0] == 30
        assert pd.isna(row["value_order"].iloc[0])

# Sort order and index reset
# -------------------------------------------------------------------

class TestSortingAndIndex:
    def test_sort_preserves_original_order(self):
        # Insert rows out of order in both frames; final sort should be deterministic
        customers = pd.DataFrame({
            "customer_id": [3, 1, 2],
            "name": ["C", "A", "B"]
        })
        orders = pd.DataFrame({
            "customer_id": [2, 3, 5],
            "item": ["y", "x", "z"]
        })
        result = join_customers_orders(customers, orders)
        # After sorting, expect order: rows from customers in original order (3,1,2) then unmatched orders in original order (5)
        # But the sort uses _customer_source_order and _order_source_order; 
        # For matched rows, the sort key is the combined order (customer_source_order takes precedence if available)
        # We'll just check that the index is reset and no duplicate index
        assert result.index.is_monotonic_increasing
        assert result.index[0] == 0

    def test_index_reset(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["A"]})
        orders = pd.DataFrame({"customer_id": [2], "item": ["x"]})
        result = join_customers_orders(customers, orders)
        assert result.index.tolist() == list(range(len(result)))

    def test_deterministic_sort_repeatable(self):
        df1 = pd.DataFrame({"customer_id": [2, 1], "name": ["B", "A"]})
        df2 = pd.DataFrame({"customer_id": [1, 3], "item": ["a", "b"]})
        r1 = join_customers_orders(df1, df2)
        r2 = join_customers_orders(df1, df2)
        assert r1.equals(r2)

# Record status correctness
# -------------------------------------------------------------------

class TestRecordStatusAccuracy:
    def test_matched_requires_both_non_null(self):
        customers = pd.DataFrame({"customer_id": [1, None], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, None], "item": ["x", "y"]})
        result = join_customers_orders(customers, orders)
        matched_rows = result[result["record_status"] == "matched"]
        # Only customer_id=1 qualifies
        assert len(matched_rows) == 1
        assert matched_rows["customer_id"].iloc[0] == 1

    def test_customer_without_order_label(self):
        customers = pd.DataFrame({"customer_id": [2], "name": ["B"]})
        orders = pd.DataFrame({"customer_id": [1], "item": ["x"]})
        result = join_customers_orders(customers, orders)
        unmatched = result[result["record_status"] == "customer_without_order"]
        assert len(unmatched) == 1
        assert unmatched["customer_id"].iloc[0] == 2

    def test_order_without_customer_label(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["A"]})
        orders = pd.DataFrame({"customer_id": [3], "item": ["y"]})
        result = join_customers_orders(customers, orders)
        unmatched = result[result["record_status"] == "order_without_customer"]
        assert len(unmatched) == 1
        assert unmatched["customer_id"].iloc[0] == 3
