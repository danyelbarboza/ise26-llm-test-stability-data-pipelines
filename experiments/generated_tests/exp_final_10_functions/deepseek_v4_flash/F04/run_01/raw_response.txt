import pytest
import pandas as pd
import numpy as np
from ise26.targets import join_customers_orders


class TestJoinCustomersOrders:
    """Suite de testes para a função join_customers_orders."""

    # ---------------------------------------------------------------
    # helpers
    # ---------------------------------------------------------------

    @staticmethod
    def _assert_dataframes_unchanged(original: pd.DataFrame, after: pd.DataFrame) -> None:
        """Verifica que um DataFrame original não foi alterado."""
        assert original.equals(after), "O DataFrame original foi modificado"

    # ---------------------------------------------------------------
    # matched cases
    # ---------------------------------------------------------------

    def test_basic_match(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["Alice", "Bob"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "product": ["A", "B"]})
        result = join_customers_orders(customers, orders)

        # ambas as linhas devem ser matched
        assert list(result["record_status"]) == ["matched", "matched"]
        # order das linhas: como ambas são matched, a ordenação será baseada em _customer_source_order
        # customers original tem índices 0,1 -> customer_source_order = 0,1 -> ordenado por 0,1
        assert result["name"].tolist() == ["Alice", "Bob"]
        # colunas sobrepostas: "product" não está em customers, então não há sufixo
        assert "product" in result.columns
        # colunas não sobrepostas: "name" só em customers, mantém "name"
        assert "name" in result.columns
        # não deve ter colunas _sufixadas desnecessariamente
        assert "name_customer" not in result.columns

    def test_match_with_suffix(self):
        """Colunas sobrepostas recebem sufixo _customer e _order."""
        customers = pd.DataFrame({"customer_id": [1], "value": [100]})
        orders = pd.DataFrame({"customer_id": [1], "value": [200]})
        result = join_customers_orders(customers, orders)

        assert "value_customer" in result.columns
        assert "value_order" in result.columns
        assert result.loc[0, "value_customer"] == 100
        assert result.loc[0, "value_order"] == 200
        assert result.loc[0, "record_status"] == "matched"

    # ---------------------------------------------------------------
    # unmatched customers
    # ---------------------------------------------------------------

    def test_customer_without_order(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["Alice", "Bob"]})
        orders = pd.DataFrame({"customer_id": [1], "product": ["A"]})
        result = join_customers_orders(customers, orders)

        # primeiro (ordem por source) deve ser matched (customer_id=1), segundo customer_without_order (id=2)
        statuses = result["record_status"].tolist()
        assert statuses[0] == "matched"
        assert statuses[1] == "customer_without_order"
        # nome do cliente sem pedido deve ser preservado
        assert result.loc[1, "name"] == "Bob"
        # colunas que só existem em orders devem ser NA na linha do sem pedido
        assert pd.isna(result.loc[1, "product"])

    # ---------------------------------------------------------------
    # unmatched orders
    # ---------------------------------------------------------------

    def test_order_without_customer(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["Alice"]})
        orders = pd.DataFrame({"customer_id": [1, 3], "product": ["A", "C"]})
        result = join_customers_orders(customers, orders)

        statuses = result["record_status"].tolist()
        # primeiro _order_source_order = 0 (customer_id=1) -> matched
        # segundo _order_source_order = 1 (customer_id=3) -> order_without_customer
        assert statuses[0] == "matched"
        assert statuses[1] == "order_without_customer"
        assert result.loc[1, "product"] == "C"
        assert pd.isna(result.loc[1, "name"])

    # ---------------------------------------------------------------
    # null keys
    # ---------------------------------------------------------------

    def test_null_key_customer(self):
        """Cliente com key nula deve ser customer_without_order, nunca matched."""
        customers = pd.DataFrame({"customer_id": [1, None], "name": ["Alice", "Bob"]})
        orders = pd.DataFrame({"customer_id": [1], "product": ["A"]})
        result = join_customers_orders(customers, orders)

        statuses = result["record_status"].tolist()
        # ordem: customer_source_order: 0 (id=1), 1 (id=None)
        # matched, customer_without_order
        assert statuses[0] == "matched"
        assert statuses[1] == "customer_without_order"
        assert result.loc[1, "name"] == "Bob"
        assert pd.isna(result.loc[1, "product"])

    def test_null_key_order(self):
        """Pedido com key nula deve ser order_without_customer, nunca matched."""
        customers = pd.DataFrame({"customer_id": [1], "name": ["Alice"]})
        orders = pd.DataFrame(
            {"customer_id": [1, None], "product": ["A", "B"]}
        )
        result = join_customers_orders(customers, orders)

        statuses = result["record_status"].tolist()
        # order_source_order: 0 (id=1), 1 (id=None) -> matched, order_without_customer
        assert statuses[0] == "matched"
        assert statuses[1] == "order_without_customer"
        assert result.loc[1, "product"] == "B"
        assert pd.isna(result.loc[1, "name"])

    def test_both_null_keys(self):
        """Ambos datasets com keys nulas: cada um aparece como seu respectivo unmatched."""
        customers = pd.DataFrame({"customer_id": [None], "name": ["NoKeyCust"]})
        orders = pd.DataFrame({"customer_id": [None], "product": ["NoKeyOrder"]})
        result = join_customers_orders(customers, orders)

        statuses = result["record_status"].tolist()
        # a ordem: primeiro customer_source_order=0, depois order_source_order=0
        # customer_without_order, order_without_customer
        assert len(result) == 2
        assert statuses[0] == "customer_without_order"
        assert statuses[1] == "order_without_customer"
        # dados preservados
        assert result.loc[0, "name"] == "NoKeyCust"
        assert result.loc[1, "product"] == "NoKeyOrder"
        # coluna ausente
        assert pd.isna(result.loc[0, "product"])
        assert pd.isna(result.loc[1, "name"])

    # ---------------------------------------------------------------
    # mixed scenario
    # ---------------------------------------------------------------

    def test_mixed_scenario(self):
        """Combinação de matched, unmatched e null keys."""
        customers = pd.DataFrame(
            {"customer_id": [1, None, 3], "name": ["A", "B", "C"]}
        )
        orders = pd.DataFrame(
            {"customer_id": [1, 2, None], "product": ["X", "Y", "Z"]}
        )
        result = join_customers_orders(customers, orders)

        # customers: source order 0 (1), 1 (None), 2 (3)
        # orders: source order 0 (1), 1 (2), 2 (None)
        # esperado ordenado por sort_order: customer_source_order para matched/cust, order_source_order para order
        # matched -> sort_order = customer_source_order 0 (também order 0)
        # customer_without_order (id=3): sort_order = customer_source_order 2 (depois de 0 e 1)
        # customer_without_order (null key): sort_order = customer_source_order 1
        # order_without_customer (id=2): sort_order = order_source_order 1
        # order_without_customer (null key): sort_order = order_source_order 2
        # Ordem real: sort_order: 0 (matched), 1 (cust null), 1 (order? cuidado: cust null e order null têm sort_order=1, mas _customer_source_order vs _order_source_order)
        # Como _sort_order é combinado, customer_null: sort_order = 1; order_null: sort_order = 2? Não: orders["customer_id"]=None tem order_source_order=2 (índice 2), então sort_order = 2.
        # Ordem: sort_order 0 -> matched
        # sort_order 1 -> customer_without_order (null key)
        # sort_order 2 -> order_without_customer (null key) e customer_without_order (id=3)? não: customer id=3 tem customer_source_order=2 -> sort_order=2; order null key tem sort_order=2.
        # Desempate por _customer_source_order, _order_source_order, record_status.
        # Para sort_order=2: customer id=3 (_cust_src=2, _ord_src=NaN) e order null (_cust_src=NaN, _ord_src=2)
        # Ordenação por _customer_source_order (NaN vs 2): NaN é tratado como maior (na_position='last'), então order null vem depois? Na prática, o sort_values usa na_position='last', então NaN fica por último.
        # Então a ordem dentro de sort_order=2 será: customer id=3 (cust_src=2) depois order null (cust_src=NaN). Record_status: "customer_without_order" vs "order_without_customer", mas como cust_src difere, essa ordenação prevalece.
        # Portanto: matched, customer_null(id=None), customer(id=3), order_null, order_without_customer(id=2)
        # Vamos verificar status apenas, não a ordem exata, mas podemos testar que cada status aparece.
        statuses = result["record_status"].tolist()
        counts = {s: statuses.count(s) for s in statuses}
        assert counts["matched"] == 1
        assert counts["customer_without_order"] == 2  # id=None e id=3
        assert counts["order_without_customer"] == 2  # id=2 e id=None
        # verificar que os registros com null key não são matched
        null_rows = result[result["record_status"] != "matched"]
        assert all(pd.isna(null_rows["customer_id"]) | (null_rows["record_status"] != "matched"))

    # ---------------------------------------------------------------
    # default key parameter
    # ---------------------------------------------------------------

    def test_default_key(self):
        """Se não passar customer_key, deve usar 'customer_id'."""
        customers = pd.DataFrame({"customer_id": [1], "name": ["Alice"]})
        orders = pd.DataFrame({"customer_id": [1], "product": ["A"]})
        result = join_customers_orders(customers, orders)
        assert result["record_status"].iloc[0] == "matched"

    def test_custom_key_name(self):
        """Usar um nome de chave diferente."""
        customers = pd.DataFrame({"cust_id": [10], "name": ["X"]})
        orders = pd.DataFrame({"cust_id": [10], "product": ["Y"]})
        result = join_customers_orders(customers, orders, customer_key="cust_id")
        assert result["record_status"].iloc[0] == "matched"
        assert "cust_id" in result.columns

    # ---------------------------------------------------------------
    # empty DataFrames
    # ---------------------------------------------------------------

    def test_empty_customers(self):
        customers = pd.DataFrame(columns=["customer_id", "name"])
        orders = pd.DataFrame({"customer_id": [1], "product": ["A"]})
        result = join_customers_orders(customers, orders)
        # apenas orders sem customer
        assert list(result["record_status"]) == ["order_without_customer"]
        assert result["product"].iloc[0] == "A"
        assert pd.isna(result["name"].iloc[0])

    def test_empty_orders(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["Alice"]})
        orders = pd.DataFrame(columns=["customer_id", "product"])
        result = join_customers_orders(customers, orders)
        assert list(result["record_status"]) == ["customer_without_order"]
        assert result["name"].iloc[0] == "Alice"

    def test_both_empty(self):
        customers = pd.DataFrame(columns=["customer_id", "name"])
        orders = pd.DataFrame(columns=["customer_id", "product"])
        result = join_customers_orders(customers, orders)
        assert len(result) == 0
        # colunas esperadas: key, customer columns, order columns, record_status
        expected_cols = {"customer_id", "name", "product", "record_status"}
        assert expected_cols.issubset(set(result.columns))

    # ---------------------------------------------------------------
    # original DataFrames unchanged
    # ---------------------------------------------------------------

    def test_customers_unchanged(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["A"]})
        orders = pd.DataFrame({"customer_id": [1], "product": ["X"]})
        cust_copy = customers.copy()
        join_customers_orders(customers, orders)
        self._assert_dataframes_unchanged(cust_copy, customers)

    def test_orders_unchanged(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["A"]})
        orders = pd.DataFrame({"customer_id": [1], "product": ["X"]})
        ord_copy = orders.copy()
        join_customers_orders(customers, orders)
        self._assert_dataframes_unchanged(ord_copy, orders)

    # ---------------------------------------------------------------
    # output structure
    # ---------------------------------------------------------------

    def test_columns_present(self):
        customers = pd.DataFrame({"customer_id": [1], "name": ["A"]})
        orders = pd.DataFrame({"customer_id": [1], "product": ["X"]})
        result = join_customers_orders(customers, orders)
        # colunas esperadas: customer_id, name, product, record_status
        assert "customer_id" in result.columns
        assert "name" in result.columns
        assert "product" in result.columns
        assert "record_status" in result.columns
        # não deve ter colunas auxiliares do processamento
        aux = {"_sort_order", "_customer_source_order", "_order_source_order", "_merge"}
        assert not aux.intersection(result.columns)

    def test_index_reset(self):
        customers = pd.DataFrame({"customer_id": [1, 2]}, index=[5, 10])
        orders = pd.DataFrame({"customer_id": [1]}, index=[100])
        result = join_customers_orders(customers, orders)
        assert list(result.index) == list(range(len(result)))

    def test_record_status_dtype_object(self):
        customers = pd.DataFrame({"customer_id": [1]})
        orders = pd.DataFrame({"customer_id": [1]})
        result = join_customers_orders(customers, orders)
        assert result["record_status"].dtype == np.object_ or result["record_status"].dtype == "object"

    # ---------------------------------------------------------------
    # deterministic order
    # ---------------------------------------------------------------

    def test_deterministic_order(self):
        """Duas execuções com os mesmos dados produzem o mesmo resultado."""
        customers = pd.DataFrame({"customer_id": [3, 1, 2], "name": ["C", "A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "product": ["X", "Y"]})
        result1 = join_customers_orders(customers, orders)
        result2 = join_customers_orders(customers, orders)
        pd.testing.assert_frame_equal(result1, result2)
