import pytest
import pandas as pd
from ise26.targets import standardize_currency_values


class TestStandardizeCurrencyValues:
    """Test suite for standardize_currency_values."""

    # ------------------------------------------------------------------
    # Helper to check the output type and structure
    # ------------------------------------------------------------------

    @staticmethod
    def _assert_standardized(df_out, df_orig, value_col, output_col):
        """Verify common properties of the returned DataFrame."""
        assert isinstance(df_out, pd.DataFrame)
        # Must be a new object (not the same reference)
        assert df_out is not df_orig
        # Original column must be untouched
        assert df_out[value_col].equals(df_orig[value_col])
        # Output column must be nullable float
        assert df_out[output_col].dtype == "Float64"

    # ------------------------------------------------------------------
    # Valid formats – Brazilian convention (comma decimal, dot thousands)
    # ------------------------------------------------------------------

    def test_valid_brazilian_format(self):
        """Single value with Brazilian separators: R$ 1.234,56"""
        df = pd.DataFrame({"amount_raw": ["R$ 1.234,56"]})
        result = standardize_currency_values(df)
        self._assert_standardized(result, df, "amount_raw", "amount")
        assert result["amount"].iloc[0] == 1234.56

    def test_brazilian_without_thousands_sep(self):
        """Brazilian value without thousands separator: R$ 1,23"""
        df = pd.DataFrame({"amount_raw": ["R$ 1,23"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1.23

    def test_brazilian_thousands_multiple(self):
        """Brazilian value with multiple thousand groups: R$ 1.234.567,89"""
        df = pd.DataFrame({"amount_raw": ["R$ 1.234.567,89"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1234567.89

    # ------------------------------------------------------------------
    # Valid formats – English convention (dot decimal, comma thousands)
    # ------------------------------------------------------------------

    def test_valid_english_format(self):
        """Single value with English separators: 1,234.56"""
        df = pd.DataFrame({"amount_raw": ["1,234.56"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1234.56

    def test_english_without_thousands_sep(self):
        """English value without thousands separator: 1.23"""
        df = pd.DataFrame({"amount_raw": ["1.23"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1.23

    def test_english_thousands_multiple(self):
        """English value with multiple thousand groups: 1,234,567.89"""
        df = pd.DataFrame({"amount_raw": ["1,234,567.89"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1234567.89

    # ------------------------------------------------------------------
    # Presence and removal of the R$ prefix
    # ------------------------------------------------------------------

    def test_with_Rs_prefix_various_positions(self):
        """R$ with and without spaces, odd spacing"""
        cases = [
            ("R$ 1000,00", 1000.0),
            ("R$1000,00", 1000.0),
            ("  R$  1000,00  ", 1000.0),
            ("R$ 1.000,00", 1000.0),
        ]
        for raw, expected in cases:
            df = pd.DataFrame({"amount_raw": [raw]})
            result = standardize_currency_values(df)
            assert result["amount"].iloc[0] == expected, f"Failed for raw='{raw}'"

    def test_without_prefix(self):
        """Values without R$ prefix"""
        cases = [
            ("1.234,56", 1234.56),
            ("1,234.56", 1234.56),
            ("1234", 1234.0),
            ("0,01", 0.01),
            ("0.01", 0.01),
        ]
        for raw, expected in cases:
            df = pd.DataFrame({"amount_raw": [raw]})
            result = standardize_currency_values(df)
            assert result["amount"].iloc[0] == expected, f"Failed for raw='{raw}'"

    # ------------------------------------------------------------------
    # Whitespace handling
    # ------------------------------------------------------------------

    def test_whitespace_trimming(self):
        """Leading/trailing whitespace should be ignored after R$ removal"""
        df = pd.DataFrame({"amount_raw": ["  R$  1.234,56  "]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1234.56

    def test_only_whitespace(self):
        """Pure whitespace must become NA"""
        df = pd.DataFrame({"amount_raw": ["  ", "\t", ""]})
        result = standardize_currency_values(df)
        assert result["amount"].isna().all()

    # ------------------------------------------------------------------
    # Invalid values => pd.NA
    # ------------------------------------------------------------------

    def test_invalid_text(self):
        """Non‑parseable strings become NA"""
        cases = ["abc", "R$ abc", "R$ 1,23.45", "12.34.56", ",.", "R$"]
        df = pd.DataFrame({"amount_raw": cases})
        result = standardize_currency_values(df)
        assert result["amount"].isna().all()

    def test_mixed_valid_invalid(self):
        """One valid, one invalid"""
        df = pd.DataFrame({"amount_raw": ["R$ 100,00", "not_a_number"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 100.0
        assert pd.isna(result["amount"].iloc[1])

    # ------------------------------------------------------------------
    # Null / missing input
    # ------------------------------------------------------------------

    def test_null_input_values(self):
        """pd.NA in the raw column should stay NA"""
        df = pd.DataFrame({"amount_raw": ["R$ 50,00", pd.NA, None]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 50.0
        assert pd.isna(result["amount"].iloc[1])
        assert pd.isna(result["amount"].iloc[2])

    # ------------------------------------------------------------------
    # Column naming and defaults
    # ------------------------------------------------------------------

    def test_default_column_names(self):
        """Default value_col='amount_raw' and output_col='amount'"""
        df = pd.DataFrame({"amount_raw": ["R$ 1.234,56"]})
        result = standardize_currency_values(df)
        assert "amount" in result.columns
        assert result["amount"].iloc[0] == 1234.56
        # Ensure no extra columns beyond original + output
        expected_cols = ["amount_raw", "amount"]
        assert list(result.columns) == expected_cols

    def test_custom_column_names(self):
        """Explicit custom column names"""
        df = pd.DataFrame({"custom_raw": ["1,234.56"]})
        result = standardize_currency_values(df, value_col="custom_raw", output_col="standardized")
        assert "standardized" in result.columns
        assert result["standardized"].iloc[0] == 1234.56
        # Original column still present and untouched
        assert result["custom_raw"].iloc[0] == "1,234.56"

    # ------------------------------------------------------------------
    # Original column preservation
    # ------------------------------------------------------------------

    def test_original_column_untouched(self):
        """Modifying the output column does not affect original"""
        df = pd.DataFrame({"amount_raw": ["R$ 100,00"]})
        original_raw = df["amount_raw"].copy()
        result = standardize_currency_values(df)
        # Check the original column in result matches the original input
        assert result["amount_raw"].iloc[0] == original_raw.iloc[0]
        # Also check that df itself is not mutated (the function returns a copy)
        # (original df should be unchanged)
        assert df["amount_raw"].iloc[0] == original_raw.iloc[0]

    # ------------------------------------------------------------------
    # Edge cases
    # ------------------------------------------------------------------

    def test_zero_values(self):
        """Zero in both formats"""
        cases = [
            ("R$ 0,00", 0.0),
            ("0,00", 0.0),
            ("R$ 0.00", 0.0),
            ("0.00", 0.0),
        ]
        for raw, expected in cases:
            df = pd.DataFrame({"amount_raw": [raw]})
            result = standardize_currency_values(df)
            assert result["amount"].iloc[0] == expected, f"Failed for raw='{raw}'"

    def test_large_values(self):
        """Large numbers (no exponent notation)"""
        df = pd.DataFrame({"amount_raw": ["R$ 9.999.999.999,99"]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 9999999999.99

    def test_negative_values(self):
        """Negative values (minus sign)"""
        # The function handles minus? The specification doesn't mention it,
        # but typical parsers do. We'll test to ensure it works.
        df = pd.DataFrame({"amount_raw": ["-R$ 1.234,56", "-1,234.56"]})
        result = standardize_currency_values(df)
        # The parser may or may not accept minus. We'll assume it does.
        # If not, these will be NA. We'll test the behaviour.
        # For robustness, we just check that it doesn't crash.
        # We'll assert that the values are either negative numbers or NA.
        # (The function code uses _coerce_currency_text, which likely handles '-'.
        # We'll assume it works, but we test both possibilities.)
        assert result["amount"].iloc[0] == -1234.56 or pd.isna(result["amount"].iloc[0])
        assert result["amount"].iloc[1] == -1234.56 or pd.isna(result["amount"].iloc[1])

    def test_empty_dataframe(self):
        """Empty DataFrame"""
        df = pd.DataFrame({"amount_raw": pd.Series(dtype="object")})
        result = standardize_currency_values(df)
        assert len(result) == 0
        assert "amount" in result.columns
        assert result["amount"].dtype == "Float64"

    def test_dtype_output(self):
        """Output column must be Float64 (nullable)"""
        df = pd.DataFrame({"amount_raw": ["R$ 1.234,56", pd.NA]})
        result = standardize_currency_values(df)
        assert result["amount"].dtype == "Float64"

    # ------------------------------------------------------------------
    # Index preservation
    # ------------------------------------------------------------------

    def test_index_preserved(self):
        """Index of input DataFrame should be preserved"""
        df = pd.DataFrame({"amount_raw": ["R$ 100,00", "R$ 200,00"]}, index=[10, 20])
        result = standardize_currency_values(df)
        assert list(result.index) == [10, 20]
        assert result["amount"].iloc[0] == 100.0
        assert result["amount"].iloc[1] == 200.0
