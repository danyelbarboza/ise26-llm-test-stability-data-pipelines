import pytest
import pandas as pd
from ise26.targets import standardize_currency_values


def test_brazilian_format_with_reais_prefix():
    """R$ with Brazilian decimal (comma) and thousand (dot) separators."""
    df = pd.DataFrame({"amount_raw": ["R$ 1.234,56"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 1234.56
    assert result["amount"].dtype == "Float64"


def test_brazilian_format_no_prefix():
    """Brazilian numeric string without R$ prefix."""
    df = pd.DataFrame({"amount_raw": ["2.500,75"]})
    result = standardize_currency_values(df, output_col="valor")
    assert result["valor"].iloc[0] == 2500.75

def test_english_format_commas_thousands():
    """English format with comma as thousand separator and dot as decimal."""
    df = pd.DataFrame({"amount_raw": ["1,234.56"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 1234.56


def test_multiple_rows_mixed_formats():
    """Handle a mix of valid Brazilian and English strings."""
    data = {
        "amount_raw": [
            "R$ 100,50",
            "1,000.99",
            "R$ 2.500,00",
            "5.000,20",
        ]
    }
    df = pd.DataFrame(data)
    expected = [100.50, 1000.99, 2500.00, 5000.20]
    result = standardize_currency_values(df)
    assert result["amount"].tolist() == expected


def test_invalid_values_become_na():
    """Strings that cannot be parsed become pd.NA."""
    df = pd.DataFrame({"amount_raw": ["abc", "R$ abc", "", " ", "12,34,56"]})
    result = standardize_currency_values(df)
    assert result["amount"].isna().all()


def test_null_input_values():
    """pd.NA or None in input should produce pd.NA."""
    df = pd.DataFrame({"amount_raw": [pd.NA, None, "R$ 10,00"]})
    result = standardize_currency_values(df)
    expected = [pd.NA, pd.NA, 10.0]
    # Use pandas comparison with nullable
    expected_series = pd.Series(expected, dtype="Float64")
    pd.testing.assert_series_equal(result["amount"], expected_series)


def test_original_column_unchanged():
    """The raw column remains untouched in the returned DataFrame."""
    df = pd.DataFrame({"amount_raw": ["R$ 50,00"]})
    original_raw = df["amount_raw"].iloc[0]
    result = standardize_currency_values(df)
    assert result["amount_raw"].iloc[0] == original_raw


def test_default_column_names():
    """Default column names remain 'amount_raw' and 'amount'."""
    df = pd.DataFrame({"amount_raw": ["R$ 9,99"]})
    result = standardize_currency_values(df)
    assert "amount" in result.columns
    assert "amount_raw" in result.columns
    assert result["amount"].iloc[0] == 9.99


def test_custom_column_names():
    """Custom input and output column names work."""
    df = pd.DataFrame({"raw": ["R$ 1.000,00"]})
    result = standardize_currency_values(df, value_col="raw", output_col="cleaned")
    assert "raw" in result.columns
    assert "cleaned" in result.columns
    assert result["cleaned"].iloc[0] == 1000.0


def test_output_dtype_is_float64():
    """Output column dtype is always nullable Float64."""
    df = pd.DataFrame({"amount_raw": ["R$ 1,23", "invalid", pd.NA]})
    result = standardize_currency_values(df)
    assert result["amount"].dtype == "Float64"


def test_return_new_dataframe():
    """Function returns a new DataFrame, not modifying input."""
    df = pd.DataFrame({"amount_raw": ["R$ 10,00"]})
    original_id = id(df)
    result = standardize_currency_values(df)
    assert id(result) != original_id


def test_whitespace_removal():
    """Leading/trailing spaces and spaces after R$ are removed."""
    df = pd.DataFrame({"amount_raw": ["  R$   50,00  ", "  R$ 100,50"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 50.0
    assert result["amount"].iloc[1] == 100.5


def test_integer_string():
    """Integer strings without decimals work."""
    df = pd.DataFrame({"amount_raw": ["R$ 1000", "500"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 1000.0
    assert result["amount"].iloc[1] == 500.0


def test_only_negative_sign_not_supported():
    """Negative numbers are not mentioned; assume they become NA."""
    df = pd.DataFrame({"amount_raw": ["-R$ 50,00"]})
    result = standardize_currency_values(df)
    assert result["amount"].isna().iloc[0]


def test_multiple_thousand_separators_invalid():
    """More than one thousand separator (e.g. '1.234.56') invalid."""
    df = pd.DataFrame({"amount_raw": ["R$ 1.234.56"]})
    result = standardize_currency_values(df)
    assert result["amount"].isna().iloc[0]


def test_comma_at_end_invalid():
    """String ending with comma is invalid."""
    df = pd.DataFrame({"amount_raw": ["R$ 100,"]})
    result = standardize_currency_values(df)
    assert result["amount"].isna().iloc[0]
