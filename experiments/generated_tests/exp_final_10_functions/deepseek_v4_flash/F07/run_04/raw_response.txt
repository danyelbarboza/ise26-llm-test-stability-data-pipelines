import pytest
import pandas as pd
import numpy as np
from ise26.targets import parse_order_items_json

class TestParseOrderItemsJson:
    """Test suite for parse_order_items_json function."""

    def test_basic_explosion(self):
        """Explode two orders with two items each."""
        data = {
            'order_id': [100, 200],
            'items_json': [
                '[{"sku":"A1","quantity":2,"unit_price":10.5},{"sku":"B2","quantity":1,"unit_price":20.0}]',
                '[{"sku":"C3","quantity":3,"unit_price":15.0},{"sku":"D4","quantity":4,"unit_price":5.5}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            'order_id': [100, 100, 200, 200],
            'sku': ['A1', 'B2', 'C3', 'D4'],
            'quantity': [2.0, 1.0, 3.0, 4.0],
            'unit_price': [10.5, 20.0, 15.0, 5.5],
            'item_total': [21.0, 20.0, 45.0, 22.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_skip_invalid_json_payloads(self):
        """Skip rows with non-list, missing, null, or empty JSON."""
        data = {
            'order_id': [1, 2, 3, 4, 5],
            'items_json': [
                '[{"sku":"X","quantity":1,"unit_price":10}]',
                None,
                '',
                '{"not":"a list"}',
                '123'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            'order_id': [1],
            'sku': ['X'],
            'quantity': [1.0],
            'unit_price': [10.0],
            'item_total': [10.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_skip_empty_list(self):
        """Skip rows where JSON is an empty list."""
        data = {
            'order_id': [10, 20],
            'items_json': [
                '[]',
                '[{"sku":"Y","quantity":2,"unit_price":3}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            'order_id': [20],
            'sku': ['Y'],
            'quantity': [2.0],
            'unit_price': [3.0],
            'item_total': [6.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_sku_becomes_na(self):
        """Missing sku should become pd.NA."""
        data = {
            'order_id': [1],
            'items_json': [
                '[{"quantity":5,"unit_price":2}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        assert result['sku'].iloc[0] is pd.NA

    def test_missing_or_invalid_quantity_unit_price_default_to_zero(self):
        """Missing or non-numeric quantity/unit_price become zero."""
        data = {
            'order_id': [1, 2, 3],
            'items_json': [
                '[{"sku":"A","unit_price":10}]',                    # missing quantity
                '[{"sku":"B","quantity":"abc","unit_price":5}]',    # invalid quantity string
                '[{"sku":"C","quantity":2}]'                        # missing unit_price
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            'order_id': [1, 2, 3],
            'sku': ['A', 'B', 'C'],
            'quantity': [0.0, 0.0, 2.0],
            'unit_price': [10.0, 0.0, 0.0],
            'item_total': [0.0, 0.0, 0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_dataframe_returns_empty_with_correct_dtypes(self):
        """Empty DataFrame should return empty DataFrame with correct columns and dtypes."""
        df = pd.DataFrame({'order_id': pd.Series(dtype='int64'), 'items_json': pd.Series(dtype='object')})
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            'order_id': pd.Series(dtype='int64'),
            'sku': pd.Series(dtype='object'),
            'quantity': pd.Series(dtype='float64'),
            'unit_price': pd.Series(dtype='float64'),
            'item_total': pd.Series(dtype='float64')
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_order_id_col(self):
        """Custom order_id_col should be used as output column name."""
        data = {
            'order_number': [99],
            'items_json': [
                '[{"sku":"Z","quantity":1,"unit_price":7}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df, order_id_col='order_number')
        assert 'order_number' in result.columns
        assert result['order_number'].iloc[0] == 99

    def test_preserves_order_id_dtype(self):
        """Order id column dtype is preserved from input."""
        data = {
            'order_id': pd.Series([12345], dtype='int64'),
            'items_json': ['[{"sku":"A","quantity":1,"unit_price":1}]']
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        assert result['order_id'].dtype == np.int64

    def test_item_total_computation(self):
        """item_total equals quantity * unit_price."""
        data = {
            'order_id': [1],
            'items_json': [
                '[{"sku":"X","quantity":3,"unit_price":4.5}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        assert result['item_total'].iloc[0] == 13.5

    def test_ordering_preserved(self):
        """Items appear in order of original rows and within each JSON list."""
        data = {
            'order_id': [4, 1],
            'items_json': [
                '[{"sku":"second","quantity":1,"unit_price":1},{"sku":"third","quantity":2,"unit_price":2}]',
                '[{"sku":"first","quantity":1,"unit_price":1}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected_sku_order = ['second', 'third', 'first']
        assert result['sku'].tolist() == expected_sku_order

    def test_all_columns_present_and_ordered(self):
        """Result has exactly columns [order_id, sku, quantity, unit_price, item_total] in that order."""
        data = {
            'order_id': [1],
            'items_json': ['[{"sku":"A","quantity":1,"unit_price":1}]']
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected_columns = ['order_id', 'sku', 'quantity', 'unit_price', 'item_total']
        assert result.columns.tolist() == expected_columns

    def test_non_numeric_quantity_unit_price_as_strings_digit(self):
        """String digits should still be coerced to numbers."""
        data = {
            'order_id': [1],
            'items_json': [
                '[{"sku":"A","quantity":"2","unit_price":"3.5"}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        assert result['quantity'].iloc[0] == 2.0
        assert result['unit_price'].iloc[0] == 3.5
        assert result['item_total'].iloc[0] == 7.0

    def test_mixed_valid_and_invalid_items_in_same_list(self):
        """Invalid items within a valid list are skipped? Actually function processes each item individually; missing keys become defaults. So all items are valid."""
        # No test needed for "skipping" an item inside a list; function does not skip items with missing fields, it defaults.
        # Instead test that missing sku is NA but item still included.
        data = {
            'order_id': [1],
            'items_json': [
                '[{"quantity":1,"unit_price":2},{"sku":"B","quantity":3,"unit_price":4}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        assert len(result) == 2
        assert pd.isna(result['sku'].iloc[0])
        assert result['sku'].iloc[1] == 'B'

    def test_extra_columns_in_items_ignored(self):
        """Extra keys in JSON object are ignored."""
        data = {
            'order_id': [1],
            'items_json': [
                '[{"sku":"A","quantity":1,"unit_price":2,"extra":"x"}]'
            ]
        }
        df = pd.DataFrame(data)
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            'order_id': [1],
            'sku': ['A'],
            'quantity': [1.0],
            'unit_price': [2.0],
            'item_total': [2.0]
        })
        pd.testing.assert_frame_equal(result, expected)
