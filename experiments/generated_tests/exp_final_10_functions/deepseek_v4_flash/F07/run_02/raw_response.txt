import pytest
import pandas as pd
from ise26.targets import parse_order_items_json


class TestParseOrderItemsJson:
    """Test suite for parse_order_items_json."""

    def test_single_order_single_item(self):
        """Basic case: one order with one valid item."""
        df = pd.DataFrame({
            "order_id": [1001],
            "items_json": ['[{"sku": "A1", "quantity": 2, "unit_price": 10.5}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1001],
            "sku": ["A1"],
            "quantity": [2.0],
            "unit_price": [10.5],
            "item_total": [21.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_multiple_orders_multiple_items(self):
        """Several orders, each with multiple items."""
        df = pd.DataFrame({
            "order_id": [1, 2],
            "items_json": [
                '[{"sku": "X", "quantity": 1, "unit_price": 20.0}, {"sku": "Y", "quantity": 3, "unit_price": 5.0}]',
                '[{"sku": "Z", "quantity": 4, "unit_price": 7.5}]'
            ]
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1, 1, 2],
            "sku": ["X", "Y", "Z"],
            "quantity": [1.0, 3.0, 4.0],
            "unit_price": [20.0, 5.0, 7.5],
            "item_total": [20.0, 15.0, 30.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_order_id_column(self):
        """Use a custom order identifier column name."""
        df = pd.DataFrame({
            "custom_id": ["abc", "def"],
            "items_json": [
                '[{"sku": "S1", "quantity": 5, "unit_price": 2.0}]',
                '[{"sku": "S2", "quantity": 1, "unit_price": 100.0}]'
            ]
        })
        result = parse_order_items_json(df, json_col="items_json", order_id_col="custom_id")
        expected = pd.DataFrame({
            "custom_id": ["abc", "def"],
            "sku": ["S1", "S2"],
            "quantity": [5.0, 1.0],
            "unit_price": [2.0, 100.0],
            "item_total": [10.0, 100.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_json_list(self):
        """JSON list is empty: no items produced."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": pd.Series(dtype="int64"),
            "sku": pd.Series(dtype="object"),
            "quantity": pd.Series(dtype="float64"),
            "unit_price": pd.Series(dtype="float64"),
            "item_total": pd.Series(dtype="float64")
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_json_payload(self):
        """Missing (NaN) JSON payload is skipped."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": [float("nan")]
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": pd.Series(dtype="float64"),
            "sku": pd.Series(dtype="object"),
            "quantity": pd.Series(dtype="float64"),
            "unit_price": pd.Series(dtype="float64"),
            "item_total": pd.Series(dtype="float64")
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_json_string(self):
        """Invalid JSON string (not a list) is skipped."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['{"not":"a list"}']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": pd.Series(dtype="int64"),
            "sku": pd.Series(dtype="object"),
            "quantity": pd.Series(dtype="float64"),
            "unit_price": pd.Series(dtype="float64"),
            "item_total": pd.Series(dtype="float64")
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_non_list_json(self):
        """JSON that is not a list (e.g., a number) is skipped."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['123']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": pd.Series(dtype="int64"),
            "sku": pd.Series(dtype="object"),
            "quantity": pd.Series(dtype="float64"),
            "unit_price": pd.Series(dtype="float64"),
            "item_total": pd.Series(dtype="float64")
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_sku_becomes_na(self):
        """Missing 'sku' in an item becomes pd.NA."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"quantity": 2, "unit_price": 10.0}]']
        })
        result = parse_order_items_json(df)
        assert result["sku"].iloc[0] is pd.NA

    def test_invalid_quantity_treated_as_zero(self):
        """Quantity that cannot be parsed (e.g., string) becomes 0."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "A", "quantity": "bad", "unit_price": 10.0}]']
        })
        result = parse_order_items_json(df)
        assert result["quantity"].iloc[0] == 0.0
        assert result["item_total"].iloc[0] == 0.0

    def test_invalid_unit_price_treated_as_zero(self):
        """Unit price that cannot be parsed becomes 0."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "A", "quantity": 5, "unit_price": "invalid"}]']
        })
        result = parse_order_items_json(df)
        assert result["unit_price"].iloc[0] == 0.0
        assert result["item_total"].iloc[0] == 0.0

    def test_missing_quantity_defaults_to_zero(self):
        """Missing 'quantity' key becomes 0."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "A", "unit_price": 10.0}]']
        })
        result = parse_order_items_json(df)
        assert result["quantity"].iloc[0] == 0.0
        assert result["item_total"].iloc[0] == 0.0

    def test_missing_unit_price_defaults_to_zero(self):
        """Missing 'unit_price' key becomes 0."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "A", "quantity": 3}]']
        })
        result = parse_order_items_json(df)
        assert result["unit_price"].iloc[0] == 0.0
        assert result["item_total"].iloc[0] == 0.0

    def test_dtype_of_output_columns(self):
        """Verify dtype of each column in the returned DataFrame."""
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "X", "quantity": 1, "unit_price": 1.0}]']
        })
        result = parse_order_items_json(df)
        assert result["order_id"].dtype == "int64"
        assert result["sku"].dtype == "object"
        assert result["quantity"].dtype == "float64"
        assert result["unit_price"].dtype == "float64"
        assert result["item_total"].dtype == "float64"

    def test_order_preserved_across_items(self):
        """Items from the same order appear consecutively, and order across orders is preserved."""
        df = pd.DataFrame({
            "order_id": [2, 1],
            "items_json": [
                '[{"sku": "B", "quantity": 1, "unit_price": 1.0}, {"sku": "C", "quantity": 2, "unit_price": 2.0}]',
                '[{"sku": "A", "quantity": 1, "unit_price": 1.0}]'
            ]
        })
        result = parse_order_items_json(df)
        # The function sorts by source order (original row index) and item order.
        # Input order: row0: order_id=2, row1: order_id=1
        # So expected order: order_id=2 items first, then order_id=1 items.
        expected = pd.DataFrame({
            "order_id": [2, 2, 1],
            "sku": ["B", "C", "A"],
            "quantity": [1.0, 2.0, 1.0],
            "unit_price": [1.0, 2.0, 1.0],
            "item_total": [1.0, 4.0, 1.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_all_invalid_rows_return_empty(self):
        """If every row has an invalid JSON, output is empty DataFrame with correct dtypes."""
        df = pd.DataFrame({
            "order_id": [1, 2],
            "items_json": ['not json', '{"not": "list"}']
        })
        result = parse_order_items_json(df)
        assert result.empty
        assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
        assert result["order_id"].dtype == "int64"
        assert result["sku"].dtype == "object"

    def test_no_data_produces_empty(self):
        """Empty input DataFrame produces empty output."""
        df = pd.DataFrame({
            "order_id": pd.Series(dtype="int64"),
            "items_json": pd.Series(dtype="object")
        })
        result = parse_order_items_json(df)
        assert result.empty
        assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
