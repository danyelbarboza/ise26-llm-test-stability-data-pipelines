import pandas as pd
import pytest
from ise26.targets import parse_order_items_json


def _approx_equal(series1, series2):
    """Check two numeric series for approximate equality (e.g., due to floating point)."""
    return (series1.fillna(0) - series2.fillna(0)).abs().max() < 1e-9


class TestParseOrderItemsJson:
    """Test suite for parse_order_items_json."""

    # ------------------------------------------------------------------
    # Valid single order with one item
    # ------------------------------------------------------------------
    def test_single_order_one_item(self):
        df = pd.DataFrame({
            "order_id": [1001],
            "items_json": ['[{"sku": "A1", "quantity": 2, "unit_price": 10.5}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1001],
            "sku": ["A1"],
            "quantity": [2.0],
            "unit_price": [10.5],
            "item_total": [21.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Single order with multiple items
    # ------------------------------------------------------------------
    def test_single_order_multiple_items(self):
        df = pd.DataFrame({
            "order_id": [2002],
            "items_json": [
                '[{"sku": "B1", "quantity": 1, "unit_price": 5.0},'
                '{"sku": "B2", "quantity": 3, "unit_price": 2.0}]'
            ]
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [2002, 2002],
            "sku": ["B1", "B2"],
            "quantity": [1.0, 3.0],
            "unit_price": [5.0, 2.0],
            "item_total": [5.0, 6.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Multiple orders, each with items
    # ------------------------------------------------------------------
    def test_multiple_orders(self):
        df = pd.DataFrame({
            "order_id": [10, 20],
            "items_json": [
                '[{"sku": "X", "quantity": 2, "unit_price": 1.0}]',
                '[{"sku": "Y", "quantity": 5, "unit_price": 0.5},'
                '{"sku": "Z", "unit_price": 3.0}]'   # missing quantity -> 0
            ]
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [10, 20, 20],
            "sku": ["X", "Y", "Z"],
            "quantity": [2.0, 5.0, 0.0],
            "unit_price": [1.0, 0.5, 3.0],
            "item_total": [2.0, 2.5, 0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Invalid JSON (not a list) -> skip row
    # ------------------------------------------------------------------
    def test_invalid_json_skips_row(self):
        df = pd.DataFrame({
            "order_id": [1, 2],
            "items_json": ['{"single": "object"}', '[{"sku": "A"}]']
        })
        result = parse_order_items_json(df)
        # Only second row yields one item
        expected = pd.DataFrame({
            "order_id": [2],
            "sku": ["A"],
            "quantity": [0.0],
            "unit_price": [0.0],
            "item_total": [0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Empty JSON list -> skip row
    # ------------------------------------------------------------------
    def test_empty_list_skips_row(self):
        df = pd.DataFrame({
            "order_id": [1, 2],
            "items_json": ['[]', '[{"sku": "A", "quantity": 2, "unit_price": 3}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [2],
            "sku": ["A"],
            "quantity": [2.0],
            "unit_price": [3.0],
            "item_total": [6.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Missing JSON (NaN) -> skip row
    # ------------------------------------------------------------------
    def test_missing_json_skips_row(self):
        df = pd.DataFrame({
            "order_id": [1, 2],
            "items_json": [None, '[{"sku": "B", "quantity": 1, "unit_price": 4}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [2],
            "sku": ["B"],
            "quantity": [1.0],
            "unit_price": [4.0],
            "item_total": [4.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Missing sku -> becomes pd.NA
    # ------------------------------------------------------------------
    def test_missing_sku_becomes_na(self):
        df = pd.DataFrame({
            "order_id": [42],
            "items_json": ['[{"quantity": 1, "unit_price": 2.5}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [42],
            "sku": [pd.NA],
            "quantity": [1.0],
            "unit_price": [2.5],
            "item_total": [2.5]
        })
        # Note: sku dtype is object; use check_dtype=True
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Invalid quantity / unit_price -> zero
    # ------------------------------------------------------------------
    def test_invalid_quantity_becomes_zero(self):
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "P", "quantity": "abc", "unit_price": 5}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1],
            "sku": ["P"],
            "quantity": [0.0],
            "unit_price": [5.0],
            "item_total": [0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_unit_price_becomes_zero(self):
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "Q", "quantity": 3, "unit_price": "N/A"}]']
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1],
            "sku": ["Q"],
            "quantity": [3.0],
            "unit_price": [0.0],
            "item_total": [0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Custom column names
    # ------------------------------------------------------------------
    def test_custom_column_names(self):
        df = pd.DataFrame({
            "order_ref": [55],
            "json_payload": ['[{"sku": "R", "quantity": 4, "unit_price": 1.25}]']
        })
        result = parse_order_items_json(
            df, json_col="json_payload", order_id_col="order_ref"
        )
        expected = pd.DataFrame({
            "order_ref": [55],
            "sku": ["R"],
            "quantity": [4.0],
            "unit_price": [1.25],
            "item_total": [5.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Empty input -> returns empty DataFrame with correct dtypes
    # ------------------------------------------------------------------
    def test_all_invalid_returns_empty_dataframe(self):
        df = pd.DataFrame({
            "order_id": [1, 2],
            "items_json": ['invalid', None]
        })
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": pd.Series(dtype="int64"),
            "sku": pd.Series(dtype="object"),
            "quantity": pd.Series(dtype="float64"),
            "unit_price": pd.Series(dtype="float64"),
            "item_total": pd.Series(dtype="float64")
        })
        pd.testing.assert_frame_equal(result, expected)

    # ------------------------------------------------------------------
    # Verify order of rows is preserved (original row order and item order)
    # ------------------------------------------------------------------
    def test_row_order_preserved(self):
        df = pd.DataFrame({
            "order_id": [3, 1, 2],
            "items_json": [
                '[{"sku": "C", "quantity": 1, "unit_price": 1}]',
                '[{"sku": "A", "quantity": 2, "unit_price": 2},'
                '{"sku": "B", "quantity": 3, "unit_price": 1}]',
                '[{"sku": "D", "quantity": 4, "unit_price": 1}]'
            ]
        })
        result = parse_order_items_json(df)
        # Expected order: items from row 0 first (order_id=3, sku C),
        # then items from row 1 (order_id=1, sku A, then B),
        # then items from row 2 (order_id=2, sku D)
        expected_order_ids = [3, 1, 1, 2]
        expected_skus = ["C", "A", "B", "D"]
        assert list(result["order_id"]) == expected_order_ids
        assert list(result["sku"]) == expected_skus

    # ------------------------------------------------------------------
    # Verify output columns are in the correct order
    # ------------------------------------------------------------------
    def test_output_column_order(self):
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "X", "quantity": 1, "unit_price": 1}]']
        })
        result = parse_order_items_json(df)
        expected_cols = ["order_id", "sku", "quantity", "unit_price", "item_total"]
        assert list(result.columns) == expected_cols

    # ------------------------------------------------------------------
    # Verify no side effects on the input DataFrame
    # ------------------------------------------------------------------
    def test_no_mutation_of_input(self):
        df = pd.DataFrame({
            "order_id": [1],
            "items_json": ['[{"sku": "A", "quantity": 2, "unit_price": 3}]']
        })
        original = df.copy()
        parse_order_items_json(df)
        pd.testing.assert_frame_equal(df, original, check_dtype=True)
