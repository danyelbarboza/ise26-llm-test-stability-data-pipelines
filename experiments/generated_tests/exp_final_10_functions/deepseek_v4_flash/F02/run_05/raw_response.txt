import pandas as pd
import numpy as np
import pytest
from ise26.targets import deduplicate_events


def test_empty_dataframe():
    """Deve retornar um DataFrame vazio quando a entrada é vazia."""
    df = pd.DataFrame()
    result = deduplicate_events(df)
    assert isinstance(result, pd.DataFrame)
    assert result.empty


def test_no_duplicates():
    """DataFrame sem duplicatas deve permanecer inalterado (exceto reset_index)."""
    df = pd.DataFrame({
        "event_id": ["A", "B", "C"],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "value": [1, 2, 3]
    })
    expected = df.copy()
    result = deduplicate_events(df)
    # Verificar que as linhas são as mesmas (ordem original e valores)
    pd.testing.assert_frame_equal(result, expected)


def test_duplicates_keep_most_recent():
    """Para mesmo id, manter a linha com timestamp mais recente."""
    df = pd.DataFrame({
        "event_id": ["X", "X", "X"],
        "updated_at": ["2023-01-01", "2023-01-03", "2023-01-02"],
        "value": [10, 20, 30]
    })
    result = deduplicate_events(df)
    # A linha mais recente (index 1) deve ser mantida
    expected = df.iloc[[1]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_timestamp_tie_keep_last_in_order():
    """Para timestamps iguais, manter a última linha na ordem original."""
    df = pd.DataFrame({
        "event_id": ["Y", "Y", "Y"],
        "updated_at": ["2023-01-01", "2023-01-01", "2023-01-01"],
        "value": [100, 200, 300]
    })
    result = deduplicate_events(df)
    # Última linha (valor 300) deve ser mantida
    expected = df.iloc[[2]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_timestamps_treated_as_older():
    """Timestamps inválidos (ex: NaT) devem ser tratados como mais antigos que válidos."""
    df = pd.DataFrame({
        "event_id": ["Z", "Z"],
        "updated_at": [pd.NaT, "2023-01-10"],
        "value": [1, 2]
    })
    result = deduplicate_events(df)
    # A linha com timestamp válido deve ser mantida
    expected = df.iloc[[1]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_mixed_invalid_and_valid_keeps_valid():
    """Se há timestamp inválido e outro válido, manter o válido."""
    df = pd.DataFrame({
        "event_id": ["A", "A", "B", "B"],
        "updated_at": ["invalid", "2023-01-01", pd.NaT, "2023-01-02"],
        "value": [10, 20, 30, 40]
    })
    result = deduplicate_events(df)
    # Para A, manter linha 1; para B, manter linha 3
    expected = df.iloc[[1, 3]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_null_ids_preserved():
    """Linhas com id nulo (NaN, None) devem ser preservadas independentemente."""
    df = pd.DataFrame({
        "event_id": [np.nan, np.nan, "C"],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "value": [1, 2, 3]
    })
    result = deduplicate_events(df)
    # As duas linhas com id nulo devem permanecer, a linha com C também
    expected = df.drop_duplicates().reset_index(drop=True)  # sem duplicatas, todas ficam
    pd.testing.assert_frame_equal(result, expected)


def test_null_ids_not_grouped():
    """Dois ids nulos diferentes (ambos NaN) não devem ser considerados duplicatas."""
    df = pd.DataFrame({
        "event_id": [np.nan, np.nan],
        "updated_at": ["2023-01-01", "2023-01-02"],
        "value": [10, 20]
    })
    result = deduplicate_events(df)
    # Ambas as linhas devem permanecer
    expected = df.reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_original_order_preserved():
    """As linhas preservadas devem manter a ordem do DataFrame original."""
    df = pd.DataFrame({
        "event_id": ["A", "B", "A", "C", "B"],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04", "2023-01-05"],
        "value": [1, 2, 3, 4, 5]
    })
    result = deduplicate_events(df)
    # Esperado: A (linha 2, valor 3), B (linha 4, valor 5), C (linha 3, valor 4)
    # Ordem original: A (2), B (4), C (3) -> mas ordem no resultado: index 0: A, 1: B, 2: C?
    # Ordem baseada em _original_order: A aparece primeiro em linha 0 (preservado), mas A duplicata foi linha 2 removida? Na verdade duplicatas: A: linhas 0 e 2; B: linhas 1 e 4. Mantém últimas: A linha 2 (ordem 2), B linha 4 (ordem 4). C única linha 3 (ordem 3). Ordenando por _original_order: ordem 2 (A), 3 (C), 4 (B). Então resultado: A,C,B.
    expected_data = {
        "event_id": ["A", "C", "B"],
        "updated_at": ["2023-01-03", "2023-01-04", "2023-01-05"],
        "value": [3, 4, 5]
    }
    expected = pd.DataFrame(expected_data)
    pd.testing.assert_frame_equal(result, expected)


def test_custom_column_names():
    """Usar nomes de coluna personalizados."""
    df = pd.DataFrame({
        "entity": ["X", "X", "Y"],
        "ts": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "info": [1, 2, 3]
    })
    result = deduplicate_events(df, id_col="entity", timestamp_col="ts")
    expected = df.iloc[[1, 2]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_input_dataframe_not_mutated():
    """A função não deve modificar o DataFrame original."""
    df = pd.DataFrame({
        "event_id": ["A", "A"],
        "updated_at": ["2023-01-01", "2023-01-02"],
        "value": [1, 2]
    })
    original = df.copy()
    deduplicate_events(df)
    pd.testing.assert_frame_equal(df, original)


def test_datetime_dtype_kept():
    """Se timestamp for datetime, após a operação o dtype deve permanecer igual (se válido)."""
    df = pd.DataFrame({
        "event_id": ["A", "A"],
        "updated_at": pd.to_datetime(["2023-01-01", "2023-01-02"]),
        "value": [1, 2]
    })
    result = deduplicate_events(df)
    assert pd.api.types.is_datetime64_any_dtype(result["updated_at"])


def test_string_timestamp_converted_back_to_object():
    """Timestamp como string permanece como object no resultado (não é convertido para datetime)."""
    df = pd.DataFrame({
        "event_id": ["A", "A"],
        "updated_at": ["2023-01-01", "2023-01-02"],
        "value": [1, 2]
    })
    result = deduplicate_events(df)
    assert result["updated_at"].dtype == object
    assert result["updated_at"].iloc[0] == "2023-01-02"


def test_mix_of_none_and_nan_ids():
    """IDs nulos podem ser None ou NaN, ambos devem ser preservados."""
    df = pd.DataFrame({
        "event_id": [None, np.nan, "A"],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "value": [1, 2, 3]
    })
    result = deduplicate_events(df)
    # Todas as linhas devem permanecer
    expected = df.reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_multiple_duplicates_same_id():
    """Várias duplicatas do mesmo id, manter a mais recente."""
    df = pd.DataFrame({
        "event_id": ["X", "X", "X", "X"],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-04", "2023-01-03"],
        "value": [10, 20, 30, 40]
    })
    result = deduplicate_events(df)
    # Mais recente é linha com 2023-01-04 (valor 30)
    expected = df.iloc[[2]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)

    """Nenhuma duplicata, mas com nulos, deve retornar o mesmo."""
