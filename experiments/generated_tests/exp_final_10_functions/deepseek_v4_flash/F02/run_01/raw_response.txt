import pandas as pd
import pytest
from ise26.targets import deduplicate_events


@pytest.fixture
def base_df():
    return pd.DataFrame({
        "event_id": [1, 2, 1, 3, None],
        "updated_at": [
            "2023-01-01",
            "2023-01-02",
            "2023-01-03",
            "2023-01-01",
            "2023-01-04",
        ],
        "value": ["a", "b", "c", "d", "e"],
    })


class TestDeduplicateEvents:
    def test_basic_deduplication(self, base_df):
        """Keep most recent row per event_id."""
        result = deduplicate_events(base_df)
        # event 1 has two rows (idx 0 and 2), keep idx 2 (newer timestamp)
        # event 2 has one row
        # event 3 has one row
        # null event_id preserved
        expected = base_df.iloc[[2, 1, 3, 4]].reset_index(drop=True)
        pd.testing.assert_frame_equal(result, expected)

    def test_timestamp_tie_keeps_last_row(self):
        """When timestamps tie, keep last row in original order."""
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-01", "2023-01-01"],
            "val": ["x", "y", "z"],
        })
        result = deduplicate_events(df)
        expected = df.iloc[[2]].reset_index(drop=True)  # last row kept
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_timestamp_treated_as_older(self):
        """Invalid timestamps are older than valid timestamps."""
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["invalid", "2023-01-01"],
            "val": ["a", "b"],
        })
        result = deduplicate_events(df)
        expected = df.iloc[[1]].reset_index(drop=True)  # valid kept
        pd.testing.assert_frame_equal(result, expected)

    def test_all_invalid_timestamps_tie(self):
        """When all timestamps invalid, last row kept."""
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["bad", "worse", "invalid"],
            "val": [10, 20, 30],
        })
        result = deduplicate_events(df)
        expected = df.iloc[[2]].reset_index(drop=True)  # last row
        pd.testing.assert_frame_equal(result, expected)

    def test_null_event_ids_preserved(self):
        """Rows with null event_id are always kept."""
        df = pd.DataFrame({
            "event_id": [None, None, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-01", "2023-01-03"],
            "val": ["a", "b", "c", "d"],
        })
        result = deduplicate_events(df)
        # null rows kept as is, event 1 deduplicated (keep newer)
        expected = df.iloc[[0, 1, 3]].reset_index(drop=True)
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_column_names(self):
        """Works with column names other than defaults."""
        df = pd.DataFrame({
            "id": [10, 10, 20],
            "ts": ["2023-01-01", "2023-01-02", "2023-01-01"],
            "data": ["x", "y", "z"],
        })
        result = deduplicate_events(df, id_col="id", timestamp_col="ts")
        expected = df.iloc[[1, 2]].reset_index(drop=True)
        pd.testing.assert_frame_equal(result, expected)

    def test_no_duplicates(self, base_df):
        """Returns same rows (same order) when no duplicates per id."""
        df = pd.DataFrame({
            "event_id": [1, 2, 3],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        })
        result = deduplicate_events(df)
        pd.testing.assert_frame_equal(result, df)

    def test_empty_dataframe(self):
        """Empty DataFrame returns empty DataFrame."""
        df = pd.DataFrame(columns=["event_id", "updated_at"])
        result = deduplicate_events(df)
        assert result.empty
        assert list(result.columns) == ["event_id", "updated_at"]

    def test_single_row(self):
        """Single row is returned unchanged."""
        df = pd.DataFrame({"event_id": [42], "updated_at": ["2023-01-01"]})
        result = deduplicate_events(df)
        pd.testing.assert_frame_equal(result, df)

    def test_does_not_mutate_input(self, base_df):
        """Original DataFrame is not modified."""
        original = base_df.copy()
        deduplicate_events(base_df)
        pd.testing.assert_frame_equal(base_df, original)

    def test_return_type_and_index(self):
        """Result is a DataFrame with default RangeIndex."""
        df = pd.DataFrame({"event_id": [1, 1], "updated_at": ["a", "b"]})
        result = deduplicate_events(df)
        assert isinstance(result, pd.DataFrame)
        assert result.index.equals(pd.RangeIndex(len(result)))

    def test_multiple_ids_with_mixed_nulls(self):
        """Complex case with multiple ids and nulls."""
        df = pd.DataFrame({
            "event_id": [1, None, 2, 1, None, 2],
            "updated_at": [
                "2023-01-01",
                "2023-01-02",
                "2023-01-01",
                "2023-01-03",
                "2023-01-04",
                "2023-01-02",
            ],
        })
        result = deduplicate_events(df)
        # event 1: keep row idx 3 (newer)
        # event 2: keep row idx 5 (newer 2023-01-02 > 2023-01-01)
        # null rows: idx 1 and 4 both kept
        expected = df.iloc[[3, 1, 5, 4]].reset_index(drop=True)
        pd.testing.assert_frame_equal(result, expected)
