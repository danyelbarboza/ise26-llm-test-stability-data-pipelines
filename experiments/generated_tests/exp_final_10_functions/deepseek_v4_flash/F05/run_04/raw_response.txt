import pytest
import pandas as pd
from ise26.targets import validate_schema


@pytest.fixture
def sample_df():
    """Create a DataFrame with various column types."""
    return pd.DataFrame({
        "a": [1, 2, 3],               # int64
        "b": [1.0, 2.5, 3.0],         # float64
        "c": ["x", "y", "z"],         # object (string)
        "d": pd.to_datetime(["2021-01-01", "2021-02-01", "2021-03-01"]),  # datetime64
        "e": [True, False, True],     # bool
        "f": [10, 20, 30],            # int64 (extra)
    })


class TestValidateSchema:
    """Test suite for validate_schema function."""

    def test_valid_schema_all_types_present(self, sample_df):
        """All columns match exactly with expected types."""
        schema = {"a": "int", "b": "float", "c": "string", "d": "datetime", "e": "bool"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_valid_schema_with_normalization(self, sample_df):
        """Schema labels should be normalized (strip and lower)."""
        schema = {"a": "  INT ", "b": "FLOAT", "c": "  string  ", "d": "DATETIME", "e": " bool "}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is True
        assert result["type_errors"] == []

    def test_valid_schema_with_number_type(self, sample_df):
        """'number' should match both int and float columns."""
        schema = {"a": "number", "b": "number"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is True
        assert result["type_errors"] == []

    def test_missing_columns(self, sample_df):
        """Columns that are in schema but not in DataFrame are reported."""
        schema = {"missing_col": "int", "a": "int", "another_missing": "string"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is False
        assert sorted(result["missing_columns"]) == sorted(["missing_col", "another_missing"])
        assert result["type_errors"] == []

    def test_type_mismatch_int_expected_float_actual(self, sample_df):
        """Column with float type and expected int should cause type error."""
        schema = {"b": "int"}  # b is float64
        result = validate_schema(sample_df, schema)
        assert result["valid"] is False
        assert len(result["type_errors"]) == 1
        err = result["type_errors"][0]
        assert err["column"] == "b"
        assert err["expected"] == "int"
        assert err["actual"] != "int"  # actual is float or similar

    def test_type_mismatch_float_expected_int_actual(self, sample_df):
        """Int column expected float is fine (int can be float?)."""
        # According to logic: int matches int, float, number? Actually "int" is specific, int column is not float.
        # We need to test actual mismatch: expected float but column is int.
        schema = {"a": "float"}  # a is int64
        result = validate_schema(sample_df, schema)
        assert result["valid"] is False
        err = result["type_errors"][0]
        assert err["column"] == "a"
        assert err["expected"] == "float"
        assert err["actual"] != "float"

    def test_type_mismatch_string_on_int(self, sample_df):
        """String type expected but column is int."""
        schema = {"a": "string"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is False
        err = result["type_errors"][0]
        assert err["column"] == "a"
        assert err["expected"] == "string"

    def test_type_mismatch_datetime_on_string(self, sample_df):
        """Datetime expected but column is string."""
        schema = {"c": "datetime"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is False
        err = result["type_errors"][0]
        assert err["column"] == "c"
        assert err["expected"] == "datetime"

    def test_bool_type_matches(self, sample_df):
        """Bool type matches bool column."""
        schema = {"e": "bool"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is True

    def test_unsupported_expected_type_raises_valueerror(self, sample_df):
        """Unsupported type label should raise ValueError."""
        schema = {"a": "list"}
        with pytest.raises(ValueError, match="Unsupported expected type: list"):
            validate_schema(sample_df, schema)

    def test_unsupported_expected_type_with_normalization(self, sample_df):
        """Unsupported type after normalization also raises error."""
        schema = {"a": " LIST "}
        with pytest.raises(ValueError, match="Unsupported expected type: list"):
            validate_schema(sample_df, schema)

    def test_multiple_errors(self, sample_df):
        """Multiple type errors and missing columns are reported."""
        schema = {"missing1": "int", "missing2": "float", "a": "string", "b": "bool"}
        result = validate_schema(sample_df, schema)
        assert result["valid"] is False
        assert sorted(result["missing_columns"]) == sorted(["missing1", "missing2"])
        assert len(result["type_errors"]) == 2
        columns_with_errors = {e["column"] for e in result["type_errors"]}
        assert columns_with_errors == {"a", "b"}

    def test_empty_dataframe(self):
        """Empty DataFrame: all schema columns are missing."""
        df = pd.DataFrame()
        schema = {"col": "int", "other": "float"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert sorted(result["missing_columns"]) == sorted(["col", "other"])
        assert result["type_errors"] == []

    def test_empty_schema(self, sample_df):
        """Empty schema: always valid."""
        result = validate_schema(sample_df, {})
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_extra_columns_ignored(self, sample_df):
        """Columns in DataFrame not in schema are ignored."""
        schema = {"a": "int"}  # b,c,d,e,f are extra
        result = validate_schema(sample_df, schema)
        assert result["valid"] is True

    def test_dataframe_with_nulls_int(self):
        """DataFrame with nullable int (Int64) should be treated as int."""
        df = pd.DataFrame({"x": pd.array([1, pd.NA, 3], dtype="Int64")})
        result = validate_schema(df, {"x": "int"})
        assert result["valid"] is True

    def test_dataframe_with_nulls_float(self):
        """DataFrame with nullable float (Float64) should be treated as float."""
        df = pd.DataFrame({"x": pd.array([1.0, pd.NA, 3.0], dtype="Float64")})
        result = validate_schema(df, {"x": "float"})
        assert result["valid"] is True

    def test_dataframe_with_nulls_string(self):
        """DataFrame with object column with None/NaN should be string."""
        df = pd.DataFrame({"x": ["a", None, "c"]})
        result = validate_schema(df, {"x": "string"})
        assert result["valid"] is True

    def test_dataframe_with_nulls_datetime(self):
        """DataFrame with datetime with NaT should be datetime."""
        df = pd.DataFrame({"x": pd.to_datetime(["2021-01-01", pd.NaT, "2021-03-01"])})
        result = validate_schema(df, {"x": "datetime"})
        assert result["valid"] is True

    def test_dataframe_with_mixed_types_object(self):
        """Object column with mixed types is still 'string' logically."""
        df = pd.DataFrame({"x": [1, "a", 3.0]})
        result = validate_schema(df, {"x": "string"})
        assert result["valid"] is True  # as per common infer: object -> string

    def test_type_error_actual_after_mismatch(self, sample_df):
        """Ensure type_errors contains 'actual' values (not just True/False)."""
        schema = {"a": "float"}
        result = validate_schema(sample_df, schema)
        assert result["type_errors"][0]["actual"] != "float"
        # actual should be a string describing the logical type
        actual = result["type_errors"][0]["actual"]
        assert isinstance(actual, str) and actual != ""
