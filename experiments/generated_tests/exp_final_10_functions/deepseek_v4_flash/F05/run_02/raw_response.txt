import pandas as pd
import pytest

from ise26.targets import validate_schema


def test_valid_schema_all_columns_present_and_types_match():
    """All required columns exist with correct logical types."""
    df = pd.DataFrame({
        "id": [1, 2, 3],
        "price": [10.5, 20.0, 30.7],
        "name": ["Alice", "Bob", "Charlie"],
        "date": pd.to_datetime(["2023-01-01", "2023-02-02", "2023-03-03"]),
        "active": [True, False, True],
    })
    schema = {
        "id": "int",
        "price": "float",
        "name": "string",
        "date": "datetime",
        "active": "bool",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_valid_schema_extra_columns_allowed():
    """Extra columns not in schema are ignored."""
    df = pd.DataFrame({
        "a": [1, 2],
        "b": [3.0, 4.0],
        "c": ["x", "y"],
    })
    schema = {"a": "int", "b": "float"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_columns_reported():
    """Columns specified in schema but missing from DataFrame are listed."""
    df = pd.DataFrame({"x": [1]})
    schema = {"x": "int", "y": "float", "z": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["y", "z"]
    assert result["type_errors"] == []


def test_type_mismatch_reported():
    """Type mismatch returns structured error entry."""
    df = pd.DataFrame({
        "col_int": [1, 2],
        "col_str": ["hello", "world"],
    })
    schema = {"col_int": "string", "col_str": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 2
    # Verify first error
    err1 = result["type_errors"][0]
    assert err1["column"] == "col_int"
    assert err1["expected"] == "string"
    assert err1["actual"] == "int"
    # Verify second error
    err2 = result["type_errors"][1]
    assert err2["column"] == "col_str"
    assert err2["expected"] == "int"
    assert err2["actual"] == "string"


def test_expected_type_normalization_strip_and_lower():
    """Schema labels are normalized with strip().lower()."""
    df = pd.DataFrame({"a": [1, 2], "b": [3.0, 4.0]})
    schema = {"a": " INT ", "b": "  Float  "}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    # Internal normalization: expected types become "int" and "float"
    # No errors expected
    assert result["type_errors"] == []

def test_unsupported_expected_type_raises_value_error():

def test_actual_type_inference_for_various_dtypes():
    """Ensure actual logical type strings are correct."""
    df = pd.DataFrame({
        "int_col": pd.Series([1, 2], dtype="int64"),
        "float_col": pd.Series([1.5, 2.5], dtype="float64"),
        "string_col": pd.Series(["a", "b"], dtype="object"),
        "datetime_col": pd.Series(pd.to_datetime(["2021-01-01", "2022-01-01"])),
        "bool_col": pd.Series([True, False], dtype="bool"),
    })
    # Make expected types intentionally mismatched to see actual inference
    schema = {
        "int_col": "string",
        "float_col": "int",
        "string_col": "float",
        "datetime_col": "int",
        "bool_col": "int",
    }
    result = validate_schema(df, schema)
    assert len(result["type_errors"]) == 5
    # Check each error has correct 'actual'
    actuals = {e["column"]: e["actual"] for e in result["type_errors"]}
    assert actuals == {
        "int_col": "int",
        "float_col": "float",
        "string_col": "string",
        "datetime_col": "datetime",
        "bool_col": "bool",
    }


def test_mixed_case_and_whitespace_in_expected_type_normalization():
    """Case and whitespace variations are normalized correctly."""
    df = pd.DataFrame({"a": [1]})
    # All valid variations
    for raw_label in ["int", "Int", "INT", " int ", " int"]:
        schema = {"a": raw_label}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    # Also for other types
    df_float = pd.DataFrame({"b": [1.0]})
    for raw_label in ["float", "Float", "FLOAT", " float "]:
        schema = {"b": raw_label}
        result = validate_schema(df_float, schema)
        assert result["valid"] is True


def test_partial_type_mismatch_and_missing_columns_combined():
    """Both missing columns and type errors are reported together."""
    df = pd.DataFrame({
        "a": [1],
        "b": ["text"],
    })
    schema = {"a": "float", "b": "int", "c": "string", "d": "bool"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["c", "d"]
    assert len(result["type_errors"]) == 2
    # Type errors for a and b
    type_cols = {e["column"] for e in result["type_errors"]}
    assert type_cols == {"a", "b"}


def test_schema_with_only_unsupported_type_raises_before_checking():
    """ValueError raised immediately, before any column checks."""
    df = pd.DataFrame({"x": [1]})
    schema = {"x": "invalid_type"}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_schema_with_mixed_supported_and_unsupported_types_raises():

    schema = {"a": "int", "b": "complex"}
