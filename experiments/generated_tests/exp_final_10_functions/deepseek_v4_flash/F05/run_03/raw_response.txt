import pandas as pd
import pytest
from ise26.targets import validate_schema


class TestValidateSchema:
    """Test suite for validate_schema function."""

    # ---------- valid cases ----------
    def test_all_columns_correct_types(self):
        df = pd.DataFrame({
            "a": [1, 2, 3],
            "b": [1.0, 2.0, 3.0],
            "c": ["x", "y", "z"],
            "d": pd.to_datetime(["2021-01-01", "2021-01-02", "2021-01-03"]),
            "e": [True, False, True],
        })
        schema = {
            "a": "int",
            "b": "float",
            "c": "string",
            "d": "datetime",
            "e": "bool",
        }
        result = validate_schema(df, schema)
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_number_type_accepts_int_and_float(self):
        df = pd.DataFrame({
            "a": [1, 2],
            "b": [1.5, 2.5],
        })
        schema = {"a": "number", "b": "number"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_extra_columns_allowed(self):
        df = pd.DataFrame({
            "x": [1, 2],
            "y": [10, 20],
            "extra1": ["a", "b"],
            "extra2": [True, False],
        })
        schema = {"x": "int", "y": "float"}
        result = validate_schema(df, schema)
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_empty_schema(self):
        df = pd.DataFrame({"a": [1, 2]})
        result = validate_schema(df, {})
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_empty_dataframe_with_schema(self):
        df = pd.DataFrame({"a": pd.Series(dtype="int64")})
        result = validate_schema(df, {"a": "int"})
        assert result["valid"] is True

    # ---------- missing columns ----------
    def test_missing_columns_reported(self):
        df = pd.DataFrame({"a": [1, 2]})
        schema = {"a": "int", "b": "float", "c": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert "b" in result["missing_columns"]
        assert "c" in result["missing_columns"]
        assert "a" not in result["missing_columns"]

    def test_only_missing_columns(self):
        df = pd.DataFrame({"a": [1]})
        schema = {"b": "int", "c": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["b", "c"]
        assert result["type_errors"] == []

    # ---------- type mismatches ----------
    def test_type_mismatch_int_expected_float_actual(self):
        df = pd.DataFrame({"a": [1.5, 2.5]})
        schema = {"a": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["type_errors"][0]["column"] == "a"
        assert result["type_errors"][0]["expected"] == "int"
        # actual type is inferred, we just check key exists
        assert "actual" in result["type_errors"][0]

    def test_type_mismatch_float_expected_int_actual(self):
        df = pd.DataFrame({"a": [1, 2]})
        schema = {"a": "float"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert len(result["type_errors"]) == 1
        assert result["type_errors"][0]["column"] == "a"
        assert result["type_errors"][0]["expected"] == "float"

    def test_type_mismatch_string_expected_int(self):
        df = pd.DataFrame({"a": ["hello", "world"]})
        schema = {"a": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert len(result["type_errors"]) == 1

    def test_multiple_type_errors(self):
        df = pd.DataFrame({
            "a": [1, 2],
            "b": ["x", "y"],
            "c": [True, False],
        })
        schema = {"a": "string", "b": "bool", "c": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert len(result["type_errors"]) == 3

    # ---------- schema normalization ----------
    @pytest.mark.parametrize(
        "raw_type, normalized",
        [
            (" int ", "int"),
            ("FLOAT ", "float"),
            ("  Number  ", "number"),
            ("STRING", "string"),
            (" datetime", "datetime"),
            ("BOOL ", "bool"),
        ],
    )
    def test_schema_type_normalization(self, raw_type, normalized):
        df = pd.DataFrame({"a": [1, 2]})
        schema = {"a": raw_type}
        result = validate_schema(df, schema)
        assert result["valid"] is True
        # Also verify that if type error occurs, expected in result is normalized
        df2 = pd.DataFrame({"a": ["x", "y"]})
        result2 = validate_schema(df2, schema)
        if result2["type_errors"]:
            assert result2["type_errors"][0]["expected"] == normalized

    # ---------- unsupported expected type ----------
    def test_unsupported_type_raises_value_error(self):
        df = pd.DataFrame({"a": [1]})
        with pytest.raises(ValueError):
            validate_schema(df, {"a": "category"})

    def test_unsupported_type_after_normalization(self):
        df = pd.DataFrame({"a": [1]})
        with pytest.raises(ValueError):
            validate_schema(df, {"a": "  INT64  "})  # not in supported set

    # ---------- edge cases ----------
    def test_dataframe_with_nan_float_column(self):
        df = pd.DataFrame({"a": [1.0, None, 3.0]})
        schema = {"a": "float"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_dataframe_with_none_in_object_column(self):
        df = pd.DataFrame({"a": [None, "hello", None]})
        schema = {"a": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_datetime_with_timezone(self):
        df = pd.DataFrame({
            "a": pd.to_datetime(["2021-01-01", "2021-01-02"]).tz_localize("UTC")
        })
        schema = {"a": "datetime"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_bool_column_with_mixed_types_not_bool(self):
        # Series with object dtype containing strings
        df = pd.DataFrame({"a": [True, False, "maybe"]})
        schema = {"a": "bool"}
        result = validate_schema(df, schema)
        assert result["valid"] is False

    def test_mixed_and_missing_columns(self):
        df = pd.DataFrame({"a": [1, 2], "b": ["x", "y"]})
        schema = {"a": "float", "b": "int", "c": "string"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["c"]
        assert len(result["type_errors"]) == 1  # a is int, expected float

    def test_return_structure_keys(self):
        df = pd.DataFrame({"a": [1]})
        schema = {"a": "int"}
        result = validate_schema(df, schema)
        assert "valid" in result
        assert "missing_columns" in result
        assert "type_errors" in result
        for err in result["type_errors"]:
            assert "column" in err
            assert "expected" in err
            assert "actual" in err

    def test_no_schema_columns(self):
        df = pd.DataFrame()
        schema = {}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_extra_columns_ignored_for_type_check(self):
        # Extra columns not in schema should not cause errors
        df = pd.DataFrame({"a": [1], "b": ["text"], "c": [2.0]})
        schema = {"a": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_number_type_with_int_series(self):
        df = pd.DataFrame({"a": [1, 2, 3]})
        schema = {"a": "number"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_number_type_with_float_series(self):
        df = pd.DataFrame({"a": [1.5, 2.5]})
        schema = {"a": "number"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_string_type_with_object_series_containing_integers(self):
        df = pd.DataFrame({"a": [1, 2]})
        schema = {"a": "string"}
        result = validate_schema(df, schema)
        # Depending on inference, object may be considered string; expect valid or not?
        # The actual _infer_series_type likely returns 'int' for integer column, so this should be a mismatch.
        assert result["valid"] is False  # integer series is not string

    def test_none_only_column_inferred_type(self):
        df = pd.DataFrame({"a": [None, None, None]})
        schema = {"a": "float"}
        # All NaN float column is float, but if None is object, it might be 'string'?
        # We assume the series dtype is object -> string => mismatch
        # Let's test expecting a type error.
        result = validate_schema(df, schema)
        # Since all None might be object dtype, expect mismatch unless inference treats as unknown.
        # But we trust implementation. Just check result structure.
        assert isinstance(result, dict)
