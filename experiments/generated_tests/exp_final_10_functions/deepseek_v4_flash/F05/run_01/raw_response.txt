import pandas as pd
import pytest
from ise26.targets import validate_schema


def test_valid_schema_all_types():
    df = pd.DataFrame({
        'a': [1, 2, 3],
        'b': [1.0, 2.5, 3.0],
        'c': ['x', 'y', 'z'],
        'd': pd.to_datetime(['2020-01-01', '2020-01-02', '2020-01-03']),
        'e': [True, False, True],
    })
    schema = {
        'a': 'int',
        'b': 'float',
        'c': 'string',
        'd': 'datetime',
        'e': 'bool',
    }
    result = validate_schema(df, schema)
    assert result == {
        'valid': True,
        'missing_columns': [],
        'type_errors': [],
    }


def test_valid_with_number_type():
    df = pd.DataFrame({
        'a': [1, 2, 3],
        'b': [1.5, 2.0, 3.7],
    })
    schema = {'a': 'number', 'b': 'number'}
    result = validate_schema(df, schema)
    assert result['valid'] is True
    assert result['missing_columns'] == []
    assert result['type_errors'] == []


def test_missing_columns():
    df = pd.DataFrame({'a': [1, 2]})
    schema = {'a': 'int', 'b': 'string', 'c': 'float'}
    result = validate_schema(df, schema)
    assert result['valid'] is False
    assert sorted(result['missing_columns']) == ['b', 'c']
    assert result['type_errors'] == []


def test_type_mismatch():
    df = pd.DataFrame({
        'a': [1, 2],
        'b': ['x', 'y'],
        'c': [1.0, 2.0],
    })
    schema = {'a': 'string', 'b': 'int', 'c': 'datetime'}
    result = validate_schema(df, schema)
    assert result['valid'] is False
    assert result['missing_columns'] == []
    # actual types: a->int, b->string, c->float
    expected_errors = [
        {'column': 'a', 'expected': 'string', 'actual': 'int'},
        {'column': 'b', 'expected': 'int', 'actual': 'string'},
        {'column': 'c', 'expected': 'datetime', 'actual': 'float'},
    ]
    assert result['type_errors'] == expected_errors


def test_extra_columns_allowed():
    df = pd.DataFrame({
        'a': [1, 2],
        'b': ['x', 'y'],
        'extra1': [10, 20],
        'extra2': [True, False],
    })
    schema = {'a': 'int', 'b': 'string'}
    result = validate_schema(df, schema)
    assert result == {
        'valid': True,
        'missing_columns': [],
        'type_errors': [],
    }


def test_invalid_expected_type_raises():
    df = pd.DataFrame({'a': [1]})
    schema = {'a': 'list'}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_normalized_expected_type():
    df = pd.DataFrame({
        'a': [1, 2],
        'b': ['x', 'y'],
    })
    schema = {'a': ' INT ', 'b': '  String  '}
    result = validate_schema(df, schema)
    assert result['valid'] is True


def test_mixed_errors():
    df = pd.DataFrame({'a': [1.5]})
    schema = {'a': 'int', 'b': 'float', 'c': 'string'}
    result = validate_schema(df, schema)
    assert result['valid'] is False
    assert result['missing_columns'] == ['b', 'c']
    # type error for column a: actual float, expected int
    assert result['type_errors'] == [{'column': 'a', 'expected': 'int', 'actual': 'float'}]


def test_empty_dataframe():
    df = pd.DataFrame({'a': pd.Series(dtype='int64'), 'b': pd.Series(dtype='float64')})
    schema = {'a': 'int', 'b': 'float'}
    result = validate_schema(df, schema)
    # Even if no rows, column presence and type are checked.
    # Empty series may have object dtype or inferred as int/float? With dtype='int64', it's int.
    assert result['valid'] is True
    assert result['missing_columns'] == []
    assert result['type_errors'] == []


def test_empty_schema():
    df = pd.DataFrame({'a': [1, 2]})
    result = validate_schema(df, {})
    assert result == {
        'valid': True,
        'missing_columns': [],
        'type_errors': [],
    }


def test_missing_and_mismatch_with_extra_columns():
    df = pd.DataFrame({
        'a': [1],
        'c': ['hello'],
        'extra': [True],
    })
    schema = {'a': 'string', 'b': 'int', 'c': 'int'}
    result = validate_schema(df, schema)
    assert result['valid'] is False
    assert result['missing_columns'] == ['b']
    # column a: actual int, expected string
    # column c: actual string, expected int
    err_a = {'column': 'a', 'expected': 'string', 'actual': 'int'}
    err_c = {'column': 'c', 'expected': 'int', 'actual': 'string'}
    assert result['type_errors'] == [err_a, err_c]


def test_unsupported_type_among_valid_ones():
    df = pd.DataFrame({'a': [1]})
    schema = {'a': 'int', 'b': 'unknown'}
    with pytest.raises(ValueError):
        validate_schema(df, schema)
