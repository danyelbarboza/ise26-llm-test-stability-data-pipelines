import pandas as pd
import pytest
from ise26.targets import validate_schema


# ========================= Fixtures (synthetic DataFrames) =========================

@pytest.fixture
def df_all_types():
    """DataFrame with one column of each supported logical type."""
    return pd.DataFrame({
        "int_col": pd.Series([1, 2, 3], dtype="int64"),
        "float_col": pd.Series([1.0, 2.5, 3.0], dtype="float64"),
        "string_col": pd.Series(["a", "b", "c"], dtype="object"),
        "datetime_col": pd.to_datetime(pd.Series(["2020-01-01", "2020-02-01", "2020-03-01"])),
        "bool_col": pd.Series([True, False, True], dtype="bool"),
    })


@pytest.fixture
def df_with_extra():
    """DataFrame with extra column not in schema."""
    return pd.DataFrame({
        "x": [1, 2],
        "y": [3.0, 4.0],
        "z": ["a", "b"],
    })


# ========================= Basic valid schema =========================

class TestValidateSchemaValid:
    def test_all_types_match(self, df_all_types):
        schema = {
            "int_col": "int",
            "float_col": "float",
            "string_col": "string",
            "datetime_col": "datetime",
            "bool_col": "bool",
        }
        result = validate_schema(df_all_types, schema)
        assert result == {
            "valid": True,
            "missing_columns": [],
            "type_errors": [],
        }

    def test_extra_columns_allowed(self, df_with_extra):
        schema = {"x": "int"}
        result = validate_schema(df_with_extra, schema)
        assert result["valid"] is True
        assert result["missing_columns"] == []
        assert result["type_errors"] == []

    def test_empty_dataframe_with_correct_columns(self):
        df = pd.DataFrame({"a": pd.Series(dtype="int64"), "b": pd.Series(dtype="float64")})
        schema = {"a": "int", "b": "float"}
        result = validate_schema(df, schema)
        assert result["valid"] is True

    def test_normalized_expected_type(self, df_all_types):
        # schema value with whitespace and mixed case
        schema = {"int_col": "  Int  ", "float_col": " FLOAT "}
        result = validate_schema(df_all_types, schema)
        # expected should be normalized to 'int', 'float'
        assert result["valid"] is True
        # also check that no type errors are reported (normalization works)
        assert result["type_errors"] == []


# ========================= Missing columns =========================

class TestValidateSchemaMissing:
    def test_one_missing_column(self, df_all_types):
        schema = {
            "int_col": "int",
            "missing_col": "string",
        }
        result = validate_schema(df_all_types, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["missing_col"]
        assert result["type_errors"] == []

    def test_multiple_missing_columns(self, df_all_types):
        schema = {"A": "int", "B": "float", "C": "string"}
        result = validate_schema(df_all_types, schema)
        assert result["valid"] is False
        assert set(result["missing_columns"]) == {"A", "B", "C"}
        assert result["type_errors"] == []

    def test_missing_and_extra_columns(self, df_all_types):
        schema = {
            "int_col": "int",
            "ghost": "bool",
        }
        result = validate_schema(df_all_types, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["ghost"]


# ========================= Type mismatches =========================

class TestValidateSchemaTypeErrors:
    def test_single_type_mismatch(self):
        df = pd.DataFrame({"x": [1.0, 2.0]})  # float actual
        schema = {"x": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == []
        assert len(result["type_errors"]) == 1
        error = result["type_errors"][0]
        assert error["column"] == "x"
        assert error["expected"] == "int"
        assert error["actual"] == "float"

    def test_multiple_type_errors(self):
        df = pd.DataFrame({
            "a": [1, 2],        # int
            "b": [True, False], # bool
        })
        schema = {"a": "float", "b": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert len(result["type_errors"]) == 2
        errors = {e["column"]: e for e in result["type_errors"]}
        assert errors["a"]["expected"] == "float"
        assert errors["a"]["actual"] == "int"
        assert errors["b"]["expected"] == "int"
        assert errors["b"]["actual"] == "bool"

    def test_mixed_missing_and_type_errors(self, df_all_types):
        schema = {
            "int_col": "float",      # type mismatch
            "missing_col": "string", # missing
        }
        result = validate_schema(df_all_types, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["missing_col"]
        assert len(result["type_errors"]) == 1
        assert result["type_errors"][0]["column"] == "int_col"
        assert result["type_errors"][0]["expected"] == "float"
        assert result["type_errors"][0]["actual"] == "int"

    def test_type_error_with_object_string(self):
        # string columns are often object dtype; actual should be 'string'
        df = pd.DataFrame({"s": ["hello", "world"]})
        schema = {"s": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["type_errors"][0]["actual"] == "string"

    def test_number_type_accepts_int_or_float(self):
        # 'number' is a valid expected type, actual may be int or float
        df_int = pd.DataFrame({"v": [1, 2]})  # int
        df_float = pd.DataFrame({"v": [1.0, 2.0]})  # float
        schema = {"v": "number"}
        # both should be valid because number matches either
        r1 = validate_schema(df_int, schema)
        r2 = validate_schema(df_float, schema)
        assert r1["valid"] is True
        assert r2["valid"] is True

    def test_type_error_when_expecting_number_but_actual_is_bool(self):
        df = pd.DataFrame({"v": [True, False]})
        schema = {"v": "number"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["type_errors"][0]["column"] == "v"
        assert result["type_errors"][0]["expected"] == "number"
        assert result["type_errors"][0]["actual"] == "bool"


# ========================= Unsupported expected type =========================

class TestValidateSchemaValueError:
    def test_unsupported_type_raises(self):
        df = pd.DataFrame({"a": [1]})
        with pytest.raises(ValueError, match=r"unsupported.*?type|not supported"):
            validate_schema(df, {"a": "category"})

    def test_unsupported_type_with_whitespace(self):
        df = pd.DataFrame({"a": [1]})
        with pytest.raises(ValueError):
            validate_schema(df, {"a": "  blob  "})

    def test_case_insensitive_normalization_for_unknown(self):
        df = pd.DataFrame({"a": [1]})
        # after strip().lower(), "LIST" -> "list", still unsupported
        with pytest.raises(ValueError):
            validate_schema(df, {"a": "LIST"})

    def test_supported_types_do_not_raise(self):
        df = pd.DataFrame({"a": [1]})
        # should not raise for any supported type
        for t in ["int", "float", "number", "string", "datetime", "bool"]:
            validate_schema(df, {"a": t})
            validate_schema(df, {"a": t.upper()})
            validate_schema(df, {"a": f"  {t}  "})
        # assert no exception
        assert True


# ========================= Edge cases =========================

class TestValidateSchemaEdge:
    def test_empty_schema(self, df_all_types):
        """Empty schema means no validation needed."""
        result = validate_schema(df_all_types, {})
        assert result == {
            "valid": True,
            "missing_columns": [],
            "type_errors": [],
        }

    def test_empty_dataframe_empty_schema(self):
        df = pd.DataFrame()
        result = validate_schema(df, {})
        assert result["valid"] is True

    def test_empty_dataframe_with_nonempty_schema(self):
        df = pd.DataFrame()
        schema = {"col": "int"}
        result = validate_schema(df, schema)
        assert result["valid"] is False
        assert result["missing_columns"] == ["col"]
