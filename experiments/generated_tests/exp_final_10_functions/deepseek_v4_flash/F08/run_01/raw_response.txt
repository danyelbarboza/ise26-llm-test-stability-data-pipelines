import pytest
import pandas as pd
from ise26.targets import calculate_conversion_rate


def test_basic_aggregation():
    df = pd.DataFrame({
        "channel": ["email", "social", "email"],
        "visits": [100, 50, 200],
        "conversions": [5, 2, 10]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["email", "social"],
        "visits": [300.0, 50.0],
        "conversions": [15.0, 2.0],
        "conversion_rate": [15.0 / 300.0, 2.0 / 50.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_missing_visits_treated_as_zero():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [None, 100],
        "conversions": [10, 20]

    result = calculate_conversion_rate(df)

        "conversions": [10.0, 20.0],

def test_mixed_scenario():
    df = pd.DataFrame({
        "channel": ["email", None, "email", "social", ""],
        "visits": [100, "bad", 200, 50, 30],
        "conversions": [5, 10, None, 1, 3]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["email", "social", "unknown"],
        "visits": [300.0, 50.0, 30.0],
        "conversions": [5.0, 1.0, 13.0],
        "conversion_rate": [5.0 / 300.0, 1.0 / 50.0, 13.0 / 30.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_column_order():
    df = pd.DataFrame({
        "channel": ["a"],
        "visits": [10],
        "conversions": [1]
    })
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]


def test_no_extra_columns_from_input():
    df = pd.DataFrame({
        "channel": ["a"],
        "visits": [10],
        "conversions": [1],
        "extra": ["x"]
    })
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]

def test_empty_dataframe():
    df = pd.DataFrame({
        "channel": pd.Series([], dtype="object"),
        "visits": pd.Series([], dtype="int64"),
        "conversions": pd.Series([], dtype="int64")
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": pd.Series([], dtype="object"),
        "visits": pd.Series([], dtype="float64"),
        "conversions": pd.Series([], dtype="float64"),
        "conversion_rate": pd.Series([], dtype="float64")
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=True)
