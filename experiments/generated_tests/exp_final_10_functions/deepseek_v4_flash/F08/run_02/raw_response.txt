import pytest
import pandas as pd
import numpy as np

from ise26.targets import calculate_conversion_rate


class TestCalculateConversionRate:
    """Test suite for calculate_conversion_rate."""

    def test_basic_aggregation(self):
        """Verify correct aggregation for normal data."""
        df = pd.DataFrame({
            "channel": ["email", "social", "email"],
            "visits": [100, 200, 300],
            "conversions": [10, 20, 30]
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["email", "social"],
            "visits": [400.0, 200.0],
            "conversions": [40.0, 20.0],
            "conversion_rate": [0.1, 0.1]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_values_in_metrics(self):
        """NaN and None in visits/conversions should be treated as zero."""
        df = pd.DataFrame({
            "channel": ["a", "b", "a"],
            "visits": [10, None, np.nan],
            "conversions": [1, 2, None]
        })
        result = calculate_conversion_rate(df)
        # channel a: visits sum = 10+0 = 10, conversions = 1+0 = 1, rate=0.1
        # channel b: visits=0, conversions=2, rate=0.0
        expected = pd.DataFrame({
            "channel": ["a", "b"],
            "visits": [10.0, 0.0],
            "conversions": [1.0, 2.0],
            "conversion_rate": [0.1, 0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_numeric_strings(self):
        """Non-numeric strings in metric columns should be treated as zero."""
        df = pd.DataFrame({
            "channel": ["x", "y"],
            "visits": ["5", "abc"],
            "conversions": ["1.5", "def"]
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["x", "y"],
            "visits": [5.0, 0.0],
            "conversions": [1.5, 0.0],
            "conversion_rate": [0.3, 0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_channel_normalized_to_unknown(self):
        """Missing/blank channels become 'unknown' and are grouped."""
        df = pd.DataFrame({
            "channel": [None, "", "  ", "email"],
            "visits": [10, 20, 30, 40],
            "conversions": [1, 2, 3, 4]
        })
        result = calculate_conversion_rate(df)
        # unknown group sums: visits=10+20+30=60, conversions=1+2+3=6, rate=0.1
        expected = pd.DataFrame({
            "channel": ["email", "unknown"],
            "visits": [40.0, 60.0],
            "conversions": [4.0, 6.0],
            "conversion_rate": [0.1, 0.1]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_alphabetical_sorting(self):
        """Result should be sorted by channel alphabetically."""
        df = pd.DataFrame({
            "channel": ["z", "a", "m"],
            "visits": [1, 2, 3],
            "conversions": [0, 1, 2]
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a", "m", "z"],
            "visits": [2.0, 3.0, 1.0],
            "conversions": [1.0, 2.0, 0.0],
            "conversion_rate": [0.5, 2/3, 0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_zero_visits_gives_zero_rate(self):
        """When total visits for a channel are zero, conversion_rate must be zero."""
        df = pd.DataFrame({
            "channel": ["a", "a"],
            "visits": [0, 0],
            "conversions": [5, 10]
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a"],
            "visits": [0.0],
            "conversions": [15.0],
            "conversion_rate": [0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_all_zero_visits(self):
        """All channels with zero visits -> conversion_rate zero everywhere."""
        df = pd.DataFrame({
            "channel": ["x", "y"],
            "visits": [0, 0],
            "conversions": [1, 2]
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["x", "y"],
            "visits": [0.0, 0.0],
            "conversions": [1.0, 2.0],
            "conversion_rate": [0.0, 0.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_single_channel(self):
        """Only one distinct channel."""
        df = pd.DataFrame({
            "channel": ["only"],
            "visits": [100],
            "conversions": [25]
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["only"],
            "visits": [100.0],
            "conversions": [25.0],
            "conversion_rate": [0.25]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_column_names(self):
        """Respect custom group_col, visits_col, conversions_col."""
        df = pd.DataFrame({
            "src": ["a", "b"],
            "imp": [200, 300],
            "conv": [20, 15]
        })
        result = calculate_conversion_rate(
            df, group_col="src", visits_col="imp", conversions_col="conv"
        )
        expected = pd.DataFrame({
            "src": ["a", "b"],
            "visits": [200.0, 300.0],
            "conversions": [20.0, 15.0],
            "conversion_rate": [0.1, 0.05]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_output_columns_exact(self):
        """Returned DataFrame must have exactly the four expected columns."""
        df = pd.DataFrame({
            "channel": ["a"],
            "visits": [5],
            "conversions": [1]
        })
        result = calculate_conversion_rate(df)
        assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]

    def test_output_dtypes(self):
        """visits, conversions, conversion_rate should be float."""
        df = pd.DataFrame({
            "channel": ["x"],
            "visits": [5],
            "conversions": [2]
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].dtype == float
        assert result["conversions"].dtype == float
        assert result["conversion_rate"].dtype == float

    def test_empty_dataframe_raises_error(self):
        """Empty DataFrame should raise an error (missing columns)."""
        # The function clones and ensures columns; empty DataFrame with proper columns is possible?
        # But with no rows, grouping yields empty DataFrame. However, if columns are missing, it raises.
        # We test that it works with no rows but correct columns.
        df = pd.DataFrame(columns=["channel", "visits", "conversions"])
        result = calculate_conversion_rate(df)
        assert len(result) == 0
        assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]

    def test_missing_required_column_raises(self):
        """Missing a required column should raise a KeyError or similar."""
        df = pd.DataFrame({"channel": ["a"], "visits": [1]})  # missing conversions_col
        with pytest.raises(KeyError):
            calculate_conversion_rate(df)
