import pytest
import pandas as pd
from ise26.targets import clean_customer_names


def test_basic_lowercase_and_spaces():
    df = pd.DataFrame({"customer_name": ["  John  Doe  ", "Alice  Smith"]})
    result = clean_customer_names(df)
    expected_names = ["john doe", "alice smith"]
    pd.testing.assert_series_equal(
        result["customer_name_clean"],
        pd.Series(expected_names, dtype="object"),
    )


def test_accent_removal():
    df = pd.DataFrame({"customer_name": ["João Silva", "Café com Leite"]})
    result = clean_customer_names(df)
    expected_names = ["joao silva", "cafe com leite"]
    pd.testing.assert_series_equal(
        result["customer_name_clean"],
        pd.Series(expected_names, dtype="object"),
    )


def test_nulls_and_blanks():
    df = pd.DataFrame(
        {"customer_name": [None, "", "   ", pd.NA, "Valid Name"]}
    )
    result = clean_customer_names(df)
    # The first four should be pd.NA, the last normalized
    expected = pd.Series(
        [pd.NA, pd.NA, pd.NA, pd.NA, "valid name"],
        dtype="object",
    )
    # We need to use pd.isna for NA values
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)


def test_original_dataframe_unchanged():
    df_original = pd.DataFrame({"customer_name": ["Test Name"]})
    df_copy = df_original.copy()
    _ = clean_customer_names(df_original)
    pd.testing.assert_frame_equal(df_original, df_copy)


def test_default_column_names():
    df = pd.DataFrame({"customer_name": ["Default Column"]})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    assert result["customer_name_clean"].iloc[0] == "default column"


def test_custom_column_names():
    df = pd.DataFrame({"name": ["Custom Output"]})
    result = clean_customer_names(df, name_col="name", output_col="clean_name")
    assert "clean_name" in result.columns
    assert "name" in result.columns
    assert result["clean_name"].iloc[0] == "custom output"


def test_mixed_cases():
    df = pd.DataFrame({"customer_name": ["UPPER CASE", "  MiXeD   SpAcEs  "]})
    result = clean_customer_names(df)
    expected_names = ["upper case", "mixed spaces"]
    pd.testing.assert_series_equal(
        result["customer_name_clean"],
        pd.Series(expected_names, dtype="object"),
    )


def test_no_mutation_of_input():
    df = pd.DataFrame({"customer_name": ["Original", None]})
    _ = clean_customer_names(df)
    # Original column must remain as is
    assert df["customer_name"].iloc[0] == "Original"
    assert pd.isna(df["customer_name"].iloc[1])
