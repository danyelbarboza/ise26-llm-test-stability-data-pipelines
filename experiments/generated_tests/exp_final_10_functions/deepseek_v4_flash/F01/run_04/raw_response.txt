import pandas as pd
import numpy as np
import pytest
from ise26.targets import clean_customer_names


def test_default_column_names():
    """Test with default column names."""
    df = pd.DataFrame({
        "customer_name": ["  João  Silva  ", "MARIA SANTOS", None, "   ", "José"]
    })
    result = clean_customer_names(df)
    # Check original is unchanged
    assert "customer_name_clean" in result.columns
    assert list(df.columns) == ["customer_name"]
    # Check normalized values
    expected = pd.Series(
        ["joão silva", "maria santos", pd.NA, pd.NA, "josé"],
        name="customer_name_clean",
        dtype="object"
    )
    # Compare using pandas testing
    pd.testing.assert_series_equal(
        result["customer_name_clean"].astype(object).fillna(pd.NA),
        expected.astype(object).fillna(pd.NA),
        check_dtype=False
    )


def test_explicit_columns():
    """Test with explicit column names."""
    df = pd.DataFrame({
        "raw": ["  ÁLVARO  ", "ana", "   ", None]
    })
    result = clean_customer_names(df, name_col="raw", output_col="clean")
    assert "clean" in result.columns
    assert "raw" in result.columns
    assert list(df.columns) == ["raw"]  # original unchanged
    expected = pd.Series(["álvaro", "ana", pd.NA, pd.NA], name="clean")
    pd.testing.assert_series_equal(
        result["clean"].astype(object).fillna(pd.NA),
        expected.astype(object).fillna(pd.NA),
        check_dtype=False
    )


def test_multiple_whitespace_collapsed():
    """Test that multiple spaces are collapsed to single space."""
    df = pd.DataFrame({"name": ["  Pedro   Paulo  "]})
    result = clean_customer_names(df, name_col="name", output_col="normalized")
    assert result["normalized"].iloc[0] == "pedro paulo"


def test_lowercasing():
    """Test that names are lowercased."""
    df = pd.DataFrame({"name": ["JoÃO Da SILVA"]})
    result = clean_customer_names(df, name_col="name")
    assert result["customer_name_clean"].iloc[0] == "joão da silva"


def test_accent_removal():
    """Test that accented characters are removed."""
    df = pd.DataFrame({
        "customer_name": [
            "café", "São Paulo", "José", "Müller"
        ]
    })
    result = clean_customer_names(df)
    expected = pd.Series(["cafe", "sao paulo", "jose", "muller"])
    # Note: Function uses unidecode? We assume it removes accents.
    # Based on code, it likely uses unicodedata.normalize + filter.
    pd.testing.assert_series_equal(
        result["customer_name_clean"].astype(str),
        expected.astype(str)
    )


def test_null_and_blank_mapped_to_na():
    """Test that None, NaN, and whitespace-only strings become pd.NA."""
    df = pd.DataFrame({
        "name": [None, np.nan, "   ", "", "  \t  ", "John"]
    })
    result = clean_customer_names(df, name_col="name")
    na_mask = result["customer_name_clean"].isna()
    # First five should be NA
    assert na_mask.iloc[:5].all()
    # Last should be "john"
    assert result["customer_name_clean"].iloc[5] == "john"


def test_original_dataframe_unchanged():
    """Test that the original DataFrame is not mutated."""
    original_df = pd.DataFrame({
        "customer_name": ["  Alice  ", "Bob", None]
    })
    df_copy = original_df.copy()
    _ = clean_customer_names(original_df)
    pd.testing.assert_frame_equal(original_df, df_copy)


def test_empty_dataframe():
    """Test with an empty DataFrame (no rows)."""
    df = pd.DataFrame({"name": pd.Series(dtype=str)})
    result = clean_customer_names(df, name_col="name")
    assert result.shape == (0, 2)
    assert "customer_name_clean" in result.columns


def test_no_customer_name_column_raises():
    """Test that missing column raises error."""
    df = pd.DataFrame({"other": [1, 2]})
    with pytest.raises(KeyError):
        clean_customer_names(df, name_col="customer_name")


def test_all_nulls():
    """Test when all names are null/blank."""
    df = pd.DataFrame({"name": [None, np.nan, "   ", ""]})
    result = clean_customer_names(df, name_col="name")
    assert result["customer_name_clean"].isna().all()


def test_mixed_types_in_column():
    """Test column may contain numbers or other types converted to string."""
    df = pd.DataFrame({"name": [123, 45.6, True]})
    result = clean_customer_names(df, name_col="name")
    # The apply should convert to string then normalize
    expected = pd.Series(["123", "45.6", "true"])
