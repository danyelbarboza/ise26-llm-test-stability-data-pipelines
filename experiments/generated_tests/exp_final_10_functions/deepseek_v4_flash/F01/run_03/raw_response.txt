import pandas as pd
import numpy as np
import pytest
from ise26.targets import clean_customer_names


class TestCleanCustomerNames:
    """Test suite for clean_customer_names function."""

    def test_typical_names(self):
        """Test that names are lowercased, stripped, and accented chars removed."""
        df = pd.DataFrame({"customer_name": ["  João  Silva ", "MÜLLER", "François"]})
        result = clean_customer_names(df, name_col="customer_name", output_col="clean")
        expected_clean = pd.Series(["joao silva", "muller", "francois"], name="clean")
        pd.testing.assert_series_equal(result["clean"], expected_clean, check_dtype=False)

    def test_extra_spaces_collapsed(self):
        """Test that multiple spaces and leading/trailing spaces are removed."""
        df = pd.DataFrame({"name": ["   John    Doe   ", "  Alice   "], "other": [1, 2]})
        result = clean_customer_names(df, name_col="name")
        expected = pd.Series(["john doe", "alice"], name="customer_name_clean")
        pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)

    def test_lowercasing(self):
        """Test that all characters are converted to lowercase."""
        df = pd.DataFrame({"n": ["JOHN", "ALICE", "bob"]})
        result = clean_customer_names(df, name_col="n", output_col="clean")
        expected = pd.Series(["john", "alice", "bob"], name="clean")
        pd.testing.assert_series_equal(result["clean"], expected, check_dtype=False)

    def test_accent_removal(self):
        """Test that accented characters are replaced with ASCII equivalents."""
        df = pd.DataFrame({"name": ["São Paulo", "José", "Café", "François"]})
        result = clean_customer_names(df)
        expected = pd.Series(["sao paulo", "jose", "cafe", "francois"], name="customer_name_clean")
        pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)

    def test_null_values_become_na(self):
        """Test that None, NaN, and pd.NA become pd.NA in the output."""
        df = pd.DataFrame({"col": [None, np.nan, pd.NA, "Alice"]})
        result = clean_customer_names(df, name_col="col", output_col="clean")
        # The normalized column must have pd.NA for nulls
        assert result["clean"].isna().tolist() == [True, True, True, False]
        # Verify the actual values
        assert result.loc[3, "clean"] == "alice"

    def test_blank_strings_become_na(self):
        """Test that empty strings or whitespace-only strings become pd.NA."""
        df = pd.DataFrame({"name": ["", "   ", "\t\n", "Bob"]})
        result = clean_customer_names(df)
        assert result["customer_name_clean"].isna().tolist() == [True, True, True, False]
        assert result.loc[3, "customer_name_clean"] == "bob"

    def test_default_column_names(self):
        """Test that default input and output columns work correctly."""
        df = pd.DataFrame({"customer_name": ["  Mary  "], "other": [1]})
        result = clean_customer_names(df)
        assert "customer_name_clean" in result.columns
        assert result["customer_name_clean"].iloc[0] == "mary"

    def test_custom_column_names(self):
        """Test that custom input and output column names are used."""
        df = pd.DataFrame({"raw": ["JOHN"], "dummy": [0]})
        result = clean_customer_names(df, name_col="raw", output_col="standard")
        assert "standard" in result.columns
        assert result["standard"].iloc[0] == "john"

    def test_original_dataframe_unchanged(self):
        """Test that the original DataFrame is not mutated."""
        df_original = pd.DataFrame({"name": ["JOHN  ", "ALICE"]})
        df_copy = df_original.copy()
        _ = clean_customer_names(df_original)
        pd.testing.assert_frame_equal(df_original, df_copy)

    def test_result_is_new_dataframe(self):
        """Test that the returned DataFrame is a different object from the input."""
        df = pd.DataFrame({"name": ["John"]})
        result = clean_customer_names(df)
        assert result is not df

    def test_other_columns_preserved(self):
        """Test that columns other than the name column are kept unchanged."""
        df = pd.DataFrame({"name": ["Alice"], "age": [30], "city": ["NYC"]})
        result = clean_customer_names(df, name_col="name")
        pd.testing.assert_frame_equal(result[["age", "city"]], df[["age", "city"]])

    def test_output_column_dtype(self):
        """Test that the output column is of object dtype (string) or nullable string."""
        df = pd.DataFrame({"name": ["Test"]})
        result = clean_customer_names(df)
        # The output column should be object dtype (since no pd.StringDtype is used)
        assert result["customer_name_clean"].dtype == object

    def test_empty_dataframe(self):
        """Test that an empty DataFrame works and returns a new DataFrame with the extra column."""
        df = pd.DataFrame({"name": pd.Series(dtype=object)})
        result = clean_customer_names(df)
        assert len(result) == 0
        assert "customer_name_clean" in result.columns
        assert result["customer_name_clean"].dtype == object

    def test_mixed_types_in_input_column(self):
        """Test that non-string values (e.g., numbers) are passed through after normalization."""
        # The function applies _normalize_optional_string which likely converts to string and normalizes.
        # Expect that numeric values become strings after lowercasing and stripping.
        df = pd.DataFrame({"name": [123, 45.6]})
        result = clean_customer_names(df)
        # After conversion to string, "123" stays, "45.6" stays.
        expected = pd.Series(["123", "45.6"], name="customer_name_clean")
        pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)

    def test_combined_edge_cases(self):
        """Test a mix of nulls, blanks, accents, and spaces in one DataFrame."""
        df = pd.DataFrame({
            "name": [
                "   João  ",
                None,
                "",
                "  müller  ",
                "\t",
                "François  ",
                pd.NA,
                "  UPPER CASE  ",
            ]
        })
        result = clean_customer_names(df)
        expected = pd.Series(
            ["joao", pd.NA, pd.NA, "muller", pd.NA, "francois", pd.NA, "upper case"],
            name="customer_name_clean",
        )
        pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)

    def test_missing_name_column_raises_error(self):
        """Test that providing a non-existing column name raises an error."""
        df = pd.DataFrame({"other": [1]})
        with pytest.raises(ValueError):
            clean_customer_names(df, name_col="nonexistent")

    # Additional check: ensure that the output column exists and is not the same as input (unless same string)
    def test_output_column_name_not_in_input(self):
        """Test that the output column is new and doesn't overwrite an existing column accidentally."""
        df = pd.DataFrame({"name": ["John"], "name_clean": ["old"]})
        result = clean_customer_names(df, name_col="name", output_col="name_clean")
        # The original "name_clean" column should be replaced?
        # The function adds a new column; if output_col already exists, it will be overwritten.
        # This is allowed behavior. Verify that the overwritten column has the normalized value.
        assert result["name_clean"].iloc[0] == "john"
