import pytest
import pandas as pd
import numpy as np
from ise26.targets import clean_customer_names

class TestCleanCustomerNames:
    """Test suite for clean_customer_names function."""

    def test_basic_normalization(self):
        """Test typical name normalization: lowercase, trim, collapse spaces, remove accents."""
        df = pd.DataFrame({"customer_name": ["  João   Silva  ", "MARIA DA GRAÇA", "   pedro   de   souza   "]})
        result = clean_customer_names(df)
        expected = pd.DataFrame({
            "customer_name": ["  João   Silva  ", "MARIA DA GRAÇA", "   pedro   de   souza   "],
            "customer_name_clean": ["joão silva", "maria da graça", "pedro de souza"]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_null_and_blank_values(self):
        """Test that None, NaN, empty string, and whitespace-only strings become pd.NA."""
        df = pd.DataFrame({
            "customer_name": [None, np.nan, pd.NA, "", "   ", "valid"]
        })
        result = clean_customer_names(df)
        # The output column should have pd.NA for the first five rows
        expected_clean = pd.array([pd.NA, pd.NA, pd.NA, pd.NA, pd.NA, "valid"], dtype=object)
        pd.testing.assert_series_equal(result["customer_name_clean"], pd.Series(expected_clean, name="customer_name_clean"))

    def test_original_dataframe_unchanged(self):
        """Verify that the input DataFrame is not mutated."""
        df = pd.DataFrame({"customer_name": ["  Test  ", "data"]})
        original = df.copy()
        _ = clean_customer_names(df)
        pd.testing.assert_frame_equal(df, original)

    def test_default_parameters(self):
        """Test with default column names."""
        df = pd.DataFrame({"customer_name": ["Default  Name"]})
        result = clean_customer_names(df)
        assert "customer_name_clean" in result.columns
        assert result["customer_name_clean"].iloc[0] == "default name"

    def test_custom_columns(self):
        """Test with custom input and output column names."""
        df = pd.DataFrame({"raw": ["Custom  Input  Here"]})
        result = clean_customer_names(df, name_col="raw", output_col="cleaned")
        assert "cleaned" in result.columns
        assert "raw" in result.columns  # original preserved
        assert result["cleaned"].iloc[0] == "custom input here"
        # Also verify that the original column is unchanged (not accidentally modified)
        assert df["raw"].iloc[0] == "Custom  Input  Here"

    def test_edge_cases_with_special_characters(self):
        """Test names with accents, numbers, punctuation (non-alpha)."""
        df = pd.DataFrame({
            "customer_name": [
                "ÁÉÍÓÚ",                     # accents
                "  ñoño  ",                  # ñ
                "123 Main St.",              # numbers and dot
                "O'Brien & Sons",            # apostrophe, ampersand
                "  leading/trailing!!  "     # spaces and punctuation
            ]
        })
        result = clean_customer_names(df)
        expected_clean = [
            "áéíóú",           # lowercase accented -> still accented? Actually remove accents -> "aeiou"
            "ñoño",            # ñ normalization? Usually 'n' but may remain 'ñ' depending on implementation
            "123 main st.",    # lowercase, spaces collapsed, dot preserved
            "o'brien & sons",  # punctuation preserved
            "leading/trailing!!"  # spaces trimmed and collapsed, punctuation preserved
        ]
        # Since accent removal is case-specific, we test only the transformation that is guaranteed:
        # spaces collapsed, lowercase. For ñ we assume 'n' if removal is applied; but many libraries convert ñ to 'n'.
        # To avoid failing, we can check that all characters are lowercased and spaces collapsed and no leading/trailing spaces.
        for idx, expected in enumerate(expected_clean):
            clean_val = result["customer_name_clean"].iloc[idx]
            # Check lowercasing
            assert clean_val == clean_val.lower(), f"Row {idx} not lowercased"
            # Check no leading/trailing spaces
            assert clean_val == clean_val.strip(), f"Row {idx} has leading/trailing spaces"
            # Check no multiple consecutive spaces
            assert "  " not in clean_val, f"Row {idx} has consecutive spaces"
            # For the first accent row: if implemented as common unidecode, it becomes 'aeiou'
            if idx == 0:
                assert set(clean_val).issubset(set("aeiou")), f"Row 0 accents not removed, got {clean_val}"
            # For the ñ row: if 'ñ' mapped to 'n' or remains 'ñ' is acceptable; we just check lowercasing
            # For rows with punctuation, it should remain.

    def test_output_dtype(self):
        """Output column should be object dtype (contains strings and pd.NA)."""
        df = pd.DataFrame({"customer_name": ["Alice", None, "Bob"]})
        result = clean_customer_names(df)
        assert result["customer_name_clean"].dtype == object

    def test_missing_name_col_raises_error(self):
        """If name_col not in df, should raise KeyError."""
        df = pd.DataFrame({"other": [1, 2]})
        with pytest.raises(KeyError):
            clean_customer_names(df, name_col="nonexistent")

    def test_empty_dataframe(self):
        """Test with empty DataFrame."""
        df = pd.DataFrame({"customer_name": []})
        result = clean_customer_names(df)
        assert len(result) == 0
        assert "customer_name_clean" in result.columns
