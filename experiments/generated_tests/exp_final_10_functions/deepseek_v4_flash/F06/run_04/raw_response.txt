import pytest
import pandas as pd
from ise26.targets import classify_payment_status


def _make_df(
    due_dates,
    paid_dates=None,
    amounts=None,
    due_col="due_date",
    paid_col="paid_date",
    amount_col="amount",
):
    data = {due_col: due_dates}
    if paid_dates is not None:
        data[paid_col] = paid_dates
    else:
        data[paid_col] = [None] * len(due_dates)
    if amounts is not None:
        data[amount_col] = amounts
    else:
        data[amount_col] = [100.0] * len(due_dates)
    return pd.DataFrame(data)


class TestClassifyPaymentStatus:

    def test_invalid_amount_nan(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=[float("nan")],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_amount_non_numeric(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=["not_a_number"],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_amount_negative(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=[-5.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_amount_zero(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=[0.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_due_date(self):
        df = _make_df(
            due_dates=["not_a_date"],
            paid_dates=["2023-01-10"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_paid_date_provided(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["bad_date"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_paid_date_empty_string_not_provided(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["   "],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        # empty string after strip -> not provided -> treat as unpaid
        # reference after due => overdue
        assert result["payment_status"].iloc[0] == "overdue"

    def test_paid_on_time_equal(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-15"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_paid_on_time_before(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_paid_late(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-20"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-25")
        assert result["payment_status"].iloc[0] == "paid_late"

    def test_unpaid_overdue(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=[pd.NA],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "overdue"

    def test_unpaid_pending(self):
        df = _make_df(
            due_dates=["2023-01-25"],
            paid_dates=[pd.NA],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "pending"

    def test_unpaid_pending_exact_reference(self):
        df = _make_df(
            due_dates=["2023-01-20"],
            paid_dates=[pd.NA],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "pending"

    def test_mixed_statuses(self):
        df = pd.DataFrame({
            "due": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04", "2023-01-05", "bad_date"],
            "paid": ["2023-01-01", "2023-01-05", pd.NA, pd.NA, None, "2023-01-06"],
            "amount": [100.0, 100.0, 100.0, 50.0, 0.0, 100.0],
        })
        result = classify_payment_status(df, reference_date="2023-01-04",
                                         due_date_col="due", paid_date_col="paid")
        expected = ["paid_on_time", "paid_late", "overdue", "pending", "invalid", "invalid"]
        assert list(result["payment_status"]) == expected

    def test_output_preserves_order(self):
        df = _make_df(
            due_dates=["2023-01-03", "2023-01-01", "2023-01-02"],
            paid_dates=[pd.NA, pd.NA, pd.NA],
            amounts=[100.0, 100.0, 100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-02")
        # reference <= due dates for first and third? Let's compute:
        # row0: due 2023-01-03 > ref 2023-01-02 => pending
        # row1: due 2023-01-01 < ref => overdue
        # row2: due 2023-01-02 == ref => pending
        assert list(result["payment_status"]) == ["pending", "overdue", "pending"]

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "dd": ["2023-01-15"],
            "pd": ["2023-01-10"],
            "amt": [100.0],
        })
        result = classify_payment_status(
            df, reference_date="2023-01-20",
            due_date_col="dd", paid_date_col="pd", amount_col="amt", output_col="status"
        )
        assert "status" in result.columns
        assert result["status"].iloc[0] == "paid_on_time"

    def test_original_dataframe_unchanged(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=[100.0],
        )
        original = df.copy()
        classify_payment_status(df, reference_date="2023-01-20")
        pd.testing.assert_frame_equal(df, original)

    def test_invalid_reference_date_raises(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=[pd.NA],
            amounts=[100.0],
        )
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, reference_date="not_a_date")

    def test_paid_date_none_missing(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=[None],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "overdue"

    def test_amount_missing(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["2023-01-10"],
            amounts=[None],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_paid_date_provided_invalid_still_invalid_even_if_due_valid(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=["invalid"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_due_date_and_paid_date_both_invalid(self):
        df = _make_df(
            due_dates=["bad"],
            paid_dates=["also_bad"],
            amounts=[100.0],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_mixed_datetime_types_for_reference(self):
        df = _make_df(
            due_dates=["2023-01-15"],
            paid_dates=[pd.NA],
            amounts=[100.0],
        )
        result1 = classify_payment_status(df, reference_date="2023-01-20")
        result2 = classify_payment_status(df, reference_date=pd.Timestamp("2023-01-20"))
        assert result1["payment_status"].iloc[0] == "overdue"
        assert result2["payment_status"].iloc[0] == "overdue"

    def test_all_invalid_produces_all_invalid(self):
        df = _make_df(
            due_dates=["bad1", "bad2"],
            paid_dates=["2023-01-10", None],
            amounts=[0.0, -1],
        )
        result = classify_payment_status(df, reference_date="2023-01-20")
        assert (result["payment_status"] == "invalid").all()
