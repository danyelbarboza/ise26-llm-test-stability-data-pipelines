import pytest
import pandas as pd
import numpy as np
from ise26.targets import classify_payment_status


class TestClassifyPaymentStatus:
    """Test suite for classify_payment_status."""

    # ------------------------------------------------------------------ #
    # Helper to create a base DataFrame with common columns
    # ------------------------------------------------------------------ #
    @staticmethod
    def _make_df(**kwargs):
        defaults = {
            "due_date": pd.to_datetime(["2023-01-15", "2023-02-20"]),
            "paid_date": pd.to_datetime(["2023-01-10", "2023-02-25"]),
            "amount": [100.0, 200.0],
        }
        defaults.update(kwargs)
        return pd.DataFrame(defaults)

    # ------------------------------------------------------------------ #
    # Reference date validation
    # ------------------------------------------------------------------ #
    def test_invalid_reference_date_raises(self):
        df = self._make_df()
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, reference_date="not_a_date")

    def test_none_reference_date_raises(self):
        df = self._make_df()
        with pytest.raises(ValueError):
            classify_payment_status(df, reference_date=None)

    def test_valid_reference_date_passes(self):
        df = self._make_df()
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert "payment_status" in result.columns

    # ------------------------------------------------------------------ #
    # Basic classification – paid on time / late / pending / overdue
    # ------------------------------------------------------------------ #
    def test_paid_on_time(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": ["2023-01-10"],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_paid_late(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": ["2023-01-20"],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "paid_late"

    def test_unpaid_overdue(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": [None],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "overdue"

    def test_unpaid_pending(self):
        df = pd.DataFrame({
            "due_date": ["2023-03-01"],
            "paid_date": [None],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "pending"

    def test_pending_on_due_date(self):
        """Reference date exactly equal to due_date => pending (<=)."""
        df = pd.DataFrame({
            "due_date": ["2023-02-01"],
            "paid_date": [None],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "pending"

    def test_overdue_when_ref_after_due(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-31"],
            "paid_date": [None],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "overdue"

        """Empty string after strip is considered not provided (treated as unpaid)."""
        df = pd.DataFrame({

    # ------------------------------------------------------------------ #
    # Edge cases – boundary dates (equal, time components)
    # ------------------------------------------------------------------ #
    def test_paid_on_time_equal_dates(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": ["2023-01-15"],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_paid_late_same_day_different_time(self):
        """Paid date with later time on same calendar day is still <= due date
        if parsed as datetime without time -> paid_date <= due_date."""
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": [pd.Timestamp("2023-01-15 23:59:59")],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        # The function parses using pd.to_datetime with mixed format; if both are dates,
        # the paid_date will be coerced to a date? Actually pd.to_datetime will keep the time.
        # Since the due_date string "2023-01-15" becomes midnight 00:00:00.
        # Then paid_date > due_date, so it becomes paid_late.
        # This is a subtle point – the function treats the dates as timestamps.
        # We'll accept the behavior as defined: strict comparison.
        assert result["payment_status"].iloc[0] == "paid_late"

    def test_pending_on_due_date_with_time(self):
        df = pd.DataFrame({
            "due_date": ["2023-02-01 12:00:00"],
            "paid_date": [None],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        # Reference date without time becomes midnight, which is <= due_date noon -> pending
        assert result["payment_status"].iloc[0] == "pending"

    # ------------------------------------------------------------------ #
    # Mixed valid and invalid rows
    # ------------------------------------------------------------------ #
    def test_mixed_classifications(self):
        df = pd.DataFrame({
            "due_date": [
                "2023-01-15",
                "2023-01-15",
                "2023-01-15",
                "2023-01-15",
                "2023-01-15",
                None,
            ],
            "paid_date": [
                "2023-01-10",  # paid_on_time
                "2023-01-20",  # paid_late
                None,          # unpaid reference 2023-02-01 -> overdue
                "invalid",     # invalid paid_date -> invalid
                "2023-01-10",  # amount zero -> invalid
                "2023-01-10",  # invalid due_date -> invalid
            ],
            "amount": [
                100.0,
                100.0,
                100.0,
                100.0,
                0.0,
                100.0,
            ],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        expected = ["paid_on_time", "paid_late", "overdue", "invalid", "invalid", "invalid"]
        assert result["payment_status"].tolist() == expected

    # ------------------------------------------------------------------ #
    # Non-default column names
    # ------------------------------------------------------------------ #
    def test_custom_column_names(self):
        df = pd.DataFrame({
            "dd": ["2023-01-15"],
            "pd": ["2023-01-10"],
            "amt": [100.0],
        })
        result = classify_payment_status(
            df,
            reference_date="2023-02-01",
            due_date_col="dd",
            paid_date_col="pd",
            amount_col="amt",
            output_col="status",
        )
        assert "status" in result.columns
        assert result["status"].iloc[0] == "paid_on_time"

    # ------------------------------------------------------------------ #
    # Preservation of original DataFrame
    # ------------------------------------------------------------------ #
    def test_does_not_modify_original(self):
        df_orig = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": ["2023-01-10"],
            "amount": [100.0],
        })
        df_copy = df_orig.copy()
        classify_payment_status(df_orig, reference_date="2023-02-01")
        pd.testing.assert_frame_equal(df_orig, df_copy)

    def test_returns_new_dataframe(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": ["2023-01-10"],
            "amount": [100.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-01")
        assert result is not df

    # ------------------------------------------------------------------ #
    # Order preservation
    # ------------------------------------------------------------------ #
    def test_order_preserved(self):
        df = pd.DataFrame({
            "due_date": ["2023-01-15", "2023-02-20", "2023-03-10"],
            "paid_date": [None, None, None],
            "amount": [100.0, 200.0, 300.0],
        })
        result = classify_payment_status(df, reference_date="2023-02-15")
        # Expected: overdue, pending, pending
        expected_status = ["overdue", "pending", "pending"]
        assert result["payment_status"].tolist() == expected_status
        # Also check row values unchanged
        assert result["due_date"].tolist() == df["due_date"].tolist()

        assert statuses.issubset(allowed)
