import pandas as pd
import numpy as np
import pytest
from ise26.targets import classify_payment_status


class TestClassifyPaymentStatus:
    """Test suite for classify_payment_status function."""

    def test_invalid_amount_missing(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [None],
            'amount': [np.nan]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'invalid'

    def test_invalid_amount_non_numeric(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [None],
            'amount': ['abc']
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'invalid'

    def test_invalid_amount_zero(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [None],
            'amount': [0]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'invalid'

    def test_invalid_amount_negative(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [None],
            'amount': [-5.0]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'invalid'

    def test_invalid_due_date(self):
        df = pd.DataFrame({
            'due_date': ['not_a_date'],
            'paid_date': [None],
            'amount': [100]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'invalid'

    def test_invalid_paid_date_provided(self):
        """paid_date is provided but invalid → invalid, even if due_date and amount are valid."""
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': ['invalid_date'],
            'amount': [100]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'invalid'

    def test_invalid_paid_date_empty_string(self):
        """Empty string in paid_date is considered not provided (coerced to NaT) and should not be invalid."""
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [''],
            'amount': [100]
        })
        result = classify_payment_status(df, reference_date='2023-01-10')
        # Unpaid, reference_date < due_date -> pending
        assert result['payment_status'].iloc[0] == 'pending'

    def test_paid_on_time(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': ['2023-01-10'],
            'amount': [200]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'paid_on_time'

    def test_paid_on_time_equal_date(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': ['2023-01-15'],
            'amount': [200]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'paid_on_time'

    def test_paid_late(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': ['2023-01-20'],
            'amount': [200]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'paid_late'

    def test_overdue(self):
        """Unpaid, reference_date > due_date."""
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [None],
            'amount': [200]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'overdue'

    def test_pending(self):
        """Unpaid, reference_date <= due_date."""
        df = pd.DataFrame({
            'due_date': ['2023-02-01'],
            'paid_date': [None],
            'amount': [200]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'pending'

    def test_pending_reference_before_due(self):
        df = pd.DataFrame({
            'due_date': ['2023-03-01'],
            'paid_date': [None],
            'amount': [200]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'pending'

    def test_reference_date_invalid_raises(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [None],
            'amount': [100]
        })
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, reference_date='not_a_date')

    def test_custom_column_names(self):
        df = pd.DataFrame({
            'dd': ['2023-01-15'],
            'pd': ['2023-01-10'],
            'amt': [100]
        })
        result = classify_payment_status(
            df,
            reference_date='2023-02-01',
            due_date_col='dd',
            paid_date_col='pd',
            amount_col='amt',
            output_col='status'
        )
        assert 'status' in result.columns
        assert result['status'].iloc[0] == 'paid_on_time'

    def test_original_dataframe_unchanged(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': ['2023-01-10'],
            'amount': [100]
        })
        original = df.copy()
        classify_payment_status(df, reference_date='2023-02-01')
        pd.testing.assert_frame_equal(df, original)

    def test_preserves_order(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15', '2023-02-01', '2023-03-01'],
            'paid_date': ['2023-01-10', None, '2023-03-05'],
            'amount': [100, 200, 300]
        })
        result = classify_payment_status(df, reference_date='2023-02-15')
        expected_statuses = ['paid_on_time', 'overdue', 'paid_late']
        assert list(result['payment_status']) == expected_statuses

    def test_mixed_scenarios(self):
        df = pd.DataFrame({
            'due_date': [
                '2023-01-15',   # valid due
                'invalid_date', # invalid due
                '2023-02-01',   # valid due
                '2023-03-01',   # valid due
                '2023-04-01',   # valid due
            ],
            'paid_date': [
                '2023-01-10',   # on time
                None,           # no paid date, but due invalid → invalid
                '2023-02-05',   # late
                None,           # unpaid, reference > due → overdue
                None,           # unpaid, reference < due → pending
            ],
            'amount': [
                100,
                200,
                -5,             # invalid amount
                400,
                500
            ]
        })
        result = classify_payment_status(df, reference_date='2023-03-15')
        expected = ['paid_on_time', 'invalid', 'invalid', 'overdue', 'pending']
        assert list(result['payment_status']) == expected

    def test_output_column_override(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': ['2023-01-10'],
            'amount': [100]
        })
        result = classify_payment_status(df, reference_date='2023-02-01', output_col='my_status')
        assert 'my_status' in result.columns
        assert result['my_status'].iloc[0] == 'paid_on_time'

    def test_paid_date_none_but_due_valid(self):
        df = pd.DataFrame({
            'due_date': pd.to_datetime(['2023-01-15']),
            'paid_date': [None],
            'amount': [100.0]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'overdue'

    def test_paid_date_nat_none(self):
        df = pd.DataFrame({
            'due_date': ['2023-01-15'],
            'paid_date': [pd.NaT],
            'amount': [100]
        })
        result = classify_payment_status(df, reference_date='2023-02-01')
        assert result['payment_status'].iloc[0] == 'overdue'
