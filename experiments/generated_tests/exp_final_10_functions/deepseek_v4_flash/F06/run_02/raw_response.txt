import pytest
import pandas as pd
from ise26.targets import classify_payment_status


def create_valid_df():
    """Return a minimal valid DataFrame for testing."""
    return pd.DataFrame({
        "due_date": ["2024-01-15", "2024-02-20", "2024-03-10", "2024-04-05"],
        "paid_date": ["2024-01-10", "2024-02-25", None, None],
        "amount": [100.0, 200.0, 300.0, 400.0],
    })


class TestClassifyPaymentStatus:
    def test_paid_on_time(self):
        df = create_valid_df()
        result = classify_payment_status(df, reference_date="2024-04-10")
        statuses = result["payment_status"]
        assert statuses.iloc[0] == "paid_on_time"

    def test_paid_late(self):
        df = create_valid_df()
        result = classify_payment_status(df, reference_date="2024-04-10")
        statuses = result["payment_status"]
        assert statuses.iloc[1] == "paid_late"

    def test_overdue(self):
        df = create_valid_df()
        result = classify_payment_status(df, reference_date="2024-04-10")
        statuses = result["payment_status"]
        # Row 2: due 2024-03-10, unpaid, reference > due => overdue
        assert statuses.iloc[2] == "overdue"

    def test_pending(self):
        df = create_valid_df()
        # Use reference_date equal to due_date of row 3 (2024-04-05) => pending
        result = classify_payment_status(df, reference_date="2024-04-05")
        statuses = result["payment_status"]
        assert statuses.iloc[3] == "pending"

    def test_invalid_amount_missing(self):
        df = create_valid_df()
        df.loc[0, "amount"] = None
        result = classify_payment_status(df, reference_date="2024-04-10")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_amount_non_numeric(self):
        df = create_valid_df()
        df.loc[1, "amount"] = "abc"
        result = classify_payment_status(df, reference_date="2024-04-10")
        assert result["payment_status"].iloc[1] == "invalid"

    def test_invalid_amount_zero(self):
        df = create_valid_df()
        df.loc[2, "amount"] = 0.0
        result = classify_payment_status(df, reference_date="2024-04-10")
        assert result["payment_status"].iloc[2] == "invalid"

    def test_invalid_amount_negative(self):
        df = create_valid_df()
        df.loc[3, "amount"] = -50.0
        result = classify_payment_status(df, reference_date="2024-04-10")
        assert result["payment_status"].iloc[3] == "invalid"

    def test_invalid_due_date(self):
        df = create_valid_df()
        df.loc[0, "due_date"] = "not-a-date"
        result = classify_payment_status(df, reference_date="2024-04-10")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_provided_paid_date(self):
        df = create_valid_df()
        df.loc[0, "paid_date"] = "bad-date"
        result = classify_payment_status(df, reference_date="2024-04-10")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_missing_paid_date_treated_as_unpaid(self):
        df = create_valid_df()
        # row 2 already has None paid_date, ensure it's pending/overdue, not invalid
        result = classify_payment_status(df, reference_date="2024-04-10")
        # due 2024-03-10, unpaid, reference > due => overdue, not invalid
        assert result["payment_status"].iloc[2] == "overdue"

    def test_paid_on_time_equal_to_due(self):
        df = pd.DataFrame({
            "due_date": ["2024-05-01"],
            "paid_date": ["2024-05-01"],
            "amount": [150.0],
        })
        result = classify_payment_status(df, reference_date="2024-05-15")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_reference_date_invalid_raises(self):
        df = create_valid_df()
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, reference_date="invalid-date")

    def test_output_column_preserves_order(self):
        df = create_valid_df()
        result = classify_payment_status(df, reference_date="2024-04-10")
        # Check that the original DataFrame index order is unchanged
        pd.testing.assert_index_equal(result.index, df.index)

    def test_original_dataframe_unchanged(self):
        df = create_valid_df()
        df_copy = df.copy()
        classify_payment_status(df, reference_date="2024-04-10")
        pd.testing.assert_frame_equal(df, df_copy)

    def test_empty_dataframe(self):
        df = pd.DataFrame({"due_date": [], "paid_date": [], "amount": []})
        result = classify_payment_status(df, reference_date="2024-01-01")
        assert len(result) == 0
        assert "payment_status" in result.columns

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "my_due": ["2024-01-01"],
            "my_paid": ["2024-01-01"],
            "my_amount": [100.0],
        })
        result = classify_payment_status(
            df,
            reference_date="2024-01-15",
            due_date_col="my_due",
            paid_date_col="my_paid",
            amount_col="my_amount",
            output_col="my_status",
        )
        assert "my_status" in result.columns
        assert result["my_status"].iloc[0] == "paid_on_time"

    def test_multiple_classifications_in_one_call(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-01", "2024-01-01", "2024-01-15", "2024-01-20"],
            "paid_date": ["2024-01-01", "2024-01-10", None, "invalid"],
            "amount": [100, 200, 300, 400],
        })
        result = classify_payment_status(df, reference_date="2024-01-20")
        expected = ["paid_on_time", "paid_late", "overdue", "invalid"]
        assert result["payment_status"].tolist() == expected

    def test_amount_numeric_string_allowed(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-01"],
            "paid_date": ["2024-01-01"],
            "amount": ["150.0"],
        })
        result = classify_payment_status(df, reference_date="2024-01-15")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_all_rows_invalid(self):
        df = pd.DataFrame({
            "due_date": [None, None],
            "paid_date": [None, None],
            "amount": [0, -1],
        })
        result = classify_payment_status(df, reference_date="2024-01-01")
        assert (result["payment_status"] == "invalid").all()

    def test_reference_date_as_datetime(self):
        from datetime import datetime
        df = create_valid_df()
        result = classify_payment_status(df, reference_date=datetime(2024, 4, 10))
        assert result["payment_status"].iloc[0] == "paid_on_time"
