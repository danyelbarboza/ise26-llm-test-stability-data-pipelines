import pytest
import pandas as pd
from ise26.targets import calculate_monthly_revenue


# Basic functionality: multiple valid orders, some canceled, different months
def test_basic_functionality():
    data = {
        "order_date": ["2023-01-15", "2023-01-20", "2023-02-10", "2023-02-25", "2023-03-05"],
        "amount": [100, 200, 150, 300, 250],
        "status": ["delivered", "canceled", "delivered", "cancelled", "shipped"],
    }
    df = pd.DataFrame(data)
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01", "2023-02", "2023-03"], dtype="string"),
        "revenue": [100.0, 150.0, 250.0],
    })
    pd.testing.assert_frame_equal(result, expected)


# Edge: empty dataframe (no rows)
def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert isinstance(result, pd.DataFrame)
    assert list(result.columns) == ["month", "revenue"]
    assert result.dtypes["month"] == pd.StringDtype()
    assert result.dtypes["revenue"] == "float64"
    assert len(result) == 0

# Edge: invalid amounts (missing, non-numeric) treated as zero
def test_invalid_amounts_treated_as_zero():
    data = {
        "order_date": ["2023-01-01", "2023-01-15", "2023-02-01"],
        "amount": ["invalid", None, 50],
        "status": ["delivered", "delivered", "canceled"],
    }
    df = pd.DataFrame(data)
    result = calculate_monthly_revenue(df)
    # First two are valid, invalid amount becomes 0, second is valid 50? Wait second amount is None? Actually second row amount is None, but status is delivered, so amount treated as 0. Third is canceled.
    # So valid rows: first (amount=0), second (amount=0), third canceled ignored.
    # Both are in January, revenue sum = 0.
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01"], dtype="string"),
        "revenue": [0.0],
    })
    pd.testing.assert_frame_equal(result, expected)

# Check that output is sorted by month
def test_sorted_output():
    data = {
        "order_date": ["2023-03-01", "2023-01-01", "2023-02-01"],
        "amount": [30, 10, 20],
        "status": ["delivered", "delivered", "delivered"],
    }
    df = pd.DataFrame(data)
    result = calculate_monthly_revenue(df)
    assert list(result["month"]) == ["2023-01", "2023-02", "2023-03"]
    assert list(result["revenue"]) == [10.0, 20.0, 30.0]


# Check that amounts are correctly summed for same month
def test_aggregation_same_month():
    data = {
        "order_date": ["2023-01-01", "2023-01-15", "2023-01-20"],
        "amount": [100, 200, 300],
        "status": ["delivered", "delivered", "canceled"],
    }
    df = pd.DataFrame(data)
    result = calculate_monthly_revenue(df)
    # Only first two are valid, sum = 300
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01"], dtype="string"),
        "revenue": [300.0],
    })
    pd.testing.assert_frame_equal(result, expected)


# Check that original DataFrame is not modified (immutability)
def test_immutability():
    data = {
        "order_date": ["2023-01-01", "2023-02-01"],
        "amount": [100, 200],
        "status": ["delivered", "canceled"],
    }
    original = pd.DataFrame(data)
    copy = original.copy()
    _ = calculate_monthly_revenue(original)
    pd.testing.assert_frame_equal(original, copy)
