import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue


def test_basic_monthly_revenue():
    df = pd.DataFrame({
        "order_date": ["2023-01-15", "2023-01-20", "2023-02-10"],
        "amount": [100.0, 200.0, 300.0],
        "status": ["completed", "shipped", "delivered"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01", "2023-02"], dtype="string"),
        "revenue": [300.0, 300.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_ignore_cancelled_orders():
    df = pd.DataFrame({
        "order_date": ["2023-01-15", "2023-01-20"],
        "amount": [100.0, 200.0],
        "status": ["cancelled", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01"], dtype="string"),
        "revenue": [200.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_cancelled_variations():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04"],
        "amount": [10.0, 20.0, 30.0, 40.0],
        "status": ["canceled", "CANCELLED", " Cancelado ", "cancelled"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series([], dtype="string"),
        "revenue": pd.Series([], dtype="float64")
    })
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_dates_ignored():
    df = pd.DataFrame({
        "order_date": ["2023-01-15", "not-a-date", "2023-02-10", None],
        "amount": [100.0, 200.0, 300.0, 400.0],
        "status": ["completed", "completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01", "2023-02"], dtype="string"),
        "revenue": [100.0, 300.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_or_missing_amounts_treated_as_zero():
    df = pd.DataFrame({
        "order_date": ["2023-01-15", "2023-01-20", "2023-02-10"],
        "amount": [100.0, "invalid", None],
        "status": ["completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01", "2023-02"], dtype="string"),
        "revenue": [100.0, 0.0]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_empty_dataframe_returns_empty_result():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series([], dtype="string"),
        "revenue": pd.Series([], dtype="float64")
    })
    pd.testing.assert_frame_equal(result, expected)

def test_column_dtypes():
    df = pd.DataFrame({
        "order_date": ["2023-01-15"],
        "amount": [100.0],
        "status": ["completed"]
    })
    result = calculate_monthly_revenue(df)
    assert result["month"].dtype == "string"
    assert result["revenue"].dtype == "float64"


def test_multiple_rows_same_month():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-15", "2023-01-31"],
        "amount": [10.0, 20.0, 30.0],
        "status": ["completed", "completed", "completed"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01"], dtype="string"),
        "revenue": [60.0]
    })
    pd.testing.assert_frame_equal(result, expected)
