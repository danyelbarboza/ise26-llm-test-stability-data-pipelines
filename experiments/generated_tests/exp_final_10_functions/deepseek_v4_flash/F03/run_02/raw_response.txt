import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue


class TestCalculateMonthlyRevenue:
    """Test suite for calculate_monthly_revenue."""

    # ---------- helper to build a DataFrame quickly ----------
    @staticmethod
    def _make_df(data, columns=None):
        if columns is None:
            columns = ["order_date", "amount", "status"]
        return pd.DataFrame(data, columns=columns)

    # ---------- basic positive cases ----------
    def test_basic_monthly_revenue(self):
        """Aggregate valid non-canceled orders into monthly revenue."""
        df = self._make_df([
            ("2024-01-15", 100.0, "completed"),
            ("2024-01-20", 50.0, "shipped"),
            ("2024-01-25", 30.0, "canceled"),
            ("2024-02-10", 200.0, "pending"),
            ("2024-02-15", 150.0, "cancelado"),
            ("2024-03-01", 300.0, "cancelled"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01", "2024-02"], dtype="string"),
            "revenue": [150.0, 200.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    def test_multiple_orders_same_month(self):
        """Sum amounts for multiple valid orders in the same month."""
        df = self._make_df([
            ("2024-01-05", 10.0, "completed"),
            ("2024-01-12", 20.0, "shipped"),
            ("2024-01-20", 30.0, "completed"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01"], dtype="string"),
            "revenue": [60.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    # ---------- edge cases: empty output ----------
    def test_all_canceled_orders(self):
        """All rows are canceled → empty DataFrame with correct dtypes."""
        df = self._make_df([
            ("2024-01-01", 100.0, "canceled"),
            ("2024-02-01", 200.0, "cancelado"),
        ])
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame({
            "month": pd.Series(dtype="string"),
            "revenue": pd.Series(dtype="float64"),
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_all_invalid_dates(self):
        """All dates are invalid → empty DataFrame."""
        df = self._make_df([
            ("not-a-date", 100.0, "completed"),
            ("also-bad", 200.0, "shipped"),
        ])
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame({
            "month": pd.Series(dtype="string"),
            "revenue": pd.Series(dtype="float64"),
        })
        pd.testing.assert_frame_equal(result, expected)

    # ---------- invalid elements are ignored/treated as zero ----------
    def test_ignore_invalid_dates(self):
        """Rows with unparseable dates are excluded from aggregation."""
        df = self._make_df([
            ("2024-01-01", 100.0, "completed"),
            ("bad-date", 50.0, "completed"),
            ("2024-02-01", 200.0, "shipped"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01", "2024-02"], dtype="string"),
            "revenue": [100.0, 200.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_amounts_treated_as_zero(self):
        """Non-numeric amounts are replaced by zero."""
        df = self._make_df([
            ("2024-01-01", "abc", "completed"),
            ("2024-01-15", 50.0, "shipped"),
            ("2024-02-01", None, "completed"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01", "2024-02"], dtype="string"),
            "revenue": [50.0, 0.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_amounts_treated_as_zero(self):
        """NaN amounts are replaced by zero."""
        df = self._make_df([
            ("2024-01-01", float("nan"), "completed"),
            ("2024-01-15", 100.0, "completed"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01"], dtype="string"),
            "revenue": [100.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    # ---------- cancel detection variations ----------
    @pytest.mark.parametrize(
        "status",
        [
            "canceled",
            "cancelled",
            "cancelado",
            "CANCELED",
            "Cancelled",
            " Cancelado ",
        ],
    )
    def test_canceled_variants_are_excluded(self, status):
        """All common canceled variants exclude the order."""
        df = self._make_df([
            ("2024-01-01", 100.0, "completed"),
            ("2024-01-15", 50.0, status),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01"], dtype="string"),
            "revenue": [100.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    def test_non_canceled_statuses_included(self):
        """Orders with non-canceled statuses are included."""
        df = self._make_df([
            ("2024-01-01", 100.0, "shipped"),
            ("2024-01-15", 50.0, "pending"),
            ("2024-02-01", 200.0, "completed"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01", "2024-02"], dtype="string"),
            "revenue": [150.0, 200.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_status_treated_as_non_canceled(self):
        """NaN status should be treated as non-canceled."""
        df = self._make_df([
            ("2024-01-01", 100.0, float("nan")),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01"], dtype="string"),
            "revenue": [100.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    # ---------- ordering and dtypes ----------
    def test_output_sorted_by_month(self):
        """Result is sorted by month in ascending order."""
        df = self._make_df([
            ("2024-03-01", 300.0, "completed"),
            ("2024-01-01", 100.0, "completed"),
            ("2024-02-01", 200.0, "completed"),
        ])
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01", "2024-02", "2024-03"], dtype="string"),
            "revenue": [100.0, 200.0, 300.0],
        })
        result = calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(result, expected)

    def test_dtypes(self):
        """Month column is string, revenue is float64."""
        df = self._make_df([
            ("2024-01-01", 100.0, "completed"),
        ])
        result = calculate_monthly_revenue(df)
        assert result["month"].dtype == "string"
        assert result["revenue"].dtype == "float64"

    # ---------- custom column names ----------
    def test_custom_column_names(self):
        """Function works with explicitly provided column names."""
        df = pd.DataFrame({
            "order_date": ["2024-01-01", "2024-02-01"],
            "amount": [100.0, 200.0],
            "status": ["completed", "canceled"],
        })
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01"], dtype="string"),
            "revenue": [100.0],
        })
        result = calculate_monthly_revenue(df, "order_date", "amount", "status")
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_column_names_different(self):
        """Works when DataFrame columns have other names."""
        df = pd.DataFrame({
            "dt": ["2024-01-01", "2024-02-01", "2024-03-01"],
            "val": [100, 200, 300],
            "sts": ["completed", "cancelado", "shipped"],
        })
        expected = pd.DataFrame({
            "month": pd.Series(["2024-01", "2024-03"], dtype="string"),
            "revenue": [100.0, 300.0],
        })
        result = calculate_monthly_revenue(df, "dt", "val", "sts")
        pd.testing.assert_frame_equal(result, expected)
