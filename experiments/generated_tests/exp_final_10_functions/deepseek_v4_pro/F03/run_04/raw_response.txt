from ise26.targets import calculate_monthly_revenue

import numpy as np
import pandas as pd
import pytest
from pandas.testing import assert_frame_equal

def test_basic_revenue_aggregation():
    df = pd.DataFrame(
        {
            "order_date": ["2023-01-15", "2023-01-20", "2023-02-10"],
            "amount": [100.0, 200.0, 50.0],
            "status": ["completed", "shipped", "delivered"],
        }
    )
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(
        {
            "month": pd.array(["2023-01", "2023-02"], dtype="string"),
            "revenue": pd.array([300.0, 50.0], dtype="float64"),
        }
    )
    assert_frame_equal(result, expected)

def test_ignores_canceled_statuses():
    df = pd.DataFrame(

    assert_frame_equal(result, expected)

def test_all_rows_filtered_returns_empty_df_with_correct_dtypes():
    df = pd.DataFrame(
        {
            "order_date": ["bad", "2023-07-01"],
            "amount": [10, 20],
            "status": ["cancelled", "ok"],
        }
    )
    # first row invalid date, second row canceled -> both filtered
    result = calculate_monthly_revenue(df)
    assert isinstance(result, pd.DataFrame)
    assert list(result.columns) == ["month", "revenue"]
    assert len(result) == 0
    assert result["month"].dtype == "string"
    assert result["revenue"].dtype == np.float64

def test_empty_input_dataframe():

def test_result_is_sorted_by_month():
    df = pd.DataFrame(
        {
            "order_date": ["2023-10-01", "2023-09-15", "2023-11-20"],
            "amount": [10, 20, 30],
            "status": ["ok"] * 3,
        }
    )
    result = calculate_monthly_revenue(df)
    expected_months = pd.array(["2023-09", "2023-10", "2023-11"], dtype="string")
    assert list(result["month"]) == list(expected_months)
    assert result["revenue"].tolist() == [20.0, 10.0, 30.0]  # revenue corresponds to sorted months

def test_missing_required_columns_raises():
    df = pd.DataFrame({"order_date": ["2024-03-01"], "amount": [100]})  # missing status column
    with pytest.raises(ValueError):
        calculate_monthly_revenue(df)

    df = pd.DataFrame({"order_date": ["2024-03-01"], "status": ["ok"]})  # missing amount
    with pytest.raises(ValueError):
        calculate_monthly_revenue(df)

    df = pd.DataFrame({"amount": [100], "status": ["ok"]})  # missing date
    with pytest.raises(ValueError):
        calculate_monthly_revenue(df)
