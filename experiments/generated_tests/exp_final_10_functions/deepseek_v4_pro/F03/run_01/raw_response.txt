import pytest
import pandas as pd
from ise26.targets import calculate_monthly_revenue


def test_basic_aggregation():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2024-01-15", "2024-02-10", "2024-02-20"],
        "amount": [100, 200, 300, 400],
        "status": ["completed", "shipped", "done", "pending"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2024-01", "2024-02"], dtype="string"),
        "revenue": pd.Series([300.0, 700.0], dtype="float64"),
    })
    pd.testing.assert_frame_equal(result, expected)


def test_canceled_orders_ignored():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2024-01-02", "2024-01-03"],
        "amount": [100, 200, 300],
        "status": ["cancelled", "canceled", "cancelado"],
    })
    result = calculate_monthly_revenue(df)
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]
    assert result["month"].dtype == "string"
    assert result["revenue"].dtype == "float64"


def test_status_case_insensitive_and_whitespace():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2024-01-02", "2024-01-03",
                       "2024-01-04", "2024-01-05"],
        "amount": [10, 20, 30, 40, 50],
        "status": ["CANCELLED", " Canceled", "cancelADO ", "Completed", "  cancelled  "],
    })
    result = calculate_monthly_revenue(df)
    # Only the "Completed" order should be counted
    assert len(result) == 1
    assert result.iloc[0]["month"] == "2024-01"
    assert result.iloc[0]["revenue"] == 40.0


def test_invalid_dates_ignored():
    df = pd.DataFrame({
        "order_date": ["not-a-date", "2024-02-30", "2024-01-01", None, ""],
        "amount": [100, 200, 300, 400, 500],
        "status": ["completed"] * 5,
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 1
    assert result.iloc[0]["month"] == "2024-01"
    assert result.iloc[0]["revenue"] == 300.0


def test_invalid_amounts_treated_as_zero():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2024-01-02", "2024-01-03", "2024-01-04"],
        "amount": [100, "invalid", None, pd.NA],
        "status": ["completed"] * 4,
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 1
    assert result.iloc[0]["revenue"] == 100.0


def test_mixed_valid_and_invalid_rows():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "bad-date", "2024-01-15",
                       "2024-02-01", "2024-02-02"],
        "amount": [100, 999, 200, "abc", 300],
        "status": ["done", "completed", "CANCELED", "shipped", "cancelled "],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2024-01", "2024-02"], dtype="string"),
        "revenue": pd.Series([100.0, 300.0], dtype="float64"),
    })
    pd.testing.assert_frame_equal(result, expected)


def test_sorting_by_month():
    df = pd.DataFrame({
        "order_date": ["2024-02-01", "2023-12-31", "2024-01-10"],
        "amount": [200, 100, 300],
        "status": ["done"] * 3,
    })
    result = calculate_monthly_revenue(df)
    expected_months = ["2023-12", "2024-01", "2024-02"]
    assert result["month"].tolist() == expected_months
    assert result["revenue"].tolist() == [100.0, 300.0, 200.0]


def test_empty_input_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]
    assert result["month"].dtype == "string"
    assert result["revenue"].dtype == "float64"


def test_custom_column_names():
    df = pd.DataFrame({
        "dt": ["2024-01-01", "2024-01-02"],
        "val": [10, 20],
        "st": ["ok", "ok"],
    })
    result = calculate_monthly_revenue(df, date_col="dt", amount_col="val", status_col="st")
    assert len(result) == 1
    assert result.iloc[0]["month"] == "2024-01"
    assert result.iloc[0]["revenue"] == 30.0


def test_missing_column_raises():
    df = pd.DataFrame({
        "order_date": ["2024-01-01"],
        "amount": [100],
    })
    with pytest.raises(Exception):
        calculate_monthly_revenue(df)


def test_original_dataframe_not_modified():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2024-01-02"],
        "amount": [100, 200],
        "status": ["done", "cancelled"],
    })
    original = df.copy()
    _ = calculate_monthly_revenue(df)
    pd.testing.assert_frame_equal(df, original)


def test_result_index_reset():
    df = pd.DataFrame({
        "order_date": ["2024-01-01", "2024-01-15"],
        "amount": [100, 200],
        "status": ["ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    expected_index = pd.RangeIndex(0, 1)
    pd.testing.assert_index_equal(result.index, expected_index)
