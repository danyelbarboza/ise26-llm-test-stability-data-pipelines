import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue


@pytest.fixture
def sample_df():
    """Basic DataFrame with mixed data for common tests."""
    return pd.DataFrame({
        "order_date": [
            "2023-01-15",
            "2023-01-20",
            "2023-02-10",
            "invalid_date",
            "2023-02-28",
            "2023-03-05",
        ],
        "amount": [100, 200, 300, 400, 500, 600],
        "status": [
            "delivered",
            "shipped",
            "cancelled",
            "delivered",
            "pending",
            "delivered",
        ],
    })


def test_basic_aggregation(sample_df):
    """Non-canceled orders with valid dates should be grouped by month and summed."""
    result = calculate_monthly_revenue(sample_df)

    assert isinstance(result, pd.DataFrame)
    expected_months = ["2023-01", "2023-03"]
    expected_revenue = [300.0, 600.0]  # Jan: 100+200, Mar: 600 (Feb canceled, invalid date ignored)
    assert result["month"].tolist() == expected_months
    assert result["revenue"].tolist() == expected_revenue
    assert result["month"].dtype == pd.StringDtype()
    assert result["revenue"].dtype == float


def test_cancelled_status_case_insensitive_and_spaces():
    """All variations of canceled status (English/Portuguese) should be ignored."""
    df = pd.DataFrame({
        "order_date": ["2023-01-01"] * 6,
        "amount": [10, 20, 30, 40, 50, 60],
        "status": [
            "Cancelled",
            " CANCELADO ",
            "canceled",
            " cAncElleD ",
            "cancelled",
            "shipped",
        ],
    })
    result = calculate_monthly_revenue(df)
    # Only shipped (60) should remain
    assert len(result) == 1
    assert result.iloc[0]["revenue"] == 60.0


def test_invalid_amounts_treated_as_zero():
    """NaN, non-numeric, and missing amounts become 0."""
    df = pd.DataFrame({
        "order_date": ["2023-06-01", "2023-06-02", "2023-06-03"],
        "amount": [None, "abc", 50],
        "status": ["done", "done", "done"],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 1
    assert result.iloc[0]["revenue"] == 50.0  # NaN -> 0, abc -> 0, 50 -> 50


def test_invalid_dates_ignored():
    """Rows with unparseable dates are excluded."""
    df = pd.DataFrame({
        "order_date": ["2023-04-01", "not_a_date", None, "2023-04-05"],
        "amount": [100, 200, 300, 400],
        "status": ["ok", "ok", "ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 1
    assert result.iloc[0]["revenue"] == 500.0  # 100+400

def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])

    assert list(result.columns) == ["month", "revenue"]
    assert len(result) == 0
    assert result["month"].dtype == pd.StringDtype()
    assert result["revenue"].dtype == float

def test_original_dataframe_not_modified():
    original = pd.DataFrame({
        "order_date": ["2023-07-01"],
        "amount": [42],
        "status": ["active"],
    })
    copy = original.copy()
    _ = calculate_monthly_revenue(original)

def test_sort_order():
    """Result must be sorted by month ascending."""
    df = pd.DataFrame({
        "order_date": ["2023-10-01", "2023-01-01", "2023-12-01", "2023-02-01"],
        "amount": [1, 2, 3, 4],
        "status": ["done"] * 4,
    })
    result = calculate_monthly_revenue(df)
    expected_months = ["2023-01", "2023-02", "2023-10", "2023-12"]
    assert result["month"].tolist() == expected_months


def test_multiple_entries_same_month():
    df = pd.DataFrame({
        "order_date": ["2023-08-01"] * 3 + ["2023-08-15"],
        "amount": [100, 50, 25, 25],
        "status": ["good"] * 4,

    result = calculate_monthly_revenue(df)

    assert result.iloc[0]["revenue"] == 200.0

def test_mixed_invalid_and_canceled():
    """Complex scenario with invalid dates, canceled statuses, and valid orders."""
    df = pd.DataFrame({
        "order_date": [
            "2023-01-01",
            "2023-01-02",
            "2023-01-03",
            "bad_date",
            "2023-02-01",
            "2023-02-02",
            "2023-02-03",
        ],
        "amount": [10, 20, 30, 40, 50, 60, 70],
        "status": [
            "delivered",
            "cancelled",
            "cancelado",
            "delivered",
            "delivered",
            "invalid_date",  # actually date is valid but status column name misleading; just a test
            " Canceled ",
        ],
    })
    result = calculate_monthly_revenue(df)
    # Valid non-canceled: Jan: 10 (only first not canceled) Feb: 50 (delivered, not canceled)
    expected = pd.DataFrame({
        "month": pd.array(["2023-01", "2023-02"], dtype="string"),
        "revenue": [10.0, 50.0],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)
