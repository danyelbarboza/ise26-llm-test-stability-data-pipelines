import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue


def test_empty_dataframe():
    """An empty DataFrame returns an empty result with correct dtypes."""
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert isinstance(result, pd.DataFrame)
    assert list(result.columns) == ["month", "revenue"]
    assert len(result) == 0
    assert pd.api.types.is_string_dtype(result["month"])
    assert result["revenue"].dtype == "float64"


def test_no_valid_rows_invalid_dates():
    """All dates are invalid so the result is empty."""
    df = pd.DataFrame({
        "order_date": ["not_a_date", "2023-02-30", "", None],
        "amount": [100, 50, 30, 10],
        "status": ["ok", "ok", "ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0


def test_no_valid_rows_all_canceled():
    """All statuses are canceled so the result is empty."""
    df = pd.DataFrame({
        "order_date": pd.to_datetime(["2023-01-01", "2023-02-01", "2023-03-01"]),
        "amount": [10, 20, 30],
        "status": ["cancelled", "canceled", "cancelado"],
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0


def test_basic_aggregation():
    """Valid orders are aggregated by month; canceled ones are ignored."""
    df = pd.DataFrame({
        "order_date": pd.to_datetime(["2023-01-05", "2023-01-20", "2023-02-10", "2023-02-14"]),
        "amount": [100, 200, 300, 400],
        "status": ["ok", "ok", "canceled", "ok"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-01", "2023-02"], dtype="string"),
        "revenue": [300.0, 400.0],
    })
    # order is guaranteed to be sorted by month
    pd.testing.assert_frame_equal(result, expected)


def test_canceled_variations():
    """Different spellings and cases of canceled status are all detected."""
    dates = pd.to_datetime(["2023-01-01"] * 8)
    amounts = [10, 20, 30, 40, 50, 60, 70, 80]
    statuses = [
        "CANCELLED",
        " cancelled ",
        "Canceled",
        "cancelado",
        "CANCELADO",
        " Cancelado ",
        "ok",
        "OK",
    ]
    df = pd.DataFrame({"order_date": dates, "amount": amounts, "status": statuses})
    result = calculate_monthly_revenue(df)
    # Only last two (70+80) are not canceled → 150
    assert result.loc[0, "month"] == "2023-01"
    assert result.loc[0, "revenue"] == 150.0


def test_invalid_dates_ignored():
    """Rows with invalid dates are excluded from the aggregation."""
    df = pd.DataFrame({
        "order_date": [
            "2023-06-15", "not_a_date", "2023-06-20", pd.NaT, "2023-07-01"
        ],
        "amount": [10, 20, 30, 40, 50],
        "status": ["ok"] * 5,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": pd.Series(["2023-06", "2023-07"], dtype="string"),
        "revenue": [40.0, 50.0],
    })
    pd.testing.assert_frame_equal(result, expected)


def test_missing_amounts_treated_as_zero():
    """Invalid or missing amounts are treated as zero."""
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-01", "2023-01-01"],
        "amount": [100, "bad_value", pd.NA],
        "status": ["ok", "ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    # Only the 100 counts; others become 0.
    assert result.loc[0, "month"] == "2023-01"
    assert result.loc[0, "revenue"] == 100.0


def test_custom_column_names():
    """Works with non‑default column names."""
    df = pd.DataFrame({
        "dt": ["2023-04-01", "2023-04-02"],
        "val": [5, 7],
        "st": ["ok", "ok"],
    })
    result = calculate_monthly_revenue(df, date_col="dt", amount_col="val", status_col="st")
    assert result.loc[0, "month"] == "2023-04"
    assert result.loc[0, "revenue"] == 12.0


def test_sorting():
    """Months are returned in ascending string order."""
    df = pd.DataFrame({
        "order_date": ["2023-12-01", "2023-01-01", "2023-02-01"],
        "amount": [10, 20, 30],
        "status": ["ok", "ok", "ok"],
    })
    result = calculate_monthly_revenue(df)
    assert list(result["month"]) == ["2023-01", "2023-02", "2023-12"]


def test_dtype_checks():
    """Output columns have the promised stable string and float64 dtypes."""
    df = pd.DataFrame({
        "order_date": ["2023-01-01"],
        "amount": [100],
        "status": ["ok"],
    })
    result = calculate_monthly_revenue(df)
    assert pd.api.types.is_string_dtype(result["month"])
    assert isinstance(result["month"].dtype, pd.StringDtype)
    assert result["revenue"].dtype == "float64"


def test_none_and_nan_status_not_canceled():
    """None/NaN status values are treated as non‑canceled."""
    df = pd.DataFrame({
        "order_date": ["2023-05-01", "2023-05-02"],
        "amount": [15, 25],
        "status": [None, float("nan")],
    })
    result = calculate_monthly_revenue(df)
    assert result.loc[0, "revenue"] == 40.0


def test_missing_columns_raise_valueerror():
    """Passing a DataFrame without one of the required columns raises ValueError."""
    df = pd.DataFrame({"order_date": ["2023-01-01"], "amount": [10]})
    with pytest.raises(ValueError):
        calculate_monthly_revenue(df)
