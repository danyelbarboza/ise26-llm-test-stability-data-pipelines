from ise26.targets import calculate_conversion_rate

import numpy as np
import pandas as pd
import pytest


class TestCalculateConversionRate:
    def test_basic_aggregation_and_rate(self):
        df = pd.DataFrame({
            "channel": ["a", "a", "b", "b"],
            "visits": [10, 20, 30, 40],
            "conversions": [1, 2, 3, 4],
        })
        result = calculate_conversion_rate(df)
        assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
        assert result["channel"].tolist() == ["a", "b"]
        assert result["visits"].tolist() == [30, 70]
        assert result["conversions"].tolist() == [3, 7]
        assert result["conversion_rate"].tolist() == [3 / 30, 7 / 70]

    def test_missing_visits_treated_as_zero(self):
        df = pd.DataFrame({
            "channel": ["x", "x"],
            "visits": [10, None],
            "conversions": [5, 5],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].iloc[0] == 10.0
        assert result["conversions"].iloc[0] == 10.0
        assert result["conversion_rate"].iloc[0] == 1.0

    def test_non_numeric_visits_treated_as_zero(self):
        df = pd.DataFrame({
            "channel": ["x"],
            "visits": ["abc"],
            "conversions": [5],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].iloc[0] == 0.0
        assert result["conversions"].iloc[0] == 5.0
        assert result["conversion_rate"].iloc[0] == 0.0

    def test_missing_conversions_treated_as_zero(self):
        df = pd.DataFrame({
            "channel": ["x", "x"],
            "visits": [10, 10],
            "conversions": [np.nan, 2],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].iloc[0] == 20.0
        assert result["conversions"].iloc[0] == 2.0
        assert result["conversion_rate"].iloc[0] == 2 / 20

    def test_blank_channels_normalized_to_unknown(self):
        df = pd.DataFrame({
            "channel": ["a", "", "  ", "b"],
            "visits": [1, 2, 3, 4],
            "conversions": [0, 1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        channels = result["channel"].tolist()
        assert "unknown" in channels
        # unknown should appear only once, summing blank/whitespace
        assert channels.count("unknown") == 1
        # unknown row visits sum of 2+3 =5
        unknown_row = result[result["channel"] == "unknown"].iloc[0]
        assert unknown_row["visits"] == 5.0
        assert unknown_row["conversions"] == 3.0

    def test_missing_channel_normalized_to_unknown(self):
        df = pd.DataFrame({
            "channel": [np.nan, "a"],
            "visits": [10, 20],
            "conversions": [1, 2],
        })
        result = calculate_conversion_rate(df)
        assert "unknown" in result["channel"].values
        unknown = result[result["channel"] == "unknown"]
        assert len(unknown) == 1
        assert unknown["visits"].iloc[0] == 10.0
        assert unknown["conversions"].iloc[0] == 1.0

    def test_output_sorted_by_channel(self):
        df = pd.DataFrame({
            "channel": ["c", "a", "b"],
            "visits": [1, 2, 3],
            "conversions": [0, 1, 2],
        })
        result = calculate_conversion_rate(df)
        assert result["channel"].tolist() == ["a", "b", "c"]

    def test_unknown_sorted_among_channels(self):
        df = pd.DataFrame({
            "channel": ["z", "a", None],
            "visits": [10, 20, 30],
            "conversions": [1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        channels = result["channel"].tolist()
        # unknown should appear in alphabetical position
        assert channels == sorted(channels)
        assert "unknown" in channels

    def test_zero_visits_yield_zero_conversion_rate(self):
        df = pd.DataFrame({
            "channel": ["a", "a"],
            "visits": [0, 0],
            "conversions": [5, 5],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].iloc[0] == 0.0
        assert result["conversions"].iloc[0] == 10.0
        assert result["conversion_rate"].iloc[0] == 0.0

    def test_negative_visits_treated_as_negative_and_rate_zero(self):
        # Negative visits are not coerced to NaN, they stay negative, but rate becomes 0 because visits <=0
        df = pd.DataFrame({
            "channel": ["a"],
            "visits": [-5],
            "conversions": [10],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].iloc[0] == -5.0
        assert result["conversion_rate"].iloc[0] == 0.0

    def test_empty_dataframe_returns_empty_with_columns(self):
        df = pd.DataFrame(columns=["channel", "visits", "conversions"])
        result = calculate_conversion_rate(df)
        assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
        assert len(result) == 0

    def test_all_channels_missing(self):
        df = pd.DataFrame({
            "channel": [np.nan, None, ""],
            "visits": [1, 2, 3],
            "conversions": [0, 0, 0],
        })
        result = calculate_conversion_rate(df)
        assert result["channel"].tolist() == ["unknown"]
        assert result["visits"].iloc[0] == 6.0

    def test_dtype_of_output_columns(self):
        df = pd.DataFrame({
            "channel": ["a"],
            "visits": [10],
            "conversions": [2],
        })
        result = calculate_conversion_rate(df)
        assert pd.api.types.is_float_dtype(result["visits"])
        assert pd.api.types.is_float_dtype(result["conversions"])
        assert pd.api.types.is_float_dtype(result["conversion_rate"])
        # channel should be string/object
        assert result["channel"].dtype == object

    def test_preserves_original_dataframe(self):
        df = pd.DataFrame({
            "channel": ["a", "b"],
            "visits": [10, 20],
            "conversions": [1, 2],
        })
        df_original = df.copy()
        calculate_conversion_rate(df)
        pd.testing.assert_frame_equal(df, df_original)

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "source": ["x", "x"],
            "clicks": [100, 200],
            "orders": [10, 20],
        })
        result = calculate_conversion_rate(df, group_col="source", visits_col="clicks", conversions_col="orders")
        assert list(result.columns) == ["source", "visits", "conversions", "conversion_rate"]
        assert result["source"].tolist() == ["x"]
        assert result["visits"].iloc[0] == 300.0
        assert result["conversions"].iloc[0] == 30.0

    def test_missing_conversions_and_visits_both_zero(self):
        df = pd.DataFrame({
            "channel": ["a"],
            "visits": [np.nan],
            "conversions": [None],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].iloc[0] == 0.0
        assert result["conversions"].iloc[0] == 0.0
        assert result["conversion_rate"].iloc[0] == 0.0
