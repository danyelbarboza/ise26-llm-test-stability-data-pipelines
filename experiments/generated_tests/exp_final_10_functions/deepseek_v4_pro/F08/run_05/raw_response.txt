from ise26.targets import calculate_conversion_rate

import numpy as np
import pandas as pd
import pytest


class TestCalculateConversionRate:
    def test_basic_aggregation_multiple_channels(self):
        df = pd.DataFrame({
            "channel": ["a", "b", "a", "b", "a"],
            "visits": [10, 20, 30, 40, 50],
            "conversions": [1, 2, 3, 4, 5],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a", "b"],
            "visits": [90.0, 60.0],
            "conversions": [9.0, 6.0],
            "conversion_rate": [9.0 / 90.0, 6.0 / 60.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_missing_numeric_values_treated_as_zero(self):
        df = pd.DataFrame({
            "channel": ["x", "x", "y", "y"],
            "visits": [10, np.nan, 20, None],
            "conversions": [np.nan, 3, None, 5],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["x", "y"],
            "visits": [10.0, 20.0],
            "conversions": [3.0, 5.0],
            "conversion_rate": [3.0 / 10.0, 5.0 / 20.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_invalid_numeric_strings_become_zero(self):
        df = pd.DataFrame({
            "channel": ["a", "a"],
            "visits": ["10", "invalid"],
            "conversions": ["2", "not_a_number"],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a"],
            "visits": [10.0],
            "conversions": [2.0],
            "conversion_rate": [2.0 / 10.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_missing_channels_normalized_to_unknown(self):
        df = pd.DataFrame({
            "channel": ["a", None, "b", np.nan],
            "visits": [1, 2, 3, 4],
            "conversions": [0, 1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a", "b", "unknown"],
            "visits": [1.0, 3.0, 6.0],
            "conversions": [0.0, 2.0, 4.0],
            "conversion_rate": [0.0, 2.0 / 3.0, 4.0 / 6.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_blank_channels_normalized_to_unknown(self):
        df = pd.DataFrame({
            "channel": ["", "  ", "a", "\t", "b"],
            "visits": [1, 2, 3, 4, 5],
            "conversions": [0, 0, 1, 0, 2],
        })
        result = calculate_conversion_rate(df)
        # expect blank/whitespace-only channels become "unknown"
        expected_channels = ["a", "b", "unknown"]
        assert list(result["channel"]) == expected_channels
        unknown_visits = 1 + 2 + 4  # rows 0,1,3
        unknown_conversions = 0 + 0 + 0
        expected = pd.DataFrame({
            "channel": expected_channels,
            "visits": [3.0, 5.0, float(unknown_visits)],
            "conversions": [1.0, 2.0, float(unknown_conversions)],
            "conversion_rate": [1.0 / 3.0, 2.0 / 5.0, 0.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_sorting_alphabetically_including_unknown(self):
        df = pd.DataFrame({
            "channel": ["z", "unknown", "a", "unknown"],
            "visits": [1, 2, 3, 4],
            "conversions": [0, 1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        assert list(result["channel"]) == ["a", "unknown", "z"]

    def test_zero_visits_gives_zero_conversion_rate(self):
        df = pd.DataFrame({
            "channel": ["x", "x", "y"],
            "visits": [0, 0, 5],
            "conversions": [0, 1, 3],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["x", "y"],
            "visits": [0.0, 5.0],
            "conversions": [1.0, 3.0],
            "conversion_rate": [0.0, 3.0 / 5.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_negative_visits_gives_zero_conversion_rate(self):
        df = pd.DataFrame({
            "channel": ["a", "a"],
            "visits": [-5, 10],
            "conversions": [2, 3],
        })
        result = calculate_conversion_rate(df)
        # sum of visits = 5, conversions = 5
        expected = pd.DataFrame({
            "channel": ["a"],
            "visits": [5.0],
            "conversions": [5.0],
            "conversion_rate": [5.0 / 5.0],  # visits > 0 now, so rate is 1.0
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)
        # test where total visits are negative
        df2 = pd.DataFrame({
            "channel": ["b"],
            "visits": [-10],
            "conversions": [1],
        })
        res2 = calculate_conversion_rate(df2)
        assert res2.loc[0, "conversion_rate"] == 0.0

    def test_output_column_order_and_dtypes(self):
        df = pd.DataFrame({
            "channel": ["a"],
            "visits": [10],
            "conversions": [2],
        })
        result = calculate_conversion_rate(df)
        assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
        assert result["visits"].dtype == float
        assert result["conversions"].dtype == float
        assert result["conversion_rate"].dtype == float

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "src": ["a", "b"],
            "v": [1, 2],
            "c": [0, 1],
        })
        result = calculate_conversion_rate(df, group_col="src", visits_col="v", conversions_col="c")
        assert list(result.columns) == ["src", "visits", "conversions", "conversion_rate"]
        expected = pd.DataFrame({
            "src": ["a", "b"],
            "visits": [1.0, 2.0],
            "conversions": [0.0, 1.0],
            "conversion_rate": [0.0, 0.5],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_empty_dataframe_returns_empty(self):
        df = pd.DataFrame({"channel": [], "visits": [], "conversions": []})
        result = calculate_conversion_rate(df)
        assert result.empty
        assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]

    def test_all_conversions_greater_than_visits(self):
        # conversion_rate can exceed 100% in this function
        df = pd.DataFrame({
            "channel": ["a", "a"],
            "visits": [10, 20],
            "conversions": [15, 25],
        })
        result = calculate_conversion_rate(df)
        assert result.loc[0, "conversion_rate"] == 40 / 30  # > 1.0

    def test_group_col_containing_only_nulls(self):
        df = pd.DataFrame({
            "channel": [None, np.nan, None],
            "visits": [10, 20, 30],
            "conversions": [1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["unknown"],
            "visits": [60.0],
            "conversions": [6.0],
            "conversion_rate": [6.0 / 60.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)

    def test_conversions_column_has_all_zeros(self):
        df = pd.DataFrame({
            "channel": ["a", "b"],
            "visits": [5, 10],
            "conversions": [0, 0],
        })
        result = calculate_conversion_rate(df)
        assert (result["conversion_rate"] == 0.0).all()
        assert (result["conversions"] == 0.0).all()

    def test_visits_and_conversions_both_missing_in_some_rows(self):
        df = pd.DataFrame({
            "channel": ["a", "a", "a"],
            "visits": [np.nan, None, 10],
            "conversions": [None, np.nan, 5],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a"],
            "visits": [10.0],
            "conversions": [5.0],
            "conversion_rate": [5.0 / 10.0],
        })
        pd.testing.assert_frame_equal(result, expected, check_exact=False)
