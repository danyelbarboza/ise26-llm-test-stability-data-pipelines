from ise26.targets import calculate_conversion_rate

import pandas as pd
import numpy as np
import pytest


def test_basic_aggregation_and_conversion_rate():
    df = pd.DataFrame({
        "channel": ["A", "B", "A", "B"],
        "visits": [100, 200, 50, 100],
        "conversions": [10, 20, 5, 30],
    })
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
    assert result["channel"].tolist() == ["A", "B"]
    assert result["visits"].tolist() == [150.0, 300.0]
    assert result["conversions"].tolist() == [15.0, 50.0]
    expected_rates = [15.0 / 150.0, 50.0 / 300.0]
    assert result["conversion_rate"].tolist() == pytest.approx(expected_rates)


def test_missing_numeric_values_treated_as_zero():
    df = pd.DataFrame({
        "channel": ["A", "A", "B", "B"],
        "visits": [100, np.nan, None, "abc"],
        "conversions": [10, 5, None, 20],
    })
    result = calculate_conversion_rate(df)
    # All invalid/missing become 0
    assert result.loc[result["channel"] == "A", "visits"].values[0] == 100.0
    assert result.loc[result["channel"] == "A", "conversions"].values[0] == 15.0
    assert result.loc[result["channel"] == "B", "visits"].values[0] == 0.0
    assert result.loc[result["channel"] == "B", "conversions"].values[0] == 20.0
    # For B visits=0, conversion_rate should be 0.0
    assert result.loc[result["channel"] == "B", "conversion_rate"].values[0] == 0.0


def test_missing_channel_normalized_to_unknown():
    df = pd.DataFrame({
        "channel": [None, np.nan, "A", "B"],
        "visits": [10, 20, 30, 40],
        "conversions": [1, 2, 3, 4],
    })
    result = calculate_conversion_rate(df)
    assert "unknown" in result["channel"].values
    unknown_row = result[result["channel"] == "unknown"]
    assert unknown_row["visits"].values[0] == 30.0
    assert unknown_row["conversions"].values[0] == 3.0
    assert unknown_row["conversion_rate"].values[0] == 3.0 / 30.0


def test_blanks_normalized_to_unknown():
    df = pd.DataFrame({
        "channel": ["", "   ", "A"],
        "visits": [5, 10, 15],
        "conversions": [1, 2, 3],
    })
    result = calculate_conversion_rate(df)
    unknown_rows = result[result["channel"] == "unknown"]
    assert len(unknown_rows) == 1
    assert unknown_rows["visits"].values[0] == 15.0
    assert unknown_rows["conversions"].values[0] == 3.0


def test_sort_by_channel():
    # Unsorted input, check alphabetical order
    df = pd.DataFrame({
        "channel": ["B", "a", "A", "c"],
        "visits": [10, 20, 30, 40],
        "conversions": [1, 2, 3, 4],
    })
    result = calculate_conversion_rate(df)
    # alphabetical: 'A' < 'B' < 'a' < 'c'
    assert result["channel"].tolist() == ["A", "B", "a", "c"]


def test_visits_zero_or_negative_yield_zero_rate():
    df = pd.DataFrame({
        "channel": ["A", "B", "C"],
        "visits": [0, -5, 0],
        "conversions": [10, 20, 0],
    })
    result = calculate_conversion_rate(df)
    # all rates should be 0.0
    assert (result["conversion_rate"] == 0.0).all()


def test_empty_dataframe():
    df = pd.DataFrame(columns=["channel", "visits", "conversions"])
    result = calculate_conversion_rate(df)
    assert len(result) == 0
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]


def test_custom_group_col():
    df = pd.DataFrame({
        "source": ["X", "Y", "X"],
        "visits": [100, 200, 50],
        "convs": [10, 20, 5],
    })
    result = calculate_conversion_rate(df, group_col="source", visits_col="visits", conversions_col="convs")
    assert "source" in result.columns
    assert result["source"].tolist() == ["X", "Y"]
    assert "channel" not in result.columns


def test_all_invalid_numeric_values_sum_zero():
    df = pd.DataFrame({
        "channel": ["A", "B"],
        "visits": ["bad", None],
        "conversions": [np.nan, "x"],
    })
    result = calculate_conversion_rate(df)
    assert result["visits"].tolist() == [0.0, 0.0]
    assert result["conversions"].tolist() == [0.0, 0.0]
    assert result["conversion_rate"].tolist() == [0.0, 0.0]


def test_output_dtypes_are_float():
    df = pd.DataFrame({
        "channel": ["A", "A"],
        "visits": [100, 50],
        "conversions": [10, 5],
    })
    result = calculate_conversion_rate(df)
    assert result["visits"].dtype == float
    assert result["conversions"].dtype == float
    assert result["conversion_rate"].dtype == float


def test_ignore_extra_columns():
    df = pd.DataFrame({
        "channel": ["A", "B"],
        "visits": [10, 20],
        "conversions": [1, 2],
        "extra": [99, 100],
    })
    result = calculate_conversion_rate(df)
    assert "extra" not in result.columns
    assert set(result.columns) == {"channel", "visits", "conversions", "conversion_rate"}


def test_negative_visits_treated_as_zero_for_rate():
    df = pd.DataFrame({
        "channel": ["A"],
        "visits": [-100],
        "conversions": [10],
    })
    result = calculate_conversion_rate(df)
    assert result.loc[0, "conversion_rate"] == 0.0
    # visits sum still negative? The code sums visits after coercing to numeric, negative values remain.
    assert result.loc[0, "visits"] == -100.0  # sum retains negative
    # conversion_rate is 0 because visits <= 0
    assert result.loc[0, "conversion_rate"] == 0.0


def test_conversion_rate_precision():
    df = pd.DataFrame({
        "channel": ["A"],
        "visits": [3],
        "conversions": [1],
    })
    result = calculate_conversion_rate(df)
    assert result.loc[0, "conversion_rate"] == pytest.approx(1.0 / 3.0)
