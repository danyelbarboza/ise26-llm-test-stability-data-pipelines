from ise26.targets import calculate_conversion_rate

import pandas as pd
import pytest


def test_basic_aggregation_sorted_output():
    df = pd.DataFrame({
        "channel": ["social", "email", "social", "email"],
        "visits": [100, 200, 50, 0],
        "conversions": [10, 25, 5, 0],
    })
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
    # alphabetical order: email, social
    assert result["channel"].tolist() == ["email", "social"]
    # email totals: visits=200, conversions=25 -> rate=0.125
    assert result.loc[result["channel"] == "email", "visits"].values[0] == 200.0
    assert result.loc[result["channel"] == "email", "conversions"].values[0] == 25.0
    assert result.loc[result["channel"] == "email", "conversion_rate"].values[0] == 0.125
    # social totals: visits=150, conversions=15 -> rate=0.1
    assert result.loc[result["channel"] == "social", "visits"].values[0] == 150.0
    assert result.loc[result["channel"] == "social", "conversions"].values[0] == 15.0
    assert result.loc[result["channel"] == "social", "conversion_rate"].values[0] == 0.1


def test_missing_values_treated_as_zero():
    df = pd.DataFrame({
        "channel": ["a", "a", "b"],
        "visits": [100, None, pd.NA],
        "conversions": [None, 10, 5],
    })
    result = calculate_conversion_rate(df)
    # a visits: 100+0=100, conversions: 0+10=10
    # b visits: 0, conversions: 5
    # rates: a=0.1, b=0.0
    assert result.loc[result["channel"] == "a", "visits"].values[0] == 100.0
    assert result.loc[result["channel"] == "a", "conversions"].values[0] == 10.0
    assert result.loc[result["channel"] == "a", "conversion_rate"].values[0] == 0.1
    assert result.loc[result["channel"] == "b", "visits"].values[0] == 0.0
    assert result.loc[result["channel"] == "b", "conversions"].values[0] == 5.0
    assert result.loc[result["channel"] == "b", "conversion_rate"].values[0] == 0.0


def test_non_numeric_strings_become_zero():
    df = pd.DataFrame({
        "channel": ["x", "x"],
        "visits": [100, "abc"],
        "conversions": ["def", 10],
    })
    result = calculate_conversion_rate(df)
    # two rows for x: visits=100+0=100, conversions=0+10=10, rate=0.1
    assert result["visits"].iloc[0] == 100.0
    assert result["conversions"].iloc[0] == 10.0
    assert result["conversion_rate"].iloc[0] == 0.1


def test_blank_channel_normalized_to_unknown():
    df = pd.DataFrame({
        "channel": ["", None, pd.NA, "social"],
        "visits": [10, 20, 30, 100],
        "conversions": [1, 2, 3, 5],
    })
    result = calculate_conversion_rate(df)
    # channels sorted: social, unknown
    assert result["channel"].tolist() == ["social", "unknown"]
    # social: visits=100, conv=5, rate=0.05
    # unknown: visits=60, conv=6, rate=0.1
    assert result.loc[result["channel"] == "social", "visits"].values[0] == 100.0
    assert result.loc[result["channel"] == "social", "conversion_rate"].values[0] == 0.05
    assert result.loc[result["channel"] == "unknown", "visits"].values[0] == 60.0
    assert result.loc[result["channel"] == "unknown", "conversion_rate"].values[0] == 0.1


def test_only_unknown_channel():
    df = pd.DataFrame({
        "channel": [None, ""],
        "visits": [100, 200],
        "conversions": [10, 20],
    })
    result = calculate_conversion_rate(df)
    assert len(result) == 1
    assert result["channel"].iloc[0] == "unknown"
    assert result["visits"].iloc[0] == 300.0
    assert result["conversions"].iloc[0] == 30.0
    assert result["conversion_rate"].iloc[0] == 0.1


def test_zero_and_negative_visits_yield_zero_conversion_rate():
    df = pd.DataFrame({
        "channel": ["a", "a", "b"],
        "visits": [0, 0, -10],
        "conversions": [5, 10, 5],
    })
    result = calculate_conversion_rate(df)
    # a visits = 0 -> rate 0
    # b visits = -10 -> <=0 -> rate 0
    assert result.loc[result["channel"] == "a", "visits"].values[0] == 0.0
    assert result.loc[result["channel"] == "a", "conversions"].values[0] == 15.0
    assert result.loc[result["channel"] == "a", "conversion_rate"].values[0] == 0.0
    assert result.loc[result["channel"] == "b", "visits"].values[0] == -10.0
    assert result.loc[result["channel"] == "b", "conversion_rate"].values[0] == 0.0


def test_custom_column_names():
    df = pd.DataFrame({
        "src": ["google", "google", "facebook"],
        "clicks": [100, 50, 200],
        "purchases": [10, 5, 20],
    })
    result = calculate_conversion_rate(df, group_col="src", visits_col="clicks", conversions_col="purchases")
    assert list(result.columns) == ["src", "visits", "conversions", "conversion_rate"]
    # facebook: visits=200, conv=20, rate=0.1
    # google: visits=150, conv=15, rate=0.1
    assert result["src"].tolist() == ["facebook", "google"]
    assert result.loc[result["src"] == "facebook", "visits"].values[0] == 200.0
    assert result.loc[result["src"] == "facebook", "conversion_rate"].values[0] == 0.1
    assert result.loc[result["src"] == "google", "visits"].values[0] == 150.0
    assert result.loc[result["src"] == "google", "conversion_rate"].values[0] == 0.1


def test_empty_dataframe():
    df = pd.DataFrame({"channel": [], "visits": [], "conversions": []})
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
    assert len(result) == 0


def test_dtype_of_returned_columns():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [1, 2],
        "conversions": [3, 4],
    })
    result = calculate_conversion_rate(df)
    assert result["visits"].dtype == float
    assert result["conversions"].dtype == float
    assert result["conversion_rate"].dtype == float


def test_sorting_stable_with_many_channels():
    df = pd.DataFrame({
        "channel": ["c", "a", "b", "a"],
        "visits": [10, 20, 30, 40],
        "conversions": [1, 2, 3, 4],
    })
    result = calculate_conversion_rate(df)
    assert result["channel"].tolist() == ["a", "b", "c"]


def test_null_group_column_name_is_preserved():
    df = pd.DataFrame({
        "channel": ["x", "x"],
        "visits": [1, 1],
        "conversions": [0, 0],
    })
    result = calculate_conversion_rate(df, group_col="channel")
    assert "channel" in result.columns
    assert "unknown" not in result.columns
