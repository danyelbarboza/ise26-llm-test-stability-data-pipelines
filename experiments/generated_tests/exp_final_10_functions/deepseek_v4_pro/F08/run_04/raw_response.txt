from ise26.targets import calculate_conversion_rate

import numpy as np
import pandas as pd
import pytest


class TestCalculateConversionRate:
    def test_basic_aggregation(self):
        df = pd.DataFrame({
            "channel": ["email", "email", "social", "social"],
            "visits": [100, 200, 150, 50],
            "conversions": [10, 30, 20, 5],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["email", "social"],
            "visits": [300.0, 200.0],
            "conversions": [40.0, 25.0],
            "conversion_rate": [40.0 / 300.0, 25.0 / 200.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_sorting_by_channel(self):
        df = pd.DataFrame({
            "channel": ["google", "facebook", "youtube"],
            "visits": [100, 200, 150],
            "conversions": [10, 20, 30],
        })
        result = calculate_conversion_rate(df)
        expected_channels = ["facebook", "google", "youtube"]
        assert list(result["channel"]) == expected_channels

    def test_missing_numeric_treated_as_zero(self):
        df = pd.DataFrame({
            "channel": ["a", "a", "b"],
            "visits": [10, np.nan, "invalid"],
            "conversions": [np.nan, "x", 5],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["a", "b"],
            "visits": [10.0, 0.0],
            "conversions": [0.0, 5.0],
            "conversion_rate": [0.0, 0.0],  # b: 5/0 -> 0; a: 0/10 -> 0
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_channel_normalized_to_unknown(self):
        df = pd.DataFrame({
            "channel": [np.nan, None, "", "social"],
            "visits": [1, 2, 3, 4],
            "conversions": [0, 1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        expected_channels = ["social", "unknown"]
        assert list(result["channel"]) == expected_channels
        unknown_row = result.loc[result["channel"] == "unknown"].iloc[0]
        assert unknown_row["visits"] == 6.0  # 1+2+3
        assert unknown_row["conversions"] == 3.0  # 0+1+2

    def test_zero_visits_rate_zero(self):
        df = pd.DataFrame({
            "channel": ["a", "a"],
            "visits": [0, 0],
            "conversions": [10, 5],
        })
        result = calculate_conversion_rate(df)
        assert result.loc[0, "conversion_rate"] == 0.0

    def test_negative_visits_rate_zero(self):
        df = pd.DataFrame({
            "channel": ["x"],
            "visits": [-5],
            "conversions": [10],
        })
        result = calculate_conversion_rate(df)
        assert result.loc[0, "conversion_rate"] == 0.0

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["channel", "visits", "conversions"])
        result = calculate_conversion_rate(df)
        expected_columns = ["channel", "visits", "conversions", "conversion_rate"]
        assert list(result.columns) == expected_columns
        assert len(result) == 0

    def test_all_nulls(self):
        df = pd.DataFrame({
            "channel": [np.nan, None],
            "visits": [np.nan, np.nan],
            "conversions": [np.nan, np.nan],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["unknown"],
            "visits": [0.0],
            "conversions": [0.0],
            "conversion_rate": [0.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_column_raises(self):
        df = pd.DataFrame({"channel": ["a"], "visits": [1]})
        with pytest.raises(ValueError):
            calculate_conversion_rate(df)

    def test_output_dtypes(self):
        df = pd.DataFrame({
            "channel": ["a", "b"],
            "visits": [10, 20],
            "conversions": [1, 2],
        })
        result = calculate_conversion_rate(df)
        assert result["visits"].dtype == float
        assert result["conversions"].dtype == float
        assert result["conversion_rate"].dtype == float
        assert result["channel"].dtype == object

    def test_custom_group_col(self):
        df = pd.DataFrame({
            "source": ["web", "mobile", "web"],
            "visits": [100, 200, 300],
            "conversions": [10, 20, 30],
        })
        result = calculate_conversion_rate(df, group_col="source")
        assert "source" in result.columns
        assert "channel" not in result.columns
        web_row = result.loc[result["source"] == "web"].iloc[0]
        assert web_row["visits"] == 400.0
        assert web_row["conversions"] == 40.0
