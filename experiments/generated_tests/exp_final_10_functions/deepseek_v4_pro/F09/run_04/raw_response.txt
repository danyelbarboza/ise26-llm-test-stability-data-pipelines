def test_cap_outliers_iqr_basic():
    df = pd.DataFrame({"amount": [10, 12, 11, 13, 100]})

def test_cap_outliers_iqr_only_low_outliers():
    df = pd.DataFrame({"amount": [-200, -150, 5, 6, 7]})
    result = cap_outliers_iqr(df)
    # sorted [-200,-150,5,6,7] -> Q1=-150? Actually Q1 = -200 + 0.75*50? Let's trust pandas.
    # Expect that -200 and -150 are capped to lower bound.
    lower = float(pd.Series([-200, -150, 5, 6, 7]).quantile(0.25)) - 1.5 * (
            float(pd.Series([-200, -150, 5, 6, 7]).quantile(0.75)) -
            float(pd.Series([-200, -150, 5, 6, 7]).quantile(0.25))
    )
    # Actually compute with actual data:
    s = pd.Series([-200, -150, 5, 6, 7])
    q1 = s.quantile(0.25, interpolation="linear")
    q3 = s.quantile(0.75, interpolation="linear")
    iqr = q3 - q1
    lower_bound = q1 - 1.5 * iqr
    upper_bound = q3 + 1.5 * iqr
    expected = pd.array([
        max(-200, lower_bound), max(-150, lower_bound), 5.0, 6.0, 7.0
    ], dtype="Float64")
    assert result["amount_capped"].equals(pd.Series(expected, name="amount_capped"))

    q3 = s.quantile(0.75, interpolation="linear")

    upper = q3 + 1.5 * iqr

def test_cap_outliers_iqr_with_nans():
    df = pd.DataFrame({"amount": [10, float("nan"), 20, None, 100]})
    result = cap_outliers_iqr(df, value_col="amount", output_col="capped")
    # NaN and None should become pd.NA (which is Float64 NA). After cap, NAs remain.
    assert result["capped"].isna().sum() == 2
    assert result["capped"][0] == 10.0
    assert result["capped"][2] == 20.0
    # 100 should be capped
    valid = pd.to_numeric(df["amount"], errors="coerce").dropna()
    q1 = valid.quantile(0.25)
    q3 = valid.quantile(0.75)
    iqr = q3 - q1
    upper = q3 + 1.5 * iqr
    assert result["capped"][4] == upper
