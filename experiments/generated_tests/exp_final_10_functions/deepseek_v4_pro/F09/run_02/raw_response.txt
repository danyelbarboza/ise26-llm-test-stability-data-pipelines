import pytest
import pandas as pd
import numpy as np
from ise26.targets import cap_outliers_iqr


def test_cap_outliers_normal():
    df = pd.DataFrame({"amount": [10, 12, 13, 15, 100]})
    result = cap_outliers_iqr(df, value_col="amount", output_col="amount_capped")

    q1 = df["amount"].quantile(0.25, interpolation="linear")
    q3 = df["amount"].quantile(0.75, interpolation="linear")
    iqr = q3 - q1
    lower = q1 - 1.5 * iqr
    upper = q3 + 1.5 * iqr

    expected_capped = [10, 12, 13, 15, min(max(100, lower), upper)]
    assert result["amount_capped"].tolist() == pytest.approx(expected_capped)
    assert result["amount"].equals(df["amount"])


def test_output_dtype_is_nullable_float():
    df = pd.DataFrame({"val": [1.0, 2.0, 3.0]})
    result = cap_outliers_iqr(df, value_col="val", output_col="capped")
    assert result["capped"].dtype == pd.Float64Dtype()


def test_missing_values_stay_missing():
    df = pd.DataFrame({"col": [1.0, np.nan, None, 2.0]})
    result = cap_outliers_iqr(df, value_col="col", output_col="col_capped")
    assert pd.isna(result.loc[1, "col_capped"])
    assert pd.isna(result.loc[2, "col_capped"])
    assert not pd.isna(result.loc[0, "col_capped"])
    assert not pd.isna(result.loc[3, "col_capped"])


def test_non_numeric_becomes_missing():
    df = pd.DataFrame({"raw": ["x", 1, 2, np.nan, "abc"]})
    result = cap_outliers_iqr(df, value_col="raw", output_col="capped")
    assert pd.isna(result.loc[0, "capped"])
    assert pd.isna(result.loc[4, "capped"])
    assert not pd.isna(result.loc[1, "capped"])
    assert not pd.isna(result.loc[2, "capped"])


def test_fewer_than_two_valid_values_returns_unchanged():
    df = pd.DataFrame({"values": [np.nan, np.nan, 5.0]})
    result = cap_outliers_iqr(df, value_col="values", output_col="capped")
    assert result["capped"].tolist() == [np.nan, np.nan, 5.0]

    df_only_nan = pd.DataFrame({"values": [np.nan, None]})
    result_nan = cap_outliers_iqr(df_only_nan, value_col="values", output_col="capped")
    assert result_nan["capped"].isna().all()


def test_exactly_two_valid_values():
    df = pd.DataFrame({"x": [1.0, 100.0]})
    result = cap_outliers_iqr(df, value_col="x", output_col="x_capped")
    assert result["x_capped"].tolist() == pytest.approx([1.0, 100.0])


def test_no_outliers():
    df = pd.DataFrame({"measure": [5.0, 6.0, 7.0, 8.0, 9.0]})
    result = cap_outliers_iqr(df, value_col="measure", output_col="measure_capped")
    assert result["measure_capped"].tolist() == pytest.approx(df["measure"].tolist())


def test_capping_large_outliers():
    df = pd.DataFrame({"value": [50, 52, 49, 1000, -500]})
    result = cap_outliers_iqr(df, value_col="value", output_col="value_capped")

    q1 = df["value"].quantile(0.25, interpolation="linear")
    q3 = df["value"].quantile(0.75, interpolation="linear")
    iqr = q3 - q1
    lower = q1 - 1.5 * iqr
    upper = q3 + 1.5 * iqr

    assert result.loc[3, "value_capped"] == upper
    assert result.loc[4, "value_capped"] == lower
    assert result.loc[0, "value_capped"] == 50.0
    assert result.loc[1, "value_capped"] == 52.0
    assert result.loc[2, "value_capped"] == 49.0


def test_original_dataframe_unchanged():
    df = pd.DataFrame({"amount": [10, 20, 30, 100]})
    original = df.copy()
    _ = cap_outliers_iqr(df, value_col="amount", output_col="amount_capped")
    assert df.equals(original)


def test_output_column_name_default():
    df = pd.DataFrame({"amount": [1, 2, 3]})
    result = cap_outliers_iqr(df, value_col="amount")
    assert "amount_capped" in result.columns
    assert "amount" in result.columns


def test_custom_output_column_name():
    df = pd.DataFrame({"val": [1, 2, 3]})
    result = cap_outliers_iqr(df, value_col="val", output_col="outliers_capped")
    assert "outliers_capped" in result.columns


def test_index_preservation():
    df = pd.DataFrame({"data": [1, 2, 3, 100, 200]}, index=[10, 20, 30, 40, 50])
    result = cap_outliers_iqr(df, value_col="data", output_col="capped")
    assert result.index.tolist() == [10, 20, 30, 40, 50]
    assert result["capped"].index.tolist() == [10, 20, 30, 40, 50]


def test_float_precision():
    df = pd.DataFrame({"f": [1.1, 1.2, 1.3, 10.0]})
    result = cap_outliers_iqr(df, value_col="f", output_col="f_capped")
    q1 = df["f"].quantile(0.25, interpolation="linear")
    q3 = df["f"].quantile(0.75, interpolation="linear")
    iqr = q3 - q1
    upper = q3 + 1.5 * iqr
    assert result.loc[3, "f_capped"] == pytest.approx(min(10.0, upper))


def test_all_invalid_missing_values():
    df = pd.DataFrame({"x": [np.nan, "bad", None]})
    result = cap_outliers_iqr(df, value_col="x", output_col="y")
    assert result["y"].isna().all()
    assert result["y"].dtype == pd.Float64Dtype()
