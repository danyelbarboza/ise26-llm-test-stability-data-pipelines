from ise26.targets import validate_schema

import pandas as pd
import pytest


def test_all_columns_match():
    df = pd.DataFrame({"a": [1, 2], "b": [1.0, 2.0]})
    schema = {"a": "int", "b": "float"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_number_matches_int_and_float():
    df = pd.DataFrame({"x": [1, 2], "y": [1.5, 2.5]})
    schema = {"x": "number", "y": "number"}
    result = validate_schema(df, schema)
    assert result["valid"] is True


def test_type_mismatch():
    df = pd.DataFrame({"col": [1, 2]})
    schema = {"col": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "col"
    assert err["expected"] == "string"
    assert err["actual"] == "int"
