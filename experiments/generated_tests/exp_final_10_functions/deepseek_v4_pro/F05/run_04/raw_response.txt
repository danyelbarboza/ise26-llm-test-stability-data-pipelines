import pytest
import pandas as pd
from ise26.targets import validate_schema


def test_all_columns_present_and_types_match():
    df = pd.DataFrame(
        {
            "id": [1, 2],
            "name": ["Alice", "Bob"],
            "score": [95.5, 87.0],
            "active": [True, False],
            "created": pd.to_datetime(["2023-01-01", "2023-06-15"]),
        }
    )
    schema = {
        "id": "int",
        "name": "string",
        "score": "float",
        "active": "bool",
        "created": "datetime",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_columns_reported():
    df = pd.DataFrame({"a": [1]})
    schema = {"a": "int", "b": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["b"]
    assert result["type_errors"] == []


def test_type_mismatch_reported():
    df = pd.DataFrame({"x": [1, 2, 3]})
    schema = {"x": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    error = result["type_errors"][0]
    assert error["column"] == "x"
    assert error["expected"] == "string"
    assert error["actual"] == "int"


def test_number_expected_matches_int():
    df = pd.DataFrame({"val": [1, 2]})
    schema = {"val": "number"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["type_errors"] == []


def test_number_expected_matches_float():
    df = pd.DataFrame({"val": [1.0, 2.5]})
    schema = {"val": "number"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["type_errors"] == []


def test_number_expected_with_string_column_mismatch():
    df = pd.DataFrame({"val": ["a", "b"]})
    schema = {"val": "number"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["type_errors"][0]["actual"] == "string"


def test_unsupported_expected_type_raises_valueerror():
    df = pd.DataFrame({"a": [1]})
    schema = {"a": "unknown_type"}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_unsupported_type_raises_even_if_column_missing():
    df = pd.DataFrame({"a": [1]})
    schema = {"b": "invalid"}
    with pytest.raises(ValueError):
        validate_schema(df, schema)
