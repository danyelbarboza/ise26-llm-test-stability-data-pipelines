from ise26.targets import validate_schema
import pandas as pd
import pytest

def test_valid_schema():
    df = pd.DataFrame({"a": [1, 2], "b": [1.5, 2.5]})
    schema = {"a": "int", "b": "float"}
    result = validate_schema(df, schema)
    assert result == {"valid": True, "missing_columns": [], "type_errors": []}

def test_missing_column():
    df = pd.DataFrame({"a": [1, 2]})
    schema = {"a": "int", "b": "float"}
    result = validate_schema(df, schema)
    assert not result["valid"]
    assert result["missing_columns"] == ["b"]
    assert result["type_errors"] == []

def test_type_mismatch():
    df = pd.DataFrame({"a": ["x", "y"]})
    schema = {"a": "int"}
    result = validate_schema(df, schema)
    assert not result["valid"]
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    error = result["type_errors"][0]
    assert error["column"] == "a"
    assert error["expected"] == "int"
    assert error["actual"] == "string"

def test_extra_columns_allowed():
    df = pd.DataFrame({"a": [1, 2], "b": [3, 4], "extra": [5, 6]})
    schema = {"a": "int", "b": "int"}
    result = validate_schema(df, schema)
    assert result["valid"]
    assert result["missing_columns"] == []
    assert result["type_errors"] == []

def test_normalization():
    df = pd.DataFrame({"a": [1.1, 2.2]})
    schema = {"a": "  Float  "}
    result = validate_schema(df, schema)
    assert result["valid"]
    assert result["missing_columns"] == []
    assert result["type_errors"] == []

def test_unsupported_type():
    df = pd.DataFrame({"a": [1]})
    with pytest.raises(ValueError):
        validate_schema(df, {"a": "unknown"})

def test_actual_value_in_type_error():
    df = pd.DataFrame({"a": pd.to_datetime(["2020-01-01"])})
    schema = {"a": "string"}
    result = validate_schema(df, schema)
    assert len(result["type_errors"]) == 1
    assert result["type_errors"][0]["actual"] == "datetime"

def test_empty_schema():
    df = pd.DataFrame({"a": [1], "b": ["x"]})
    schema = {}
    result = validate_schema(df, schema)
    assert result == {"valid": True, "missing_columns": [], "type_errors": []}
