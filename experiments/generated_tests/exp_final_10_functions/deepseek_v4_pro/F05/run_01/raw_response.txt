from ise26.targets import validate_schema

import pandas as pd
import pytest


def test_valid_all_columns_match():
    df = pd.DataFrame({
        "col_int": pd.array([1, 2], dtype="Int64"),
        "col_float": [1.0, 2.0],
        "col_str": ["a", "b"],
        "col_bool": [True, False],
        "col_dt": pd.to_datetime(["2020-01-01", "2020-01-02"]),
    })
    schema = {
        "col_int": "int",
        "col_float": "float",
        "col_str": "string",
        "col_bool": "bool",
        "col_dt": "datetime",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_extra_columns_allowed():
    df = pd.DataFrame({"a": [1], "b": [2]})
    schema = {"a": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_columns_reported():
    df = pd.DataFrame({"a": [1]})
    schema = {"a": "int", "b": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["b"]
    assert result["type_errors"] == []


def test_type_mismatch_single():
    df = pd.DataFrame({"a": [1, 2]})
    schema = {"a": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == []
    assert len(result["type_errors"]) == 1
    error = result["type_errors"][0]
    assert error["column"] == "a"
    assert error["expected"] == "string"
    assert error["actual"] != "string"


def test_combined_missing_and_type_errors():
    df = pd.DataFrame({"x": [1.5], "y": ["hello"]})
    schema = {"x": "string", "z": "bool"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert "z" in result["missing_columns"]
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "x"
    assert err["expected"] == "string"
    assert err["actual"] != "string"


def test_schema_normalization_whitespace_and_case():
    df = pd.DataFrame({"col": [1]})
    schema = {"col": "  InT  "}
    result = validate_schema(df, schema)
    assert result["valid"] is True


def test_number_type_accepts_int_and_float():
    df_int = pd.DataFrame({"a": [1, 2]})
    df_float = pd.DataFrame({"a": [1.0, 2.0]})
    schema = {"a": "number"}
    assert validate_schema(df_int, schema)["valid"] is True
    assert validate_schema(df_float, schema)["valid"] is True

    df_str = pd.DataFrame({"a": ["1"]})
    assert validate_schema(df_str, schema)["valid"] is False


def test_string_matches_object_dtype():
    df = pd.DataFrame({"s": ["a", "b"]})
    schema = {"s": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
