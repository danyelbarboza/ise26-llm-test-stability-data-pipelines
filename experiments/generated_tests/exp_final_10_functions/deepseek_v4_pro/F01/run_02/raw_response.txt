import numpy as np
import pandas as pd
import pytest
from ise26.targets import clean_customer_names

def test_basic_normalization():
    df = pd.DataFrame({
        "customer_name": [
            "  Maria  Teresa  ",
            "JOSÉ ÁNGEL",
            "François",
            "Normal Name"
        ]
    })
    result = clean_customer_names(df)
    expected = pd.Series([
        "maria teresa",
        "jose angel",
        "francois",
        "normal name"
    ], name="customer_name_clean")
    pd.testing.assert_series_equal(
        result["customer_name_clean"].reset_index(drop=True),
        expected.reset_index(drop=True),
        check_dtype=False
    )

def test_preserves_original_dataframe():
    original = pd.DataFrame({
        "customer_name": ["  Anna "],
        "other_col": [42]
    })
    reference = original.copy()
    result = clean_customer_names(original)
    # original must be unchanged
    pd.testing.assert_frame_equal(original, reference)
    # result must contain all original columns plus the new one
    assert set(result.columns) == set(["customer_name", "other_col", "customer_name_clean"])
    # result values unchanged for original columns
    assert result["customer_name"].iloc[0] == "  Anna "
    assert result["other_col"].iloc[0] == 42

def test_whitespace_collapse():
    df = pd.DataFrame({
        "customer_name": [
            "leading space",
            "trailing space ",
            "   very   messy    text   "
        ]
    })
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"].tolist()
    assert cleaned == ["leading space", "trailing space", "very messy text"]

def test_lowercasing():
    df = pd.DataFrame({
        "customer_name": ["UPPERCASE", "MixedCase", "lowercase"]
    })
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"].tolist()
    assert cleaned == ["uppercase", "mixedcase", "lowercase"]

def test_accent_removal():
    df = pd.DataFrame({
        "customer_name": [
            "áéíóúý",
            "ÁÉÍÓÚÝ",
            "ñçãõü",
            "López-Díaz",
            "São Paulo"
        ]
    })
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"].tolist()
    expected = [
        "aeiouy",
        "aeiouy",
        "ncaou",
        "lopez-diaz",
        "sao paulo"
    ]
    assert cleaned == expected

def test_null_and_blank_mapping():
    df = pd.DataFrame({
        "customer_name": [None, np.nan, "", "   ", "actual"]
    })
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"]
    assert cleaned.iloc[0] is pd.NA
    assert cleaned.iloc[1] is pd.NA
    assert cleaned.iloc[2] is pd.NA
    assert cleaned.iloc[3] is pd.NA
    assert cleaned.iloc[4] == "actual"

def test_custom_column_names():
    df = pd.DataFrame({"raw": ["  María "], "id": [1]})
    result = clean_customer_names(df, name_col="raw", output_col="normalized")
    assert "normalized" in result.columns
    assert result["normalized"].iloc[0] == "maria"
    # preserved
    assert result["raw"].iloc[0] == "  María "

def test_missing_column_raises():
    df = pd.DataFrame({"wrong_column": ["test"]})
    with pytest.raises(Exception):
        clean_customer_names(df, name_col="customer_name")

def test_empty_dataframe():
    df = pd.DataFrame(columns=["customer_name"])
    result = clean_customer_names(df)
    assert result.empty
    assert "customer_name_clean" in result.columns
    # original column still present
    assert list(result.columns) == ["customer_name", "customer_name_clean"]

def test_non_string_values():
    df = pd.DataFrame({
        "customer_name": [123, 45.0, True, 0]
    })
    # The function should safely convert these to strings and normalise
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"].tolist()
    # Expect strings after normalisation; bool True -> "true", float 45.0 -> "45.0", int 0 -> "0", etc.
    assert cleaned == ["123", "45.0", "true", "0"]
