import unicodedata

import numpy as np
import pandas as pd
import pytest
from ise26.targets import clean_customer_names


def _remove_accents(text: str) -> str:
    """Remove diacritical marks, keep base letters (e.g., 'é' -> 'e')."""
    nfkd = unicodedata.normalize("NFKD", text)
    return "".join(c for c in nfkd if not unicodedata.combining(c))


class TestCleanCustomerNames:
    """Tests for clean_customer_names."""

    def test_original_df_unchanged(self):
        """Original DataFrame is not modified."""
        df = pd.DataFrame({"customer_name": ["John   Doe", "Anna"]})
        original = df.copy(deep=True)
        _ = clean_customer_names(df)
        pd.testing.assert_frame_equal(df, original)

    def test_returns_new_dataframe(self):
        """Returned object is a new DataFrame, not a view of the original."""
        df = pd.DataFrame({"customer_name": ["X"]})
        result = clean_customer_names(df)
        assert result is not df
        assert "customer_name_clean" in result.columns
        assert "customer_name_clean" not in df.columns

    def test_adds_normalized_column_with_default_names(self):
        """Default output column is added and normalized correctly."""
        df = pd.DataFrame({"customer_name": [" Alice  ", "BOB"]})
        result = clean_customer_names(df)
        assert "customer_name_clean" in result.columns
        assert list(result.columns) == ["customer_name", "customer_name_clean"]
        # original data preserved
        assert result["customer_name"].tolist() == [" Alice  ", "BOB"]

    def test_lowercase(self):
        """Values are lowercased."""
        df = pd.DataFrame({"customer_name": ["ALICE", "Bob"]})
        result = clean_customer_names(df)
        assert result["customer_name_clean"].tolist() == ["alice", "bob"]

    def test_collapse_whitespace(self):
        """Multiple spaces are collapsed to a single space; leading/trailing whitespace is preserved."""
        df = pd.DataFrame({"customer_name": ["  Hello    World  ", "   A   B   "]})
        result = clean_customer_names(df)
        expected = [" hello world ", " a b "]
        assert result["customer_name_clean"].tolist() == expected

    def test_accent_removal(self):
        """Diacritical marks are removed (base letters kept)."""
        df = pd.DataFrame({"customer_name": ["José François", "Äë ÏöÜ"]})
        result = clean_customer_names(df)
        # Expected: accents removed, lowercased
        expected_raw = ["José François", "Äë ÏöÜ"]
        expected_clean = [
            _remove_accents(raw).lower() for raw in expected_raw
        ]
        assert result["customer_name_clean"].tolist() == expected_clean

    def test_null_input_maps_to_NA(self):
        """None and NaN become pd.NA."""
        df = pd.DataFrame({"customer_name": [None, np.nan, "valid"]})
        result = clean_customer_names(df)
        assert pd.isna(result["customer_name_clean"].iloc[0])
        assert result["customer_name_clean"].iloc[0] is pd.NA
        assert pd.isna(result["customer_name_clean"].iloc[1])
        assert result["customer_name_clean"].iloc[1] is pd.NA
        assert result["customer_name_clean"].iloc[2] == "valid"
