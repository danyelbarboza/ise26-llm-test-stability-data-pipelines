import pandas as pd
import numpy as np
import pytest
from ise26.targets import clean_customer_names


def _check_pd_na(val):
    """Helper to assert a value is pd.NA."""
    assert val is pd.NA


def test_preserves_original_dataframe():
    df = pd.DataFrame({"customer_name": [" João ", "MARIA"]})
    original = df.copy()
    result = clean_customer_names(df)
    # Original should be identical to the copy before any operation
    pd.testing.assert_frame_equal(df, original)
    # Result should be a different object
    assert result is not df


def test_default_columns_add_output():
    df = pd.DataFrame({"customer_name": [" Alice ", " BRIAN "]})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    expected = pd.Series(["alice", "brian"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)


def test_custom_output_column():
    df = pd.DataFrame({"customer_name": [" Alice "]})
    result = clean_customer_names(df, output_col="normalized")
    assert "normalized" in result.columns
    assert result["normalized"].iloc[0] == "alice"


def test_custom_name_column():
    df = pd.DataFrame({"raw": [" JEAN "], "other": [42]})
    result = clean_customer_names(df, name_col="raw", output_col="clean")
    assert "clean" in result.columns
    assert result["clean"].iloc[0] == "jean"


def test_missing_name_column_raises():
    df = pd.DataFrame({"foo": ["bar"]})
    with pytest.raises((KeyError, ValueError)):
        clean_customer_names(df, name_col="customer_name")


def test_overwrite_existing_output_column():
    df = pd.DataFrame({"customer_name": [" Ana "], "clean": [1]})
    original = df.copy()
    result = clean_customer_names(df, output_col="clean")
    assert "clean" in result.columns
    assert result["clean"].iloc[0] == "ana"
    # original unchanged (side-effect check)
    assert original["clean"].iloc[0] == 1


def test_null_handling_None():
    df = pd.DataFrame({"customer_name": [None, "Valid"]})
    result = clean_customer_names(df)
    _check_pd_na(result["customer_name_clean"].iloc[0])
    assert result["customer_name_clean"].iloc[1] == "valid"


def test_null_handling_np_nan():
    df = pd.DataFrame({"customer_name": [np.nan, "Valid"]})
    result = clean_customer_names(df)
    _check_pd_na(result["customer_name_clean"].iloc[0])


def test_null_handling_pd_NA():
    # pd.NA already present
    df = pd.DataFrame({"customer_name": [pd.NA, "Valid"]})
    result = clean_customer_names(df)
    _check_pd_na(result["customer_name_clean"].iloc[0])


def test_blank_empty_string():
    df = pd.DataFrame({"customer_name": ["", "Valid"]})
    result = clean_customer_names(df)
    _check_pd_na(result["customer_name_clean"].iloc[0])


def test_blank_whitespace_only():
    df = pd.DataFrame({"customer_name": ["   \t ", "Valid"]})
    result = clean_customer_names(df)
    _check_pd_na(result["customer_name_clean"].iloc[0])


def test_collapse_multiple_spaces():
    df = pd.DataFrame({"customer_name": ["  Jane   Doe  "]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].iloc[0] == "jane doe"


def test_lowercase_conversion():
    df = pd.DataFrame({"customer_name": ["UPPER", "Lower", "MiXeD"]})
    result = clean_customer_names(df)
    expected = pd.Series(["upper", "lower", "mixed"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)


def test_accent_removal():
    test_cases = [
        ("José", "jose"),
        ("Mário", "mario"),
        ("Niña", "nina"),
        ("François", "francois"),
        ("Über", "uber"),
        ("São Paulo", "sao paulo"),
        ("façade", "facade"),
        ("Zoë", "zoe"),
    ]
    for raw, expected in test_cases:
        df = pd.DataFrame({"customer_name": [raw]})
        result = clean_customer_names(df)
        assert result["customer_name_clean"].iloc[0] == expected


def test_mixed_operations_on_single_value():
    df = pd.DataFrame({"customer_name": ["   Érica  Müller  "]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].iloc[0] == "erica muller"


def test_empty_dataframe():
    df = pd.DataFrame({"customer_name": []})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    assert len(result["customer_name_clean"]) == 0


def test_all_nulls():
    df = pd.DataFrame({"customer_name": [None, np.nan, pd.NA, ""]})
    result = clean_customer_names(df)
    for val in result["customer_name_clean"]:
        _check_pd_na(val)


def test_does_not_modify_other_columns():
    df = pd.DataFrame({"customer_name": [" Alice "], "id": [42], "score": [3.14]})
    result = clean_customer_names(df)
    assert list(result.columns) == ["customer_name", "id", "score", "customer_name_clean"]
    assert result["id"].iloc[0] == 42
    assert result["score"].iloc[0] == 3.14


def test_original_unchanged_when_output_column_existed():
    df = pd.DataFrame({"customer_name": [" João "], "x": [100], "customer_name_clean": ["original"]})
    original = df.copy()
    result = clean_customer_names(df)
    assert result["customer_name_clean"].iloc[0] == "joao"
    # original should still have the old value
    assert original["customer_name_clean"].iloc[0] == "original"
    # and no extra column added
    pd.testing.assert_frame_equal(df, original)
