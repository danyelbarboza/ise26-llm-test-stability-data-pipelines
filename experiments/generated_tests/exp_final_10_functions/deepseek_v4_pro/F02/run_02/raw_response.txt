import pytest
import pandas as pd
from ise26.targets import deduplicate_events


def create_test_df(data, id_col="event_id", timestamp_col="updated_at"):
    """Helper to create a DataFrame with given rows and consistent column order."""
    return pd.DataFrame(data, columns=[id_col, timestamp_col])


class TestDeduplicateEvents:
    def test_preserves_null_identifiers(self):
        """Rows with null event_id are all kept."""
        df = create_test_df(
            [
                [None, "2021-01-01"],
                ["a", "2021-01-01"],
                [None, "2021-02-01"],
                ["b", "2021-01-01"],
                [None, "2021-03-01"],
            ]
        )
        result = deduplicate_events(df)
        # All three null rows should be present, plus deduplicated non-null rows
        expected_ids = [None, "a", None, "b", None]
        assert list(result["event_id"]) == expected_ids

    def test_deduplicate_keeps_most_recent_timestamp(self):
        """For the same non-null event_id, only the row with latest valid timestamp remains."""
        df = create_test_df(
            [
                ["ev1", "2021-01-01"],
                ["ev1", "2021-03-01"],
                ["ev1", "2021-02-01"],
                ["ev2", "2021-01-01"],
            ]
        )
        result = deduplicate_events(df)
        # Should keep row with "2021-03-01" for ev1 (index 1) and the single ev2 row
        expected = [
            ["ev1", "2021-03-01"],
            ["ev2", "2021-01-01"],
        ]
        assert result.reset_index(drop=True).values.tolist() == expected

    def test_invalid_timestamps_treated_as_older(self):
        """Unparseable timestamps become NaT and are considered older than any valid timestamp;
        among multiple NaTs, keep the last in original order."""
        df = create_test_df(
            [
                ["ev1", "invalid_date"],           # 0 – NaT
                ["ev1", "2021-01-01"],              # 1 – valid
                ["ev1", "not_a_date"],              # 2 – NaT
                ["ev1", "2021-06-01"],              # 3 – valid
            ]
        )
        result = deduplicate_events(df)
        # The valid timestamp "2021-06-01" (index 3) is most recent -> kept
        assert result.shape[0] == 1
        assert result.iloc[0]["event_id"] == "ev1"
        assert result.iloc[0]["updated_at"] == "2021-06-01"

    def test_timestamp_tie_keeps_last_row_by_original_order(self):
        """When two rows share the same event_id and identical valid timestamp,
        keep the one that appears later in the original DataFrame."""
        df = create_test_df(
            [
                ["ev1", "2021-01-01"],   # index 0 – should be dropped
                ["ev1", "2021-01-01"],   # index 1 – kept because later
            ]
        )
        result = deduplicate_events(df)
        assert len(result) == 1
        # The kept row is the second one, so its original data.
        assert result.iloc[0]["updated_at"] == "2021-01-01"
        # Check that the row corresponds to the second occurrence (any extra column would help,
        # but here they are identical) – we rely on the tie-breaking logic.

        # Test with an extra column to distinguish rows
        df2 = pd.DataFrame(
            {
                "event_id": ["ev1", "ev1"],
                "updated_at": ["2021-01-01", "2021-01-01"],
                "info": ["first", "second"],
            }
        )
        result2 = deduplicate_events(df2[["event_id", "updated_at", "info"]])
        # With default args but extra column, order still follows original.
        # The row with "second" should be kept.
        assert result2.shape[0] == 1
        assert result2.iloc[0]["info"] == "second"

    def test_original_order_preserved_after_dedup(self):
        """The output rows appear in the same relative order as the originals, after removing duplicates."""
        df = pd.DataFrame(
            {
                "event_id": [None, "a", "b", "a", None, "b", "c"],
                "updated_at": [
                    "2021-01-01",
                    "2021-01-01",
                    "2021-03-01",
                    "2021-02-01",
                    "2021-02-01",
                    "2021-01-01",
                    "2021-01-01",
                ],
            }
        )
        result = deduplicate_events(df)
        # Expected kept rows (with original indices):
        # None (0), a (3, most recent for a), b (2, most recent for b vs index 5), None (4), c (6)
        expected_ids = [None, "a", "b", None, "c"]
        expected_timestamps = ["2021-01-01", "2021-02-01", "2021-03-01", "2021-02-01", "2021-01-01"]
        assert list(result["event_id"]) == expected_ids
        assert list(result["updated_at"]) == expected_timestamps

    def test_input_df_not_mutated(self):
        """The original DataFrame is not modified by the function."""
        df = pd.DataFrame(
            {"event_id": ["a", "a"], "updated_at": ["2021-01-01", "2021-02-01"]}
        )
        df_copy = df.copy()
        deduplicate_events(df)
        pd.testing.assert_frame_equal(df, df_copy)

    def test_empty_dataframe(self):
        """An empty DataFrame returns an empty DataFrame with the same columns."""
        df = pd.DataFrame(columns=["event_id", "updated_at"])
        result = deduplicate_events(df)
        assert result.empty
        pd.testing.assert_index_equal(result.columns, df.columns)

    def test_all_null_ids(self):
        """When all event_ids are null, no deduplication should happen."""
        df = create_test_df(
            [
                [None, "2021-01-01"],
                [None, "2021-02-01"],
                [None, "2021-01-01"],
            ]
        )
        result = deduplicate_events(df)
        assert len(result) == 3
        # All rows kept, original order preserved
        assert list(result["event_id"]) == [None, None, None]
        assert list(result["updated_at"]) == ["2021-01-01", "2021-02-01", "2021-01-01"]

    def test_all_valid_ids_no_duplicates(self):
        """With no duplicates among non-null ids, all rows are kept and order preserved,
        but index is reset."""
        df = create_test_df(
            [
                ["a", "2021-01-01"],
                ["b", "2021-02-01"],
                ["c", "2021-03-01"],
            ]
        )
        result = deduplicate_events(df)
        assert len(result) == 3
        assert list(result["event_id"]) == ["a", "b", "c"]
        assert list(result["updated_at"]) == ["2021-01-01", "2021-02-01", "2021-03-01"]
        # Index should be reset
        assert list(result.index) == [0, 1, 2]

    def test_column_dtypes_preserved(self):
        """Original column dtypes should remain unchanged."""
        df = pd.DataFrame(
            {
                "event_id": pd.Series(["a", "b", "a", None], dtype="object"),
                "updated_at": pd.Series(
                    ["2021-01-01", "2021-02-01", "2021-03-01", "2021-04-01"],
                    dtype="string",
                ),
                "extra": pd.Series([1, 2, 3, 4], dtype="int64"),
            }
        )
        result = deduplicate_events(df)
        assert result["event_id"].dtype == object
        # updated_at was string dtype, should remain string (not datetime)
        assert isinstance(result["updated_at"].dtype, pd.StringDtype)
        assert result["extra"].dtype == "int64"

    def test_missing_required_columns(self):
        """A ValueError is raised if id_col or timestamp_col are missing."""
        df = pd.DataFrame({"col1": [1, 2], "col2": [3, 4]})
        with pytest.raises(ValueError, match="deduplicate_events"):
            deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        with pytest.raises(ValueError, match="deduplicate_events"):
            deduplicate_events(df, id_col="col1", timestamp_col="missing_col")

    def test_additional_columns_preserved(self):
        """Columns beyond id and timestamp are preserved and appear correctly."""
        df = pd.DataFrame(
            {
                "event_id": ["a", "a", "b"],
                "updated_at": ["2021-02-01", "2021-01-01", "2021-03-01"],
                "value": [10, 20, 30],
            }
        )
        result = deduplicate_events(df)
        # For "a", keep the most recent timestamp "2021-02-01" (value 10)
        # For "b", keep value 30
        expected_values = [10, 30]
        assert list(result["value"]) == expected_values
        # Ensure columns are in original order (event_id, updated_at, value)
        assert list(result.columns) == ["event_id", "updated_at", "value"]
