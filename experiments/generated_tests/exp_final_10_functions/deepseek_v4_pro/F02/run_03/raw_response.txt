import pandas as pd
import numpy as np
from ise26.targets import deduplicate_events


def test_keeps_most_recent_valid_timestamp():
    """For the same non-null id, the row with the highest valid timestamp is kept."""
    df = pd.DataFrame({
        "event_id": [1, 1, 2],
        "updated_at": ["2020-01-02", "2020-01-01", "2020-06-15"],
        "value": [10, 20, 30],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    expected_ids = [1, 2]
    assert list(result["event_id"]) == expected_ids
    # For event_id=1 the most recent is "2020-01-02" -> value 10
    row_1 = result[result["event_id"] == 1].iloc[0]
    assert row_1["value"] == 10
    assert row_1["updated_at"] == "2020-01-02"


def test_invalid_timestamps_treated_as_older():
    """Rows with unparseable timestamps are considered older than any valid timestamp."""
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["not-a-date", "2020-01-01", "invalid"],
        "value": [1, 2, 3],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Only one row for id=1, the valid one "2020-01-01" -> value 2
    assert len(result) == 1
    assert result.iloc[0]["value"] == 2
    assert result.iloc[0]["updated_at"] == "2020-01-01"


def test_all_invalid_timestamps_keeps_last_row_in_original_order():
    """When all timestamps for an id are invalid, the last row in original order is kept."""
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["bad", "garbage", "ugly"],
        "value": [10, 20, 30],  # order: 10, 20, 30 — last is 30
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert len(result) == 1
    assert result.iloc[0]["value"] == 30


def test_null_identifiers_preserved():
    """Rows with null event_id are never deduplicated — all are kept."""
    df = pd.DataFrame({
        "event_id": [np.nan, np.nan, 1],
        "updated_at": ["2020-01-01", "2020-01-02", "2020-01-03"],
        "value": [1, 2, 3],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Two null rows remain
    assert result["event_id"].isna().sum() == 2
    # The valid id=1 is kept
    assert len(result) == 3
    # Values of the two null rows are both present
    null_values = set(result.loc[result["event_id"].isna(), "value"])
    assert null_values == {1, 2}


def test_mixed_null_and_non_null_with_dedup():
    """Non-null ids are deduplicated; nulls are unaffected."""
    df = pd.DataFrame({
        "event_id": [np.nan, 2, 2, np.nan, 2],
        "updated_at": ["2020-01-01", "2020-01-02", "2020-01-03", "2020-01-04", "2020-01-01"],
        "value": [10, 20, 30, 40, 50],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # For id=2, most recent is "2020-01-03" -> value 30
    assert result[result["event_id"] == 2]["value"].iloc[0] == 30
    # Two null rows remain (values 10 and 40)
    assert len(result[result["event_id"].isna()]) == 2
    assert set(result.loc[result["event_id"].isna(), "value"]) == {10, 40}
    assert len(result) == 3


def test_no_duplicates_returns_all_rows():
    """When no non-null id appears more than once, all rows are returned."""
    df = pd.DataFrame({
        "event_id": [1, 2, 3],
        "updated_at": ["2022-01-01", "2022-02-02", "2022-03-03"],
        "val": ["a", "b", "c"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Same rows as input (order by original order)
    assert result["event_id"].tolist() == [1, 2, 3]
    assert result["val"].tolist() == ["a", "b", "c"]


def test_empty_dataframe():
    """An empty DataFrame should return an empty DataFrame with same columns."""
    df = pd.DataFrame(columns=["event_id", "updated_at", "x"])
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert len(result) == 0
    assert list(result.columns) == ["event_id", "updated_at", "x"]


def test_input_not_mutated():
    """Original DataFrame remains unchanged."""
    df = pd.DataFrame({
        "event_id": [0, 1, 2],
        "updated_at": [pd.Timestamp("2022-01-01"), pd.Timestamp("2022-01-02"), pd.Timestamp("2022-01-03")],
        "value": [100, 200, 300],
    })
    original = df.copy()
    _ = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    pd.testing.assert_frame_equal(df, original)


def test_custom_column_names():
    """Works with non-default id_col and timestamp_col names."""
    df = pd.DataFrame({
        "uuid": ["a", "a", "b"],
        "ts": ["2023-05-01", "2023-05-02", "2023-05-01"],
        "data": [1, 2, 3],
    })
    result = deduplicate_events(df, id_col="uuid", timestamp_col="ts")
    # For uuid='a', most recent is '2023-05-02' -> data=2
    assert len(result[result["uuid"] == "a"]) == 1
    assert result[result["uuid"] == "a"]["data"].iloc[0] == 2


def test_tiebreaker_last_row_in_original_order():
    """When timestamps tie, the one later in the original DataFrame is kept."""
    df = pd.DataFrame({
        "event_id": [10, 10, 10],
        "updated_at": ["2020-01-01"] * 3,
        "value": ["first", "second", "third"],  # order: first, second, third — keep third
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert len(result) == 1
    assert result.iloc[0]["value"] == "third"


def test_original_order_preserved():
    """Output rows appear in the same relative order as they did originally."""
    df = pd.DataFrame({
        "event_id": [2, np.nan, 1, 2, 1],
        "updated_at": [
            "2022-03-01",   # id=2 older
            "2022-01-01",   # null
            "2022-04-01",   # id=1 later
            "2022-05-01",   # id=2 later (kept)
            "2022-01-01",   # id=1 older
        ],
        "info": [10, 20, 30, 40, 50],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Expected: row0? Row0 id=2 kept (40), row1 NaN kept (20), row2 id=1 kept (30).
    # Then row3 id=2 duplicate (dropped), row4 id=1 duplicate (dropped).
    # So order: row0, row1, row2 (original indices 0,1,2)
    expected_info = [40, 20, 30]  # order check
    assert result["info"].tolist() == expected_info


def test_output_index_reset():
    """The returned DataFrame has a clean default RangeIndex."""
    df = pd.DataFrame({
        "event_id": [5, 5, 7],
        "updated_at": ["2021-06-01", "2021-06-02", "2021-07-04"],
        "desc": ["old", "new", "unique"],
    }, index=[100, 200, 300])
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert result.index.equals(pd.RangeIndex(start=0, stop=len(result), step=1))


def test_all_columns_preserved():
    """Extra columns are kept in output."""
    df = pd.DataFrame({
        "event_id": [10, 10],
        "updated_at": ["2018-01-01", "2018-01-02"],
        "col_x": [True, False],
        "col_y": ["foo", "bar"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert set(result.columns) == {"event_id", "updated_at", "col_x", "col_y"}
    # The more recent row (2018-01-02) is kept -> col_x=False, col_y="bar"
    row = result.iloc[0]
    assert row["col_x"] == False
    assert row["col_y"] == "bar"


def test_nulls_in_non_key_columns_preserved():
    """Null values in non-id, non-timestamp columns are preserved."""
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2020-10-10", "2020-10-11"],
        "desc": [None, "valid"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert len(result) == 1
    assert result.iloc[0]["desc"] == "valid"


def test_mixed_valid_invalid_with_null_ids():
    """Null ids are kept, valid ids deduped by valid timestamps, invalid timestamps removed."""
    df = pd.DataFrame({
        "event_id": [np.nan, 3, 3, 3, np.nan],
        "updated_at": ["2020-01-01", "bad", "2020-02-02", "2020-03-03", "garbage"],
        "value": [1, 2, 3, 4, 5],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Null rows: values 1 and 5 both kept
    assert result[result["event_id"].isna()]["value"].tolist() == [1, 5]
    # For id=3, most recent valid is "2020-03-03" -> value 4
    assert result[result["event_id"] == 3]["value"].iloc[0] == 4
    assert len(result) == 3


def test_many_rows_with_one_id():
    """A single id appears many times, only the latest valid is kept."""
    n = 100
    ids = [1] * n
    # timestamps increasing, so last is most recent
    dates = pd.date_range(start="2000-01-01", periods=n, freq="D")
    vals = list(range(n))
    df = pd.DataFrame({"event_id": ids, "updated_at": dates, "val": vals})
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    assert len(result) == 1
    assert result.iloc[0]["val"] == n - 1
