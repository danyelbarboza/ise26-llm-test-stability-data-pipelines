from ise26.targets import deduplicate_events
import pandas as pd
import pytest
from pandas.testing import assert_frame_equal
import numpy as np

def test_keep_most_recent_timestamp():
    df = pd.DataFrame({
        "event_id": [1, 1, 2],
        "updated_at": ["2023-01-02", "2023-01-03", "2023-01-01"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "updated_at": ["2023-01-03", "2023-01-01"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_preserve_null_event_ids():
    df = pd.DataFrame({
        "event_id": [None, None, 1],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [None, None, 1],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_invalid_timestamp_treated_as_older():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": [None, "2023-01-03"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-03"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_invalid_timestamp_treated_as_older_reversed_order():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-03", None]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-03"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_timestamp_tie_keep_last_row():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-01", "2023-01-01"],
        "value": ["A", "B"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-01"],
        "value": ["B"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_input_df_not_mutated():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-02", "2023-01-03"]
    })
    original_df = df.copy(deep=True)
    deduplicate_events(df)
    assert_frame_equal(df, original_df)

def test_custom_column_names():
    df = pd.DataFrame({
        "id": [1, 1, 2],
        "ts": ["2023-01-02", "2023-01-03", "2023-01-01"]
    })
    result = deduplicate_events(df, id_col="id", timestamp_col="ts")
    expected = pd.DataFrame({
        "id": [1, 2],
        "ts": ["2023-01-03", "2023-01-01"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_empty_dataframe():
    df = pd.DataFrame({"event_id": [], "updated_at": []})
    result = deduplicate_events(df)
    assert len(result) == 0
    assert list(result.columns) == ["event_id", "updated_at"]

def test_single_row():
    df = pd.DataFrame({"event_id": [1], "updated_at": ["2023-01-01"]})
    result = deduplicate_events(df)
    assert len(result) == 1
    assert result["event_id"].iloc[0] == 1

def test_all_null_event_ids():
    df = pd.DataFrame({
        "event_id": [None, None, None],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"]
    })
    result = deduplicate_events(df)
    assert len(result) == 3
    assert list(result["event_id"].isna()) == [True, True, True]

def test_all_invalid_timestamps_same_id():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": [pd.NaT, pd.NaT]
    })
    result = deduplicate_events(df)
    # Both invalid, tie broken by original order (last row kept)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": [pd.NaT]
    })
    assert len(result) == 1
    assert result["event_id"].iloc[0] == 1
    assert pd.isna(result["updated_at"].iloc[0])

def test_mixed_valid_invalid_timestamps_for_same_id():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2023-01-01", pd.NaT, "2023-01-03"],
        "val": [10, 20, 30]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-03"],
        "val": [30]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_na_timestamp_treated_as_oldest():
    # Even when the valid timestamp is older than the NaT's potential value, NaT is older
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2020-01-01", pd.NaT]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2020-01-01"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_output_index_reset():
    df = pd.DataFrame({
        "event_id": [2, 1, 1],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"]
    })
    result = deduplicate_events(df)
    assert result.index.tolist() == [0, 1]  # Two rows: id 2 and id 1 (most recent)

def test_no_helper_columns_in_output():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-01", "2023-01-02"]
    })
    result = deduplicate_events(df)
    assert "_original_order" not in result.columns
    assert "_parsed_timestamp" not in result.columns

def test_preserve_null_ids_with_mixed_timestamps():
    df = pd.DataFrame({
        "event_id": [None, None, 1, 1],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-01", "2023-01-02"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [None, None, 1],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-02"]
    })
    assert_frame_equal(result.reset_index(drop=True), expected)

def test_different_dtypes_preserved():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-01", "2023-01-02"],
        "metric": [0.5, 0.6]
    })
    result = deduplicate_events(df)
    assert result["metric"].dtype == np.float64
