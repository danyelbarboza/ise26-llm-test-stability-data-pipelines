import numpy as np
import pandas as pd
import pytest
from ise26.targets import deduplicate_events


def test_keeps_most_recent_valid_timestamp():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2021-01-01", "2021-01-03", "2021-01-02"],
        "value": [10, 30, 20],
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2021-01-03"],
        "value": [30],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_null_identifiers_preserved():
    df = pd.DataFrame({
        "event_id": [np.nan, 1, np.nan, 1],
        "updated_at": ["2021-01-01", "2021-01-02", "2021-01-03", "2021-01-02"],
        "value": [1, 2, 3, 4],
    })
    result = deduplicate_events(df)
    # Identified id=1 should keep the most recent (2021-01-02, value=4)
    # Null ids: both kept.
    expected = pd.DataFrame({
        "event_id": [np.nan, np.nan, 1],
        "updated_at": ["2021-01-01", "2021-01-03", "2021-01-02"],
        "value": [1, 3, 4],
    }, index=[0,2,3])  # original order positions: row0, row2, row3 (row1 duplicate removed)
    # Note: after sorting by original_order, order is 0->2->3, so this expected order matches.
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected.reset_index(drop=True))


def test_invalid_timestamps_treated_as_oldest():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["bad", "2021-01-01", "2021-01-02"],
        "value": [1, 2, 3],
    })
    result = deduplicate_events(df)
    # NaT (bad) is oldest, then 2021-01-01, then 2021-01-02 -> keep last (row with 2021-01-02)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2021-01-02"],
        "value": [3],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_all_invalid_timestamps_tie_keeps_last():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["bad", "bad"],
        "value": [10, 20],
    })
    result = deduplicate_events(df)
    # Both NaT, keep last row in original order -> row with value=20
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["bad"],
        "value": [20],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_valid_timestamp_ties_keep_last():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2021-01-01", "2021-01-01", "2021-01-01"],
        "value": [30, 20, 10],
    })
    result = deduplicate_events(df)
    # All same timestamp, keep last row based on original order (highest index)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2021-01-01"],
        "value": [10],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_preserves_original_order():
    df = pd.DataFrame({
        "event_id": [np.nan, 1, 2, np.nan, 1, 2],
        "updated_at": [
            "2021-01-01", "2021-01-01", "2021-01-01",
            "2021-01-02", "2021-01-03", "2021-01-03",
        ],
        "value": [1, 2, 3, 4, 5, 6],
    })
    result = deduplicate_events(df)
    # Expected: nulls retained (rows 0 and 3), identified duplicates removed (1: keep row4, 2: keep row5)
    # Order by original position: row0, row3, row4, row5
    expected = pd.DataFrame({
        "event_id": [np.nan, np.nan, 1, 2],
        "updated_at": ["2021-01-01", "2021-01-02", "2021-01-03", "2021-01-03"],
        "value": [1, 4, 5, 6],
    }, index=[0, 3, 4, 5])
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected.reset_index(drop=True))


def test_does_not_mutate_input():
    df = pd.DataFrame({
        "event_id": [1, 1, np.nan],
        "updated_at": ["2021-01-02", "2021-01-01", "2021-01-03"],
        "value": [10, 20, 30],
    })
    original = df.copy(deep=True)
    _ = deduplicate_events(df)
    pd.testing.assert_frame_equal(df, original)


def test_empty_dataframe():
    df = pd.DataFrame(columns=["event_id", "updated_at"])
    result = deduplicate_events(df)
    expected = pd.DataFrame(columns=["event_id", "updated_at"])
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_extra_columns_preserved():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2021-01-02", "2021-01-01"],
        "x": [1.5, 2.5],
        "y": ["a", "b"],
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2021-01-02"],
        "x": [1.5],
        "y": ["a"],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_custom_column_names():
    df = pd.DataFrame({
        "id": [1, 1, 2],
        "ts": ["2021-01-02", "2021-01-01", "2021-01-03"],
        "data": [100, 200, 300],
    })
    result = deduplicate_events(df, id_col="id", timestamp_col="ts")
    expected = pd.DataFrame({
        "id": [1, 2],
        "ts": ["2021-01-02", "2021-01-03"],
        "data": [100, 300],
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)
