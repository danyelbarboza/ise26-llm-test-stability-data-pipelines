from ise26.targets import deduplicate_events
import pandas as pd
import pytest

def test_preserves_null_event_ids():
    df = pd.DataFrame({
        "event_id": [None, None, 1, 1, None],
        "updated_at": ["2022-01-01", "2022-01-02", "2022-01-03", "2022-01-04", "2022-01-05"],
        "value": [10, 20, 30, 40, 50],
    })
    result = deduplicate_events(df)
    # null ids rows are all preserved; non-null 1 deduplicates to keep the last (id=1, updated_at=2022-01-04)
    expected_ids = [None, None, 1, None]  # rows corresponding to input indices 0,1,3,4 in that order
    expected_values = [10, 20, 40, 50]
    assert result["event_id"].tolist() == expected_ids
    assert result["value"].tolist() == expected_values

def test_keep_most_recent_timestamp():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2022-01-01", "2022-01-03", "2022-01-02"],
        "data": ["a", "b", "c"],
    })
    result = deduplicate_events(df)
    # Should keep row with 2022-01-03
    assert result.shape[0] == 1
    assert result["updated_at"].iloc[0] == "2022-01-03"
    assert result["data"].iloc[0] == "b"

def test_timestamp_tie_keeps_last_in_original_order():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2022-01-01", "2022-01-01", "2022-01-02"],
        "value": [10, 20, 30],
    })
    result = deduplicate_events(df)
    # only one row for id=1, timestamp tie for first two, but most recent is 2022-01-02, so that's kept
    assert result.shape[0] == 1
    assert result["value"].iloc[0] == 30
    # Now tie across all: same timestamp
    df2 = pd.DataFrame({
        "event_id": [2, 2, 2],
        "updated_at": ["2022-01-01", "2022-01-01", "2022-01-01"],
        "value": [100, 200, 300],
    })
    result2 = deduplicate_events(df2)
    assert result2.shape[0] == 1
    assert result2["value"].iloc[0] == 300  # last in original order

def test_invalid_timestamp_treated_as_older():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["invalid", "2022-01-01", "2022-01-02"],
        "value": [10, 20, 30],
    })
    result = deduplicate_events(df)
    # The invalid becomes NaT and sorts first, valid later; keep last => 2022-01-02
    assert result.shape[0] == 1
    assert result["value"].iloc[0] == 30

def test_all_invalid_timestamps_for_an_id_keeps_last_original():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["bad", "also_bad", "garbage"],
        "value": [5, 6, 7],
    })
    result = deduplicate_events(df)
    assert result.shape[0] == 1
    # All parse to NaT, sorted with na_position='first' then by original_order ascending, keep last => row with index 2
    assert result["value"].iloc[0] == 7

def test_does_not_mutate_input():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2022-01-01", "2022-01-02"],
        "value": [10, 20],
    })
    original_shape = df.shape
    original_columns = list(df.columns)
    _ = deduplicate_events(df)
    assert df.shape == original_shape
    assert list(df.columns) == original_columns
    # also content unchanged
    assert df["value"].tolist() == [10, 20]

def test_empty_dataframe():
    df = pd.DataFrame(columns=["event_id", "updated_at"])
    result = deduplicate_events(df)
    assert result.empty
    assert list(result.columns) == ["event_id", "updated_at"]

def test_missing_columns_raises_error():
    df = pd.DataFrame({"a": [1]})
    with pytest.raises(KeyError):
        deduplicate_events(df)

def test_custom_column_names():
    df = pd.DataFrame({
        "ident": [10, 10, None, 20],
        "ts": [pd.Timestamp("2022-01-01"), pd.Timestamp("2022-01-02"), pd.Timestamp("2022-01-03"), pd.Timestamp("2022-01-01")],
        "info": ["x", "y", "z", "w"],
    })
    result = deduplicate_events(df, id_col="ident", timestamp_col="ts")
    # id=10: two rows, keep most recent ts: 2022-01-02 => row index 1 (info="y")
    # id=20: only one row, keep it
    # None: keep row index 2 (info="z")
    expected_info = ["y", "z", "w"]
    assert result["info"].tolist() == expected_info

def test_output_index_reset():
    df = pd.DataFrame({
        "event_id": [1, 2, 2],
        "updated_at": ["2022-01-01", "2022-01-01", "2022-01-02"],
        "value": [10, 20, 30],
    })
    result = deduplicate_events(df)
    # expected rows: id=1 (value=10), id=2 (value=30)
    assert result.index.tolist() == [0, 1]

def test_order_of_output_matches_original_order():
    # rows: null, id=2 old, id=1 new, id=2 new, null, id=1 old
    df = pd.DataFrame({
        "event_id": [None, 2, 1, 2, None, 1],
        "updated_at": [
            "2022-01-10",
            "2022-01-01",
            "2022-01-05",
            "2022-01-03",
            "2022-01-06",
            "2022-01-01",
        ],
        "value": [1, 2, 3, 4, 5, 6],
    })
    result = deduplicate_events(df)
    # expected:
    # null (value=1, index0) kept
    # id=2: rows 1 and 3 -> keep most recent 2022-01-03 (value=4)
    # id=1: rows 2 and 5 -> keep most recent 2022-01-05 (value=3)
    # null (value=5, index4) kept
    # final order by original order: 0,1? Actually after dedup, rows sorted by _original_order. So order: null1, id2(value=4, orig_idx=3), id1(value=3, orig_idx=2?), wait original order: 0 null1, 1 id2 old, 2 id1 new, 3 id2 new, 4 null5, 5 id1 old.
    # After dedup: keep null1 (orig0), id2 new (orig3), id1 new (orig2), null5 (orig4). Sorted by orig: 0,2,3,4 -> values: 1,3,4,5.
    expected_values = [1, 3, 4, 5]
    assert result["value"].tolist() == expected_values

def test_multiple_ids_with_null_segment():
    df = pd.DataFrame({
        "event_id": [None, None, 1, 1, None],
        "updated_at": [
            "2022-01-01",
            "2022-01-02",
            "2022-01-03",
            "2022-01-04",
            "2022-01-05",
        ],
        "data": ["A", "B", "C", "D", "E"],
    })
    result = deduplicate_events(df)
    # null rows preserved; id=1 dedup to keep D (most recent)
    assert result["data"].tolist() == ["A", "B", "D", "E"]
