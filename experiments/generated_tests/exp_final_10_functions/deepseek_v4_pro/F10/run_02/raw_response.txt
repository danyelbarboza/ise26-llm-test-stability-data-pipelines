import pandas as pd
import pytest
from ise26.targets import standardize_currency_values


def test_basic_brazilian_format():
    df = pd.DataFrame({"raw": ["1.234,56"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == 1234.56
    assert result["amount"].dtype == "Float64"


def test_basic_english_format():
    df = pd.DataFrame({"raw": ["1,234.56"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == 1234.56
    assert result["amount"].dtype == "Float64"


def test_removes_r_prefix_space():
    df = pd.DataFrame({"raw": ["R$ 100.00"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == 100.0


def test_removes_r_prefix_no_space():
    df = pd.DataFrame({"raw": ["R$100"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == 100.0


def test_handles_whitespace():
    df = pd.DataFrame({"raw": ["   200.75   "]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == 200.75


def test_invalid_text_returns_na():
    df = pd.DataFrame({"raw": ["abc", "", "12..34"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert pd.isna(result.loc[0, "amount"])
    assert pd.isna(result.loc[1, "amount"])
    assert pd.isna(result.loc[2, "amount"])


def test_none_value_returns_na():
    df = pd.DataFrame({"raw": [None]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert pd.isna(result.loc[0, "amount"])


def test_mixed_valid_invalid():
    df = pd.DataFrame({"raw": ["1.000,00", "invalid", "R$ 50.99", ""]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = [1000.0, pd.NA, 50.99, pd.NA]
    for idx, exp in enumerate(expected):
        if pd.isna(exp):
            assert pd.isna(result.loc[idx, "amount"])
        else:
            assert result.loc[idx, "amount"] == exp


def test_original_column_unchanged():
    original_data = ["R$ 1.234,56", "abc"]
    df = pd.DataFrame({"raw": original_data.copy()})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result["raw"].tolist() == original_data
    assert df["raw"].tolist() == original_data  # source df untouched?


def test_output_dtype_nullable_float():
    df = pd.DataFrame({"raw": ["100"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result["amount"].dtype == "Float64"


def test_returns_new_dataframe():
    df = pd.DataFrame({"raw": ["10"]})
    result = standardize_currency_values(df)
    assert result is not df


def test_custom_column_names():
    df = pd.DataFrame({"val": ["250.50"]})
    result = standardize_currency_values(df, value_col="val", output_col="num")
    assert "num" in result.columns
    assert result.loc[0, "num"] == 250.50


def test_brazilian_thousands_and_decimal():
    df = pd.DataFrame({"raw": ["1.234.567,89"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == 1234567.89


def test_negative_value():
    df = pd.DataFrame({"raw": ["-1.000,50"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.loc[0, "amount"] == -1000.5


def test_missing_column_raises():
    df = pd.DataFrame({"other": ["100"]})
    with pytest.raises(Exception):
        standardize_currency_values(df, value_col="raw", output_col="amount")


def test_empty_dataframe():
    df = pd.DataFrame({"raw": []}, dtype="object")
    result = standardize_currency_values(df)
    assert result.empty
    assert "amount" in result.columns
    assert result["amount"].dtype == "Float64"
