from ise26.targets import standardize_currency_values

import pandas as pd
import pytest


class TestStandardizeCurrencyValues:

    def test_basic_english_format(self):
        df = pd.DataFrame({"amount_raw": ["R$ 1,234.56", "R$ 7,890.01"]})
        result = standardize_currency_values(df)
        expected = pd.Series([1234.56, 7890.01], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)
        assert "amount_raw" in result.columns
        assert df["amount_raw"].equals(result["amount_raw"])  # verificacao extra de copia

    def test_brazilian_format(self):
        df = pd.DataFrame({"amount_raw": ["R$ 1.234,56", "R$ 5.000,75"]})
        result = standardize_currency_values(df)
        expected = pd.Series([1234.56, 5000.75], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_mixed_separators_last_dot_wins(self):
        # Quando ha multiplos separadores, o ultimo e considerado decimal
        df = pd.DataFrame({"amount_raw": ["1,234.56", "1.234,56", "1234.56"]})
        result = standardize_currency_values(df)
        expected = pd.Series([1234.56, 1234.56, 1234.56], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_integer_no_decimals(self):
        df = pd.DataFrame({"amount_raw": ["R$ 1500", "R$ 99"]})
        result = standardize_currency_values(df)
        expected = pd.Series([1500.0, 99.0], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_negative_values(self):
        df = pd.DataFrame({"amount_raw": ["-R$ 100.00", "-R$ 2.500,30", "-150,75"]})
        result = standardize_currency_values(df)
        expected = pd.Series([-100.0, -2500.30, -150.75], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_no_currency_prefix(self):
        df = pd.DataFrame({"amount_raw": ["100.00", "1,250.50", "500"]})
        result = standardize_currency_values(df)
        expected = pd.Series([100.0, 1250.50, 500.0], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_whitespace_and_extra_spaces(self):
        df = pd.DataFrame({"amount_raw": ["  R$  10,000.00  ", "\tR$ 5.50\n"]})
        result = standardize_currency_values(df)
        expected = pd.Series([10000.0, 5.5], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_zero_cases(self):
        df = pd.DataFrame({"amount_raw": ["R$ 0.00", "0", "0,00", "0.000"]})
        result = standardize_currency_values(df)
        expected = pd.Series([0.0, 0.0, 0.0, 0.0], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_invalid_values_become_na(self):
        df = pd.DataFrame({"amount_raw": ["abc", "R$ ", "", None, "1,2,3", "1..2", "R$ X"]})
        result = standardize_currency_values(df)
        expected = pd.Series([pd.NA]*7, dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)
        # Verificar que os valores originais permanecem
        assert df["amount_raw"].tolist() == ["abc", "R$ ", "", None, "1,2,3", "1..2", "R$ X"]

    def test_missing_value_none_and_nan(self):
        df = pd.DataFrame({"amount_raw": ["10.00", None, "20.00", float("nan")]})
        # Nota: float("nan") vira NaN em pandas, que e enviado como str? Provavelmente sera tratado como invalido
        result = standardize_currency_values(df)
        expected = pd.Series([10.0, pd.NA, 20.0, pd.NA], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_output_column_name_custom(self):
        df = pd.DataFrame({"valor": ["R$ 50,50"]})
        result = standardize_currency_values(df, value_col="valor", output_col="limpo")
        assert "limpo" in result.columns
        assert result["limpo"].iloc[0] == 50.50

    def test_output_dtype_is_float64(self):
        df = pd.DataFrame({"amount_raw": ["R$ 123.45"]})
        result = standardize_currency_values(df)
        assert result["amount"].dtype == "Float64"

    def test_original_dataframe_unchanged(self):
        original = pd.DataFrame({"amount_raw": ["R$ 10.00", "abc"]})
        copy_original = original.copy()
        result = standardize_currency_values(original)
        # DataFrame original nao deve conter a coluna de saida
        assert "amount" not in original.columns
        # Os dados da coluna original nao foram alterados
        assert original["amount_raw"].equals(copy_original["amount_raw"])
        # Resultado e um novo objeto
        assert result is not original

    def test_missing_required_column_raises(self):
        df = pd.DataFrame({"outra_coluna": ["x"]})
        with pytest.raises(ValueError):  # _ensure_columns_present levanta ValueError
            standardize_currency_values(df, value_col="amount_raw")

    def test_large_numbers(self):
        df = pd.DataFrame({"amount_raw": ["R$ 9,999,999,999.99", "R$ 1.000.000.000,00"]})
        result = standardize_currency_values(df)
        expected = pd.Series([9999999999.99, 1000000000.00], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_single_comma_decimal(self):
        df = pd.DataFrame({"amount_raw": ["R$ 12,5", "9,99"]})
        result = standardize_currency_values(df)
        expected = pd.Series([12.5, 9.99], dtype="Float64", name="amount")
        pd.testing.assert_series_equal(result["amount"], expected)

    def test_value_with_currency_in_middle(self):
        # Nao especificado mas o stripping de prefixo remove "R$" lider, nao em qualquer lugar
        df = pd.DataFrame({"amount_raw": ["xR$ 10.00"]})
        result = standardize_currency_values(df)
        # Provavelmente nao e reconhecido, vira NA
        assert pd.isna(result["amount"].iloc[0])

    def test_preserve_index(self):
        df = pd.DataFrame({"amount_raw": ["10.00", "20.00"]}, index=[5, 7])
        result = standardize_currency_values(df)
        assert result.index.tolist() == [5, 7]
