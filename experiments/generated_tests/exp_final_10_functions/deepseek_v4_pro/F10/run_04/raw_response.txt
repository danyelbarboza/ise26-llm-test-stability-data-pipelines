from ise26.targets import standardize_currency_values

import pytest
import pandas as pd
import numpy as np


def test_brazilian_format():
    df = pd.DataFrame({"amount_raw": ["1.234,56", "0,99", "10.000,00"]})
    result = standardize_currency_values(df)
    expected = pd.array([1234.56, 0.99, 10000.0], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected)


def test_english_format():
    df = pd.DataFrame({"amount_raw": ["1,234.56", "0.99", "10,000.00"]})
    result = standardize_currency_values(df)
    expected = pd.array([1234.56, 0.99, 10000.0], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected)


def test_rs_prefix_is_removed():
    df = pd.DataFrame({"amount_raw": ["R$ 1,234.56", "R$1.234,56", "R$0.99"]})
    result = standardize_currency_values(df)
    expected = pd.array([1234.56, 1234.56, 0.99], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected)


def test_whitespace_trimmed():
    df = pd.DataFrame({"amount_raw": ["   1,234.56  ", " 1.234,56 "]})
    result = standardize_currency_values(df)
    expected = pd.array([1234.56, 1234.56], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected)


def test_invalid_string_returns_na():
    df = pd.DataFrame({"amount_raw": ["invalid", "abc", "R$--", "123,abc"]})
    result = standardize_currency_values(df)
    assert result["amount"].isna().all()


def test_none_input_returns_na():
    df = pd.DataFrame({"amount_raw": [None, "1,234.56"]})
    result = standardize_currency_values(df)
    assert pd.isna(result.loc[0, "amount"])
    assert result.loc[1, "amount"] == 1234.56


def test_nan_input_returns_na():
    df = pd.DataFrame({"amount_raw": [np.nan, "100.00"]})
    result = standardize_currency_values(df)
    assert pd.isna(result.loc[0, "amount"])
    assert result.loc[1, "amount"] == 100.00


def test_empty_string_returns_na():
    df = pd.DataFrame({"amount_raw": ["", "1.50"]})
    result = standardize_currency_values(df)
    assert pd.isna(result.loc[0, "amount"])
    assert result.loc[1, "amount"] == 1.5


def test_preserves_original_column():
    original_data = ["R$ 1,234.56", "1.234,56"]
    df = pd.DataFrame({"amount_raw": original_data})
    result = standardize_currency_values(df)
    assert list(result["amount_raw"]) == original_data


def test_returns_new_dataframe():
    df = pd.DataFrame({"amount_raw": ["10.00"]})
    result = standardize_currency_values(df)
    assert result is not df


def test_output_dtype_is_nullable_float():
    df = pd.DataFrame({"amount_raw": ["1.50", "2.000,00"]})
    result = standardize_currency_values(df)
    assert result["amount"].dtype == pd.Float64Dtype()


def test_custom_column_names():
    df = pd.DataFrame({"raw": ["1,234.56"], "other": [1]})
    result = standardize_currency_values(df, value_col="raw", output_col="clean")
    assert "clean" in result.columns
    assert result["clean"].iloc[0] == 1234.56
    assert result.loc[0, "other"] == 1


def test_mixed_valid_and_invalid():
    df = pd.DataFrame({"amount_raw": ["1,234.56", "invalid", "R$ 0.99", None, "   "]})
    result = standardize_currency_values(df)
    expected_values = pd.array([1234.56, pd.NA, 0.99, pd.NA, pd.NA], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected_values)


def test_plain_number_string_parsed():
    df = pd.DataFrame({"amount_raw": ["1234", "0"]})
    result = standardize_currency_values(df)
    expected = pd.array([1234.0, 0.0], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected)


def test_missing_value_col_raises():
    df = pd.DataFrame({"other_col": ["1.50"]})
    # _ensure_columns_present raises KeyError for missing column
    with pytest.raises(KeyError):
        standardize_currency_values(df, value_col="missing")


def test_empty_dataframe():
    df = pd.DataFrame({"amount_raw": pd.Series([], dtype=str)})
    result = standardize_currency_values(df)
    assert len(result) == 0
    assert "amount" in result.columns
    assert result["amount"].dtype == pd.Float64Dtype()


def test_negative_amount():
    df = pd.DataFrame({"amount_raw": ["-1,234.56", "-R$ 500.00"]})
    result = standardize_currency_values(df)
    expected = pd.array([-1234.56, -500.0], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"], expected)


def test_boolean_column_unchanged():
    df = pd.DataFrame({"amount_raw": ["10.00", "20.00"], "flag": [True, False]})
    result = standardize_currency_values(df)
    assert list(result["flag"]) == [True, False]
