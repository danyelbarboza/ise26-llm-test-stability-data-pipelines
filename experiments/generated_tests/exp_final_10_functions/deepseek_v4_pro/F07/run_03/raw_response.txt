from ise26.targets import parse_order_items_json

import pandas as pd
import pytest


def test_basic_valid_payload():
    """A simple valid JSON array with all fields present yields correct rows."""
    df = pd.DataFrame({
        "order_id": [101, 102],
        "items_json": [
            '[{"sku": "A", "quantity": 2, "unit_price": 10.5}]',
            '[{"sku": "B", "quantity": "3", "unit_price": "20.0"}]',
        ],
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [101, 102],
        "sku": ["A", "B"],
        "quantity": [2.0, 3.0],
        "unit_price": [10.5, 20.0],
        "item_total": [21.0, 60.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_missing_sku_becomes_na():
    """When the 'sku' key is missing, sku becomes pd.NA."""
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['[{"quantity": 1, "unit_price": 5.0}]'],
    })
    result = parse_order_items_json(df)
    assert result.loc[0, "sku"] is pd.NA
    # other columns fine
    assert result.loc[0, "quantity"] == 1.0
    assert result.loc[0, "unit_price"] == 5.0
    assert result.loc[0, "item_total"] == 5.0


def test_sku_none_value_becomes_na():

def test_non_list_payload_is_skipped():
    """JSON that is not a list (e.g., object, string, number) is skipped entirely."""
    # Object
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['{"sku": "A", "quantity": 1, "unit_price": 2.0}'],
    })
    result = parse_order_items_json(df)
    assert len(result) == 0

    # String
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['"string_not_list"'],
    })
    result = parse_order_items_json(df)
    assert len(result) == 0

    # Number
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['123'],
    })
    result = parse_order_items_json(df)
    assert len(result) == 0


def test_empty_list_payload_skipped():

    })
    result = parse_order_items_json(df)

    """None or NaN in json_col is skipped."""
    df = pd.DataFrame({

        "items_json": [None, float("nan")],
    })
    result = parse_order_items_json(df)
    assert len(result) == 0


def test_mixed_valid_and_invalid_rows():
    """Only valid list payloads are processed, invalid ones are omitted."""
    df = pd.DataFrame({
        "order_id": [1, 2, 3, 4],
        "items_json": [
            '[{"sku": "A", "quantity": 1, "unit_price": 5.0}]',  # valid
            'not_json',                                             # invalid JSON (should skip)
            '[{"sku": "B", "quantity": 2, "unit_price": 10.0}]',  # valid
            '{"x": 1}',                                             # object, skip
        ],
    })
    result = parse_order_items_json(df)
    expected_order_ids = [1, 3]  # only orders 1 and 3 produce rows
    assert list(result["order_id"]) == expected_order_ids
    assert list(result["sku"]) == ["A", "B"]


def test_output_columns_when_no_rows():
    """When no valid rows, output empty DataFrame with correct columns and dtypes."""
    df = pd.DataFrame({"order_id": [], "items_json": []})
    result = parse_order_items_json(df)
    assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
    assert len(result) == 0


def test_default_column_names():
    """If json_col and order_id_col are not specified, defaults are used."""
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['[{"sku": "X", "quantity": 1, "unit_price": 10.0}]'],
    })
    result = parse_order_items_json(df)  # uses defaults
    assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
    assert result.loc[0, "order_id"] == 1


def test_custom_order_id_column_name_preserved():
    """When order_id_col differs, output uses that exact name."""
    df = pd.DataFrame({
        "purchase_id": [5],
        "items_json": ['[{"sku": "Y", "quantity": 4, "unit_price": 2.0}]'],
    })
    result = parse_order_items_json(df, order_id_col="purchase_id")
    assert "purchase_id" in result.columns
    assert "order_id" not in result.columns  # default no longer present
    assert list(result.columns) == ["purchase_id", "sku", "quantity", "unit_price", "item_total"]
    assert result.loc[0, "purchase_id"] == 5


def test_custom_json_column_name():
    """When json_col is different, it is used."""
    df = pd.DataFrame({
        "order_id": [1],
        "items": ['[{"sku": "Z", "quantity": 1, "unit_price": 99.0}]'],
    })
    result = parse_order_items_json(df, json_col="items")
    assert result.loc[0, "sku"] == "Z"


def test_missing_required_columns_raises():
    """If json_col or order_id_col is missing from DataFrame, raises exception."""
    df = pd.DataFrame({"x": [1]})
    with pytest.raises(Exception):  # either KeyError or ValueError from internal check
        parse_order_items_json(df)


def test_preserves_order_of_items_within_list():
    """Items from the same list must appear in their original JSON order."""
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['[{"sku": "first", "quantity": 1, "unit_price": 1}, {"sku": "second", "quantity": 2, "unit_price": 2}]'],
    })
    result = parse_order_items_json(df)
    assert list(result["sku"]) == ["first", "second"]


def test_multiple_orders_preserve_row_order():
    """Output rows are sorted by the order of the source rows then by item order."""
    df = pd.DataFrame({
        "order_id": [2, 1],
        "items_json": [
            '[{"sku": "B1", "quantity": 1, "unit_price": 1}, {"sku": "B2", "quantity": 2, "unit_price": 2}]',
            '[{"sku": "A1", "quantity": 3, "unit_price": 3}]',
        ],
    })
    result = parse_order_items_json(df)
    # source order: first row (order_id=2) then second row (order_id=1)
    assert list(result["order_id"]) == [2, 2, 1]
    assert list(result["sku"]) == ["B1", "B2", "A1"]
