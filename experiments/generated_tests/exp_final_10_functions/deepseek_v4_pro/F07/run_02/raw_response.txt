import pytest
import pandas as pd
from ise26.targets import parse_order_items_json


class TestParseOrderItemsJson:
    """Tests for parse_order_items_json, respecting its contract and edge cases."""

    def test_basic_explosion(self):
        """A typical valid JSON list of items should expand to one row per item."""
        df = pd.DataFrame({
            "items_json": ['[{"sku": "A", "quantity": 2, "unit_price": 10.5}]'],
            "order_id": [1]
        })
        result = parse_order_items_json(df)

        expected = pd.DataFrame({
            "order_id": pd.Series([1], dtype="int64"),
            "sku": pd.Series(["A"], dtype="object"),
            "quantity": pd.Series([2.0], dtype="float64"),
            "unit_price": pd.Series([10.5], dtype="float64"),
            "item_total": pd.Series([21.0], dtype="float64"),
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_sku_becomes_null(self):
        """Missing 'sku' key or None should become pd.NA."""
        df = pd.DataFrame({
            "items_json": ['[{"quantity": 1, "unit_price": 5}]',
                           '[{"sku": null, "quantity": 2, "unit_price": 6}]'],
            "order_id": [1, 2]
        })
        result = parse_order_items_json(df)
        assert pd.isna(result.loc[0, "sku"])  # first row missing key
        assert pd.isna(result.loc[1, "sku"])  # second row explicit null
        # Both should be pd.NA, dtype should be object
        assert result["sku"].dtype == "object"

    def test_invalid_quantity_and_unit_price_become_zero(self):
        """Non‑numeric quantity or unit_price must be converted to zero."""
        df = pd.DataFrame({
            "items_json": ['[{"sku":"X","quantity":"abc","unit_price":"def"},
                             {"sku":"Y","quantity":3,"unit_price":null},
                             {"sku":"Z","quantity":null,"unit_price":4.5}]'],
            "order_id": [100]
        })
        result = parse_order_items_json(df)

        # Row 0: both invalid → 0.0, 0.0 => item_total 0
        assert result.loc[0, "quantity"] == 0.0
        assert result.loc[0, "unit_price"] == 0.0
        assert result.loc[0, "item_total"] == 0.0

        # Row 1: quantity valid, unit_price null → 0.0, so item_total 0
        assert result.loc[1, "quantity"] == 3.0
        assert result.loc[1, "unit_price"] == 0.0
        assert result.loc[1, "item_total"] == 0.0

        # Row 2: quantity null → 0.0, unit_price 4.5 -> item_total 0
        assert result.loc[2, "quantity"] == 0.0
        assert result.loc[2, "unit_price"] == 4.5
        assert result.loc[2, "item_total"] == 0.0

    def test_empty_json_list_skipped(self):
        """An empty JSON list produces no output rows."""
        df = pd.DataFrame({
            "items_json": ["[]", '[]'],
            "order_id": [1, 2]
        })
        result = parse_order_items_json(df)
        assert len(result) == 0
        # Column schema must still be present
        assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]

    def test_missing_json_payload_skipped(self):
        """None, NaN, or non‑string payloads are skipped."""
        df = pd.DataFrame({
            "items_json": [None, float("nan"), 123, '{"sku":"a"}'],
            "order_id": [1, 2, 3, 4]
        })
        result = parse_order_items_json(df)
        # Only the last entry is a JSON list? No, it's a JSON object, not a list.
        # So all should be skipped.
        assert len(result) == 0

    def test_non_list_json_payload_skipped(self):
        """JSON objects, numbers, strings, invalid JSON are skipped."""
        df = pd.DataFrame({
            "items_json": ['{"sku":"X"}', "42", "not json", "true"],
            "order_id": [10, 20, 30, 40]
        })
        result = parse_order_items_json(df)
        assert len(result) == 0

    def test_custom_column_names(self):
        """Custom json_col and order_id_col should be respected in output."""
        df = pd.DataFrame({
            "json_data": ['[{"sku":"B","quantity":1,"unit_price":2.0}]'],
            "order_ref": ["ORD-001"]
        })
        result = parse_order_items_json(df, json_col="json_data", order_id_col="order_ref")
        assert "order_ref" in result.columns
        assert result["order_ref"].iloc[0] == "ORD-001"
        # order_id column should not exist
        assert "order_id" not in result.columns

    def test_order_preservation(self):
        """
        Output rows must be ordered by the original DataFrame row,
        then by the position inside the JSON list.
        """
        df = pd.DataFrame({
            "items_json": [
                '[{"sku":"C","quantity":1},{"sku":"D","quantity":2}]',
                '[{"sku":"E","quantity":3}]',
                '[{"sku":"F","quantity":4},{"sku":"G","quantity":5}]'
            ],
            "order_id": [1, 2, 3]
        })
        result = parse_order_items_json(df)
        # Expected order: row 0: C, D; row 1: E; row 2: F, G
        expected_skus = ["C", "D", "E", "F", "G"]
        assert result["sku"].tolist() == expected_skus

    def test_empty_dataframe(self):
        """An empty DataFrame returns an empty result with correct schema."""
        df = pd.DataFrame({"items_json": pd.Series(dtype="object"),
                           "order_id": pd.Series(dtype="int64")})
        result = parse_order_items_json(df)
        assert len(result) == 0
        assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
        # Check dtypes of an empty frame
        assert result["order_id"].dtype == "int64"
        assert result["sku"].dtype == "object"
        assert result["quantity"].dtype == "float64"
        assert result["unit_price"].dtype == "float64"
        assert result["item_total"].dtype == "float64"

    def test_missing_required_column_raises(self):
        """Missing json_col or order_id_col must raise an error."""
        df = pd.DataFrame({"order_id": [1]})  # items_json missing
        with pytest.raises(KeyError):
            parse_order_items_json(df)

        df2 = pd.DataFrame({"items_json": ['[{"sku":"a"}]']})  # order_id missing
        with pytest.raises(KeyError):
            parse_order_items_json(df2)

    def test_output_dtypes(self):
        """Verify exact dtypes of the returned columns."""
        df = pd.DataFrame({
            "items_json": ['[{"sku":"H","quantity":2,"unit_price":3.5}]'],
            "order_id": pd.Series([42], dtype="int32")  # custom dtype
        })
        result = parse_order_items_json(df)
        assert result["order_id"].dtype == "int32"          # preserved from input
        assert result["sku"].dtype == "object"
        assert result["quantity"].dtype == "float64"
        assert result["unit_price"].dtype == "float64"
        assert result["item_total"].dtype == "float64"

        df = pd.DataFrame({
            "items_json": ['[{"sku":"I","quantity":3,"unit_price":7.5},
                             {"sku":"J","quantity":0,"unit_price":10},
                             {"sku
