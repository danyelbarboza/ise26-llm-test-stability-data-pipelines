import pandas as pd
import pytest
from ise26.targets import classify_payment_status


def test_invalid_amount_missing():
    df = pd.DataFrame({
        "due_date": ["2023-01-01"],
        "paid_date": [None],
        "amount": [None],
    })
    result = classify_payment_status(df, "2023-06-01")
    assert result["payment_status"].tolist() == ["invalid"]

    })
    result = classify_payment_status(df, "2023-06-15")
    assert result["payment_status"].tolist() == ["overdue"]

def test_mixed_scenarios_preserves_order():
    df = pd.DataFrame({
        "due_date": ["2023-01-15", "2023-02-01", "not-a-date", "2023-03-01"],
        "paid_date": [None, "2023-02-05", None, "2023-02-28"],
        "amount": [-10, 200, 100, 300],
    })
    expected = ["invalid", "paid_late", "invalid", "paid_on_time"]
    result = classify_payment_status(df, "2023-04-01")
    assert result["payment_status"].tolist() == expected
    # Check original columns present and order unchanged
    assert list(result.columns) == list(df.columns) + ["payment_status"]
    assert result.index.tolist() == df.index.tolist()

def test_invalid_reference_date_raises_valueerror():
    df = pd.DataFrame({
        "due_date": ["2023-01-01"],
        "paid_date": [None],
        "amount": [100],
    })
    with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
        classify_payment_status(df, "not-a-date")


def test_output_column_name_custom():
    df = pd.DataFrame({
        "due_date": ["2023-01-01"],
        "paid_date": [None],
        "amount": [100],
    })
    result = classify_payment_status(df, "2023-02-01", output_col="status")
    assert "status" in result.columns
    assert "payment_status" not in result.columns
    assert result["status"].tolist() == ["overdue"]


def test_dtype_of_status_column():
    df = pd.DataFrame({
        "due_date": ["2023-01-01"],
        "paid_date": ["2023-01-01"],
        "amount": [50],
    })
    result = classify_payment_status(df, "2023-01-01")
    # The status column should contain strings, but dtype may be object
    assert result["payment_status"].dtype == object
    assert isinstance(result["payment_status"].iloc[0], str)
