import pandas as pd
import pytest
from datetime import date, datetime
from ise26.targets import classify_payment_status


def test_raises_value_error_on_invalid_reference_date():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100.0],
    })
    with pytest.raises(ValueError):
        classify_payment_status(df, "not-a-date")


def test_valid_reference_date_accepts_various_types():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100.0],
    })
    # string
    result = classify_payment_status(df, "2023-02-01")
    assert result["payment_status"].iloc[0] == "paid_on_time"
    # datetime.date
    result = classify_payment_status(df, date(2023, 2, 1))
    assert result["payment_status"].iloc[0] == "paid_on_time"
    # pd.Timestamp
    result = classify_payment_status(df, pd.Timestamp("2023-02-01"))
    assert result["payment_status"].iloc[0] == "paid_on_time"


def test_invalid_amount_missing_or_non_numeric():
    # missing (None/NaN)
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [None],
    })
    result = classify_payment_status(df, "2023-01-20")
    assert result["payment_status"].iloc[0] == "invalid"

    # non-numeric string
    df["amount"] = ["abc"]
    result = classify_payment_status(df, "2023-01-20")
    assert result["payment_status"].iloc[0] == "invalid"

def test_invalid_paid_date_provided():
    # provided but unparseable
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["bad-date"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, "2023-01-20")
    assert result["payment_status"].iloc[0] == "invalid"

    # empty string treated as missing (not invalid)
    df["paid_date"] = [""]
    result = classify_payment_status(df, "2023-02-01")  # ref > due -> overdue
    assert result["payment_status"].iloc[0] == "overdue"


def test_paid_on_time():

def test_mixed_scenarios():
    df = pd.DataFrame({
        "due_date": ["2023-01-10", "2023-01-20", "2023-02-01", "2023-02-10", "2023-02-15", "2023-02-20"],
        "paid_date": ["2023-01-05", "2023-01-25", None, "bad", "", "2023-02-20"],
        "amount": [10, 20, 30, 0, 40, -5],
    })
    # row0: paid on time; row1: paid late; row2: unpaid overdue? ref 2023-02-01 > due 2023-02-01?? equal -> not >, so pending. Let's set ref to 2023-02-02 to make overdue for row2.
    # We'll sort: reference_date = "2023-02-02"
    # row0: due 2023-01-10, paid 2023-01-05 -> paid_on_time
    # row1: due 2023-01-20, paid 2023-01-25 -> paid_late
    # row2: due 2023-02-01, paid None -> ref 2023-02-02 > due -> overdue
    # row3: amount 0 -> invalid
    # row4: paid "" (empty, missing) -> unpaid, due 2023-02-15, ref 2023-02-02 <= due -> pending
    # row5: amount -5 -> invalid
    result = classify_payment_status(df, "2023-02-02")
    expected = ["paid_on_time", "paid_late", "overdue", "invalid", "pending", "invalid"]
    assert result["payment_status"].tolist() == expected


def test_empty_dataframe():
    df = pd.DataFrame(columns=["due_date", "paid_date", "amount"])
    result = classify_payment_status(df, "2023-01-01")
    assert result.empty
    assert "payment_status" in result.columns


def test_missing_columns_raises_error():
    df = pd.DataFrame({"due_date": ["2023-01-01"]})
    with pytest.raises(KeyError):
        classify_payment_status(df, "2023-01-01")
    df = pd.DataFrame({"paid_date": ["2023-01-01"]})
    with pytest.raises(KeyError):
        classify_payment_status(df, "2023-01-01", due_date_col="due_date")
    # missing amount column
    df = pd.DataFrame({"due_date": ["2023-01-01"], "paid_date": ["2023-01-01"]})
    with pytest.raises(KeyError):
        classify_payment_status(df, "2023-01-01")


def test_output_column_name_default_and_custom():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, "2023-01-20")
    assert "payment_status" in result.columns

    result_custom = classify_payment_status(df, "2023-01-20", output_col="status")
    assert "status" in result_custom.columns
    assert "payment_status" not in result_custom.columns


def test_original_dataframe_immutable():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100.0],
    })
    original_cols = df.columns.tolist()
    _ = classify_payment_status(df, "2023-01-20")
    assert df.columns.tolist() == original_cols
    assert "payment_status" not in df.columns


def test_preserves_index_and_order():
    df = pd.DataFrame({
        "due_date": ["2023-01-15", "2023-02-15", "2023-03-15"],
        "paid_date": ["2023-01-10", None, "2023-04-01"],
        "amount": [100, 200, 300],
    }, index=[10, 20, 30])
    result = classify_payment_status(df, "2023-02-01")
    assert result.index.tolist() == [10, 20, 30]
    # row order preserved
    assert result["payment_status"].tolist() == ["paid_on_time", "pending", "paid_late"]
