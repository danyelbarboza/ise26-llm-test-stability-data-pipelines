import pytest
import pandas as pd
import numpy as np
from datetime import date, datetime
from ise26.targets import classify_payment_status


class TestClassifyPaymentStatus:
    @pytest.fixture
    def basic_df(self):
        """Minimal valid DataFrame with one row per classification."""
        df = pd.DataFrame({
            "due_date": pd.date_range("2024-01-15", periods=4, freq="MS"),
            "paid_date": [
                "2024-01-15",  # on time
                "2024-02-02",  # late
                None,          # unpaid (pending)
                None           # unpaid (overdue)
            ],
            "amount": [100, 200, 300, 400]
        })
        return df

    def test_valid_combinations(self, basic_df):
        reference_date = "2024-02-01"
        result = classify_payment_status(basic_df, reference_date)

        expected_statuses = ["paid_on_time", "paid_late", "pending", "overdue"]
        assert list(result["payment_status"]) == expected_statuses

    def test_invalid_due_date(self):
        df = pd.DataFrame({
            "due_date": ["not-a-date", "2024-01-01"],
            "paid_date": [None, "2024-01-01"],
            "amount": [100, 100]
        })
        result = classify_payment_status(df, "2024-02-01")
        assert result["payment_status"].tolist() == ["invalid", "paid_on_time"]

    def test_invalid_amount(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-15", "2024-01-15", "2024-01-15", "2024-01-15"],
            "paid_date": [None, "2024-01-10", "2024-01-10", "2024-01-10"],
            "amount": [None, "abc", 0, -50]
        })
        result = classify_payment_status(df, "2024-02-01")
        assert all(status == "invalid" for status in result["payment_status"])

    def test_invalid_paid_date_provided(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-15"],
            "paid_date": ["2023-13-01"],  # invalid date
            "amount": [100]
        })
        result = classify_payment_status(df, "2024-02-01")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_paid_date_not_provided(self):
        """Missing paid date (None, empty string, whitespace) -> unpaid."""
        df = pd.DataFrame({
            "due_date": ["2024-01-15", "2024-01-15", "2024-01-15"],
            "paid_date": [None, "", "   "],
            "amount": [100, 100, 100]
        })
        # reference_date > due_date -> overdue
        result_overdue = classify_payment_status(df, "2024-02-01")
        assert result_overdue["payment_status"].tolist() == ["overdue", "overdue", "overdue"]

        # reference_date <= due_date -> pending
        result_pending = classify_payment_status(df, "2024-01-15")
        assert result_pending["payment_status"].tolist() == ["pending", "pending", "pending"]

    def test_reference_date_invalid_raises(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-15"],
            "paid_date": [None],
            "amount": [100]
        })
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, "not-a-date")

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "my_due": ["2024-01-15", "2024-01-15"],
            "my_paid": ["2024-01-10", "2024-01-20"],
            "my_amount": [100, 200]
        })
        result = classify_payment_status(
            df,
            "2024-02-01",
            due_date_col="my_due",
            paid_date_col="my_paid",
            amount_col="my_amount",
            output_col="custom_status"
        )
        assert list(result.columns) == ["my_due", "my_paid", "my_amount", "custom_status"]
        assert result["custom_status"].tolist() == ["paid_on_time", "paid_late"]

    def test_preserves_original_order(self):
        # Shuffle rows to check order is preserved
        original = pd.DataFrame({
            "due_date": ["2024-03-01", "2024-01-15", "2024-02-01"],
            "paid_date": [None, "2024-01-15", None],
            "amount": [100, 200, 300]
        })
        result = classify_payment_status(original, "2024-03-02")
        # Order must match
        assert result.index.tolist() == original.index.tolist()
        # Check status order
        assert result["payment_status"].tolist() == ["pending", "paid_on_time", "overdue"]

    def test_dataframe_not_modified(self, basic_df):
        original_copy = basic_df.copy(deep=True)
        classify_payment_status(basic_df, "2024-02-01")
        pd.testing.assert_frame_equal(basic_df, original_copy)

    def test_edge_dates_exact_comparison(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-15"],
            "paid_date": ["2024-01-15"],  # exactly equal -> on time
            "amount": [100]
        })
        result = classify_payment_status(df, "2024-02-01")
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_amount_infinity_edge(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-15", "2024-01-15"],
            "paid_date": ["2024-01-10", "2024-01-10"],
            "amount": [float("inf"), float("-inf")]
        })
        result = classify_payment_status(df, "2024-02-01")
        # inf is >0 and numeric -> valid, -inf is <=0 -> invalid
        assert result["payment_status"].tolist() == ["paid_on_time", "invalid"]

    def test_result_column_dtype(self, basic_df):
        result = classify_payment_status(basic_df, "2024-02-01")
        assert result["payment_status"].dtype == object

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["due_date", "paid_date", "amount"])
        result = classify_payment_status(df, "2024-02-01")
        assert len(result) == 0
        assert "payment_status" in result.columns

    def test_missing_required_column_raises(self):
        df = pd.DataFrame({
            "due_date": ["2024-01-15"],
            "paid_date": [None]
            # missing amount
        })
        # The internal function should raise an error (likely KeyError)
        with pytest.raises(Exception):
            classify_payment_status(df, "2024-02-01")
