import pytest
import pandas as pd
import numpy as np
from ise26.targets import join_customers_orders


def test_basic_join_all_statuses():
    customers = pd.DataFrame({
        "customer_id": [1, 2, 3, 4],
        "customer_name": ["Alice", "Bob", "Charlie", "David"]
    })
    orders = pd.DataFrame({
        "customer_id": [2, 3, 5, 6],
        "order_amount": [100, 200, 300, 400]
    })
    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame({
        "customer_id": [1, 2, 3, 4, 5, 6],
        "customer_name": ["Alice", "Bob", "Charlie", "David", pd.NA, pd.NA],
        "order_amount": [pd.NA, 100, 200, pd.NA, 300, 400],
        "record_status": [
            "customer_without_order",
            "matched",
            "matched",
            "customer_without_order",
            "order_without_customer",
            "order_without_customer",
        ]
    })

    # compare ignoring row order (test content, determinism tested separately)
    result_sorted = result.sort_values("customer_id").reset_index(drop=True)
    expected_sorted = expected.sort_values("customer_id").reset_index(drop=True)
    pd.testing.assert_frame_equal(result_sorted, expected_sorted, check_dtype=False)
