    if sort_by is not None:
        result_sorted = result.sort_values(by=sort_by, na_position="last").reset_index(drop=True)
        expected_sorted = expected.sort_values(by=sort_by, na_position="last").reset_index(drop=True)
    else:
        result_sorted = result.reset_index(drop=True)
        expected_sorted = expected.reset_index(drop=True)

def test_matched_customers_and_orders():
    """Rows sharing a non‑null key are marked as 'matched'."""
    customers = pd.DataFrame({"customer_id": [1, 2], "name": ["Alice", "Bob"]})
    orders = pd.DataFrame({"customer_id": [1, 2], "amount": [100, 200]})
    result = join_customers_orders(customers, orders)

    expected = pd.DataFrame(
        {
            "customer_id": [1, 2],
            "name": ["Alice", "Bob"],
            "amount": [100, 200],
            "record_status": ["matched", "matched"],
        }
    )
    # Order: source order of customers (1 then 2) becomes matched -> sorted by
    # combined _sort_order = _customer_source_order -> 0, 1.
    pd.testing.assert_frame_equal(result, expected)
