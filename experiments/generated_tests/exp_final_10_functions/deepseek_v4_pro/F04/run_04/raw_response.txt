import numpy as np
import pandas as pd
import pytest
from ise26.targets import join_customers_orders


class TestJoinCustomersOrders:
    @pytest.fixture
    def simple_customers(self):
        return pd.DataFrame({"customer_id": [1, 2, 3], "name": ["a", "b", "c"]})

    @pytest.fixture
    def simple_orders(self):
        return pd.DataFrame({"customer_id": [1, 3, 4], "amount": [10, 30, 40]})

    def test_basic_join(self, simple_customers, simple_orders):
        result = join_customers_orders(simple_customers, simple_orders)
        # id=1 matched, id=2 customer_without_order, id=3 matched, id=4 order_without_customer
        assert set(result["record_status"].unique()) == {
            "matched",
            "customer_without_order",
            "order_without_customer",
        }
        matched = result[result["record_status"] == "matched"]
        assert len(matched) == 2
        assert sorted(matched["customer_id"]) == [1, 3]
        # customer 2
        left_only = result[result["record_status"] == "customer_without_order"]
        assert len(left_only) == 1
        assert left_only.iloc[0]["customer_id"] == 2
        assert left_only.iloc[0]["name"] == "b"
        # order 4
        right_only = result[result["record_status"] == "order_without_customer"]
        assert len(right_only) == 1
        assert right_only.iloc[0]["customer_id"] == 4.0  # may become float?
        assert right_only.iloc[0]["amount"] == 40
        # index reset
        assert result.index.equals(pd.RangeIndex(len(result)))

    def test_null_keys_not_matched(self):
        customers = pd.DataFrame({"customer_id": [np.nan, 1.0, np.nan], "name": ["n1", "a", "n2"]})
        orders = pd.DataFrame({"customer_id": [np.nan, 1.0, np.nan], "quantity": [10, 20, 30]})
        result = join_customers_orders(customers, orders)

        # Rows with null keys should not be matched
        null_customers = result[result["record_status"] == "customer_without_order"]
        null_orders = result[result["record_status"] == "order_without_customer"]
        matched = result[result["record_status"] == "matched"]

        # Two null customer rows
        assert len(null_customers) == 2
        assert all(pd.isna(null_customers["customer_id"]))
        # Two null order rows
        assert len(null_orders) == 2
        assert all(pd.isna(null_orders["customer_id"]))
        # One matched row with id=1
        assert len(matched) == 1
        assert matched.iloc[0]["customer_id"] == 1.0

    def test_empty_customers(self):
        customers = pd.DataFrame(columns=["customer_id", "name"])
        orders = pd.DataFrame({"customer_id": [1, 2], "value": [10, 20]})
        result = join_customers_orders(customers, orders)
        assert len(result) == 2
        assert all(result["record_status"] == "order_without_customer")
        assert set(result.columns) >= {"customer_id", "value", "record_status"}

    def test_empty_orders(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["a", "b"]})
        orders = pd.DataFrame(columns=["customer_id", "value"])
        result = join_customers_orders(customers, orders)
        assert len(result) == 2
        assert all(result["record_status"] == "customer_without_order")
        assert set(result.columns) >= {"customer_id", "name", "record_status"}

    def test_empty_both(self):
        customers = pd.DataFrame(columns=["customer_id", "name"])
        orders = pd.DataFrame(columns=["customer_id", "value"])
        result = join_customers_orders(customers, orders)
        assert len(result) == 0
        assert "record_status" in result.columns

    def test_original_dataframes_unchanged(self, simple_customers, simple_orders):
        cust_before = simple_customers.copy()
        ord_before = simple_orders.copy()
        _ = join_customers_orders(simple_customers, simple_orders)
        pd.testing.assert_frame_equal(cust_before, simple_customers)
        pd.testing.assert_frame_equal(ord_before, simple_orders)

    def test_overlapping_column_suffixes(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "info": ["c1", "c2"], "extra": [10, 20]})
        orders = pd.DataFrame({"customer_id": [1, 3], "info": ["o1", "o3"], "extra": [1.0, 3.0]})
        result = join_customers_orders(customers, orders)

        # Overlapping non-key columns 'info' and 'extra' must have suffixes
        assert "info_customer" in result.columns
        assert "info_order" in result.columns
        assert "extra_customer" in result.columns
        assert "extra_order" in result.columns
        assert "customer_id" in result.columns

        # Non-overlapping column 'extra'? Actually both have extra, so suffixes apply.
        # If only one had a column, it should not be suffixed. Let's test that
        customers2 = pd.DataFrame({"customer_id": [1], "only_customer": [100]})
        orders2 = pd.DataFrame({"customer_id": [1], "only_order": [200]})
        result2 = join_customers_orders(customers2, orders2)
        assert "only_customer" in result2.columns
        assert "only_order" in result2.columns

    def test_deterministic_sorting(self):
        customers = pd.DataFrame({"customer_id": [2, 1, 3], "val": [20, 10, 30]})
        orders = pd.DataFrame({"customer_id": [1, 2, 4], "qty": [100, 200, 400]})
        result = join_customers_orders(customers, orders)
        # Expected order:
        # - matched rows in customer source order: first customer id=2 -> order id=2 (matched),
        #   then customer id=1 -> order id=1 (matched),
        #   then customer id=3 -> no order -> customer_without_order
        # - order_without_order: order id=4 after orders, because its source order is after matched ones?
        #   Actually order id=4 is third order row (source order 2). Sorting order:
        #   For matched rows, sort by _sort_order = customer_source_order (since combine_first uses customer order for matched),
        #   then tie-break with _customer_source_order, then _order_source_order, then record_status.
        #   Unmatched right rows sorted by _order_source_order (which is their _sort_order).
        # So output sequence: matched(2->2), matched(1->1), customer_without_order(3), order_without_order(4)
        expected_status = ["matched", "matched", "customer_without_order", "order_without_customer"]
        # But careful: customer 2 matched row includes order info; id values should be 2,2 and 1,1.
        statuses = list(result["record_status"])
        assert statuses == expected_status
        assert result.iloc[0]["customer_id"] == 2
        assert result.iloc[1]["customer_id"] == 1
        assert result.iloc[2]["customer_id"] == 3
