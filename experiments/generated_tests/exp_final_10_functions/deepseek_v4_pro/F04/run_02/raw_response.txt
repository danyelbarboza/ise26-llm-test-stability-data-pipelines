from ise26.targets import join_customers_orders

import pandas as pd
import pytest


def test_basic_matched_and_unmatched():
    customers = pd.DataFrame({
        "customer_id": [1, 2, 3],
        "name": ["Alice", "Bob", "Charlie"],
        "city": ["NY", "LA", "SF"]
    })
    orders = pd.DataFrame({
        "customer_id": [2, 3, 4],
        "amount": [100, 200, 300]
    })
    result = join_customers_orders(customers, orders, "customer_id")

    # Check statuses
    expected_statuses = {
        "matched",
        "customer_without_order",
        "order_without_customer"
    }
    assert set(result["record_status"]) == expected_statuses

    # Check matched rows (customer 2 and 3)
    matched = result[result["record_status"] == "matched"]
    assert matched["customer_id"].tolist() == [2, 3]
    assert matched["name"].tolist() == ["Bob", "Charlie"]

    # Customer without order (customer 1)
    cust_wo = result[result["record_status"] == "customer_without_order"]
    assert cust_wo["customer_id"].iloc[0] == 1
    assert cust_wo["name"].iloc[0] == "Alice"
    assert pd.isna(cust_wo["amount"].iloc[0])

    # Order without customer (customer 4)
    order_wo = result[result["record_status"] == "order_without_customer"]
    assert order_wo["customer_id"].iloc[0] == 4
    assert pd.isna(order_wo["name"].iloc[0])
    assert order_wo["amount"].iloc[0] == 300

    # No _merge column
    assert "_merge" not in result.columns


def test_null_keys_customers():
    customers = pd.DataFrame({
        "customer_id": [1, None, 3],
        "name": ["A", "B", "C"]
    })
    orders = pd.DataFrame({
        "customer_id": [1, 3, 4],
        "amt": [10, 20, 30]
    })
    result = join_customers_orders(customers, orders, "customer_id")

    # Customer with null id should be customer_without_order, never matched
    null_cust = result[(result["name"] == "B") & (pd.isna(result["customer_id"]))]
    assert len(null_cust) == 1
    assert null_cust["record_status"].iloc[0] == "customer_without_order"
    # No order columns data for that row
    assert pd.isna(null_cust["amt"].iloc[0])


def test_null_keys_orders():
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["X", "Y"]
    })
    orders = pd.DataFrame({
        "customer_id": [1, None],
        "amt": [10, 20]
    })
    result = join_customers_orders(customers, orders, "customer_id")

    # Order with null id should be order_without_customer
    null_order = result[(pd.isna(result["customer_id"])) & (result["record_status"] == "order_without_customer")]
    assert len(null_order) == 1
    assert null_order["amt"].iloc[0] == 20
    assert pd.isna(null_order["name"].iloc[0])


def test_null_keys_both_sides():
    customers = pd.DataFrame({
        "customer_id": [1, None],
        "name": ["A", "B"]
    })
    orders = pd.DataFrame({
        "customer_id": [1, None],
        "amt": [10, 20]
    })
    result = join_customers_orders(customers, orders, "customer_id")

    # Matched non-null
    matched = result[(result["customer_id"] == 1) & (result["record_status"] == "matched")]
    assert len(matched) == 1
    assert matched["name"].iloc[0] == "A"
    assert matched["amt"].iloc[0] == 10

    # Null customer row
    null_cust = result[(pd.isna(result["customer_id"])) & (result["name"] == "B")]
    assert len(null_cust) == 1
    assert null_cust["record_status"].iloc[0] == "customer_without_order"

    # Null order row
    null_order = result[(pd.isna(result["customer_id"])) & (result["amt"] == 20) & (result["record_status"] == "order_without_customer")]
    assert len(null_order) == 1


def test_overlapping_non_key_columns():
    customers = pd.DataFrame({
        "customer_id": [1, 2],
        "name": ["Alice", "Bob"],
        "city": ["NY", "LA"]
    })
    orders = pd.DataFrame({
        "customer_id": [1, 3],
        "name": ["Alice", "OrderCharlie"],  # overlapping 'name'
        "amount": [100, 200]
    })
    result = join_customers_orders(customers, orders, "customer_id")

    # Check suffixes on overlapping non-key columns
    assert "name_customer" in result.columns
    assert "name_order" in result.columns
    assert "city" in result.columns
    assert "amount" in result.columns

    # Row for customer 1 (matched)
    row1 = result[(result["customer_id"] == 1) & (result["record_status"] == "matched")]
    assert row1["name_customer"].iloc[0] == "Alice"
    assert row1["name_order"].iloc[0] == "Alice"
    assert row1["city"].iloc[0] == "NY"
    assert row1["amount"].iloc[0] == 100


def test_original_dataframes_unchanged():
    customers = pd.DataFrame({"customer_id": [1, 2], "val": [10, 20]})
    orders = pd.DataFrame({"customer_id": [2, 3], "qty": [5, 15]})
    cust_copy = customers.copy()
    order_copy = orders.copy()

    join_customers_orders(customers, orders, "customer_id")

    pd.testing.assert_frame_equal(customers, cust_copy)
    pd.testing.assert_frame_equal(orders, order_copy)


def test_index_reset():
    customers = pd.DataFrame({"customer_id": [5, 6]})
    orders = pd.DataFrame({"customer_id": [6, 7]})
    result = join_customers_orders(customers, orders, "customer_id")
    assert result.index.tolist() == list(range(len(result)))


def test_deterministic_order():
    customers = pd.DataFrame({
        "customer_id": [3, 1, 2],
        "data": ["c3", "c1", "c2"]
    })
    orders = pd.DataFrame({
        "customer_id": [1, 4, 2],
        "info": ["o1", "o4", "o2"]
    })
    res1 = join_customers_orders(customers, orders, "customer_id")
    res2 = join_customers_orders(customers, orders, "customer_id")
    pd.testing.assert_frame_equal(res1, res2)


def test_empty_dataframes():
    empty_cust = pd.DataFrame(columns=["customer_id", "name"])
    empty_orders = pd.DataFrame(columns=["customer_id", "amount"])

    result_both_empty = join_customers_orders(empty_cust, empty_orders, "customer_id")
    assert result_both_empty.empty
    assert set(result_both_empty.columns) == {"customer_id", "name", "amount", "record_status"}

    cust_with_data = pd.DataFrame({"customer_id": [1], "name": ["A"]})
    result_cust_only = join_customers_orders(cust_with_data, empty_orders, "customer_id")
    assert len(result_cust_only) == 1
    assert result_cust_only["record_status"].iloc[0] == "customer_without_order"

    orders_with_data = pd.DataFrame({"customer_id": [1], "amount": [10]})
    result_order_only = join_customers_orders(empty_cust, orders_with_data, "customer_id")
    assert len(result_order_only) == 1
    assert result_order_only["record_status"].iloc[0] == "order_without_customer"


def test_duplicate_keys():
    customers = pd.DataFrame({
        "customer_id": [1, 1, 2],
        "name": ["Alice1", "Alice2", "Bob"]
    })
    orders = pd.DataFrame({
        "customer_id": [1, 1, 3],
        "amount": [100, 200, 300]
    })
    result = join_customers_orders(customers, orders, "customer_id")

    # All combinations of key=1 should be matched
    matched_1 = result[(result["customer_id"] == 1) & (result["record_status"] == "matched")]
    assert len(matched_1) == 4

    # Customer 2 is unmatched, order 3 is unmatched
    assert any((result["customer_id"] == 2) & (result["record_status"] == "customer_without_order"))
    assert any((result["customer_id"] == 3) & (result["record_status"] == "order_without_customer"))


def test_custom_key_name():
    customers = pd.DataFrame({
        "cust_id": [1, 2],
        "info": ["c1", "c2"]
    })
    orders = pd.DataFrame({
        "cust_id": [1, 3],
        "value": [10, 20]
    })
    result = join_customers_orders(customers, orders, "cust_id")

    assert result["cust_id"].tolist() == [1, 2, 3]
    matched = result[result["record_status"] == "matched"]
    assert matched["cust_id"].iloc[0] == 1 and matched["info"].iloc[0] == "c1"
