```python
import pandas as pd
import pytest
from pandas.testing import assert_series_equal
from ise26.targets import standardize_currency_values


def test_basic_english():
    """Test conversion of simple English-format currency strings."""
    df = pd.DataFrame({"amount_raw": ["1234.56", "0.99", "1000.00"]})
    expected = pd.Series([1234.56, 0.99, 1000.00], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_basic_brazilian():
    """Test conversion of Brazilian-format currency strings (comma decimal, dot thousand)."""
    df = pd.DataFrame({"amount_raw": ["1.234,56", "0,99", "1.000,00"]})
    expected = pd.Series([1234.56, 0.99, 1000.00], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_english_with_thousands_separator():
    """Test English format with thousands separator comma and decimal dot."""
    df = pd.DataFrame({"amount_raw": ["1,234.56", "10,000.99", "1,000,000.00"]})
    expected = pd.Series([1234.56, 10000.99, 1000000.00], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_invalid_values():
    """Test invalid inputs produce pd.NA."""
    df = pd.DataFrame({"amount_raw": ["abc", "", "1.2.3", "12-34"]})
    expected = pd.Series([pd.NA, pd.NA, pd.NA, pd.NA], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_null_input():
    """Test pd.NA in input produces pd.NA in output."""
    df = pd.DataFrame({"amount_raw": [pd.NA, "123.45", pd.NA]})
    expected = pd.Series([pd.NA, 123.45, pd.NA], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_mixed_valid_invalid():
    """Test DataFrame with mixture of valid and invalid strings."""
    df = pd.DataFrame({"amount_raw": ["1.234,56", "invalid", "999.99", ""]})
    expected = pd.Series([1234.56, pd.NA, 999.99, pd.NA], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_original_column_preserved():
    """Verify original raw column remains unchanged in result."""
    df = pd.DataFrame({"amount_raw": ["1.234,56", "999.99"]})
    original_raw = df["amount_raw"].copy()
    result = standardize_currency_values(df)
    pd.testing.assert_series_equal(result["amount_raw"], original_raw)


def test_empty_dataframe():
    """Test empty DataFrame returns empty nullable column."""
    df = pd.DataFrame({"amount_raw": pd.Series([], dtype="object")})
    result = standardize_currency_values(df)
    expected = pd.Series([], dtype="Float64", name="amount")
    assert_series_equal(result["amount"], expected)


def test_duplicate_values():
    """Test duplicate input values are all parsed correctly."""
    df = pd.DataFrame({"amount_raw": ["1.234,56", "1.234,56", "999.99"]})
    expected = pd.Series([1234.56, 1234.56, 999.99], dtype="Float64", name="amount")
    result = standardize_currency_values(df)
    assert_series_equal(result["amount"], expected)


def test_custom_column_names():
    """Test function works with non-default column names."""
    df = pd.DataFrame({"custom_raw": ["1.234,56", "999.99"]})
    result = standardize_currency_values(
        df, value_col="custom_raw", output_col="processed"
    )
    expected = pd.Series([1234.56, 999.99], dtype="Float64", name="processed")
    assert_series_equal(result["processed"], expected)
```
