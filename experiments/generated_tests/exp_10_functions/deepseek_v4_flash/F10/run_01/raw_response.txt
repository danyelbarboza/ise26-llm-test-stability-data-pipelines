import pandas as pd
import numpy as np
import pytest
from ise26.targets import standardize_currency_values


@pytest.fixture
def sample_df():
    """Base DataFrame with typical currency strings."""
    return pd.DataFrame(
        {
            "amount_raw": [
                "1.234,56",       # Brazilian
                "1,234.56",       # English
                "1234.56",        # simple dot
                "1234,56",        # simple comma
                "R$ 2.000,00",    # with symbol
                "$ 3,500.75",     # with dollar
                "1.000.000,00",   # large BR
                "1,000,000.00",   # large US
                "0,50",           # BR zero
                "0.50",           # US zero
                "-1.234,56",      # negative BR
                "-1,234.56",      # negative US
                "abc",            # invalid
                "12,34.56",       # mixed separators
                "",               # empty string
                None,             # None
                pd.NA,            # pandas NA
                "  1.234,56  ",   # with spaces
                "  R$ 500,00  ",  # spaces and symbol
            ]
        }
    )


class TestStandardizeCurrencyValues:

    def test_returns_dataframe(self, sample_df):
        result = standardize_currency_values(sample_df)
        assert isinstance(result, pd.DataFrame)

    def test_original_column_preserved(self, sample_df):
        result = standardize_currency_values(sample_df)
        assert "amount_raw" in result.columns
        assert result["amount_raw"].equals(sample_df["amount_raw"])

    def test_output_column_exists_and_dtype(self, sample_df):
        result = standardize_currency_values(sample_df)
        assert "amount" in result.columns
        assert result["amount"].dtype == pd.Float64Dtype()
        # check first few values are nullable numeric
        assert pd.isna(result["amount"].iloc[12])  # "abc" should be NA
        assert result["amount"].iloc[0] == pytest.approx(1234.56)

    @pytest.mark.parametrize(
        "input_raw, expected",
        [
            ("1.234,56", 1234.56),
            ("1,234.56", 1234.56),
            ("1234.56", 1234.56),
            ("1234,56", 1234.56),
            ("R$ 2.000,00", 2000.00),
            ("$ 3,500.75", 3500.75),
            ("1.000.000,00", 1000000.00),
            ("1,000,000.00", 1000000.00),
            ("0,50", 0.50),
            ("0.50", 0.50),
            ("-1.234,56", -1234.56),
            ("-1,234.56", -1234.56),
            ("  1.234,56  ", 1234.56),  # spaces trimmed
            ("  R$ 500,00  ", 500.00),  # spaces and symbol
        ],
    )
    def test_valid_conversions(self, input_raw, expected):
        df = pd.DataFrame({"amount_raw": [input_raw]})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == pytest.approx(expected)

    @pytest.mark.parametrize(
        "input_raw",
        [
            "abc",
            "12,34.56",
            "",
            " ",
            "1.234,56.78",
            "1,234.56.78",
            "R$ abc",
            "$12,34.56",
        ],
    )
    def test_invalid_returns_na(self, input_raw):
        df = pd.DataFrame({"amount_raw": [input_raw]})
        result = standardize_currency_values(df)
        assert pd.isna(result["amount"].iloc[0])

    def test_none_input_returns_na(self):
        df = pd.DataFrame({"amount_raw": [None]})
        result = standardize_currency_values(df)
        assert pd.isna(result["amount"].iloc[0])

    def test_pdna_input_returns_na(self):
        df = pd.DataFrame({"amount_raw": [pd.NA]})
        result = standardize_currency_values(df)
        assert pd.isna(result["amount"].iloc[0])

    def test_nan_input_returns_na(self):
        df = pd.DataFrame({"amount_raw": [np.nan]})
        result = standardize_currency_values(df)
        assert pd.isna(result["amount"].iloc[0])

    def test_custom_column_names(self):
        df = pd.DataFrame({"my_values": ["1.200,50"]})
        result = standardize_currency_values(df, value_col="my_values", output_col="my_output")
        assert "my_output" in result.columns
        assert result["my_output"].iloc[0] == pytest.approx(1200.50)
        assert "my_values" in result.columns  # original preserved

    def test_does_not_modify_input(self, sample_df):
        original = sample_df.copy()
        _ = standardize_currency_values(sample_df)
        pd.testing.assert_frame_equal(sample_df, original)

    def test_multiple_rows(self):
        data = ["1.000,00", "2,000.00", "invalid", None]
        df = pd.DataFrame({"amount_raw": data})
        result = standardize_currency_values(df)
        expected_values = [1000.00, 2000.00, pd.NA, pd.NA]
        for i, (res, exp) in enumerate(zip(result["amount"], expected_values)):
            if exp is pd.NA:
                assert pd.isna(res) or res is pd.NA
            else:
                assert res == pytest.approx(exp)

    def test_duplicate_values(self):
        data = ["1.234,56", "1.234,56"]
        df = pd.DataFrame({"amount_raw": data})
        result = standardize_currency_values(df)
        assert result["amount"].iloc[0] == 1234.56
        assert result["amount"].iloc[1] == 1234.56
