import pytest
import pandas as pd
import numpy as np
from ise26.targets import calculate_monthly_revenue


class TestCalculateMonthlyRevenue:
    """Test suite for calculate_monthly_revenue function."""

    def test_normal_case(self):
        """Valid orders with multiple months and some canceled."""
        df = pd.DataFrame({
            'order_date': ['2023-01-15', '2023-01-20', '2023-02-10', '2023-02-25'],
            'amount': [100.0, 200.0, 300.0, 400.0],
            'status': ['completed', 'cancelled', 'shipped', 'completed']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01', '2023-02'],
            'revenue': [100.0, 700.0]  # second order canceled, 300+400=700
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_dates_ignored(self):
        """Rows with unparseable dates are ignored."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', 'not-a-date', '2023-01-15', None],
            'amount': [50.0, 100.0, 150.0, 200.0],
            'status': ['completed', 'completed', 'completed', 'completed']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [200.0]  # 50 + 150
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_amounts_treated_as_zero(self):
        """Invalid or missing amounts become zero."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-01-02', '2023-01-03'],
            'amount': [100.0, 'abc', None],
            'status': ['completed', 'completed', 'completed']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [100.0]  # 100 + 0 + 0
        })
        pd.testing.assert_frame_equal(result, expected)

    @pytest.mark.parametrize("status_value", [
        " cancelled ",
        "CANCELED",
        "cancelado",
        "Cancelado",
        "cancelled",
        "CANCELADO",
        "  Canceled  "
    ])
    def test_canceled_status_variants(self, status_value):
        """Various forms of canceled status are ignored."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-01-01'],
            'amount': [100.0, 200.0],
            'status': ['completed', status_value]
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [100.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_non_canceled_statuses_included(self):
        """All non-canceled statuses contribute revenue."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-01-01', '2023-01-01'],
            'amount': [10.0, 20.0, 30.0],
            'status': ['completed', 'shipped', 'processing']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [60.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_all_orders_canceled_returns_empty(self):
        """Only canceled orders -> empty result."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-02-01'],
            'amount': [100.0, 200.0],
            'status': ['cancelled', 'cancelado']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': pd.Series([], dtype='string'),
            'revenue': pd.Series([], dtype='float64')
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_dataframe(self):
        """Empty input DataFrame returns empty result."""
        df = pd.DataFrame({'order_date': [], 'amount': [], 'status': []})
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': pd.Series([], dtype='string'),
            'revenue': pd.Series([], dtype='float64')
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_duplicate_orders_sum(self):
        """Multiple orders in same month are summed."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-01-05', '2023-01-31'],
            'amount': [50.0, 50.0, 100.0],
            'status': ['completed', 'completed', 'completed']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [200.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_month_sorting(self):
        """Result is sorted by month (chronological)."""
        df = pd.DataFrame({
            'order_date': ['2023-02-01', '2023-01-01', '2023-03-01'],
            'amount': [100.0, 200.0, 300.0],
            'status': ['completed', 'completed', 'completed']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01', '2023-02', '2023-03'],
            'revenue': [200.0, 100.0, 300.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_column_names(self):
        """Works with different column names."""
        df = pd.DataFrame({
            'date': ['2023-01-01', '2023-02-01'],
            'value': [100.0, 200.0],
            'state': ['completed', 'cancelled']
        })
        result = calculate_monthly_revenue(df, date_col='date', amount_col='value', status_col='state')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [100.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_mixed_valid_and_invalid_amounts(self):
        """Mix of valid, invalid, and missing amounts."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-01-02', '2023-01-03', '2023-01-04'],
            'amount': [10.0, 'nope', None, 40.0],
            'status': ['completed', 'completed', 'completed', 'cancelado']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [10.0]  # 10 + 0 + 0, last canceled ignored
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_result_types(self):
        """Verify month is string (object) and revenue is float."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01'],
            'amount': [100.0],
            'status': ['completed']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        assert result['month'].dtype == 'string' or result['month'].dtype == 'object'
        assert result['revenue'].dtype == 'float64'

    def test_no_status_column_uses_default(self):
        """Default column names are used when not provided."""
        df = pd.DataFrame({
            'order_date': ['2023-01-01', '2023-01-01'],
            'amount': [100.0, 200.0],
            'status': ['completed', 'cancelado']
        })
        result = calculate_monthly_revenue(df)  # uses defaults
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [100.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_dates_already_datetime64(self):
        """Works if date column is already datetime."""
        df = pd.DataFrame({
            'order_date': pd.to_datetime(['2023-01-01', '2023-02-01']),
            'amount': [100.0, 200.0],
            'status': ['shipped', 'canceled']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01'],
            'revenue': [100.0]
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_revenue_total_over_multiple_months(self):
        """More complex scenario with multiple months and statuses."""
        df = pd.DataFrame({
            'order_date': ['2023-01-15', '2023-02-10', '2023-02-20', '2023-03-05', '2023-03-15'],
            'amount': [100, 200, 'invalid', 300, None],
            'status': ['completed', 'cancelled', 'completed', 'shipped', 'cancelado']
        })
        result = calculate_monthly_revenue(df, 'order_date', 'amount', 'status')
        expected = pd.DataFrame({
            'month': ['2023-01', '2023-02', '2023-03'],
            'revenue': [100.0, 0.0, 300.0]  # Feb: second order canceled (ignored), third invalid amount => 0; Mar: last canceled ignored, 300 from shipped
        })
        pd.testing.assert_frame_equal(result, expected)
