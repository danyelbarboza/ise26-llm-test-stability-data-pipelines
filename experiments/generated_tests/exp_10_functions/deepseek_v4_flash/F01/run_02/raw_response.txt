from ise26.targets import clean_customer_names
import pandas as pd
import pytest


class TestCleanCustomerNames:
    """Test suite for clean_customer_names function."""

    def test_basic_normalization(self):
        """Standard lowercasing and whitespace collapsing."""
        df = pd.DataFrame({"customer_name": ["  João  Silva  ", "MARIA   SOUZA"]})
        result = clean_customer_names(df, "customer_name", "name_clean")
        assert result["name_clean"].tolist() == ["joão silva", "maria souza"]

    def test_accent_removal(self):
        """Accented characters should be normalized."""
        df = pd.DataFrame({"name": ["Café", "São Paulo", "Ação"]})
        result = clean_customer_names(df, "name", "clean")
        expected = ["cafe", "sao paulo", "acao"]
        assert result["clean"].tolist() == expected

    def test_null_values(self):
        """None, NaN and empty strings become pd.NA."""
        df = pd.DataFrame({"raw": ["Pedro", None, "", float("nan"), "   "]})
        result = clean_customer_names(df, "raw", "clean")
        assert result["clean"].iloc[0] == "pedro"
        assert pd.isna(result["clean"].iloc[1])
        assert pd.isna(result["clean"].iloc[2])
        assert pd.isna(result["clean"].iloc[3])
        assert pd.isna(result["clean"].iloc[4])

    def test_output_column_default_name(self):
        """Default output column is 'customer_name_clean'."""
        df = pd.DataFrame({"customer_name": ["Ana"]})
        result = clean_customer_names(df)
        assert "customer_name_clean" in result.columns

    def test_original_dataframe_unchanged(self):
        """The input DataFrame must not be mutated."""
        original = pd.DataFrame({"customer_name": ["  Teste  "]})
        original_copy = original.copy()
        _ = clean_customer_names(original, "customer_name", "clean")
        pd.testing.assert_frame_equal(original, original_copy)

    def test_returns_new_dataframe(self):
        """The returned DataFrame must be a different object."""
        df = pd.DataFrame({"customer_name": ["A"]})
        result = clean_customer_names(df)
        assert result is not df

    def test_mixed_multiple_rows(self):
        """Combination of whitespace, accents, nulls and normal names."""
        df = pd.DataFrame({
            "name": [
                "  ÀLvaro   Costa  ",
                "  ",
                None,
                "maria do céu",
                "   João   ",
            ]
        })
        result = clean_customer_names(df, "name", "cleaned")
        expected = ["alvaro costa", pd.NA, pd.NA, "maria do ceu", "joão"]
        for i, val in enumerate(expected):
            if pd.isna(val):
                assert pd.isna(result["cleaned"].iloc[i])
            else:
                assert result["cleaned"].iloc[i] == val

    def test_duplicate_names(self):
        """Duplicates in source should be duplicated cleanly."""
        df = pd.DataFrame({"n": ["  João  ", "  João  "]})
        result = clean_customer_names(df, "n", "n_clean")
        assert result["n_clean"].tolist() == ["joão", "joão"]

    def test_custom_input_output_column_names(self):
        """Arbitrary column names should work."""
        df = pd.DataFrame({"INPUT": ["  Foo  "]})
        result = clean_customer_names(df, "INPUT", "OUTPUT")
        assert result["OUTPUT"].iloc[0] == "foo"

    def test_dataframe_with_additional_columns(self):
        """Other columns in the DataFrame must be preserved unchanged."""
        df = pd.DataFrame({
            "id": [1, 2],
            "customer_name": ["  Pedro  ", "  Maria  "],
            "value": [100.0, 200.0],
        })
        result = clean_customer_names(df, "customer_name", "clean")
        assert list(result.columns) == ["id", "customer_name", "value", "clean"]
        assert result["id"].tolist() == [1, 2]
        assert result["value"].tolist() == [100.0, 200.0]

    def test_empty_dataframe(self):
        """An empty DataFrame should return a copy with the new column."""
        df = pd.DataFrame({"customer_name": pd.Series(dtype=str)})
        result = clean_customer_names(df, "customer_name", "clean")
        assert result.shape[0] == 0
        assert "clean" in result.columns

    def test_no_extra_whitespace_left(self):
        """Leading/trailing whitespace must be stripped and internal spaces a single space."""
        df = pd.DataFrame({"name": ["   multiple    spaces   here   "]})
        result = clean_customer_names(df, "name", "clean")
        assert result["clean"].iloc[0] == "multiple spaces here"
