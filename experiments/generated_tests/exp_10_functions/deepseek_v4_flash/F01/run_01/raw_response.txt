import pandas as pd
import numpy as np
import pytest
from ise26.targets import clean_customer_names


def test_basic_normalization():
    df = pd.DataFrame({"name": ["Alice", "Bob", "Charlie"]})
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    expected = df.copy()
    expected["name_clean"] = ["alice", "bob", "charlie"]
    pd.testing.assert_frame_equal(result, expected)


def test_whitespace_collapse():
    df = pd.DataFrame({"name": ["  Alice  ", "Bob   Smith", "\tCharlie\nJones  "]})
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    expected = df.copy()
    expected["name_clean"] = ["alice", "bob smith", "charlie jones"]
    pd.testing.assert_frame_equal(result, expected)


def test_lowercase():
    df = pd.DataFrame({"name": ["ALICE", "Bob", "cHARLIE"]})
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    expected = df.copy()
    expected["name_clean"] = ["alice", "bob", "charlie"]
    pd.testing.assert_frame_equal(result, expected)


def test_accent_removal():
    df = pd.DataFrame({"name": ["François", "José", "Müller", "naïve"]})
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    expected = df.copy()
    expected["name_clean"] = ["francois", "jose", "muller", "naive"]
    pd.testing.assert_frame_equal(result, expected)


def test_null_and_blank():
    df = pd.DataFrame({
        "name": [None, np.nan, pd.NA, "", "   ", "Alice"]
    })
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    expected = df.copy()
    expected["name_clean"] = [pd.NA, pd.NA, pd.NA, pd.NA, pd.NA, "alice"]
    # Ensure the NA values are properly recognized
    pd.testing.assert_frame_equal(result, expected)


def test_original_dataframe_not_mutated():
    original = pd.DataFrame({"name": ["  Alice  ", "Bob"]})
    original_copy = original.copy()
    clean_customer_names(original, name_col="name", output_col="name_clean")
    pd.testing.assert_frame_equal(original, original_copy)


def test_default_parameters():
    df = pd.DataFrame({"customer_name": ["  Alice  ", "Bob"]})
    result = clean_customer_names(df)
    expected = df.copy()
    expected["customer_name_clean"] = ["alice", "bob"]
    pd.testing.assert_frame_equal(result, expected)


def test_empty_dataframe():
    df = pd.DataFrame({"name": pd.Series(dtype=str)})
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    assert "name_clean" in result.columns
    assert result.empty


def test_missing_column_raises_error():
    df = pd.DataFrame({"other": ["value"]})
    with pytest.raises(KeyError):
        clean_customer_names(df, name_col="nonexistent", output_col="name_clean")


def test_keeps_original_columns_and_adds_new():
    df = pd.DataFrame({"a": [1, 2], "name": ["Alice", "Bob"]})
    result = clean_customer_names(df, name_col="name", output_col="name_clean")
    assert "a" in result.columns
    assert "name" in result.columns
    assert "name_clean" in result.columns
    assert list(result.columns) == ["a", "name", "name_clean"]
