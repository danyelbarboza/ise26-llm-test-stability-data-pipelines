import pytest
import pandas as pd
import numpy as np
from ise26.targets import clean_customer_names


def test_normalize_basic():
    """Test basic white space collapsing and lowercasing."""
    df = pd.DataFrame({"customer_name": ["  JOHN   DOE  ", "Alice  Smith", "  "]})
    result = clean_customer_names(df)
    expected = df.copy()
    expected["customer_name_clean"] = pd.Series(["john doe", "alice smith", pd.NA], dtype="object")
    pd.testing.assert_frame_equal(result, expected)


def test_accent_removal():
    """Test removal of diacritics/accents."""
    df = pd.DataFrame({"customer_name": ["José María", "Müller", "São Paulo"]})
    result = clean_customer_names(df)
    expected = df.copy()
    expected["customer_name_clean"] = pd.Series(
        ["jose maria", "muller", "sao paulo"], dtype="object"
    )
    pd.testing.assert_frame_equal(result, expected)


def test_nulls_and_blanks():
    """Test null, NaN, None, and blank strings become pd.NA."""
    df = pd.DataFrame({
        "customer_name": [np.nan, None, pd.NA, "", "   ", "valid"]
    })
    result = clean_customer_names(df)
    expected = df.copy()
    expected_clean = [pd.NA, pd.NA, pd.NA, pd.NA, pd.NA, "valid"]
    expected["customer_name_clean"] = pd.Series(expected_clean, dtype="object")
    pd.testing.assert_frame_equal(result, expected)


def test_original_df_unchanged():
    """Verify the original DataFrame is not mutated."""
    original = pd.DataFrame({"customer_name": [" John ", "Mary"]})
    original_copy = original.copy()
    _ = clean_customer_names(original)
    assert original.equals(original_copy), "Original DataFrame was modified"


def test_default_columns():
    """Test default parameter values for name_col and output_col."""
    df = pd.DataFrame({"customer_name": ["Alice  Smith"]})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    assert result["customer_name_clean"].iloc[0] == "alice smith"


def test_custom_column_names():
    """Test custom values for name_col and output_col."""
    df = pd.DataFrame({"raw_name": ["  John   Doe  "]})
    result = clean_customer_names(df, name_col="raw_name", output_col="clean")
    assert "clean" in result.columns
    assert result["clean"].iloc[0] == "john doe"
    # original column preserved
    assert "raw_name" in result.columns
    assert result["raw_name"].iloc[0] == "  John   Doe  "


def test_extra_columns_preserved():
    """Test that other columns in the DataFrame are not lost."""
    df = pd.DataFrame({
        "id": [1, 2],
        "customer_name": [" John ", "Mary"],
        "extra": ["x", "y"]
    })
    result = clean_customer_names(df)
    expected = df.copy()
    expected["customer_name_clean"] = pd.Series(["john", "mary"], dtype="object")
    pd.testing.assert_frame_equal(result, expected)


def test_output_is_copy():
    """Ensure the returned DataFrame is a new object, not the original."""
    df = pd.DataFrame({"customer_name": ["Alice"]})
    result = clean_customer_names(df)
    assert result is not df


def test_duplicate_names():
    """Test that identical names are normalized identically."""
    df = pd.DataFrame({"customer_name": ["  JOHN  ", "  john  ", "JOHN"]})
    result = clean_customer_names(df)
    norm_col = result["customer_name_clean"]
    assert norm_col.nunique() == 1
    assert norm_col.iloc[0] == "john"


def test_missing_column_raises():
    """Test that an error is raised if name_col does not exist."""
    df = pd.DataFrame({"wrong_col": ["a"]})
    with pytest.raises(KeyError):
        clean_customer_names(df)


def test_multiple_rows_with_mixed_types():
    """Test a larger DataFrame with various cases."""
    df = pd.DataFrame({
        "customer_name": [
            "   ALICE   MÜLLER  ",
            "bob",
            None,
            "  ",
            " Ção ",
            "José  Pedro   ",
        ]
    })
    result = clean_customer_names(df)
    expected = df.copy()
    expected["customer_name_clean"] = pd.Series([
        "alice muller",
        "bob",
        pd.NA,
        pd.NA,
        "cao",
        "jose pedro",
    ], dtype="object")
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)
