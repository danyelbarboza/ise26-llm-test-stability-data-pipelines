from typing import Any

import pandas as pd
import pytest

from ise26.targets import classify_payment_status


def test_basic_paid_on_time() -> None:
    df = pd.DataFrame({
        "due_date": ["2023-01-15", "2023-02-20"],
        "paid_date": ["2023-01-10", "2023-02-18"],
        "amount": [100.0, 200.0],
    })
    result = classify_payment_status(df, reference_date="2023-03-01")
    expected = pd.Series(["paid_on_time", "paid_on_time"], name="payment_status")
    pd.testing.assert_series_equal(result["payment_status"], expected, check_names=True)

def test_mixed_statuses() -> None:
    df = pd.DataFrame({
        "due_date": [
            "2023-01-15",  # paid_on_time
            "2023-01-15",  # paid_late
            "2023-01-15",  # overdue
            "2023-05-15",  # pending
            "2023-01-15",  # invalid (amount missing)
            "2023-01-15",  # invalid (paid_date provided but invalid)
            "2023-01-15",  # invalid (due_date invalid)
        ],
        "paid_date": [
            "2023-01-10",
            "2023-01-20",
            None,
            None,
            "2023-01-10",
            "bad_date",
            "2023-01-10",
        ],
        "amount": [
            100.0,
            100.0,
            100.0,
            100.0,
            None,
            100.0,
            100.0,
        ],
    })
    # Overwrite due_date for last row to be invalid
    df.loc[6, "due_date"] = None
    result = classify_payment_status(df, reference_date="2023-03-01")
    expected = pd.Series(
        ["paid_on_time", "paid_late", "overdue", "pending", "invalid", "invalid", "invalid"],
        name="payment_status",
    )
    pd.testing.assert_series_equal(result["payment_status"], expected, check_names=True)

def test_custom_column_names() -> None:
    df = pd.DataFrame({
        "dd": ["2023-01-15"],
        "pd": ["2023-01-10"],
        "amt": [100.0],
    })
    result = classify_payment_status(
        df,
        reference_date="2023-03-01",
        due_date_col="dd",
        paid_date_col="pd",
        amount_col="amt",
        output_col="status",
    )
    assert "status" in result.columns
    assert result["status"].iloc[0] == "paid_on_time"

def test_boundary_dates() -> None:
    # paid_date == due_date => paid_on_time
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-15"],
        "amount": [100.0],
    })
    result = classify_payment_status(df, reference_date="2023-03-01")
    assert result["payment_status"].iloc[0] == "paid_on_time"

    # reference_date == due_date => pending (not overdue)
    df2 = pd.DataFrame({
        "due_date": ["2023-03-01"],
        "paid_date": [None],
        "amount": [100.0],
    })
    result2 = classify_payment_status(df2, reference_date="2023-03-01")
    assert result2["payment_status"].iloc[0] == "pending"

    # reference_date just after due_date => overdue
    df3 = pd.DataFrame({
        "due_date": ["2023-02-28"],
        "paid_date": [None],
        "amount": [100.0],
    })
    result3 = classify_payment_status(df3, reference_date="2023-03-01")
    assert result3["payment_status"].iloc[0] == "overdue"
