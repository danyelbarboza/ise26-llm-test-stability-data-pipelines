def test_returns_new_dataframe():

        {
            "due_date": ["2024-01-01"],
            "paid_date": ["2024-01-01"],
            "amount": [100.0],

    )

    assert result is not df

def test_invalid_paid_date_na_but_provided():
    # paid_date column is not null but contains NA? Actually if it's None,
    # it should be treated as missing. We want invalid only if a non-null
    # value is unparseable. We'll test with a numeric value that fails parsing.
    df = pd.DataFrame(
        {
            "due_date": ["2024-01-15"],
            "paid_date": ["2024-02-30"],  # invalid calendar date
            "amount": [100.0],
        }
    )
    result = classify_payment_status(df, "2024-01-20")
    # '2024-02-30' is parsed by pd.to_datetime with errors='coerce' -> NaT
    # and paid_date is provided (not null) => invalid
    assert result["payment_status"].iloc[0] == "invalid"


def test_invalid_paid_date_empty_string():

def test_mixed_dataframe():
    # Use the default df and check all rows
    df = _make_default_df()
    result = classify_payment_status(df, "2024-06-20")
    expected_statuses = [
        "paid_on_time",  # row0
        "paid_late",     # row1
        "paid_on_time",  # row2: paid exactly on due
        "overdue",       # row3: unpaid, due < reference
        "paid_on_time",  # row4: paid before due
        "invalid",       # row5: due_date missing
        "invalid",       # row6: due_date invalid
        "pending",       # row7: unpaid, due == reference? reference is 2024-06-20, due is 2024-06-15 => overdue? Wait due 2024-06-15 < ref 2024-06-20 => overdue, but we need to recalc: row7 has due 2024-06-15, unpaid => overdue.
        "invalid",       # row8: paid_date invalid
        "invalid",       # row9: amount negative
    ]
    # Let's recompute carefully for row7 and row8
    # Row7 (index7): due_date 2024-06-15, paid_date None, amount 350.0 -> unpaid, due < reference => overdue. But we had predicted pending incorrectly.
    # Row8: due_date 2024-06-15, paid_date "bad_date" -> invalid
    # Row9: due 2024-07-01, paid 2024-07-01, amount -10 -> invalid
    # So let's set expected properly.
    expected = [
        "paid_on_time",
        "paid_late",
        "paid_on_time",
        "overdue",
        "paid_on_time",
        "invalid",
        "invalid",
        "overdue",   # row7
        "invalid",   # row8
        "invalid",   # row9
    ]
    assert list(result["payment_status"]) == expected

def test_overdue_edge_equal_reference():
    # Overdue condition: reference > due. If reference == due, it's pending.
    df = pd.DataFrame(
        {
            "due_date": ["2024-01-15"],
            "paid_date": [None],
            "amount": [100.0],
        }
    )
    result = classify_payment_status(df, "2024-01-15")
    assert result["payment_status"].iloc[0] == "pending"
    result2 = classify_payment_status(df, "2024-01-16")
    assert result2["payment_status"].iloc[0] == "overdue"
