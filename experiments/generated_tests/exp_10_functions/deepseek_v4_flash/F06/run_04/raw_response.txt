class TestClassifyPaymentStatus:
    """Suite of tests for classify_payment_status."""

    def test_empty_dataframe(self):
        """Empty DataFrame should return a copy with the new column."""
        df = pd.DataFrame({"due_date": [], "paid_date": [], "amount": []})
        result = classify_payment_status(df, "2023-06-01")
        assert isinstance(result, pd.DataFrame)
        assert "payment_status" in result.columns
        assert len(result) == 0

    def test_basic_payment_statuses(self, df_basic):
        """Test the five main statuses on a simple dataset."""
        reference_date = "2023-04-15"
        result = classify_payment_status(df_basic, reference_date)
        expected = ["paid_on_time", "paid_late", "pending", "invalid", "overdue"]
        assert list(result["payment_status"]) == expected

    def test_edge_cases_without_reference_override(self, df_edge_cases):
        """Default reference date; the fixture uses specific reference later."""
        # Using a fixed reference date to match some expected statuses
        reference_date = "2023-07-15"
        result = classify_payment_status(df_edge_cases, reference_date)
        statuses = result["payment_status"].tolist()
        # Expected statuses based on the fixture and reference date
        expected = [
            "paid_on_time",        # 0
            "paid_late",           # 1
            "pending",             # 2: reference (Jul 15) > due (Mar 10)?? Actually due 2023-03-10, reference Jul 15 => overdue? Wait: unpaid + reference > due => overdue.
            "invalid",             # 3: invalid due_date
            "overdue",             # 4: unpaid, due 2023-04-05, reference Jul 15 > due
            "pending",             # 5: empty paid_date treated as unpaid, due May 1, reference Jul 15 > due => overdue.
            "invalid",             # 6: invalid paid_date
            "invalid",             # 7: amount <=0
            "invalid",             # 8: amount ==0
            "invalid",             # 9: missing amount
            "invalid",             # 10: non-numeric amount
            "pending",             # 11: unpaid, due Nov 1, reference Jul 15 < due => pending
            "overdue",             # 12: unpaid, due Dec 1, reference Jul 15 < due? Actually Jul < Dec => pending not overdue.
        ]
        # Let's recompute carefully: For row 2: due 2023-03-10, unpaid, reference 2023-07-15 => reference > due => overdue
        # Row 5: due 2023-05-01, unpaid (empty string), reference Jul 15 > May => overdue
        # Row 11: due 2023-11-01, unpaid, reference Jul < Nov => pending
        # Row 12: due 2023-12-01, unpaid, reference Jul < Dec => pending
        expected = [
            "paid_on_time",        # 0
            "paid_late",           # 1
            "overdue",             # 2 (was pending, but reference > due)
            "invalid",             # 3
            "overdue",             # 4
            "overdue",             # 5
            "invalid",             # 6
            "invalid",             # 7
            "invalid",             # 8
            "invalid",             # 9
            "invalid",             # 10
            "pending",             # 11
            "pending",             # 12
        ]
        assert statuses == expected

    def test_reference_date_edge(self):
        """Test when reference_date equals due_date for unpaid rows."""
        df = pd.DataFrame({
            "due_date": ["2023-06-01", "2023-06-01"],
            "paid_date": [None, None],
            "amount": [100.0, 100.0],
        })
        # reference equal to due: should be pending (reference <= due)
        result = classify_payment_status(df, "2023-06-01")
        assert list(result["payment_status"]) == ["pending", "pending"]

        # reference one day after due: should be overdue
        result = classify_payment_status(df, "2023-06-02")
        assert list(result["payment_status"]) == ["overdue", "overdue"]

    def test_paid_date_missing_as_nan_and_empty_string(self):
        """Both NaN and empty string should be treated as unpaid, not invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01", "2023-01-01"],
            "paid_date": [None, ""],
            "amount": [100.0, 100.0],
        })
        result = classify_payment_status(df, "2023-01-15")
        # Both should be overdue (reference > due)
        assert list(result["payment_status"]) == ["overdue", "overdue"]

    def test_invalid_paid_date_marks_invalid(self):
        """A provided but unparseable paid_date should yield invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01"],
            "paid_date": ["not_a_date"],
            "amount": [100.0],
        })
        result = classify_payment_status(df, "2023-01-15")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_amount_zero_or_negative_invalid(self):
        """Amount <= 0 should be invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01", "2023-01-01"],
            "paid_date": ["2023-01-01", None],
            "amount": [0.0, -5.0],
        })
        result = classify_payment_status(df, "2023-01-15")
        assert list(result["payment_status"]) == ["invalid", "invalid"]

    def test_missing_amount_invalid(self):
        """Missing amount should be invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01"],
            "paid_date": ["2023-01-01"],
            "amount": [None],
        })
        result = classify_payment_status(df, "2023-01-15")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_non_numeric_amount_invalid(self):
        """Non-numeric amount (string) should be invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01"],
            "paid_date": ["2023-01-01"],
            "amount": ["one hundred"],
        })
        result = classify_payment_status(df, "2023-01-15")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_invalid_due_date_invalid(self):
        """Invalid due_date should be invalid, regardless of paid_date."""
        df = pd.DataFrame({
            "due_date": ["bad_date"],
            "paid_date": ["2023-01-01"],
            "amount": [100.0],
        })
        result = classify_payment_status(df, "2023-01-15")
        assert result["payment_status"].iloc[0] == "invalid"

    def test_reference_date_as_datetime(self):
        """Reference date can be a datetime or Timestamp."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01"],
            "paid_date": [None],
            "amount": [100.0],
        })
        # datetime object
        result = classify_payment_status(df, datetime(2023, 1, 15))
        assert result["payment_status"].iloc[0] == "overdue"

        # Timestamp
        result = classify_payment_status(df, pd.Timestamp("2023-01-15"))
        assert result["payment_status"].iloc[0] == "overdue"

    def test_invalid_reference_date_raises_valueerror(self):
        """An unparseable reference_date should raise ValueError."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01"],
            "paid_date": [None],
            "amount": [100.0],
        })
        with pytest.raises(ValueError, match="must be a valid date-like value"):
            classify_payment_status(df, "not_a_date")

    def test_non_default_column_names(self):
        """Function should accept custom column names."""
        df = pd.DataFrame({
            "dd": ["2023-01-01"],
            "pd": ["2023-01-01"],
            "amt": [150.0],
        })
        result = classify_payment_status(
            df,
            reference_date="2023-01-15",
            due_date_col="dd",
            paid_date_col="pd",
            amount_col="amt",
            output_col="status",
        )
        assert "status" in result.columns
        assert result["status"].iloc[0] == "paid_on_time"

    def test_original_dataframe_not_mutated(self):
        """The original DataFrame should not be modified."""
        original_df = pd.DataFrame(
            {"due_date": ["2023-01-01"], "paid_date": ["2023-01-01"], "amount": [100.0]}
        )
        df_copy = original_df.copy()
        result = classify_payment_status(original_df, "2023-01-15")
        assert original_df.equals(df_copy), "Original DataFrame was mutated"
        # Also check that the returned DataFrame has the new column but original does not
        assert "payment_status" in result.columns
        assert "payment_status" not in original_df.columns

    def test_duplicated_index(self):
        """Function should handle DataFrames with non-unique indices."""
        df = pd.DataFrame(
            {
                "due_date": ["2023-01-01", "2023-01-01"],
                "paid_date": ["2023-01-01", None],
                "amount": [100.0, 100.0],
            },
            index=[0, 0],
        )
        result = classify_payment_status(df, "2023-01-15")
        assert result.index.tolist() == [0, 0]
        assert list(result["payment_status"]) == ["paid_on_time", "overdue"]

    def test_paid_exactly_on_due_date(self):
        """paid_date equal to due_date should be paid_on_time."""
        df = pd.DataFrame({
