import pandas as pd
import pytest
from ise26.targets import calculate_conversion_rate


def test_basic_conversion_calculation():
    df = pd.DataFrame({
        "channel": ["email", "social", "search"],
        "visits": [100, 200, 150],
        "conversions": [5, 12, 8],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["email", "search", "social"],
        "visits": [100.0, 150.0, 200.0],
        "conversions": [5.0, 8.0, 12.0],
        "conversion_rate": [0.05, 0.05333333333333334, 0.06],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_null_visits_treated_as_zero():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [None, 100],
        "conversions": [10, 5],
    })
    result = calculate_conversion_rate(df)
    assert result.loc[result["channel"] == "a", "visits"].iloc[0] == 0.0
    assert result.loc[result["channel"] == "a", "conversion_rate"].iloc[0] == 0.0

    })
    result = calculate_conversion_rate(df)
    assert result["conversion_rate"].iloc[0] == 0.0


def test_duplicate_channels_aggregated():
    df = pd.DataFrame({
        "channel": ["email", "email", "social"],
        "visits": [100, 200, 150],
        "conversions": [5, 10, 8],
    })
    result = calculate_conversion_rate(df)
    email_row = result[result["channel"] == "email"]
    assert email_row["visits"].iloc[0] == 300.0
    assert email_row["conversions"].iloc[0] == 15.0
    assert email_row["conversion_rate"].iloc[0] == 0.05


def test_all_nan_or_missing():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [None, None],
        "conversions": [None, None],
    })
    result = calculate_conversion_rate(df)
    assert result["visits"].sum() == 0.0
    assert result["conversions"].sum() == 0.0
    assert (result["conversion_rate"] == 0.0).all()


def test_empty_dataframe():
    df = pd.DataFrame(columns=["channel", "visits", "conversions"])
    result = calculate_conversion_rate(df)
    assert result.empty
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]


def test_single_row():
    df = pd.DataFrame({
        "channel": ["only"],
        "visits": [100],
        "conversions": [1],
    })
    result = calculate_conversion_rate(df)
    assert result["conversion_rate"].iloc[0] == 0.01


def test_dtype_of_columns():
    df = pd.DataFrame({
        "channel": ["a"],
        "visits": [10],
        "conversions": [2],
    })
    result = calculate_conversion_rate(df)
    assert result["visits"].dtype == float
    assert result["conversions"].dtype == float
    assert result["conversion_rate"].dtype == float


def test_custom_column_names():
    df = pd.DataFrame({
        "group": ["a", "b"],
        "views": [100, None],
        "purchases": [3, 5],
    })
    result = calculate_conversion_rate(df, group_col="group", visits_col="views", conversions_col="purchases")
    assert "group" in result.columns
    assert "views" not in result.columns
    assert result.loc[result["group"] == "b", "visits"].iloc[0] == 0.0
    assert result.loc[result["group"] == "b", "conversions"].iloc[0] == 5.0
    assert result.loc[result["group"] == "a", "conversion_rate"].iloc[0] == 0.03


def test_sorting_by_channel_alphabetically():
    df = pd.DataFrame({
        "channel": ["z", "a", "m"],
        "visits": [1, 2, 3],
        "conversions": [0, 1, 0],
    })
    result = calculate_conversion_rate(df)
    assert result["channel"].tolist() == ["a", "m", "z"]


def test_zero_conversions_divide_by_nonzero():
    df = pd.DataFrame({
        "channel": ["x"],
        "visits": [100],
        "conversions": [0],
    })
    result = calculate_conversion_rate(df)
    assert result["conversion_rate"].iloc[0] == 0.0


def test_high_precision_rate():
    df = pd.DataFrame({
        "channel": ["x"],
        "visits": [3],
        "conversions": [1],
    })
    result = calculate_conversion_rate(df)
    assert result["conversion_rate"].iloc[0] == pytest.approx(1/3)
