import pandas as pd
import pytest
from ise26.targets import calculate_conversion_rate


def test_basic_aggregation():
    """Multiple channels with valid numeric values."""
    df = pd.DataFrame({
        "channel": ["email", "social", "email", "search"],
        "visits": [100, 200, 150, 300],
        "conversions": [5, 10, 8, 15],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["email", "search", "social"],
        "visits": [250.0, 300.0, 200.0],
        "conversions": [13.0, 15.0, 10.0],
        "conversion_rate": [13.0 / 250.0, 15.0 / 300.0, 10.0 / 200.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_nulls_treated_as_zero():
    """Missing values in visits or conversions become zero."""
    df = pd.DataFrame({
        "channel": ["a", "b", "a"],
        "visits": [None, 100, 50],
        "conversions": [10, None, 5],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [50.0, 100.0],
        "conversions": [15.0, 0.0],
        "conversion_rate": [15.0 / 50.0, 0.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_invalid_numeric_strings_treated_as_zero():
    """Non-numeric strings in numeric columns become zero."""
    df = pd.DataFrame({
        "channel": ["x", "y"],
        "visits": ["abc", 50],
        "conversions": [10, "xyz"],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["x", "y"],
        "visits": [0.0, 50.0],
        "conversions": [10.0, 0.0],
        "conversion_rate": [0.0, 0.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_zero_visits():
    """When visits are zero, conversion_rate is zero."""
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0, 20],
        "conversions": [5, 2],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0.0, 20.0],
        "conversions": [5.0, 2.0],
        "conversion_rate": [0.0, 2.0 / 20.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_duplicate_channels_summed():
    """Rows with same channel are aggregated."""
    df = pd.DataFrame({
        "channel": ["c", "c", "c"],
        "visits": [10, 20, 30],
        "conversions": [1, 2, 3],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["c"],
        "visits": [60.0],
        "conversions": [6.0],
        "conversion_rate": [6.0 / 60.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_empty_dataframe():
    """Empty input yields empty DataFrame with correct columns."""
    df = pd.DataFrame(columns=["channel", "visits", "conversions"])
    result = calculate_conversion_rate(df)
    assert result.empty
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]


def test_single_row():
    """Single row works correctly."""
    df = pd.DataFrame({
        "channel": ["only"],
        "visits": [100],
        "conversions": [5],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["only"],
        "visits": [100.0],
        "conversions": [5.0],
        "conversion_rate": [0.05],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_sorting_alphabetical():
    """Channels are sorted alphabetically in the output."""
    df = pd.DataFrame({
        "channel": ["z", "a", "m"],
        "visits": [10, 20, 30],
        "conversions": [1, 2, 3],
    })
    result = calculate_conversion_rate(df)
    assert list(result["channel"]) == ["a", "m", "z"]


def test_custom_column_names():
    """Uses custom column names passed as arguments."""
    df = pd.DataFrame({
        "grp": ["x", "y", "x"],
        "vst": [100, 200, 150],
        "cvr": [10, 20, 15],
    })
    result = calculate_conversion_rate(
        df, group_col="grp", visits_col="vst", conversions_col="cvr"
    )
    expected = pd.DataFrame({
        "grp": ["x", "y"],
        "visits": [250.0, 200.0],
        "conversions": [25.0, 20.0],
        "conversion_rate": [25.0 / 250.0, 20.0 / 200.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_all_zeros_and_nans():
    """All visits and conversions zero or NaN yields zero rate."""
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0, None],
        "conversions": [None, 0],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0.0, 0.0],
        "conversions": [0.0, 0.0],
        "conversion_rate": [0.0, 0.0],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)
