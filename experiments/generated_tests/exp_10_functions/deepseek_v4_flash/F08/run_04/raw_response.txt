import pandas as pd
import pytest
from ise26.targets import calculate_conversion_rate


def test_basic_conversion_rate():
    df = pd.DataFrame({
        "channel": ["email", "social", "email", "search"],
        "visits": [100, 200, 150, 50],
        "conversions": [10, 30, 15, 2]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["email", "search", "social"],
        "visits": [250.0, 50.0, 200.0],
        "conversions": [25.0, 2.0, 30.0],
        "conversion_rate": [0.1, 0.04, 0.15]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_missing_values_treated_as_zero():
    df = pd.DataFrame({
        "channel": ["a", "b", "a", "b"],
        "visits": [100, None, 200, 150],
        "conversions": [10, 20, None, 30]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [300.0, 150.0],
        "conversions": [10.0, 50.0],
        "conversion_rate": [10/300, 50/150]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_numeric_values_treated_as_zero():
    df = pd.DataFrame({
        "channel": ["x", "y", "x"],
        "visits": ["abc", "100", "200"],
        "conversions": ["10", "def", "20"]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["x", "y"],
        "visits": [200.0, 100.0],
        "conversions": [30.0, 0.0],
        "conversion_rate": [0.15, 0.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_zero_visits_returns_zero_rate():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0, 100],
        "conversions": [5, 10]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0.0, 100.0],
        "conversions": [5.0, 10.0],
        "conversion_rate": [0.0, 0.1]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_duplicate_channels_aggregation():
    df = pd.DataFrame({
        "channel": ["a", "b", "a", "b", "a"],
        "visits": [10, 20, 30, 40, 50],
        "conversions": [1, 2, 3, 4, 5]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [90.0, 60.0],
        "conversions": [9.0, 6.0],
        "conversion_rate": [0.1, 0.1]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_sorted_alphabetically():
    df = pd.DataFrame({
        "channel": ["z", "a", "m"],
        "visits": [100, 200, 150],
        "conversions": [10, 20, 15]
    })
    result = calculate_conversion_rate(df)
    assert list(result["channel"]) == ["a", "m", "z"]


def test_custom_column_names():
    df = pd.DataFrame({
        "group": ["x", "y", "x"],
        "visit_cnt": [100, 200, 300],
        "conv_cnt": [10, 20, 30]
    })
    result = calculate_conversion_rate(
        df, group_col="group", visits_col="visit_cnt", conversions_col="conv_cnt"
    )
    expected = pd.DataFrame({
        "group": ["x", "y"],
        "visits": [400.0, 200.0],
        "conversions": [40.0, 20.0],
        "conversion_rate": [0.1, 0.1]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_empty_dataframe():
    df = pd.DataFrame(columns=["channel", "visits", "conversions"])
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(columns=["channel", "visits", "conversions", "conversion_rate"])
    pd.testing.assert_frame_equal(result, expected)


def test_single_row():
    df = pd.DataFrame({
        "channel": ["only"],
        "visits": [50],
        "conversions": [5]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["only"],
        "visits": [50.0],
        "conversions": [5.0],
        "conversion_rate": [0.1]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_large_numbers():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [1e9, 2e9],
        "conversions": [1e6, 2e6]
    })
    result = calculate_conversion_rate(df)
    expected_rate_a = 1e6 / 1e9
    expected_rate_b = 2e6 / 2e9
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [1e9, 2e9],
        "conversions": [1e6, 2e6],
        "conversion_rate": [expected_rate_a, expected_rate_b]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_all_zeros():
    df = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0, 0],
        "conversions": [0, 0]
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["a", "b"],
        "visits": [0.0, 0.0],
        "conversions": [0.0, 0.0],
        "conversion_rate": [0.0, 0.0]
    })
    pd.testing.assert_frame_equal(result, expected)
