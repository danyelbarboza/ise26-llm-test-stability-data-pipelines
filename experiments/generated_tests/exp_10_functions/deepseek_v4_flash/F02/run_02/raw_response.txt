from ise26.targets import deduplicate_events
import pandas as pd
import pytest


def test_keeps_most_recent_for_non_null_id():
    """For a given event_id, keep the row with the most recent timestamp."""
    df = pd.DataFrame({
        "event_id": ["A", "A", "A"],
        "updated_at": ["2024-01-01", "2024-01-03", "2024-01-02"],
        "data": [1, 2, 3],
    })
    result = deduplicate_events(df)
    expected = df.iloc[[1]].reset_index(drop=True)  # row with timestamp 2024-01-03
    pd.testing.assert_frame_equal(result, expected)


def test_preserves_rows_with_null_id():
    """Rows with null event_id are kept as-is (no deduplication)."""
    df = pd.DataFrame({
        "event_id": [None, None, "A"],
        "updated_at": ["2024-01-01", "2024-01-02", "2024-01-01"],
        "data": [10, 20, 30],
    })
    result = deduplicate_events(df)
    # Expected: both null rows preserved, one 'A' kept (most recent)
    expected = pd.DataFrame({
        "event_id": [None, None, "A"],
        "updated_at": ["2024-01-01", "2024-01-02", "2024-01-01"],
        "data": [10, 20, 30],
    })
    pd.testing.assert_frame_equal(result, expected)


def test_null_id_not_treated_as_duplicate_group():
    """Null IDs should not be grouped; each null row stays."""
    df = pd.DataFrame({
        "event_id": [None, None],
        "updated_at": ["2024-01-01", "2024-01-02"],
        "data": [1, 2],
    })
    result = deduplicate_events(df)
    expected = df.copy()
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_timestamp_treated_as_oldest():
    """Invalid timestamps should be considered older than valid ones."""
    df = pd.DataFrame({
        "event_id": ["A", "A"],
        "updated_at": ["not_a_date", "2024-01-01"],
        "data": [1, 2],
    })
    result = deduplicate_events(df)
    # Valid timestamp row should be kept
    expected = df.iloc[[1]].reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_timestamp_tie_keeps_last_row_in_original_order():
    """When timestamps are equal, keep the row that appears last in the DataFrame."""
    df = pd.DataFrame({
        "event_id": ["A", "A", "A"],
        "updated_at": ["2024-01-01", "2024-01-01", "2024-01-01"],
        "data": [1, 2, 3],
    })
    result = deduplicate_events(df)
    expected = df.iloc[[2]].reset_index(drop=True)  # last among equal timestamps
    pd.testing.assert_frame_equal(result, expected)


def test_original_input_not_mutated():
    """The function should not modify the input DataFrame."""
    df = pd.DataFrame({
        "event_id": ["A", "A"],
        "updated_at": ["2024-01-01", "2024-01-02"],
        "data": [1, 2],
    })
    original = df.copy()
    deduplicate_events(df)
    pd.testing.assert_frame_equal(df, original)


def test_no_duplicates_preserves_all_non_null_rows():
    """If no duplicate event_ids exist, all rows are kept unchanged."""
    df = pd.DataFrame({
        "event_id": ["A", "B", "C"],
        "updated_at": ["2024-01-01", "2024-01-02", "2024-01-03"],
        "data": [1, 2, 3],
    })
    result = deduplicate_events(df)
    expected = df.copy()
    pd.testing.assert_frame_equal(result, expected)


def test_multiple_groups_keep_most_recent_per_group():
    """Multiple event_id groups each keep their most recent row."""
    df = pd.DataFrame({
        "event_id": ["A", "A", "B", "B", "C"],
        "updated_at": ["2024-01-01", "2024-01-03", "2024-01-02", "2024-01-01", "2024-01-02"],
        "data": [1, 2, 3, 4, 5],
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": ["A", "B", "C"],
        "updated_at": ["2024-01-03", "2024-01-02", "2024-01-02"],
        "data": [2, 3, 5],
    }).reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_mixed_null_and_non_null_with_duplicates():
    """Null rows preserved, non‑null deduplicated."""
    df = pd.DataFrame({
        "event_id": [None, None, "A", "A", "B", None],
        "updated_at": ["2024-01-01", "2024-01-02", "2024-01-01", "2024-01-03", "2024-01-02", "2024-01-03"],
        "data": [1, 2, 3, 4, 5, 6],
    })
    result = deduplicate_events(df)
    # null rows: indices 0,1,5 ; A keeps index3, B keeps index4
    expected = pd.DataFrame({
        "event_id": [None, None, "A", "B", None],
        "updated_at": ["2024-01-01", "2024-01-02", "2024-01-03", "2024-01-02", "2024-01-03"],
        "data": [1, 2, 4, 5, 6],
    }).reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_empty_dataframe():
    """Empty DataFrame should return empty DataFrame."""
    df = pd.DataFrame(columns=["event_id", "updated_at", "data"])
    result = deduplicate_events(df)
    expected = df.copy()
    pd.testing.assert_frame_equal(result, expected)


def test_all_invalid_timestamps():
    """If all timestamps are invalid, keep last row per ID."""
    df = pd.DataFrame({
        "event_id": ["A", "A", "B"],
        "updated_at": ["bad1", "bad2", "bad3"],
        "data": [1, 2, 3],
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": ["A", "B"],
        "updated_at": ["bad2", "bad3"],  # last rows per group
        "data": [2, 3],
    }).reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_custom_column_names():
    """Function works with non‑default column names."""
    df = pd.DataFrame({
        "my_id": ["X", "X", "Y"],
        "my_ts": ["2024-01-01", "2024-01-03", "2024-01-02"],
        "value": [10, 20, 30],
    })
    result = deduplicate_events(df, id_col="my_id", timestamp_col="my_ts")
    expected = pd.DataFrame({
        "my_id": ["X", "Y"],
        "my_ts": ["2024-01-03", "2024-01-02"],
        "value": [20, 30],
    }).reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_order_of_column_columns_preserved():
    """Output DataFrame should have same columns as input, order preserved."""
    df = pd.DataFrame({
        "event_id": ["A", "A"],
        "updated_at": ["2024-01-01", "2024-01-02"],
        "data": [1, 2],
        "extra": [True, False],
    })
    result = deduplicate_events(df)
    expected_columns = list(df.columns)
    assert list(result.columns) == expected_columns
