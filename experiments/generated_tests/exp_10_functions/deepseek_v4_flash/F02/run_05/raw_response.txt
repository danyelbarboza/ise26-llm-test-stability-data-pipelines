import pandas as pd
import pytest
from ise26.targets import deduplicate_events


def test_no_duplicates():
    df = pd.DataFrame({
        "event_id": [1, 2, 3],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "data": ["a", "b", "c"]
    })
    result = deduplicate_events(df)
    expected = df.copy()
    pd.testing.assert_frame_equal(result, expected)


def test_duplicates_keep_most_recent():
    df = pd.DataFrame({
        "event_id": [1, 1, 2],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "data": ["old", "new", "c"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "updated_at": ["2023-01-02", "2023-01-03"],
        "data": ["new", "c"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_null_ids_preserved():
    df = pd.DataFrame({
        "event_id": [1, None, None, 2],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04"],
        "data": ["a", "b", "c", "d"]
    })
    result = deduplicate_events(df)
    # event_id 1 kept most recent (only one), nulls preserved, event_id 2 kept
    expected = pd.DataFrame({
        "event_id": [1, None, None, 2],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04"],
        "data": ["a", "b", "c", "d"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_duplicates_with_null_ids_kept():
    df = pd.DataFrame({
        "event_id": [1, 1, None, None, 2, 2],
        "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04", "2023-01-05", "2023-01-06"],
        "data": ["a", "b", "c", "d", "e", "f"]
    })
    result = deduplicate_events(df)
    # For non-null: keep most recent per id: id1->b, id2->f
    # Nulls preserved
    expected = pd.DataFrame({
        "event_id": [1, None, None, 2],
        "updated_at": ["2023-01-02", "2023-01-03", "2023-01-04", "2023-01-06"],
        "data": ["b", "c", "d", "f"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_timestamp_treated_as_older():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["invalid", "2023-01-02"],
        "data": ["old", "new"]
    })
    result = deduplicate_events(df)
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-02"],
        "data": ["new"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_timestamp_older_than_all():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["invalid", "also_invalid"],
        "data": ["a", "b"]
    })
    result = deduplicate_events(df)
    # Both invalid, both treated as NaT, sort stable keeps last
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["also_invalid"],
        "data": ["b"]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_mixed_scenario():
    df = pd.DataFrame({
        "event_id": [1, 2, 3, 1, 2, None, None, 3],
        "updated_at": ["2023-01-01", "2023-01-02", "invalid", "2023-01-03", "2023-01-04", "2023-01-05", None, "2023-01-06"],
        "data": ["a", "b", "c", "d", "e", "f", "g", "h"]
    })
    result = deduplicate_events(df)
    # id1: timestamps 2023-01-01 and 2023-01-03 -> keep 2023-01-03 (data d)
    # id2: 2023-01-02 and 2023-01-04 -> keep 2023-01-04 (data e)
    # id3: invalid and 2023-01-06 -> keep 2023-01-06 (data h)
    # nulls preserved: row5 (2023-01-05, data f) and row6 (NaT, data g) because they are null
    expected = pd.DataFrame({
        "event_id": [1, 2, 3, None, None],
        "updated_at": ["2023-01-03", "2023-01-04", "2023-01-06", "2023-01-05", None],
        "data": ["d", "e", "h", "f", "g"]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_custom_column_names():
    df = pd.DataFrame({
        "id": [10, 10, 20],
        "ts": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "value": [1, 2, 3]
    })
    result = deduplicate_events(df, id_col="id", timestamp_col="ts")
    expected = pd.DataFrame({
        "id": [10, 20],
        "ts": ["2023-01-02", "2023-01-03"],
        "value": [2, 3]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_input_unchanged():

        "updated_at": ["2023-01-01", "2023-01-02"],
        "data": ["a", "b"]

def test_duplicate_across_custom_columns():
    df = pd.DataFrame({
        "code": ["x", "x", "y"],
        "date": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "val": [10, 20, 30]
    })
    result = deduplicate_events(df, id_col="code", timestamp_col="date")
    expected = pd.DataFrame({
        "code": ["x", "y"],
        "date": ["2023-01-02", "2023-01-03"],
        "val": [20, 30]
    })
    pd.testing.assert_frame_equal(result, expected)
