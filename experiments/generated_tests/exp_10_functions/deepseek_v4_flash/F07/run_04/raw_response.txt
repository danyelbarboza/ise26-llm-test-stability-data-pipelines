```python
import pytest
import pandas as pd
import numpy as np
from ise26.targets import parse_order_items_json


# Helper to create a clean DataFrame for testing
def _build_df(items_json_list, order_ids=None, json_col="items_json", order_id_col="order_id"):
    if order_ids is None:
        order_ids = list(range(len(items_json_list)))
    data = {order_id_col: order_ids, json_col: items_json_list}
    return pd.DataFrame(data)


class TestParseOrderItemsJson:

    def test_basic_expansion(self):
        """One order with two items: expands correctly and computes totals."""
        df = _build_df([
            '[{"sku": "A", "quantity": 2, "unit_price": 10.5}, {"sku": "B", "quantity": 1, "unit_price": 20}]'
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [0, 0],
            "sku": ["A", "B"],
            "quantity": [2.0, 1.0],
            "unit_price": [10.5, 20.0],
            "item_total": [21.0, 20.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_multiple_orders(self):
        """Multiple orders with different numbers of items."""
        df = _build_df([
            '[{"sku": "X", "quantity": 1, "unit_price": 5.0}]',
            '[{"sku": "Y", "quantity": 3, "unit_price": 7.5}, {"sku": "Z", "quantity": 2, "unit_price": 3.0}]',
            '[{"sku": "W", "quantity": 4, "unit_price": 2.0}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [0, 1, 1, 2],
            "sku": ["X", "Y", "Z", "W"],
            "quantity": [1.0, 3.0, 2.0, 4.0],
            "unit_price": [5.0, 7.5, 3.0, 2.0],
            "item_total": [5.0, 22.5, 6.0, 8.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_invalid_json_skipped(self):
        """Rows with invalid JSON are ignored."""
        df = _build_df([
            'not valid json',
            '[{"sku": "A", "quantity": 1, "unit_price": 10.0}]',
            None,
            '[{"sku": "B", "quantity": 2, "unit_price": 5.0}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1, 3],
            "sku": ["A", "B"],
            "quantity": [1.0, 2.0],
            "unit_price": [10.0, 5.0],
            "item_total": [10.0, 10.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_json_list_skipped(self):
        """Row with an empty JSON list is skipped."""
        df = _build_df([
            '[]',
            '[{"sku": "C", "quantity": 1, "unit_price": 1.0}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1],
            "sku": ["C"],
            "quantity": [1.0],
            "unit_price": [1.0],
            "item_total": [1.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_missing_quantity_and_price_default_to_zero(self):
        """Items lacking quantity or unit_price fields use zero."""
        df = _build_df([
            '[{"sku": "D"}, {"sku": "E", "quantity": "abc"}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [0, 0],
            "sku": ["D", "E"],
            "quantity": [0.0, 0.0],
            "unit_price": [0.0, 0.0],
            "item_total": [0.0, 0.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_non_numeric_strings_default_to_zero(self):
        """Strings that are not numeric in quantity/unit_price become zero."""
        df = _build_df([
            '[{"sku": "F", "quantity": "two", "unit_price": "free"}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [0],
            "sku": ["F"],
            "quantity": [0.0],
            "unit_price": [0.0],
            "item_total": [0.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_zero_quantity_and_price(self):
        """Valid zero values are kept."""
        df = _build_df([
            '[{"sku": "G", "quantity": 0, "unit_price": 100.0}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [0],
            "sku": ["G"],
            "quantity": [0.0],
            "unit_price": [100.0],
            "item_total": [0.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_none_json_column(self):
        """Row with None in json_col is skipped."""
        df = _build_df([
            None,
            '[{"sku": "H", "quantity": 1, "unit_price": 2.0}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [1],
            "sku": ["H"],
            "quantity": [1.0],
            "unit_price": [2.0],
            "item_total": [2.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_dataframe(self):
        """Empty input returns empty DataFrame with correct columns."""
        df = pd.DataFrame(columns=["order_id", "items_json"])
        result = parse_order_items_json(df)
        expected = pd.DataFrame(columns=["order_id", "sku", "quantity", "unit_price", "item_total"])
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_column_names(self):
        """Works with non-default json_col and order_id_col."""
        df = pd.DataFrame({
            "order_ref": [101, 102],
            "items": [
                '[{"sku": "X", "quantity": 1, "unit_price": 10.0}]',
                '[{"sku": "Y", "quantity": 2, "unit_price": 5.5}]',
            ]
        })
        result = parse_order_items_json(df, json_col="items", order_id_col="order_ref")
        expected = pd.DataFrame({
            "order_ref": [101, 102],
            "sku": ["X", "Y"],
            "quantity": [1.0, 2.0],
            "unit_price": [10.0, 5.5],
            "item_total": [10.0, 11.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_preserves_order_of_items_and_orders(self):
        """Items appear in the same order as in JSON lists and as rows."""
        df = _build_df([
            '[{"sku": "C"}, {"sku": "A"}]',
            '[{"sku": "B"}]',
        ])
        result = parse_order_items_json(df)
        # Order: first row items: C, A; then row: B
        expected = pd.DataFrame({
            "order_id": [0, 0, 1],
            "sku": ["C", "A", "B"],
            "quantity": [0.0, 0.0, 0.0],
            "unit_price": [0.0, 0.0, 0.0],
            "item_total": [0.0, 0.0, 0.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_sku_can_be_none(self):
        """sku field may be None in JSON; output keeps None."""
        df = _build_df([
            '[{"quantity": 1, "unit_price": 2.0}]',
        ])
        result = parse_order_items_json(df)
        expected = pd.DataFrame({
            "order_id": [0],
            "sku": [None],
            "quantity": [1.0],
            "unit_price": [2.0],
            "item_total": [2.0],
        })
        pd.testing.assert_frame_equal(result, expected)

    def test_item_total_is_float(self):
        """Item_total should be float (can be fractional)."""
        df = _build_df([
            '[{"sku": "X", "quantity": 3, "unit_price": 2.5}]',
        ])
        result = parse_order_items_json(df)
        assert result["item_total"].dtype == np.float64
        assert result["item_total"].iloc[0] == 7.5
```
