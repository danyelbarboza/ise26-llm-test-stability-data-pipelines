import pytest
import pandas as pd
from ise26.targets import join_customers_orders


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _check_original_unchanged(original, after_call):
    """Assert that `original` (DataFrame) has not been mutated."""
    pd.testing.assert_frame_equal(original, after_call)

# Tests
# ---------------------------------------------------------------------------

class TestJoinCustomersOrders:
    """Suite of tests for join_customers_orders."""

    def test_basic_full_outer_join(self, customers_basic, orders_basic):
        """Basic join – matched, customer only, order only."""
        result = join_customers_orders(customers_basic, orders_basic, customer_key="customer_id")
        # Expected rows:
        # customer 1 + order 101 -> matched
        # customer 2 + order 102 -> matched
        # customer 3 (no order) -> customer_without_order
        # order 103 (no customer) -> order_without_customer
        assert len(result) == 4
        assert "record_status" in result.columns
        statuses = result["record_status"].tolist()
        # The exact order is deterministic; we check counts
        assert statuses.count("matched") == 2
        assert statuses.count("customer_without_order") == 1
        assert statuses.count("order_without_customer") == 1
        # Check data correctness
        matched_rows = result[result["record_status"] == "matched"]
        assert matched_rows["name"].tolist() == ["Alice", "Bob"]
        assert matched_rows["order_id"].tolist() == [101, 102]
        # Check customer without order
        cust_only = result[result["record_status"] == "customer_without_order"]
        assert cust_only.iloc[0]["customer_id"] == 3
        assert cust_only.iloc[0]["name"] == "Charlie"
        assert pd.isna(cust_only.iloc[0]["order_id"])
        # Check order without customer
        order_only = result[result["record_status"] == "order_without_customer"]
        assert order_only.iloc[0]["customer_id"] == 4
        assert order_only.iloc[0]["order_id"] == 103

    def test_null_join_keys(self, customers_with_nulls, orders_with_nulls):
        """Rows with null keys must never be classified as matched."""
        result = join_customers_orders(customers_with_nulls, orders_with_nulls)
        # customer_null_rows: id = None (row index 2 -> Nully) -> customer_without_order
        # order_null_rows: id = None (row 1) -> order_without_customer
        # Also matched: customer 1 + order 1 -> matched
        # customer 3 (no order) -> customer_without_order
        # order 4 (no customer) -> order_without_customer
        # Check that matched rows have non-null keys
        matched = result[result["record_status"] == "matched"]
        assert matched["customer_id"].notna().all()
        # Count statuses
        assert result["record_status"].value_counts().to_dict() == {
            "matched": 1,
            "customer_without_order": 2,  # customer 3 and null-key customer
            "order_without_customer": 2,  # order 4 and null-key order
        }
        # Check null-key rows status
        null_cust_rows = result[result["name"] == "Nully"]
        assert null_cust_rows["record_status"].iloc[0] == "customer_without_order"
        null_order_rows = result[result["order_id"] == 102]
        assert null_order_rows["record_status"].iloc[0] == "order_without_customer"

    def test_duplicate_keys(self, customers_duplicates, orders_duplicates):
        """Duplicates should lead to a cartesian product within same key."""
        result = join_customers_orders(customers_duplicates, orders_duplicates, customer_key="customer_id")
        # Key 1: customers have 2 rows, orders have 2 rows -> 4 matched rows
        # Key 2: 1 customer, 1 order -> 1 matched row
        # No unmatched rows expected
        assert len(result) == 5
        assert (result["record_status"] == "matched").all()
        # Check that each combination appears
        # For key 1: customer names "Alice" and "Alice2" paired with order_ids 101 and 103
        combos = set(zip(result["name"], result["order_id"]))
        assert combos == {("Alice", 101), ("Alice", 103), ("Alice2", 101), ("Alice2", 103), ("Bob", 102)}

    def test_missing_columns_filled_with_na(self, customers_basic, orders_basic):
        """Unmatched rows should have missing columns filled with NA."""
        result = join_customers_orders(customers_basic, orders_basic)
        # order_without_customer row has no name, customer_without_order has no order_id/amount
        order_only = result[result["record_status"] == "order_without_customer"]
        assert order_only["name"].isna().all()
        cust_only = result[result["record_status"] == "customer_without_order"]
        assert cust_only["order_id"].isna().all()
        assert cust_only["amount"].isna().all()

    def test_deterministic_sorting(self, customers_basic, orders_basic):
        """The result should be sorted in a deterministic order."""
        result1 = join_customers_orders(customers_basic, orders_basic)
        result2 = join_customers_orders(customers_basic, orders_basic)
        pd.testing.assert_frame_equal(result1, result2)

    def test_index_reset(self, customers_basic, orders_basic):
        """The result index should be a reset RangeIndex."""
        result = join_customers_orders(customers_basic, orders_basic)
        assert isinstance(result.index, pd.RangeIndex)
        assert result.index[0] == 0
        assert len(result.index) == len(result)

    def test_original_datasets_unchanged(self, customers_basic, orders_basic):
        """Input DataFrames must not be mutated."""
        cust_orig = customers_basic.copy()
        ord_orig = orders_basic.copy()
        join_customers_orders(customers_basic, orders_basic)
        _check_original_unchanged(cust_orig, customers_basic)
        _check_original_unchanged(ord_orig, orders_basic)

    def test_empty_dataframes(self, customers_empty, orders_empty):
        """Both DataFrames empty -> empty result."""
        result = join_customers_orders(customers_empty, orders_empty)
        assert len(result) == 0
        assert "record_status" in result.columns

    def test_empty_customers(self, customers_empty, orders_basic):
        """Empty customers, non-empty orders -> all rows are order_without_customer."""
        result = join_customers_orders(customers_empty, orders_basic)
        expected_len = len(orders_basic)
        assert len(result) == expected_len
        assert (result["record_status"] == "order_without_customer").all()

    def test_empty_orders(self, customers_basic, orders_empty):
        """Non-empty customers, empty orders -> all rows are customer_without_order."""
        result = join_customers_orders(customers_basic, orders_empty)
        expected_len = len(customers_basic)
        assert len(result) == expected_len
        assert (result["record_status"] == "customer_without_order").all()

    def test_custom_key_name(self, customers_basic, orders_basic):
        """Use a different key column name."""
        cust = customers_basic.rename(columns={"customer_id": "key"})
        ords = orders_basic.rename(columns={"customer_id": "key"})
        result = join_customers_orders(cust, ords, customer_key="key")
        assert "key" in result.columns
        assert result.loc[0, "key"] == 1  # first row (deterministic)

    def test_partial_key_missing_in_one_df(self):
        """Key column exists in customers but not in orders – should raise."""
        cust = pd.DataFrame({"customer_id": [1], "name": ["A"]})
        orders = pd.DataFrame({"order_id": [10]})
        with pytest.raises(KeyError):
            join_customers_orders(cust, orders, customer_key="customer_id")

    def test_all_null_keys(self):
        """All keys are null – no matches possible."""
        cust = pd.DataFrame({"customer_id": [None, None], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [None], "order_id": [1]})
        result = join_customers_orders(cust, orders, customer_key="customer_id")
        # 2 customers + 1 order = 3 rows, all unmatched
        assert len(result) == 3
        for _, row in result.iterrows():
            assert row["record_status"] != "matched"

    def test_no_null_keys_all_match(self):
        """All keys present and matching – all rows should be matched."""
        cust = pd.DataFrame({"customer_id": [1, 2], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "order_id": [10, 20]})
        result = join_customers_orders(cust, orders)
        assert len(result) == 2
        assert (result["record_status"] == "matched").all()
