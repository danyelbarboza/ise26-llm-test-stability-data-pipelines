import pytest
import pandas as pd
from ise26.targets import join_customers_orders

class TestJoinCustomersOrders:
    """Test suite for join_customers_orders function."""

    # ----------------------------------------------------------------
    # Basic full outer join scenarios
    # ----------------------------------------------------------------
    def test_basic_full_outer_join_with_matched_only(self):
        customers = pd.DataFrame({
            'customer_id': [1, 2, 3],
            'name': ['Alice', 'Bob', 'Charlie']
        })
        orders = pd.DataFrame({
            'customer_id': [1, 2, 3],
            'order_id': [101, 102, 103],
            'amount': [100, 200, 300]
        })
        result = join_customers_orders(customers, orders)
        assert result.shape[0] == 3
        assert set(result['record_status'].unique()) == {'matched'}
        assert 'name' in result.columns
        assert 'order_id' in result.columns
        assert 'amount' in result.columns

    def test_matched_and_unmatched_customers(self):
        customers = pd.DataFrame({
            'customer_id': [1, 2],
            'name': ['Alice', 'Bob']
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101]
        })
        result = join_customers_orders(customers, orders)
        # customer 1 matched, customer 2 without order
        assert result.shape[0] == 2
        statuses = result.set_index('customer_id')['record_status'].to_dict()
        assert statuses[1] == 'matched'
        assert statuses[2] == 'customer_without_order'

    def test_matched_and_unmatched_orders(self):
        customers = pd.DataFrame({
            'customer_id': [1],
            'name': ['Alice']
        })
        orders = pd.DataFrame({
            'customer_id': [1, 2],
            'order_id': [101, 102]
        })
        result = join_customers_orders(customers, orders)
        assert result.shape[0] == 2
        statuses = result.set_index('customer_id')['record_status'].to_dict()
        assert statuses[1] == 'matched'
        # order with customer_id=2 has no customer -> order_without_customer
        # but the join key is 2, which is not null, and appears in orders only.
        # The status for that row (customer_id=2) should be 'order_without_customer'
        # However, the row may have customer_id = 2, name = NaN
        # We need to check the status for the row with customer_id = 2
        row2 = result[result['customer_id'] == 2]
        assert len(row2) == 1
        assert row2.iloc[0]['record_status'] == 'order_without_customer'

    def test_mixed_all_statuses(self):
        customers = pd.DataFrame({
            'customer_id': [1, 2, 4],
            'name': ['Alice', 'Bob', 'Diana']
        })
        orders = pd.DataFrame({
            'customer_id': [1, 3, 4],
            'order_id': [101, 102, 103]
        })
        result = join_customers_orders(customers, orders)
        # Expected:
        # 1 -> matched (both)
        # 2 -> customer_without_order
        # 3 -> order_without_customer
        # 4 -> matched (both)
        assert result.shape[0] == 4
        statuses = result.set_index('customer_id')['record_status'].to_dict()
        assert statuses[1] == 'matched'
        assert statuses[2] == 'customer_without_order'
        assert statuses[3] == 'order_without_customer'
        assert statuses[4] == 'matched'

    # ----------------------------------------------------------------
    # Null key handling
    # ----------------------------------------------------------------
    def test_null_customer_key_in_customers(self):
        customers = pd.DataFrame({
            'customer_id': [1, None],
            'name': ['Alice', 'Blank']
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101]
        })
        result = join_customers_orders(customers, orders)
        # customer_id=1 matched, null-key row should be customer_without_order
        assert result.shape[0] == 2
        null_rows = result[result['customer_id'].isna()]
        assert len(null_rows) == 1
        assert null_rows.iloc[0]['record_status'] == 'customer_without_order'
        # Check that the null key row is NOT matched (even though there is an order with customer_id=1? no, it's null vs 1)
        # Ensure original customers unchanged (length)
        assert len(customers) == 2

    def test_null_order_key_in_orders(self):
        customers = pd.DataFrame({
            'customer_id': [1],
            'name': ['Alice']
        })
        orders = pd.DataFrame({
            'customer_id': [1, None],
            'order_id': [101, 102]
        })
        result = join_customers_orders(customers, orders)
        # customer_id=1 matched, null-key order row should be order_without_customer
        assert result.shape[0] == 2
        null_rows = result[result['customer_id'].isna()]
        assert len(null_rows) == 1
        assert null_rows.iloc[0]['record_status'] == 'order_without_customer'

    def test_null_keys_in_both(self):
        customers = pd.DataFrame({
            'customer_id': [None],
            'name': ['Alice']
        })
        orders = pd.DataFrame({
            'customer_id': [None],
            'order_id': [101]
        })
        result = join_customers_orders(customers, orders)
        # Two separate rows: one customer_without_order, one order_without_customer
        assert result.shape[0] == 2
        statuses = result['record_status'].tolist()
        assert sorted(statuses) == ['customer_without_order', 'order_without_customer']
        # Both rows have NaN in customer_id
        assert result['customer_id'].isna().all()

    # ----------------------------------------------------------------
    # Duplicate keys
    # ----------------------------------------------------------------
    def test_duplicate_customer_key_in_customers(self):
        customers = pd.DataFrame({
            'customer_id': [1, 1],
            'name': ['Alice', 'Alicia']
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101]
        })
        result = join_customers_orders(customers, orders)
        # Both customers match the same order -> cartesian product -> 2 rows matched
        assert result.shape[0] == 2
        assert all(result['record_status'] == 'matched')

    def test_duplicate_order_key_in_orders(self):
        customers = pd.DataFrame({
            'customer_id': [1],
            'name': ['Alice']
        })
        orders = pd.DataFrame({
            'customer_id': [1, 1],
            'order_id': [101, 102]
        })
        result = join_customers_orders(customers, orders)
        assert result.shape[0] == 2
        assert all(result['record_status'] == 'matched')

    def test_duplicate_keys_both_sides(self):
        customers = pd.DataFrame({
            'customer_id': [1, 1],
            'name': ['Alice', 'Alicia']
        })
        orders = pd.DataFrame({
            'customer_id': [1, 1],
            'order_id': [101, 102]
        })
        result = join_customers_orders(customers, orders)
        # Cartesian: 2*2=4 rows, all matched
        assert result.shape[0] == 4
        assert all(result['record_status'] == 'matched')

    # ----------------------------------------------------------------
    # Empty dataframes
    # ----------------------------------------------------------------
    def test_empty_customers(self):
        customers = pd.DataFrame(columns=['customer_id', 'name'])
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101]
        })
        result = join_customers_orders(customers, orders)
        # Only the order row appears as order_without_customer
        assert result.shape[0] == 1
        assert result.iloc[0]['record_status'] == 'order_without_customer'

    def test_empty_orders(self):
        customers = pd.DataFrame({
            'customer_id': [1],
            'name': ['Alice']
        })
        orders = pd.DataFrame(columns=['customer_id', 'order_id'])
        result = join_customers_orders(customers, orders)
        assert result.shape[0] == 1
        assert result.iloc[0]['record_status'] == 'customer_without_order'

    def test_both_empty(self):
        customers = pd.DataFrame(columns=['customer_id', 'name'])
        orders = pd.DataFrame(columns=['customer_id', 'order_id'])
        result = join_customers_orders(customers, orders)
        assert result.shape[0] == 0

    # ----------------------------------------------------------------
    # Custom key column name
    # ----------------------------------------------------------------
    def test_custom_key_column(self):
        customers = pd.DataFrame({
            'cid': [1, 2],
            'name': ['Alice', 'Bob']
        })
        orders = pd.DataFrame({
            'cid': [1, 3],
            'item': ['A', 'B']
        })
        result = join_customers_orders(customers, orders, customer_key='cid')
        assert result.shape[0] == 3
        statuses = result.set_index('cid')['record_status'].to_dict()
        assert statuses[1] == 'matched'
        assert statuses[2] == 'customer_without_order'
        assert statuses[3] == 'order_without_customer'

    # ----------------------------------------------------------------
    # Preservation of original DataFrames
    # ----------------------------------------------------------------
    def test_original_dataframes_unchanged(self):
        customers = pd.DataFrame({
            'customer_id': [1, 2],
            'name': ['Alice', 'Bob']
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101]
        })
        customers_copy = customers.copy()
        orders_copy = orders.copy()
        _ = join_customers_orders(customers, orders)
        pd.testing.assert_frame_equal(customers, customers_copy)
        pd.testing.assert_frame_equal(orders, orders_copy)

    # ----------------------------------------------------------------
    # Column integrity and structure
    # ----------------------------------------------------------------
    def test_output_columns_include_all_input_columns(self):
        customers = pd.DataFrame({
            'customer_id': [1],
            'name': ['Alice'],
            'email': ['a@b.com']
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101],
            'amount': [100.0]
        })
        result = join_customers_orders(customers, orders)
        expected_cols = {'customer_id', 'name', 'email', 'order_id', 'amount', 'record_status'}
        assert set(result.columns) == expected_cols

    def test_output_has_reset_index(self):
        customers = pd.DataFrame({
            'customer_id': [2, 1],
            'name': ['Bob', 'Alice']
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_id': [101]
        })
        result = join_customers_orders(customers, orders)
        # Index should be 0,1,2...
        assert result.index.tolist() == list(range(len(result)))

    def test_record_status_column_present(self):
        customers = pd.DataFrame({'customer_id': [1]})
        orders = pd.DataFrame({'customer_id': [1]})
        result = join_customers_orders(customers, orders)
        assert 'record_status' in result.columns

    # ----------------------------------------------------------------
    # Deterministic sorting
    # ----------------------------------------------------------------
    def test_deterministic_sorting(self):
        customers = pd.DataFrame({
            'customer_id': [3, 1, 2],
            'name': ['C', 'A', 'B']
        })
        orders = pd.DataFrame({
            'customer_id': [2, 1, 3],
            'order_id': [200, 100, 300]
        })
        result1 = join_customers_orders(customers, orders)
        result2 = join_customers_orders(customers, orders)
        pd.testing.assert_frame_equal(result1, result2)

    # ----------------------------------------------------------------
    # Date columns (just to ensure no error)
    # ----------------------------------------------------------------
    def test_with_date_columns(self):
        customers = pd.DataFrame({
            'customer_id': [1],
            'signup_date': [pd.Timestamp('2020-01-01')]
        })
        orders = pd.DataFrame({
            'customer_id': [1],
            'order_date': [pd.Timestamp('2021-06-15')]
        })
        result = join_customers_orders(customers, orders)
        assert len(result) == 1
        assert result.iloc[0]['record_status'] == 'matched'
        assert result.iloc[0]['signup_date'] == pd.Timestamp('2020-01-01')
        assert result.iloc[0]['order_date'] == pd.Timestamp('2021-06-15')
