def _check_unchanged(original, after):
    """Assert that original DataFrame has not been mutated."""
    assert original is not after, "Input DataFrame mutated (same object)"
    # Check values are still the same (deep equality)
    pd.testing.assert_frame_equal(original, original)

class TestJoinCustomersOrders:
    """Test suite for join_customers_orders."""

    def test_basic_full_outer_join(self, basic_customers, basic_orders):
        customers = basic_customers.copy()
        orders = basic_orders.copy()
        result = join_customers_orders(customers, orders, customer_key="customer_id")

        # Check basic structure
        assert isinstance(result, pd.DataFrame)
        assert result.index.is_monotonic_increasing and result.index[0] == 0  # reset index
        assert "record_status" in result.columns
        assert set(result["record_status"].unique()) == {
            "matched",
            "customer_without_order",
            "order_without_customer",
        }

        # Check specific rows
        matched = result[result["record_status"] == "matched"]
        assert len(matched) == 2  # customer_id 1 and 2
        assert matched["customer_id"].tolist() == [1, 2]

        cust_only = result[result["record_status"] == "customer_without_order"]
        assert len(cust_only) == 2
        assert set(cust_only["customer_id"]) == {3, 4}

        order_only = result[result["record_status"] == "order_without_customer"]
        assert len(order_only) == 2
        assert set(order_only["customer_id"]) == {5, 6}

        # Check that matched rows have order details
        for _, row in matched.iterrows():
            assert not pd.isna(row["order_id"])
            assert not pd.isna(row["amount"])

        # Check original DataFrames unchanged
        _check_unchanged(basic_customers, customers)
        _check_unchanged(basic_orders, orders)

    def test_null_join_keys(self, customers_with_null_keys, orders_with_null_keys):
        customers = customers_with_null_keys.copy()
        orders = orders_with_null_keys.copy()
        result = join_customers_orders(customers, orders, customer_key="customer_id")

        # Check that no null-key rows are marked "matched"
        null_cust_rows = result[result["customer_id"].isna() & (result["record_status"] == "customer_without_order")]
        null_order_rows = result[result["customer_id"].isna() & (result["record_status"] == "order_without_customer")]

        # Two customers with null keys (index 1 and 3)
        assert len(null_cust_rows) == 2
        # Two orders with null keys (index 2 and 3)
        assert len(null_order_rows) == 2

        # Non-null key rows should still be matched where exists
        matched = result[(result["customer_id"] == 1) & (result["record_status"] == "matched")]
        assert len(matched) == 1

        _check_unchanged(customers_with_null_keys, customers)
        _check_unchanged(orders_with_null_keys, orders)

    def test_all_matched(self):
        customers = pd.DataFrame({"customer_id": [1, 2], "name": ["A", "B"]})
        orders = pd.DataFrame({"customer_id": [1, 2], "order_id": [101, 102]})
        result = join_customers_orders(customers, orders)
        assert all(result["record_status"] == "matched")
        assert len(result) == 2

    def test_no_matches(self):
        customers = pd.DataFrame({"customer_id": [1, 2]})
        orders = pd.DataFrame({"customer_id": [3, 4]})
        result = join_customers_orders(customers, orders)
        assert set(result["record_status"]) == {"customer_without_order", "order_without_customer"}
        assert len(result) == 4

    def test_duplicate_keys(self, duplicate_customers, duplicate_orders):
        customers = duplicate_customers.copy()
        orders = duplicate_orders.copy()
        result = join_customers_orders(customers, orders, customer_key="customer_id")

        # There are two customers with id=1, two orders with id=1
        # Full outer join should produce 2*2 = 4 matched rows for id=1
        matched_id1 = result[(result["customer_id"] == 1) & (result["record_status"] == "matched")]
        assert len(matched_id1) == 4

        # Customer with id=2 unmatched (left only)
        cust2 = result[(result["customer_id"] == 2) & (result["record_status"] == "customer_without_order")]
        assert len(cust2) == 1

        # Order with id=3 unmatched (right only)
        order3 = result[(result["customer_id"] == 3) & (result["record_status"] == "order_without_customer")]
        assert len(order3) == 1

        _check_unchanged(duplicate_customers, customers)
        _check_unchanged(duplicate_orders, orders)

    def test_empty_dataframes(self, empty_dataframe):

        result = join_customers_orders(empty_dataframe.copy(), empty_dataframe.copy())
        assert isinstance(result, pd.DataFrame)

        assert "record_status" in result.columns

        # Only one empty

        result = join_customers_orders(customers, empty_dataframe.copy())
        assert len(result) == 1
        assert result.iloc[0]["record_status"] == "customer_without_order"

        result = join_customers_orders(empty_dataframe.copy(), customers)
        assert len(result) == 1
        assert result.iloc[0]["record_status"] == "order_without_customer"
