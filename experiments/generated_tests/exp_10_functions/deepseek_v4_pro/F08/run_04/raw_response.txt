import pandas as pd
import pytest
from pandas.testing import assert_frame_equal
from ise26.targets import calculate_conversion_rate


class TestCalculateConversionRate:
    def test_basic_calculation(self):
        df = pd.DataFrame({
            "channel": ["A", "B", "A", "B"],
            "visits": [100, 200, 50, 300],
            "conversions": [10, 20, 5, 30],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["A", "B"],
            "visits": [150.0, 500.0],
            "conversions": [15.0, 50.0],
            "conversion_rate": [0.1, 0.1],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_missing_values_treated_as_zero(self):
        df = pd.DataFrame({
            "channel": ["A", "A", "B"],
            "visits": [100, None, 200],
            "conversions": [10, None, None],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["A", "B"],
            "visits": [100.0, 200.0],
            "conversions": [10.0, 0.0],
            "conversion_rate": [0.1, 0.0],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_invalid_strings_coerced_to_zero(self):
        df = pd.DataFrame({
            "channel": ["X"],
            "visits": ["abc"],
            "conversions": ["def"],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["X"],
            "visits": [0.0],
            "conversions": [0.0],
            "conversion_rate": [0.0],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_zero_visits_yields_zero_rate(self):
        df = pd.DataFrame({
            "channel": ["A", "A"],
            "visits": [0, 0],
            "conversions": [5, 3],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["A"],
            "visits": [0.0],
            "conversions": [8.0],
            "conversion_rate": [0.0],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_negative_visits_treated_as_valid_but_rate_zero(self):
        df = pd.DataFrame({
            "channel": ["A"],
            "visits": [-10],
            "conversions": [5],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["A"],
            "visits": [-10.0],
            "conversions": [5.0],
            "conversion_rate": [0.0],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_sort_order_alphabetical(self):
        df = pd.DataFrame({
            "channel": ["C", "A", "B"],
            "visits": [1, 2, 3],
            "conversions": [0, 0, 0],
        })
        result = calculate_conversion_rate(df)
        assert list(result["channel"]) == ["A", "B", "C"]

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "source": ["S1", "S2"],
            "clicks": [100, 200],
            "signups": [10, 30],
        })
        result = calculate_conversion_rate(
            df, group_col="source", visits_col="clicks", conversions_col="signups"
        )
        expected = pd.DataFrame({
            "source": ["S1", "S2"],
            "visits": [100.0, 200.0],
            "conversions": [10.0, 30.0],
            "conversion_rate": [0.1, 0.15],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["channel", "visits", "conversions"])
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame(columns=["channel", "visits", "conversions", "conversion_rate"])
        assert_frame_equal(result, expected)

    def test_all_null_channel_gives_nan_group(self):
        df = pd.DataFrame({
            "channel": [None, None],
            "visits": [10, 20],
            "conversions": [1, 2],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": [np.nan],
            "visits": [30.0],
            "conversions": [3.0],
            "conversion_rate": [0.1],
        })
        assert_frame_equal(result, expected, check_dtype=False, check_names=False)

    def test_conversion_rate_precision(self):
        df = pd.DataFrame({
            "channel": ["A"],
            "visits": [3],
            "conversions": [1],
        })
        result = calculate_conversion_rate(df)
        assert result["conversion_rate"].iloc[0] == pytest.approx(1/3)

    def test_output_column_types_are_float_except_group(self):
        df = pd.DataFrame({"channel": ["X"], "visits": [10], "conversions": [1]})
        result = calculate_conversion_rate(df)
        assert result["visits"].dtype == float
        assert result["conversions"].dtype == float
        assert result["conversion_rate"].dtype == float
        # group column can be object (string), that's fine

    def test_duplicate_rows_aggregated(self):
        df = pd.DataFrame({
            "channel": ["A", "A", "A"],
            "visits": [10, 20, 30],
            "conversions": [1, 2, 3],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["A"],
            "visits": [60.0],
            "conversions": [6.0],
            "conversion_rate": [0.1],
        })
        assert_frame_equal(result, expected, check_dtype=False)

    def test_mixed_numeric_types(self):
        df = pd.DataFrame({
            "channel": ["A"],
            "visits": ["100.5", 200],
            "conversions": [10, 20.2],
        })
        result = calculate_conversion_rate(df)
        expected = pd.DataFrame({
            "channel": ["A"],
            "visits": [300.5],
            "conversions": [30.2],
            "conversion_rate": [30.2 / 300.5],
        })
        assert_frame_equal(result, expected, check_dtype=False)
