import pytest
import pandas as pd
import numpy as np
from ise26.targets import calculate_conversion_rate


def test_basic_aggregation():
    # Test normal case: multiple rows per channel, sum visits and conversions,
    # compute conversion rate = conversions / visits.
    df = pd.DataFrame({
        "channel": ["A", "A", "B", "B", "C"],
        "visits": [10, 20, 5, 5, 0],
        "conversions": [2, 3, 1, 1, 0],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["A", "B", "C"],
        "visits": [30.0, 10.0, 0.0],
        "conversions": [5.0, 2.0, 0.0],
        "conversion_rate": [5/30, 2/10, 0.0],
    })
    # Sort by channel is already alphabetical
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_sorted_output():
    # Channels should be sorted alphabetically.
    df = pd.DataFrame({
        "channel": ["Z", "A", "M"],
        "visits": [1, 2, 3],
        "conversions": [0, 1, 1],
    })
    result = calculate_conversion_rate(df)
    assert list(result["channel"]) == ["A", "M", "Z"]


def test_visits_zero_conversion_zero():
    # Visits = 0, conversions = 0 -> conversion_rate = 0.
    df = pd.DataFrame({
        "channel": ["X"],
        "visits": [0],
        "conversions": [0],
    })
    result = calculate_conversion_rate(df)
    assert result.loc[0, "conversion_rate"] == 0.0
    assert result.loc[0, "visits"] == 0.0
    assert result.loc[0, "conversions"] == 0.0


def test_visits_zero_conversions_positive():
    # If visits = 0 but conversions > 0, conversion_rate = 0.
    df = pd.DataFrame({
        "channel": ["X"],
        "visits": [0],
        "conversions": [5],
    })
    result = calculate_conversion_rate(df)
    assert result.loc[0, "conversion_rate"] == 0.0


def test_missing_values_treated_as_zero():
    # NaN in numeric columns should be treated as zero.
    df = pd.DataFrame({
        "channel": ["A", "A"],
        "visits": [np.nan, 10],
        "conversions": [2, np.nan],
    })
    result = calculate_conversion_rate(df)
    # Aggregated: visits = 0 + 10 = 10, conversions = 2 + 0 = 2
    expected = pd.DataFrame({
        "channel": ["A"],
        "visits": [10.0],
        "conversions": [2.0],
        "conversion_rate": [0.2],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_invalid_string_treated_as_zero():
    # Non-numeric strings in numeric columns -> coerce to NaN -> fillna(0)
    df = pd.DataFrame({
        "channel": ["A", "A"],
        "visits": ["bad", 10],
        "conversions": [2, "invalid"],
    })
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame({
        "channel": ["A"],
        "visits": [10.0],
        "conversions": [2.0],
        "conversion_rate": [0.2],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_nan_channel_included():
    # NaN channel is kept due to dropna=False, appears last in sorting.
    df = pd.DataFrame({
        "channel": [np.nan, "B", "A"],
        "visits": [1, 2, 3],
        "conversions": [0.1, 0.2, 0.3],
    })
    result = calculate_conversion_rate(df)
    # Sorted: "A", "B", NaN (NaN at end by default)
    channels = list(result["channel"])
    assert channels[0] == "A"
    assert channels[1] == "B"
    assert pd.isna(channels[2])
    # Check conversion rates
    assert result.loc[result["channel"] == "A", "conversion_rate"].values[0] == pytest.approx(0.1)
    assert result.loc[result["channel"] == "B", "conversion_rate"].values[0] == pytest.approx(0.1)
    nan_row = result.loc[result["channel"].isna()]
    assert nan_row["conversion_rate"].values[0] == pytest.approx(0.1)


def test_original_dataframe_unchanged():
    df = pd.DataFrame({
        "channel": ["A", "A"],
        "visits": [10, 20],
        "conversions": [2, 3],
    })
    df_copy = df.copy()
    _ = calculate_conversion_rate(df)
    pd.testing.assert_frame_equal(df, df_copy)


def test_custom_column_names():
    df = pd.DataFrame({
        "src": ["X", "X", "Y"],
        "clicks": [100, 200, 50],
        "buys": [10, 20, 5],
    })
    result = calculate_conversion_rate(
        df,
        group_col="src",
        visits_col="clicks",
        conversions_col="buys",
    )
    expected = pd.DataFrame({
        "src": ["X", "Y"],
        "clicks": [300.0, 50.0],
        "buys": [30.0, 5.0],
        "conversion_rate": [0.1, 0.1],
    })
    pd.testing.assert_frame_equal(
        result, expected,
        check_dtype=False,
        check_column_type=False,
    )


def test_empty_dataframe():
    # Function should handle empty input gracefully.
    df = pd.DataFrame(columns=["channel", "visits", "conversions"])
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(columns=["channel", "visits", "conversions", "conversion_rate"])
    # dtypes may vary, just check rows are empty and columns match
    assert list(result.columns) == list(expected.columns)
    assert len(result) == 0


def test_negative_visits():
    # Negative visits sum to <=0 triggers zero conversion rate.
    df = pd.DataFrame({
        "channel": ["A", "A"],
        "visits": [5, -5],  # sum = 0
        "conversions": [2, 1],  # sum = 3
    })
    result = calculate_conversion_rate(df)
    assert result.loc[0, "visits"] == 0.0
    assert result.loc[0, "conversions"] == 3.0
    assert result.loc[0, "conversion_rate"] == 0.0


def test_multiple_channels_with_nan_and_missing():
    # Complex mix: NaN channel, invalid numbers, negative.
    df = pd.DataFrame({
        "channel": ["A", np.nan, "A", "B", "B"],
        "visits": [100, "???", -50, 0, np.nan],
        "conversions": [10, 5, np.nan, 2, -2],
    })
    result = calculate_conversion_rate(df)
    # Expected aggregation:
    # A: visits = 100 + 0 + (-50) = 50; conversions = 10 + 0 + 0 = 10
    # B: visits = 0 + 0 = 0; conversions = 2 + (-2) = 0
    # NaN: visits = 0; conversions = 5 + 0? Actually NaN row: "???" -> NaN -> 0, conversions 5 -> 5
    # But note: conversions for NaN row: original conversions = 5, not NaN. So conversions sum = 5.
    # So: channel A: visits=50, conversions=10 -> rate 0.2
    # B: visits=0, conversions=0 -> rate 0.0
    # NaN: visits=0, conversions=5 -> rate 0.0
    expected = pd.DataFrame({
        "channel": ["A", "B", np.nan],
        "visits": [50.0, 0.0, 0.0],
        "conversions": [10.0, 0.0, 5.0],
        "conversion_rate": [0.2, 0.0, 0.0],
    })
    # Sort order: "A", "B", NaN (NaNs last)
    pd.testing.assert_frame_equal(
        result, expected,
        check_dtype=False,
        check_names=False,  # because column names may be generic
    )


def test_conversion_rate_column_type():
    df = pd.DataFrame({
        "channel": ["A", "A"],
        "visits": [10, 20],
        "conversions": [2, 3],
    })
    result = calculate_conversion_rate(df)
    assert result["conversion_rate"].dtype == float
    assert result["visits"].dtype == float
    assert result["conversions"].dtype == float


def test_output_columns_no_extra():
    df = pd.DataFrame({
        "channel": ["A"],
        "visits": [1],
        "conversions": [0],
    })
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]


def test_no_rows_with_partial_groups_handled_correctly():
    # Single row per group.
    df = pd.DataFrame({
        "channel": ["X", "Y", "Z"],
        "visits": [1, 2, 3],
        "conversions": [0, 1, 2],
    })
    result = calculate_conversion_rate(df)
    assert len(result) == 3
    assert result["conversion_rate"].tolist() == [0.0, 0.5, 2/3]
