import pytest
import pandas as pd
import numpy as np
from ise26.targets import calculate_conversion_rate


def test_basic_aggregation():
    df = pd.DataFrame(
        {
            "channel": ["A", "A", "B", "B"],
            "visits": [100, 200, 300, 400],
            "conversions": [10, 20, 30, 40],
        }
    )
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(
        {
            "channel": ["A", "B"],
            "visits": [300.0, 700.0],
            "conversions": [30.0, 70.0],
            "conversion_rate": [0.1, 0.1],
        }
    )
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_zero_visits_yields_zero_rate():
    df = pd.DataFrame(
        {
            "channel": ["A", "A"],
            "visits": [0, 0],
            "conversions": [10, 20],
        }
    )
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(
        {
            "channel": ["A"],
            "visits": [0.0],
            "conversions": [30.0],
            "conversion_rate": [0.0],
        }
    )
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_negative_visits_considered_zero():
    df = pd.DataFrame(
        {
            "channel": ["A", "A"],
            "visits": [-5, 10],
            "conversions": [2, 3],
        }
    )
    result = calculate_conversion_rate(df)
    # visits sum: -5 + 10 = 5 -> positive, so rate = conversions/visits = 5/5 = 1.0
    # but we need a case where sum is <=0. Let's do both rows negative.
    df2 = pd.DataFrame(
        {
            "channel": ["B"],
            "visits": [-10, -20],
            "conversions": [5, 5],
        }
    )
    result2 = calculate_conversion_rate(df2)
    expected2 = pd.DataFrame(
        {
            "channel": ["B"],
            "visits": [-30.0],
            "conversions": [10.0],
            "conversion_rate": [0.0],
        }
    )
    pd.testing.assert_frame_equal(result2, expected2, check_dtype=False)


def test_missing_values_treated_as_zero():
    df = pd.DataFrame(
        {
            "channel": ["A", "A", "B"],
            "visits": [np.nan, 100, 200],
            "conversions": [10, np.nan, 30],
        }
    )
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(
        {
            "channel": ["A", "B"],
            "visits": [100.0, 200.0],
            "conversions": [10.0, 30.0],
            "conversion_rate": [0.1, 0.15],
        }
    )
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)

def test_alphabetical_sorting():
    df = pd.DataFrame(
        {
            "channel": ["Z", "A", "M", "a", "B"],
            "visits": [1, 1, 1, 1, 1],
            "conversions": [0, 0, 0, 0, 0],
        }
    )
    result = calculate_conversion_rate(df)
    expected_channels = ["A", "B", "M", "Z", "a"]  # default string sort
    assert list(result["channel"]) == expected_channels


def test_custom_column_names():
    df = pd.DataFrame(
        {
            "source": ["X", "X", "Y"],
            "clicks": [100, 200, 300],
            "sales": [10, 20, 30],
        }
    )
    result = calculate_conversion_rate(
        df, group_col="source", visits_col="clicks", conversions_col="sales"
    )
    expected = pd.DataFrame(
        {
            "source": ["X", "Y"],
            "visits": [300.0, 300.0],
            "conversions": [30.0, 30.0],
            "conversion_rate": [0.1, 0.1],
        }
    )
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_channel_with_nan_value():
    df = pd.DataFrame(
        {
            "channel": [np.nan, "A", "A"],
            "visits": [10, 100, 200],
            "conversions": [1, 10, 20],
        }
    )
    result = calculate_conversion_rate(df)
    # NaN group appears last in default sort
    expected = pd.DataFrame(
        {
            "channel": ["A", np.nan],
            "visits": [300.0, 10.0],
            "conversions": [30.0, 1.0],
            "conversion_rate": [0.1, 0.1],
        }
    )
    # Since sorting places NaN at end, the order is A, NaN
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_empty_dataframe():
    df = pd.DataFrame(columns=["channel", "visits", "conversions"])
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(columns=["channel", "visits", "conversions", "conversion_rate"])
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_output_columns_and_types():
    df = pd.DataFrame(
        {
            "channel": ["A"],
            "visits": [100],
            "conversions": [10],
        }
    )
    result = calculate_conversion_rate(df)
    assert list(result.columns) == ["channel", "visits", "conversions", "conversion_rate"]
    assert result["visits"].dtype == float
    assert result["conversions"].dtype == float
    assert result["conversion_rate"].dtype == float


def test_does_not_mutate_original():
    df = pd.DataFrame(
        {
            "channel": ["A", "A"],
            "visits": [10, 20],
            "conversions": [1, 2],
        }
    )
    original = df.copy()
    calculate_conversion_rate(df)
    pd.testing.assert_frame_equal(df, original)
