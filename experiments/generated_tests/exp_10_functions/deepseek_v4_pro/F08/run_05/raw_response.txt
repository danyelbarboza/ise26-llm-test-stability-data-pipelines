import pytest
import pandas as pd
from ise26.targets import calculate_conversion_rate


@pytest.fixture
def basic_df():
    return pd.DataFrame(
        {
            "channel": ["A", "B", "A", "C", "B", "B"],
            "visits": [10, 20, 15, 5, 30, 25],
            "conversions": [2, 4, 1, 1, 6, 5],
        }
    )


def test_basic_aggregation(basic_df):
    result = calculate_conversion_rate(basic_df)
    expected = pd.DataFrame(
        {
            "channel": ["A", "B", "C"],
            "visits": [25.0, 75.0, 5.0],
            "conversions": [3.0, 15.0, 1.0],
            "conversion_rate": [0.12, 0.2, 0.2],
        }
    )
    pd.testing.assert_frame_equal(result, expected)


def test_missing_values_treated_as_zero():
    df = pd.DataFrame(
        {
            "channel": ["X", "X", "Y", "Y"],
            "visits": [10, None, 0, 5],
            "conversions": [None, 2, 3, None],
        }
    )
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(
        {
            "channel": ["X", "Y"],
            "visits": [10.0, 5.0],
            "conversions": [2.0, 3.0],
            "conversion_rate": [0.2, 0.6],
        }
    )
    pd.testing.assert_frame_equal(result, expected)

def test_sorting_alphabetical():
    df = pd.DataFrame(
        {
            "channel": ["C", "B", "A", "B"],
            "visits": [1, 2, 3, 4],
            "conversions": [0, 1, 2, 3],
        }
    )
    result = calculate_conversion_rate(df)
    assert list(result["channel"]) == ["A", "B", "C"]


def test_sorting_with_nan_group():
    df = pd.DataFrame(
        {
            "channel": [None, "A", None, "B"],
            "visits": [10, 20, 5, 30],
            "conversions": [1, 2, 0, 3],
        }
    )
    result = calculate_conversion_rate(df)
    # NaN should appear last in sorted order (by default na_position='last')
    channels = list(result["channel"])
    assert channels[:2] == ["A", "B"]  # sorted alpha
    assert pd.isna(channels[2])  # NaN at end


def test_empty_dataframe():
    df = pd.DataFrame({"channel": [], "visits": [], "conversions": []})
    result = calculate_conversion_rate(df)
    expected = pd.DataFrame(
        columns=["channel", "visits", "conversions", "conversion_rate"],
        dtype=float,
    )
    # channel column may be object, not float; we'll create expected accordingly
    expected = pd.DataFrame(
        {"channel": pd.Series(dtype=str), "visits": pd.Series(dtype=float),
         "conversions": pd.Series(dtype=float), "conversion_rate": pd.Series(dtype=float)}
    )
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_custom_column_names():
    df = pd.DataFrame(
        {
            "source": ["x", "y", "x"],
            "clicks": [100, 200, 150],
            "sales": [10, 20, 15],
        }
    )
    result = calculate_conversion_rate(
        df, group_col="source", visits_col="clicks", conversions_col="sales"
    )
    expected = pd.DataFrame(
        {
            "source": ["x", "y"],
            "visits": [250.0, 200.0],
            "conversions": [25.0, 20.0],
            "conversion_rate": [0.1, 0.1],
        }
    )
    pd.testing.assert_frame_equal(
        result.rename(columns={"source": "channel"}), expected.rename(columns={"source": "channel"})
    )
    # Actually the output columns are always "channel", "visits", "conversions", "conversion_rate"
    # regardless of custom names? The function uses group_col for grouping, but output column name is group_col.
    # So with group_col="source", the result will have "source", "visits", ... not "channel".
    # The docstring says returns columns with ``channel``, but the implementation returns grouped[[group_col, ...]].
    # So it returns the provided group_col.
    # Let's test directly:
    expected_correct = pd.DataFrame(
        {
            "source": ["x", "y"],
            "visits": [250.0, 200.0],
            "conversions": [25.0, 20.0],
            "conversion_rate": [0.1, 0.1],
        }
    )
    pd.testing.assert_frame_equal(result, expected_correct)


def test_no_modification_of_original():
    df = pd.DataFrame(
        {
            "channel": ["A", "B"],
            "visits": [10, 20],
            "conversions": [1, 2],
        }
    )
    df_original = df.copy()
    _ = calculate_conversion_rate(df)
    pd.testing.assert_frame_equal(df, df_original)


def test_column_types():
    df = pd.DataFrame(
        {
            "channel": ["A"],
            "visits": [10],
            "conversions": [3],
        }
    )
    result = calculate_conversion_rate(df)
    assert result["visits"].dtype == float
    assert result["conversions"].dtype == float
    assert result["conversion_rate"].dtype == float
    assert result["channel"].dtype == object


def test_all_channels_zero_visits():
    df = pd.DataFrame(
        {
            "channel": ["A", "B", "B"],
            "visits": [0, 0, 0],
            "conversions": [1, 2, 3],
        }

        {
            "channel": ["A", "B"],
            "visits": [0.0, 0.0],
            "conversions": [1.0, 5.0],
            "conversion_rate": [0.0, 0.0],
        }
    )
    pd.testing.assert_frame_equal(result, expected)
