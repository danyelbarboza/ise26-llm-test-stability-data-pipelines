from ise26.targets import clean_customer_names
import pandas as pd
import pytest


def test_basic_normalization():
    """Normalization removes accents, collapses spaces, and lowercases."""
    df = pd.DataFrame({"customer_name": ["  José   María  ", "Ánna-Käthe"]})
    result = clean_customer_names(df)
    expected = pd.Series(["jose maria", "anna-kathe"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)


def test_null_values():
    """None/NaN become pd.NA in the normalized column."""
    df = pd.DataFrame({"customer_name": [None, float("nan"), "valid"]})
    result = clean_customer_names(df)
    expected = pd.Series([pd.NA, pd.NA, "valid"], name="customer_name_clean")
    # pd.NA equality is tricky; we can compare with pd.isna
    assert pd.isna(result["customer_name_clean"][0]) and pd.isna(result["customer_name_clean"][1])
    assert result["customer_name_clean"][2] == "valid"


def test_blank_values():
    """Empty strings and whitespace-only strings become pd.NA."""
    df = pd.DataFrame({"customer_name": ["", "   ", "\t\n", "a"]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"][0] is pd.NA
    assert result["customer_name_clean"][1] is pd.NA
    assert result["customer_name_clean"][2] is pd.NA
    assert result["customer_name_clean"][3] == "a"


def test_preserves_original_dataframe():
    """Original DataFrame is unchanged: no new columns, data intact."""
    original = pd.DataFrame({"customer_name": ["Anne", "   Bob   "]})
    orig_cols = original.columns.copy()
    orig_data = original.copy()
    _ = clean_customer_names(original)
    # check columns unchanged
    pd.testing.assert_index_equal(original.columns, orig_cols)
    # check values unchanged (no normalization leaks)
    pd.testing.assert_frame_equal(original, orig_data)


def test_custom_column_names():
    """Use user-supplied name_col and output_col."""
    df = pd.DataFrame({"raw_name": ["  Cláre  ", "NULL"]})
    result = clean_customer_names(df, name_col="raw_name", output_col="norm_name")
    assert "norm_name" in result.columns
    assert result["norm_name"][0] == "clare"
    assert result["norm_name"][1] is pd.NA


def test_multiple_internal_spaces_collapse():
    """Repeated whitespace inside the string collapses to a single space."""
    df = pd.DataFrame({"customer_name": ["Mary   Jane", "  John    Doe  "]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"][0] == "mary jane"
    # leading/trailing spaces? If stripped, expected "john doe"
    # Based on typical behavior, we expect stripping as well.
    assert result["customer_name_clean"][1] == "john doe"


def test_accent_removal():
    """Accents are removed (NFD decomposition or equivalent)."""
    df = pd.DataFrame({"customer_name": ["Clément", "Åge", "Niño"]})
    result = clean_customer_names(df)
    expected = ["clement", "age", "nino"]
    assert result["customer_name_clean"].tolist() == expected


def test_duplicate_rows():
    """Normalization works row‑wise, duplicates remain as separate rows."""
    df = pd.DataFrame({"customer_name": [" Alice ", " Alice ", "Bob"]})
    result = clean_customer_names(df)
    expected = pd.Series(["alice", "alice", "bob"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)
