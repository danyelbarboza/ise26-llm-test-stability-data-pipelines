import pytest
import pandas as pd
import numpy as np
from ise26.targets import clean_customer_names

def test_basic_normalization():
    df = pd.DataFrame({"customer_name": ["  John   Doe  "]})
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    assert result["customer_name_clean"].iloc[0] == "john doe"
    # Original unchanged
    assert df.equals(pd.DataFrame({"customer_name": ["  John   Doe  "]}))

def test_accent_removal():
    df = pd.DataFrame({"customer_name": ["José", "François", "Müller", "Sofía"]})
    result = clean_customer_names(df)
    expected = ["jose", "francois", "muller", "sofia"]
    actual = result["customer_name_clean"].tolist()
    assert actual == expected

def test_mixed_content_and_spaces():
    df = pd.DataFrame({"customer_name": ["   María   Luz  ", "  Ángel  ", "Pèpè"]})
    result = clean_customer_names(df)
    # After accent removal: maria luz, angel, pepe
    expected = ["maria luz", "angel", "pepe"]
    actual = result["customer_name_clean"].tolist()
    assert actual == expected

def test_blank_and_null_values():
    df = pd.DataFrame({"customer_name": [None, np.nan, "", "   ", "Alice"]})
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"]
    assert cleaned.iloc[0] is pd.NA
    assert cleaned.iloc[1] is pd.NA
    assert cleaned.iloc[2] is pd.NA
    assert cleaned.iloc[3] is pd.NA
    assert cleaned.iloc[4] == "alice"

def test_preserve_original_dataframe():
    original = pd.DataFrame({"customer_name": ["Bob"], "other": [1]})
    original_copy = original.copy()
    result = clean_customer_names(original, "customer_name", "clean_name")
    assert original.equals(original_copy)
    assert "clean_name" not in original.columns

def test_custom_column_names():
    df = pd.DataFrame({"raw": ["  Eva  "]})
    result = clean_customer_names(df, name_col="raw", output_col="normalized")
    assert "normalized" in result.columns
    assert result["normalized"].iloc[0] == "eva"

def test_non_existent_name_col():
    df = pd.DataFrame({"other": ["a"]})
    with pytest.raises(KeyError):
        clean_customer_names(df, name_col="missing")

def test_multiple_columns_and_preservation():
    df = pd.DataFrame({
        "id": [1, 2, 3],
        "customer_name": ["  Ana  ", "  BOB  ", None],
        "extra": [10, 20, 30]
    })
    expected_df = df.copy()
    result = clean_customer_names(df)
    # Check that id and extra columns are unchanged in result
    assert result[["id", "extra"]].equals(expected_df[["id", "extra"]])
    assert result["customer_name_clean"].tolist() == ["ana", "bob", pd.NA]
    # Original df unchanged
    assert df.equals(expected_df)

def test_only_whitespace_strings_become_NA():
    df = pd.DataFrame({"customer_name": ["\t\n  ", "  \t", " "]})
    result = clean_customer_names(df)
    cleaned = result["customer_name_clean"]
    for val in cleaned:
        assert val is pd.NA

def test_already_normalized():
    df = pd.DataFrame({"customer_name": ["alice", "bob"]})
    result = clean_customer_names(df)
    assert result["customer_name_clean"].tolist() == ["alice", "bob"]
