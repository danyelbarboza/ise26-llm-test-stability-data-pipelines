import pytest
import pandas as pd
from ise26.targets import clean_customer_names


def test_basic_normalization():
    df = pd.DataFrame({"customer_name": ["  John   DOE  ", "  jane doe"]})
    result = clean_customer_names(df)
    expected = pd.Series(["john doe", "jane doe"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)


def test_accent_removal():
    df = pd.DataFrame({"customer_name": ["José", "Ñoño", "café"]})
    result = clean_customer_names(df)
    expected = pd.Series(["jose", "nono", "cafe"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)


def test_null_values():
    df = pd.DataFrame({"customer_name": [None, pd.NA, pd.NA, "valid"]})
    result = clean_customer_names(df)
    expected = pd.Series([pd.NA, pd.NA, pd.NA, "valid"], name="customer_name_clean", dtype="object")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)


def test_blank_strings():
    df = pd.DataFrame({"customer_name": ["", "   ", "\t \n", "a"]})
    result = clean_customer_names(df)
    expected = pd.Series([pd.NA, pd.NA, pd.NA, "a"], name="customer_name_clean", dtype="object")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected)


def test_whitespace_collapse():
    df = pd.DataFrame({"customer_name": ["hello     world", "  a  b  "]})
    result = clean_customer_names(df)
    expected = pd.Series(["hello world", "a b"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)


def test_original_dataframe_unchanged():
    df = pd.DataFrame({"customer_name": ["Foo"]})
    df_initial_cols = df.columns.tolist()
    df_initial_data = df.copy()
    _ = clean_customer_names(df)
    assert df.columns.tolist() == df_initial_cols
    pd.testing.assert_frame_equal(df, df_initial_data)


def test_immutability_returned_dataframe():
    df = pd.DataFrame({"customer_name": ["Foo"]})
    result = clean_customer_names(df)
    result.loc[0, "customer_name"] = "changed"
    assert df.loc[0, "customer_name"] == "Foo"


def test_custom_column_names():
    df = pd.DataFrame({"raw": ["  BAR "], "other": [1]})
    result = clean_customer_names(df, name_col="raw", output_col="normalized")
    assert "normalized" in result.columns
    assert result.loc[0, "normalized"] == "bar"
    assert "raw" in result.columns


def test_missing_input_column_raises():
    df = pd.DataFrame({"wrong_name": ["test"]})
    with pytest.raises(KeyError):
        clean_customer_names(df)


def test_with_numbers():
    df = pd.DataFrame({"customer_name": [123, 45.6, "  valid "], "x": [1, 2, 3]})
    result = clean_customer_names(df)
    # numbers are converted to strings and normalized (no spaces, lowercased) -> '123', '45.6'
    expected = pd.Series(["123", "45.6", "valid"], name="customer_name_clean")
    pd.testing.assert_series_equal(result["customer_name_clean"], expected, check_dtype=False)


def test_only_extra_column_added():
    df = pd.DataFrame({"customer_name": ["name"]})
    result = clean_customer_names(df)
    expected_columns = df.columns.tolist() + ["customer_name_clean"]
    assert result.columns.tolist() == expected_columns


def test_empty_dataframe():
    df = pd.DataFrame({"customer_name": []}, dtype=str)
    result = clean_customer_names(df)
    assert "customer_name_clean" in result.columns
    assert len(result) == 0
    assert result["customer_name_clean"].dtype == object
