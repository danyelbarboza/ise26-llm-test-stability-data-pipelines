import pytest
import pandas as pd
import numpy as np
from ise26.targets import clean_customer_names


def test_basic_normalization():
    """Collapse whitespace, lowercases, and removes accents."""
    df = pd.DataFrame({
        "customer_name": [
            " João   da   Silva ",
            "MARIA",
            "José",
            "   naïve   ",
            "CAFÉ",
        ]
    })
    expected_names = [
        "joao da silva",
        "maria",
        "jose",
        "naive",
        "cafe",
    ]

    result = clean_customer_names(df)

    assert "customer_name_clean" in result.columns
    assert result["customer_name_clean"].tolist() == expected_names


def test_null_and_blank_mapping():
    """None, np.nan, empty string, and whitespace-only become pd.NA."""
    df = pd.DataFrame({
        "customer_name": [
            None,
            np.nan,
            "",
            "   ",
            "valid name",
        ]
    })

    result = clean_customer_names(df)

    clean_col = result["customer_name_clean"]
    # first four entries should be pd.NA (pandas' NA scalar)
    for i in range(4):
        assert clean_col[i] is pd.NA

    assert clean_col[4] == "valid name"


def test_original_dataframe_not_mutated():
    """Ensure input DataFrame is not changed and a new object is returned."""
    original_data = [" João ", " maria ", None, "JOSÉ"]
    df = pd.DataFrame({"customer_name": original_data.copy()})
    df_original_view = df.copy(deep=True)

    result = clean_customer_names(df)

    # input unchanged (values and column set)
    pd.testing.assert_frame_equal(df, df_original_view)
    # returned object is not the same DataFrame
    assert result is not df


def test_custom_column_names():
    """Works with non‑default input and output column names."""
    df = pd.DataFrame({"raw_name": ["  Carlos  ", "   "], "other": [1, 2]})
    result = clean_customer_names(df, name_col="raw_name", output_col="clean_name")

    assert "clean_name" in result.columns
    assert result["clean_name"].tolist() == ["carlos", pd.NA]
    # original column name is still present
    assert "raw_name" in result.columns
    # the original df is unchanged
    assert "clean_name" not in df.columns


def test_accent_removal_various_characters():
    """Accents from different languages are removed correctly."""
    df = pd.DataFrame({
        "customer_name": [
            "Smörgåsbord",
            "élève",
            "çağatay",
            "André",
            "Fiança",
        ]
    })
    expected = [
        "smorgasbord",
        "eleve",
        "cagatay",
        "andre",
        "fianca",
    ]

    result = clean_customer_names(df)
    assert result["customer_name_clean"].tolist() == expected


def test_no_side_effects_on_other_columns():
    """Other columns remain untouched in the returned DataFrame."""
    df = pd.DataFrame({
        "customer_name": [" Alice "],
        "id": [42],
        "active": [True],
    })

    result = clean_customer_names(df)

    # the new DataFrame has original columns plus the clean column
    assert list(result.columns) == ["customer_name", "id", "active", "customer_name_clean"]
    # values of original columns unchanged
    assert result["id"].tolist() == [42]
    assert result["active"].tolist() == [True]
    assert result["customer_name"].tolist() == [" Alice "]  # unchanged


def test_output_column_overwrite_if_exists():
    """If output column already exists it is overwritten (current behavior)."""
    df = pd.DataFrame({
        "customer_name": [" Bob "],
        "customer_name_clean": ["will be replaced"]
    })

    result = clean_customer_names(df)

    assert result["customer_name_clean"].tolist() == ["bob"]
    # original column still unchanged
    assert df["customer_name_clean"].tolist() == ["will be replaced"]


def test_all_null_or_blank_column():
    """When all values are missing or blank the output column is all pd.NA."""
    df = pd.DataFrame({
        "customer_name": [None, "", "   ", pd.NA]
    })

    result = clean_customer_names(df)

    clean_col = result["customer_name_clean"]
    assert all(v is pd.NA for v in clean_col)
