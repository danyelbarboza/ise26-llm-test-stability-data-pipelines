import pandas as pd
import pytest
from ise26.targets import clean_customer_names

class TestCleanCustomerNames:
    def test_basic_cleaning(self):
        df = pd.DataFrame({"customer_name": ["  João da   Silva  "]})
        result = clean_customer_names(df)
        assert result["customer_name_clean"].iloc[0] == "joao da silva"

    def test_accent_removal(self):
        df = pd.DataFrame({"customer_name": ["Él Niño", "François", "Müller"]})
        result = clean_customer_names(df)
        expected = ["el nino", "francois", "muller"]
        actual = result["customer_name_clean"].tolist()
        assert actual == expected

    def test_null_and_blank(self):
        df = pd.DataFrame({"customer_name": [None, "", "   ", "normal"]})
        result = clean_customer_names(df)
        assert pd.isna(result["customer_name_clean"].iloc[0])
        assert pd.isna(result["customer_name_clean"].iloc[1])
        assert pd.isna(result["customer_name_clean"].iloc[2])
        assert result["customer_name_clean"].iloc[3] == "normal"

    def test_preserve_original(self):
        original = pd.DataFrame({"customer_name": ["Ana   Maria"]})
        df_copy = original.copy()
        result = clean_customer_names(original)
        # original DataFrame must not have the new column
        assert "customer_name_clean" not in original.columns
        # original values unchanged
        assert original.equals(df_copy)

    def test_custom_column_names(self):
        df = pd.DataFrame({"raw_name": ["  MARIA  "]})
        result = clean_customer_names(df, name_col="raw_name", output_col="clean")
        assert "clean" in result.columns
        assert result["clean"].iloc[0] == "maria"

    def test_multiple_rows_mixed(self):
        df = pd.DataFrame({"customer_name": [
            "  José   ", None, "   ", "  María  Dolores  ", "", "NORMAL"
        ]})
        result = clean_customer_names(df)
        expected = ["jose", pd.NA, pd.NA, "maria dolores", pd.NA, "normal"]
        # Compare using list, handling pd.NA via pd.isna per element
        actual = result["customer_name_clean"].tolist()
        assert len(expected) == len(actual)
        for e, a in zip(expected, actual):
            if pd.isna(e):
                assert pd.isna(a)
            else:
                assert e == a

    def test_no_side_effects(self):
        # verify that the original DataFrame is not mutated in any column
        original = pd.DataFrame({"customer_name": ["  Test  "], "other": [1]})
        result = clean_customer_names(original)
        # original columns same as before
        assert list(original.columns) == ["customer_name", "other"]
        assert original["customer_name"].iloc[0] == "  Test  "
        # result is a different object
        assert result is not original
