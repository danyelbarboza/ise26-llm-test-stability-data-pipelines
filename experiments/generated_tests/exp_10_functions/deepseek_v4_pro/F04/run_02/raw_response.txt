import pandas as pd
import pytest
from ise26.targets import join_customers_orders


def test_basic_join():
    """Test a typical join with matches, unmatched, and null keys."""
    customers = pd.DataFrame({
        "customer_id": [1, 2, 3, None, 5],
        "name": ["Alice", "Bob", "Charlie", "NullUser", "Eve"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 2, None, 4],
        "item": ["apple", "book", "null_order", "dice"],
    })
    original_customers = customers.copy()
    original_orders = orders.copy()

    result = join_customers_orders(customers, orders, customer_key="customer_id")

    # original DataFrames unchanged
    assert customers.equals(original_customers)
    assert orders.equals(original_orders)

    # result is a DataFrame with expected columns
    assert isinstance(result, pd.DataFrame)
    expected_columns = {"customer_id", "name", "item", "record_status"}
    assert set(result.columns) == expected_columns

    # record_status values present
    assert set(result["record_status"].unique()) <= {"matched", "customer_without_order", "order_without_customer"}

    # rows with null customer_id should never be matched
    null_rows = result[result["customer_id"].isna()]
    assert not any(null_rows["record_status"] == "matched")

    # check counts of statuses based on logic
    status_counts = result["record_status"].value_counts()
    # matched: customer_id 1 and 2 appear in both -> 2 rows
    assert status_counts.get("matched", 0) == 2
    # customer_without_order: customers with id 3, 5, and None -> 3 rows
    assert status_counts.get("customer_without_order", 0) == 3
    # order_without_customer: orders with id None and 4 -> 2 rows
    assert status_counts.get("order_without_customer", 0) == 2

    # total rows = matched(2) + unmatched_left(3) + unmatched_right(2) = 7
    assert len(result) == 7

    # index is reset
    assert result.index.equals(pd.RangeIndex(len(result)))


def test_empty_dataframes():
    """Test behavior with empty inputs."""
    empty_cust = pd.DataFrame(columns=["customer_id", "name"])
    empty_orders = pd.DataFrame(columns=["customer_id", "item"])

    # both empty
    result_both = join_customers_orders(empty_cust, empty_orders, customer_key="customer_id")
    assert len(result_both) == 0
    assert set(result_both.columns) == {"customer_id", "name", "item", "record_status"}

    # only customers empty
    orders = pd.DataFrame({"customer_id": [1], "item": ["a"]})
    result_left_empty = join_customers_orders(empty_cust, orders, customer_key="customer_id")
    assert len(result_left_empty) == 1
    assert result_left_empty.iloc[0]["record_status"] == "order_without_customer"
    assert set(result_left_empty.columns) == {"customer_id", "name", "item", "record_status"}

    # only orders empty
    customers = pd.DataFrame({"customer_id": [1], "name": ["b"]})
    result_right_empty = join_customers_orders(customers, empty_orders, customer_key="customer_id")
    assert len(result_right_empty) == 1
    assert result_right_empty.iloc[0]["record_status"] == "customer_without_order"
    assert set(result_right_empty.columns) == {"customer_id", "name", "item", "record_status"}


def test_all_null_keys():
    """When all keys are null, no rows should be matched."""
    customers = pd.DataFrame({
        "customer_id": [None, None],
        "name": ["N1", "N2"],
    })
    orders = pd.DataFrame({
        "customer_id": [None, None, None],
        "item": ["o1", "o2", "o3"],
    })

    result = join_customers_orders(customers, orders, customer_key="customer_id")

    # All statuses should be unmatched
    assert set(result["record_status"].unique()) == {"customer_without_order", "order_without_customer"}
    assert "matched" not in result["record_status"].values

    # counts
    status_counts = result["record_status"].value_counts()
    assert status_counts.get("customer_without_order", 0) == 2
    assert status_counts.get("order_without_customer", 0) == 3
    assert len(result) == 5


def test_duplicate_keys():
    """Duplicate join keys should produce multiple matches and preserve all rows."""
    customers = pd.DataFrame({
        "customer_id": [1, 1, 2],
        "name": ["A1", "A2", "B"],
    })
    orders = pd.DataFrame({
        "customer_id": [1, 2],
        "item": ["x", "y"],
    })
    original_customers = customers.copy()
    original_orders = orders.copy()

    result = join_customers_orders(customers, orders, customer_key="customer_id")

    assert customers.equals(original_customers)
    assert orders.equals(original_orders)

    status_counts = result["record_status"].value_counts()
    # matched: customer_id=1 (2 in customers x 1 in orders => 2 matches), customer_id=2 (1 match) => total 3 matched
    assert status_counts.get("matched", 0) == 3
    # everyone is matched, no unmatched
    assert status_counts.get("customer_without_order", 0) == 0
    assert status_counts.get("order_without_customer", 0) == 0
    assert len(result) == 3
