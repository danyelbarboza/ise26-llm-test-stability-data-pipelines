import pytest
import pandas as pd
import numpy as np

from ise26.targets import join_customers_orders


class TestBasicJoin:
    def test_matched_and_unmatched(self):
        """Rows with matching keys get status 'matched', others get appropriate statuses."""
        customers = pd.DataFrame({
            "customer_id": [1, 2, 3],
            "name": ["Alice", "Bob", "Charlie"]
        })
        orders = pd.DataFrame({
            "customer_id": [2, 4],
            "amount": [100, 200]
        })
        result = join_customers_orders(customers, orders)

        # Assert all rows present
        assert len(result) == 4
        # Check statuses
        status_counts = result["record_status"].value_counts().to_dict()
        assert status_counts == {
            "matched": 1,
            "customer_without_order": 2,
            "order_without_customer": 1
        }
        # Check specific row contents
        bob_row = result[(result["customer_id"] == 2) & (result["record_status"] == "matched")]
        assert bob_row.iloc[0]["name"] == "Bob"
        assert bob_row.iloc[0]["amount"] == 100

    def test_original_dfs_unchanged(self):
        """Original DataFrames must not be modified."""
        customers = pd.DataFrame({
            "customer_id": [1, 2],
            "name": ["A", "B"]
        })
        orders = pd.DataFrame({
            "customer_id": [2],
            "amount": [50]
        })
        cust_copy = customers.copy()
        ord_copy = orders.copy()
        _ = join_customers_orders(customers, orders)
        pd.testing.assert_frame_equal(customers, cust_copy)
        pd.testing.assert_frame_equal(orders, ord_copy)


class TestNullKeys:
    def test_null_keys_never_matched(self):
        """Null keys are isolated and should never produce 'matched' status."""
        customers = pd.DataFrame({
            "customer_id": [1, None, np.nan],
            "name": ["Alice", "Bob", "Charlie"]
        })
        orders = pd.DataFrame({
            "customer_id": [1, None],
            "amount": [100, 200]
        })
        result = join_customers_orders(customers, orders)
        # Only key=1 is matched
        assert result[result["record_status"] == "matched"].shape[0] == 1
        # Null key from customers -> customer_without_order
        assert result[(result["customer_id"].isna()) & (result["record_status"] == "customer_without_order")].shape[0] == 2
        # Null key from orders -> order_without_customer
        assert result[(result["customer_id"].isna()) & (result["record_status"] == "order_without_customer")].shape[0] == 1
        # No null key row should be matched
        assert len(result[(result["customer_id"].isna()) & (result["record_status"] == "matched")]) == 0


class TestDuplicates:
    def test_duplicate_keys_produce_cartesian(self):
        """Duplicate keys produce all combinations and are all 'matched'."""
        customers = pd.DataFrame({
            "customer_id": [1, 1],
            "name": ["A1", "A2"]
        })
        orders = pd.DataFrame({
            "customer_id": [1, 1],
            "amount": [10, 20]
        })
        result = join_customers_orders(customers, orders)
        # expected combinations: 2*2 = 4 matched rows
        assert len(result) == 4
        assert all(result["record_status"] == "matched")
        # All combinations of name and amount present
        names = set(result["name"])
        amounts = set(result["amount"])
        assert names == {"A1", "A2"}
        assert amounts == {10, 20}


class TestDeterministicOrder:
    def test_order_follows_source_order(self):
        """Output order must be deterministic based on source position."""
        customers = pd.DataFrame({
            "customer_id": [1, 2, 3],
            "name": ["C1", "C2", "C3"]
        })
        orders = pd.DataFrame({
            "customer_id": [2, 4],
            "amount": [100, 200]
        })
        result = join_customers_orders(customers, orders)
        # Expected order based on internal source orders:
        # 1) customer_without_order id=1 (cust source order 0)
        # 2) matched id=2 (cust source order 1)
        # 3) order_without_customer id=4 (order source order 1)
        # 4) customer_without_order id=3 (cust source order 2)
        expected_statuses = [
            "customer_without_order",  # C1
            "matched",                 # C2
            "order_without_customer",  # order id=4
            "customer_without_order"   # C3
        ]
        actual_statuses = result["record_status"].tolist()
        assert actual_statuses == expected_statuses
        # Additionally check that specific row contents are in expected positions
        assert result.iloc[0]["customer_id"] == 1
        assert result.iloc[1]["customer_id"] == 2
        assert result.iloc[2]["customer_id"] == 4  # This is from orders
        assert result.iloc[3]["customer_id"] == 3


class TestEdgeCases:
    def test_empty_inputs(self):
        """Empty DataFrames return empty DataFrame with correct columns."""
        customers = pd.DataFrame(columns=["customer_id", "name"])
        orders = pd.DataFrame(columns=["customer_id", "amount"])
        result = join_customers_orders(customers, orders)
        assert len(result) == 0
        # Columns should be union + record_status
        expected_cols = {"customer_id", "name", "amount", "record_status"}
        assert set(result.columns) == expected_cols

    def test_one_empty(self):
        """One empty input returns all rows from the other with appropriate status."""
        customers = pd.DataFrame({"customer_id": [1], "name": ["X"]})
        orders = pd.DataFrame(columns=["customer_id", "amount"])
        result = join_customers_orders(customers, orders)
        assert len(result) == 1
        assert result.iloc[0]["record_status"] == "customer_without_order"

    def test_overlapping_columns(self):
        """Overlapping non-key columns are handled with suffix (_x, _y)."""
        customers = pd.DataFrame({
            "customer_id": [1],
            "info": "cust_info"
        })
        orders = pd.DataFrame({
            "customer_id": [1],
            "info": "order_info"
        })
        result = join_customers_orders(customers, orders)
        # Both info columns present with suffixes
        assert "info_x" in result.columns
        assert "info_y" in result.columns
        assert result.iloc[0]["info_x"] == "cust_info"
        assert result.iloc[0]["info_y"] == "order_info"

    def test_missing_key_column_raises(self):
        """If the specified key is missing from either DataFrame, a KeyError is raised."""
        customers = pd.DataFrame({"customer_id": [1]})
        orders = pd.DataFrame({"amount": [100]})
        with pytest.raises(KeyError):
            join_customers_orders(customers, orders, customer_key="customer_id")

    def test_output_columns_no_internal_leftovers(self):
        """No internal columns (_sort_order, _merge, etc.) should appear."""
        customers = pd.DataFrame({"customer_id": [1]})
        orders = pd.DataFrame({"customer_id": [1]})
        result = join_customers_orders(customers, orders)
        internal_cols = {"_sort_order", "_customer_source_order", "_order_source_order", "_merge"}
        assert internal_cols.isdisjoint(set(result.columns))

    def test_index_was_reset(self):
        """Result must have a fresh RangeIndex."""
        customers = pd.DataFrame({"customer_id": [3, 1]})
        orders = pd.DataFrame({"customer_id": [1, 2]})
        result = join_customers_orders(customers, orders)
        assert result.index.equals(pd.RangeIndex(len(result)))

    def test_record_status_values_only_valid(self):
        """record_status column must contain only the three allowed labels."""
        customers = pd.DataFrame({"customer_id": [1, 2, None]})
        orders = pd.DataFrame({"customer_id": [2, 3, None]})
        result = join_customers_orders(customers, orders)
        valid = {"matched", "customer_without_order", "order_without_customer"}
        assert set(result["record_status"].unique()).issubset(valid)
