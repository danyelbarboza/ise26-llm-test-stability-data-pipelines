import pytest
import pandas as pd
from pandas.testing import assert_frame_equal

from ise26.targets import join_customers_orders


def _make_df(data, columns):
    """Helper to create a DataFrame."""
    return pd.DataFrame(data, columns=columns)


class TestJoinCustomersOrders:
    # ---------- helper assertions ----------
    def assert_result_schema(self, result, expected_non_status_cols):
        """Check that result has record_status and all expected columns, no extra internal columns."""
        assert "record_status" in result.columns
        for col in expected_non_status_cols:
            assert col in result.columns
        # no leftover internal columns
        internal = {"_sort_order", "_customer_source_order", "_order_source_order", "_merge"}
        assert set(result.columns).isdisjoint(internal)

    def assert_record_status_values(self, series):
        allowed = {"matched", "customer_without_order", "order_without_customer"}
        assert series.isin(allowed).all()

    def assert_no_null_matched(self, df, key_col):
        """Rows with null key must not be 'matched'."""
        null_mask = df[key_col].isna()
        if null_mask.any():
            assert (df.loc[null_mask, "record_status"] != "matched").all()

    def assert_original_unchanged(self, orig, copy):
        """Check that original DataFrames are not mutated."""
        pd.testing.assert_frame_equal(orig, copy)

    # ---------- basic join ----------
    def test_basic_join_all_three_statuses(self):
        customers = _make_df([[1, "Alice"], [2, "Bob"], [3, "Charlie"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"], [3, "Order3"], [4, "Order4"]], ["customer_id", "order_desc"])

        customers_orig = customers.copy()
        orders_orig = orders.copy()

        result = join_customers_orders(customers, orders)

        # check schema
        expected_cols = {"customer_id", "name", "order_desc"}
        self.assert_result_schema(result, expected_cols)
        self.assert_record_status_values(result["record_status"])

        # verify statuses
        # customer 1 -> matched
        assert result.loc[result["customer_id"] == 1, "record_status"].iloc[0] == "matched"
        # customer 2 -> unmatched customer
        assert result.loc[result["customer_id"] == 2, "record_status"].iloc[0] == "customer_without_order"
        # order 4 -> unmatched order
        assert result.loc[result["customer_id"] == 4, "record_status"].iloc[0] == "order_without_customer"
        # customer 3 matched
        assert result.loc[result["customer_id"] == 3, "record_status"].iloc[0] == "matched"

        # index reset
        assert result.index.equals(pd.RangeIndex(len(result)))

        # original DataFrames unchanged
        self.assert_original_unchanged(customers, customers_orig)
        self.assert_original_unchanged(orders, orders_orig)

    # ---------- null keys ----------
    def test_null_keys_not_matched(self):
        customers = _make_df([[1, "Alice"], [None, "Bob"], [3, "Charlie"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"], [None, "OrderX"], [3, "Order3"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)

        # null key rows must not be matched
        assert result.loc[result["customer_id"].isna(), "record_status"].isin(["customer_without_order", "order_without_customer"]).all()
        self.assert_no_null_matched(result, "customer_id")

        # specific checks
        null_cust = result[(result["customer_id"].isna()) & (result["name"] == "Bob")]
        assert (null_cust["record_status"] == "customer_without_order").all()
        null_ord = result[(result["customer_id"].isna()) & (result["order_desc"] == "OrderX")]
        assert (null_ord["record_status"] == "order_without_customer").all()

    def test_only_null_keys_both_sides(self):
        customers = _make_df([[None, "Alice"], [None, "Bob"]], ["customer_id", "name"])
        orders = _make_df([[None, "Order1"], [None, "Order2"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)

        # all rows should have null customer_id and status not matched
        assert result["customer_id"].isna().all()
        assert (result["record_status"] != "matched").all()
        # counts: customers rows become customer_without_order, orders rows become order_without_customer
        assert (result["record_status"] == "customer_without_order").sum() == 2
        assert (result["record_status"] == "order_without_customer").sum() == 2

    # ---------- duplicate keys ----------
    def test_duplicate_keys_generate_matched_combinations(self):
        customers = _make_df([[1, "Alice"], [1, "Alicia"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"], [1, "Order2"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)

        # 2 customers and 2 orders with same id -> 4 matched rows
        matched = result[result["record_status"] == "matched"]
        assert len(matched) == 4
        # ensure all are matched
        assert (result["record_status"] == "matched").sum() == 4
        self.assert_no_null_matched(result, "customer_id")

    # ---------- no common keys ----------
    def test_no_common_keys(self):
        customers = _make_df([[1, "Alice"], [2, "Bob"]], ["customer_id", "name"])
        orders = _make_df([[3, "Order3"], [4, "Order4"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        assert "matched" not in result["record_status"].values
        assert set(result["record_status"]) == {"customer_without_order", "order_without_customer"}
        assert len(result) == 4

    # ---------- empty DataFrames ----------
    def test_empty_customers(self):
        customers = _make_df([], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        assert (result["record_status"] == "order_without_customer").all()
        assert len(result) == 1
        self.assert_result_schema(result, {"customer_id", "name", "order_desc"})

    def test_empty_orders(self):
        customers = _make_df([[1, "Alice"]], ["customer_id", "name"])
        orders = _make_df([], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        assert (result["record_status"] == "customer_without_order").all()
        assert len(result) == 1

    def test_both_empty(self):
        customers = _make_df([], ["customer_id", "name"])
        orders = _make_df([], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        assert len(result) == 0
        assert "record_status" in result.columns

    # ---------- column conflicts ----------
    def test_column_conflict_suffixes(self):
        customers = _make_df([[1, "Alice", "A"]], ["customer_id", "name", "extra"])
        orders = _make_df([[1, "Order1", "B"]], ["customer_id", "order_desc", "extra"])

        result = join_customers_orders(customers, orders)
        # default merge suffixes _x and _y
        assert "extra_x" in result.columns
        assert "extra_y" in result.columns
        assert result["record_status"].iloc[0] == "matched"

    # ---------- deterministic order ----------
    def test_output_order_is_deterministic(self):
        customers = _make_df([[2, "Bob"], [1, "Alice"], [3, "Charlie"]], ["customer_id", "name"])
        orders = _make_df([[3, "O3"], [1, "O1"]], ["customer_id", "order_desc"])

        result1 = join_customers_orders(customers, orders)
        result2 = join_customers_orders(customers, orders)
        assert_frame_equal(result1, result2)

    # ---------- different key data types ----------
    def test_join_with_string_key(self):
        customers = _make_df([["a", "Alice"], ["b", "Bob"]], ["customer_id", "name"])
        orders = _make_df([["a", "OrderA"], ["c", "OrderC"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders, customer_key="customer_id")
        assert result.loc[result["customer_id"] == "a", "record_status"].iloc[0] == "matched"
        assert result.loc[result["customer_id"] == "b", "record_status"].iloc[0] == "customer_without_order"
        assert result.loc[result["customer_id"] == "c", "record_status"].iloc[0] == "order_without_customer"

    def test_join_with_datetime_key(self):
        from datetime import datetime
        d1 = datetime(2025, 1, 1)
        d2 = datetime(2025, 1, 2)
        d3 = datetime(2025, 1, 3)

        customers = _make_df([[d1, "Alice"], [d2, "Bob"]], ["customer_id", "name"])
        orders = _make_df([[d1, "O1"], [d3, "O3"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        assert result.loc[result["customer_id"] == d1, "record_status"].iloc[0] == "matched"
        assert result.loc[result["customer_id"] == d2, "record_status"].iloc[0] == "customer_without_order"
        assert result.loc[result["customer_id"] == d3, "record_status"].iloc[0] == "order_without_customer"

    # ---------- NaN in non-key columns ----------
    def test_nan_in_non_key_columns_preserved(self):
        customers = _make_df([[1, None], [2, "Bob"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        matched = result[result["record_status"] == "matched"]
        assert pd.isna(matched.iloc[0]["name"])  # NaN preserved
        assert len(result) == 2  # Bob unmatched

    # ---------- edge cases for null handling ----------
    def test_null_key_customer_matches_nothing(self):
        customers = _make_df([[1, "Alice"], [None, "Ghost"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"], [None, "GhostOrder"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        # null key rows: customer null -> customer_without_order, order null -> order_without_customer
        null_cust = result[(result["customer_id"].isna()) & (result["name"] == "Ghost")]
        assert null_cust["record_status"].values[0] == "customer_without_order"
        null_ord = result[(result["customer_id"].isna()) & (result["order_desc"] == "GhostOrder")]
        assert null_ord["record_status"].values[0] == "order_without_customer"
        # matched row remains
        assert result.loc[result["customer_id"] == 1, "record_status"].iloc[0] == "matched"

    def test_null_key_duplicate_with_non_null_mix(self):
        customers = _make_df([[1, "Alice"], [None, "Ghost"], [None, "Phantom"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"], [None, "GhostOrder"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        # null customers: 2 rows -> customer_without_order for both
        null_customers = result[(result["customer_id"].isna()) & (result["name"].isin(["Ghost", "Phantom"]))]
        assert all(null_customers["record_status"] == "customer_without_order")
        # null order -> order_without_customer
        null_order = result[(result["customer_id"].isna()) & (result["order_desc"] == "GhostOrder")]
        assert all(null_order["record_status"] == "order_without_customer")

    def test_index_reset(self):
        customers = _make_df([[2, "Bob"], [1, "Alice"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"]], ["customer_id", "order_desc"])

        result = join_customers_orders(customers, orders)
        assert result.index.tolist() == list(range(len(result)))

    def test_original_dataframes_not_modified(self):
        customers = _make_df([[1, "Alice"]], ["customer_id", "name"])
        orders = _make_df([[1, "Order1"]], ["customer_id", "order_desc"])

        customers_orig = customers.copy()
        orders_orig = orders.copy()

        _ = join_customers_orders(customers, orders)

        assert_frame_equal(customers, customers_orig)
        assert_frame_equal(orders, orders_orig)
