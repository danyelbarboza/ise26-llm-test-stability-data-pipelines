from ise26.targets import parse_order_items_json

import pytest
import pandas as pd
import math
import json


def test_normal_explosion():
    df = pd.DataFrame({
        "order_id": [101, 102],
        "items_json": [
            json.dumps([
                {"sku": "A", "quantity": 2, "unit_price": 10},
                {"sku": "B", "quantity": 3, "unit_price": 5.5}
            ]),
            json.dumps([
                {"sku": "C", "quantity": 1, "unit_price": 20}
            ])
        ]
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [101, 101, 102],
        "sku": ["A", "B", "C"],
        "quantity": [2.0, 3.0, 1.0],
        "unit_price": [10.0, 5.5, 20.0],
        "item_total": [20.0, 16.5, 20.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_custom_column_names():
    df = pd.DataFrame({
        "ord_id": [1],
        "stuff": [json.dumps([{"sku": "X", "quantity": 5, "unit_price": 2}])]
    })
    result = parse_order_items_json(df, json_col="stuff", order_id_col="ord_id")
    expected = pd.DataFrame({
        "ord_id": [1],
        "sku": ["X"],
        "quantity": [5.0],
        "unit_price": [2.0],
        "item_total": [10.0]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_empty_or_invalid_json_skipped():
    df = pd.DataFrame({
        "order_id": [1, 2, 3, 4],
        "items_json": [
            "",
            "not json",
            json.dumps({}),        # dict not list
            json.dumps([{"sku": "skip", "quantity": "a", "unit_price": "b"}])
        ]
    })
    result = parse_order_items_json(df)
    # Row 4's item has invalid quantity/unit_price -> coerced to 0, so it should appear with zeros
    expected = pd.DataFrame({
        "order_id": [4],
        "sku": ["skip"],
        "quantity": [0.0],
        "unit_price": [0.0],
        "item_total": [0.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_quantity_and_unit_price_coercion():
    # Mixed numeric and non-numeric
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": [json.dumps([
            {"sku": "P", "quantity": "x", "unit_price": "y"},
            {"sku": "Q", "quantity": None, "unit_price": None},
            {"sku": "R", "quantity": 3.5, "unit_price": "4.2"},
            {"sku": "S", "quantity": "2", "unit_price": "5"}
        ])]
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [1, 1, 1, 1],
        "sku": ["P", "Q", "R", "S"],
        "quantity": [0.0, 0.0, 3.5, 2.0],
        "unit_price": [0.0, 0.0, 4.2, 5.0],
        "item_total": [0.0, 0.0, 14.7, 10.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_order_preservation():
    df = pd.DataFrame({
        "order_id": [3, 1, 2],
        "items_json": [
            json.dumps([{"sku": "C", "quantity": 1, "unit_price": 1}]),
            json.dumps([{"sku": "A", "quantity": 1, "unit_price": 1}, {"sku": "Aa", "quantity": 1, "unit_price": 1}]),
            json.dumps([{"sku": "B", "quantity": 1, "unit_price": 1}])
        ]
    })
    result = parse_order_items_json(df)
    # The function sorts by source order index, which is the order of rows in df (0,1,2). So output should be order 3, then 1, then 2.
    expected_order_ids = [3, 1, 1, 2]
    assert list(result["order_id"]) == expected_order_ids
    # Check sku ordering within each order
    expected_skus = ["C", "A", "Aa", "B"]
    assert list(result["sku"]) == expected_skus


def test_all_rows_invalid_yields_empty_dataframe():
    df = pd.DataFrame({
        "order_id": [1, 2],
        "items_json": [None, "invalid"]
    })
    result = parse_order_items_json(df)
    assert result.empty
    assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]

def test_mixed_valid_and_invalid_within_json():
    # One item with missing sku? The _normalize_optional_text would make it '' or keep None? The output column "sku" will contain whatever normalization does. We'll just test presence.
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": [json.dumps([
            {"sku": "ok", "quantity": 1, "unit_price": 2},
            {"quantity": 1, "unit_price": 3},  # missing sku
        ])]
    })
    result = parse_order_items_json(df)
    assert len(result) == 2
    # Check that sku for second row is something (probably None or empty string depending on helper). We'll just test existence.
    assert result.iloc[0]["sku"] == "ok"
    # Second row's sku - not testing specific string, just that no error.
    assert "sku" in result.columns
    # quantity and unit_price should be parsed as 1 and 3, item_total=3
    assert result.iloc[1]["quantity"] == 1.0
    assert result.iloc[1]["unit_price"] == 3.0
    assert result.iloc[1]["item_total"] == 3.0


def test_single_order_with_duplicate_items():
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": [json.dumps([
            {"sku": "X", "quantity": 2, "unit_price": 10},
            {"sku": "X", "quantity": 2, "unit_price": 10}
        ])]
    })
    result = parse_order_items_json(df)
    assert len(result) == 2
    # Both rows identical
    expected = pd.DataFrame({
        "order_id": [1, 1],
        "sku": ["X", "X"],
        "quantity": [2.0, 2.0],
        "unit_price": [10.0, 10.0],
        "item_total": [20.0, 20.0]
    })
    pd.testing.assert_frame_equal(result, expected)
