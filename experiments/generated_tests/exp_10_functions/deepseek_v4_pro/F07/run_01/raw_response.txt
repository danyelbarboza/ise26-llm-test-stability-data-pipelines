import pandas as pd
import pytest
from ise26.targets import parse_order_items_json


def test_basic_two_items():
    df = pd.DataFrame({
        "order_id": [100],
        "items_json": ['[{"sku": "A01", "quantity": 2, "unit_price": 9.99}, {"sku": "B02", "quantity": 1, "unit_price": 4.50}]'],
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [100, 100],
        "sku": ["A01", "B02"],
        "quantity": [2, 1],
        "unit_price": [9.99, 4.50],
        "item_total": [19.98, 4.50],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_multiple_orders():

def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_id", "items_json"])
    result = parse_order_items_json(df)
    assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
    assert len(result) == 0


def test_nan_json_skipped():
    df = pd.DataFrame({
        "order_id": [1, 2],

def test_quantity_unit_price_string_conversion():
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['[{"sku": "A", "quantity": "2", "unit_price": "3.5"}]'],

def test_output_column_order_and_sorting():
    df = pd.DataFrame({
        "order_id": [2, 1],
        "items_json": [
            '[{"sku": "B", "quantity": 1, "unit_price": 2}]',
            '[{"sku": "A", "quantity": 3, "unit_price": 1}]',
        ],
    })
    result = parse_order_items_json(df)
    assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
    # Row order must correspond to input row order (0,1) and item order (0)
    assert result.iloc[0]["order_id"] == 2
    assert result.iloc[1]["order_id"] == 1


def test_duplicate_order_ids():
    df = pd.DataFrame({
        "order_id": [5, 5],
        "items_json": [
            '[{"sku": "C", "quantity": 1, "unit_price": 1}]',
            '[{"sku": "D", "quantity": 2, "unit_price": 2}]',
        ],
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [5, 5],
        "sku": ["C", "D"],
        "quantity": [1, 2],
        "unit_price": [1, 2],
        "item_total": [1, 4],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_float_and_int_mixing():
    df = pd.DataFrame({

        "unit_price": [2.5, 3.0],

    pd.testing.assert_frame_equal(result, expected, check_dtype=False)
