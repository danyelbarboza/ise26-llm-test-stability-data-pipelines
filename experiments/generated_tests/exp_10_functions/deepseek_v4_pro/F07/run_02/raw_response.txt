from ise26.targets import parse_order_items_json

import pandas as pd
import pytest

# -----------------------------------------------
# Helpers to build synthetic DataFrames
# -----------------------------------------------
def make_df(data, col_json="items_json", col_order="order_id"):
    """Create a DataFrame with given rows as list of dicts."""
    return pd.DataFrame(data).rename(columns={col_json: "items_json", col_order: "order_id"} if col_json != "items_json" or col_order != "order_id" else {})

def json_str_single_item(sku="A", qty=1, price=10.0):
    import json
    return json.dumps([{"sku": sku, "quantity": qty, "unit_price": price}])

def json_str_items(items):
    import json
    return json.dumps(items)

@pytest.fixture
def default_df():
    return pd.DataFrame({
        "order_id": [1],
        "items_json": [json_str_single_item("SKU1", 2, 15.0)]
    })

# -----------------------------------------------
# Basic success cases
# -----------------------------------------------
def test_single_order_single_item(default_df):
    result = parse_order_items_json(default_df)
    assert list(result.columns) == ["order_id", "sku", "quantity", "unit_price", "item_total"]
    assert len(result) == 1
    row = result.iloc[0]
    assert row["order_id"] == 1
    assert row["sku"] == "SKU1"
    assert row["quantity"] == 2
    assert row["unit_price"] == 15.0
    assert row["item_total"] == 30.0

def test_multiple_items_single_order():
    items = [
        {"sku": "A", "quantity": 1, "unit_price": 10},
        {"sku": "B", "quantity": 3, "unit_price": 5.5}
    ]
    df = pd.DataFrame({
        "order_id": [100],
        "items_json": [json_str_items(items)]
    })
    result = parse_order_items_json(df)
    assert len(result) == 2
    assert list(result["sku"]) == ["A", "B"]
    assert list(result["order_id"]) == [100, 100]
    assert list(result["quantity"]) == [1, 3]
    assert list(result["unit_price"]) == [10.0, 5.5]
    assert list(result["item_total"]) == [10.0, 16.5]

def test_multiple_orders():
    df = pd.DataFrame({
        "order_id": [10, 20],
        "items_json": [
            json_str_items([{"sku": "X", "quantity": 1, "unit_price": 10}]),
            json_str_items([{"sku": "Y", "quantity": 2, "unit_price": 5}])
        ]
    })
    result = parse_order_items_json(df)
    assert len(result) == 2
    assert set(result["order_id"]) == {10, 20}
    # order preserved: first order 10, then 20 (stable sort)
    assert list(result["order_id"]) == [10, 20]

def test_custom_column_names():
    df = pd.DataFrame({
        "ord_num": [1],
        "payload": [json_str_single_item("P", 1, 100)]
    })
    result = parse_order_items_json(df, json_col="payload", order_id_col="ord_num")
    assert list(result.columns) == ["ord_num", "sku", "quantity", "unit_price", "item_total"]
    assert result.iloc[0]["ord_num"] == 1

# -----------------------------------------------
# Deterministic order
# -----------------------------------------------
def test_output_order_preserves_source_order_and_item_order():
    import json
    def make_items(*skus):
        return [{"sku": s, "quantity": 1, "unit_price": 1} for s in skus]
    df = pd.DataFrame({
        "order_id": [2, 1],
        "items_json": [
            json.dumps(make_items("B1", "B2")),
            json.dumps(make_items("A1", "A2", "A3"))
        ]
    })
    result = parse_order_items_json(df)
    # Expected order: first row (order_id=2) items in order B1, B2; then second row (order_id=1) items A1,A2,A3
    expected_skus = ["B1", "B2", "A1", "A2", "A3"]
    assert list(result["sku"]) == expected_skus
