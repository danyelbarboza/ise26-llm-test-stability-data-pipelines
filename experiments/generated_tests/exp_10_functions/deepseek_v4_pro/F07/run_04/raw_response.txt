import pandas as pd
import pytest
from ise26.targets import parse_order_items_json


class TestParseOrderItemsJson:
    def test_basic_single_item(self):
        df = pd.DataFrame(
            {
                "order_id": [1],
                "items_json": ['[{"sku": "A", "quantity": 2, "unit_price": 10.5}]'],
            }
        )
        result = parse_order_items_json(df)
        expected = pd.DataFrame(
            {
                "order_id": [1],
                "sku": ["A"],
                "quantity": [2],
                "unit_price": [10.5],
                "item_total": [21.0],
            }
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_multiple_items_one_order(self):
        df = pd.DataFrame(
            {
                "order_id": [42],
                "items_json": [
                    '[{"sku": "B", "quantity": 1, "unit_price": 20.0}, {"sku": "C", "quantity": 3, "unit_price": 5.5}]'
                ],
            }
        )
        result = parse_order_items_json(df)
        expected = pd.DataFrame(
            {
                "order_id": [42, 42],
                "sku": ["B", "C"],
                "quantity": [1, 3],
                "unit_price": [20.0, 5.5],
                "item_total": [20.0, 16.5],
            }
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_multiple_orders_preserve_order(self):
        df = pd.DataFrame(
            {
                "order_id": [10, 20],
                "items_json": [
                    '[{"sku": "X", "quantity": 1, "unit_price": 10}]',
                    '[{"sku": "Y", "quantity": 2, "unit_price": 5}]',
                ],
            }
        )
        result = parse_order_items_json(df)
        expected = pd.DataFrame(
            {
                "order_id": [10, 20],
                "sku": ["X", "Y"],
                "quantity": [1, 2],
                "unit_price": [10, 5],
                "item_total": [10, 10],
            }
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_json_array_skipped(self):
        df = pd.DataFrame({"order_id": [1], "items_json": ["[]"]})
        result = parse_order_items_json(df)
        assert result.empty
        assert list(result.columns) == [
            "order_id",
            "sku",
            "quantity",
            "unit_price",
            "item_total",
        ]

    def test_null_json_skipped(self):
        df = pd.DataFrame({"order_id": [1], "items_json": [None]})
        result = parse_order_items_json(df)
        assert result.empty

    def test_nan_json_skipped(self):
        df = pd.DataFrame({"order_id": [1], "items_json": [float("nan")]})
        result = parse_order_items_json(df)
        assert result.empty

    def test_invalid_json_skipped(self):
        df = pd.DataFrame({"order_id": [1], "items_json": ["not a json"]})
        result = parse_order_items_json(df)
        assert result.empty

    def test_non_list_json_skipped(self):
        df = pd.DataFrame({"order_id": [1], "items_json": ['{"sku": "A"}']})
        result = parse_order_items_json(df)
        assert result.empty

    def test_missing_quantity_becomes_zero(self):
        df = pd.DataFrame(
            {
                "order_id": [1],
                "items_json": [
                    '[{"sku": "A", "unit_price": 10}, {"sku": "B", "quantity": 5}]'
                ],
            }
        )
        result = parse_order_items_json(df)
        expected = pd.DataFrame(
            {
                "order_id": [1, 1],
                "sku": ["A", "B"],
                "quantity": [0, 5],
                "unit_price": [10, 0],
                "item_total": [0, 0],  # first: 0*10=0, second: 5*0=0
            }
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_non_numeric_quantity_becomes_zero(self):
        df = pd.DataFrame(
            {
                "order_id": [1],
                "items_json": [
                    '[{"sku": "A", "quantity": "abc", "unit_price": 10}]'
                ],
            }
        )
        result = parse_order_items_json(df)
        assert result.loc[0, "quantity"] == 0
        assert result.loc[0, "item_total"] == 0

    def test_missing_sku_becomes_empty_string(self):
        df = pd.DataFrame(
            {
                "order_id": [1],
                "items_json": ['[{"quantity": 2, "unit_price": 5}]'],
            }
        )
        result = parse_order_items_json(df)
        assert result.loc[0, "sku"] == ""

    def test_custom_column_names(self):
        df = pd.DataFrame(
            {"id": [100], "json": ['[{"sku": "D", "quantity": 3, "unit_price": 7}]']}
        )
        result = parse_order_items_json(df, json_col="json", order_id_col="id")
        expected = pd.DataFrame(
            {
                "id": [100],
                "sku": ["D"],
                "quantity": [3],
                "unit_price": [7],
                "item_total": [21],
            }
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_output_column_names_default(self):
        df = pd.DataFrame(
            {"order_id": [1], "items_json": ['[{"sku": "Z", "quantity": 1, "unit_price": 1}]']}
        )
        result = parse_order_items_json(df)
        assert list(result.columns) == [
            "order_id",
            "sku",
            "quantity",
            "unit_price",
            "item_total",
        ]

    def test_output_preserves_order_id_type(self):
        df = pd.DataFrame(
            {"order_id": ["O-1"], "items_json": ['[{"sku": "P", "quantity": 1, "unit_price": 2}]']}
        )
        result = parse_order_items_json(df)
        assert result["order_id"].dtype == object

    def test_empty_input_dataframe(self):
        df = pd.DataFrame({"order_id": pd.Series([], dtype=int), "items_json": pd.Series([], dtype=str)})
        result = parse_order_items_json(df)
        assert result.empty
        assert list(result.columns) == [
            "order_id",
            "sku",
            "quantity",
            "unit_price",
            "item_total",
        ]

    def test_all_items_invalid_some_orders(self):
        df = pd.DataFrame(
            {
                "order_id": [1, 2],
                "items_json": [
                    "invalid",
                    '[{"sku": "OK", "quantity": 1, "unit_price": 1}]',
                ],
            }
        )
        result = parse_order_items_json(df)
        assert len(result) == 1
        assert result.iloc[0]["order_id"] == 2
        assert result.iloc[0]["sku"] == "OK"

    def test_item_order_within_order_preserved(self):
        df = pd.DataFrame(
            {
                "order_id": [1],
                "items_json": [
                    '[{"sku": "first", "quantity": 1, "unit_price": 1}, {"sku": "second", "quantity": 1, "unit_price": 1}]'
                ],
            }
        )
        result = parse_order_items_json(df)
        assert list(result["sku"]) == ["first", "second"]

    def test_negative_quantity_and_price_allowed(self):
        df = pd.DataFrame(
            {
                "order_id": [1],
                "items_json": [
                    '[{"sku": "N", "quantity": -2, "unit_price": -3.5}]'
                ],
            }
        )
        result = parse_order_items_json(df)
        expected = pd.DataFrame(
            {
                "order_id": [1],
                "sku": ["N"],
                "quantity": [-2],
                "unit_price": [-3.5],
                "item_total": [7.0],
            }
        )
        pd.testing.assert_frame_equal(result, expected)
