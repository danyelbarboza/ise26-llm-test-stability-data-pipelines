import pandas as pd
import pytest
from ise26.targets import parse_order_items_json


def test_basic_explosion():
    df = pd.DataFrame({
        "order_id": [1],
        "items_json": ['[{"sku": "A", "quantity": 2, "unit_price": 10.5}]']
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [1],
        "sku": ["A"],
        "quantity": [2],
        "unit_price": [10.5],
        "item_total": [21.0]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_multiple_items_single_order():
    df = pd.DataFrame({
        "order_id": [101],
        "items_json": ['[{"sku": "X", "quantity": 1, "unit_price": 5.0}, '
                       '{"sku": "Y", "quantity": 3, "unit_price": 2.5}]']
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [101, 101],
        "sku": ["X", "Y"],
        "quantity": [1, 3],
        "unit_price": [5.0, 2.5],
        "item_total": [5.0, 7.5]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_mixed_valid_invalid_rows():
    df = pd.DataFrame({
        "order_id": [1, 2, 3],
        "items_json": [
            '[{"sku": "A", "quantity": 2, "unit_price": 10}]',
            "not_json",
            '[{"sku": "B", "quantity": 1, "unit_price": 20}]'
        ]
    })
    result = parse_order_items_json(df)
    expected = pd.DataFrame({
        "order_id": [1, 3],
        "sku": ["A", "B"],
        "quantity": [2, 1],
        "unit_price": [10, 20],
        "item_total": [20, 20]
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

def test_duplicate_order_ids():
    df = pd.DataFrame({
        "order_id": [1, 1],
        "items_json": [
            '[{"sku": "A", "quantity": 1, "unit_price": 10}]',
            '[{"sku": "B", "quantity": 2, "unit_price": 5}]'
        ]
    })
    result = parse_order_items_json(df)
    # Both orders should produce items; order of rows respects input row order
    expected = pd.DataFrame({
        "order_id": [1, 1],
        "sku": ["A", "B"],
        "quantity": [1, 2],
        "unit_price": [10, 5],
        "item_total": [10, 10]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_sort_order_multiple_rows():
    df = pd.DataFrame({
        "order_id": [10, 20],
        "items_json": [
            '[{"sku": "Y", "quantity": 1, "unit_price": 1}, '
            '{"sku": "Z", "quantity": 2, "unit_price": 2}]',
            '[{"sku": "X", "quantity": 3, "unit_price": 3}]'
        ]
    })
    result = parse_order_items_json(df)
    # output should preserve source row order: first order (10) items, then order (20)
    expected_order_ids = [10, 10, 20]
    expected_skus = ["Y", "Z", "X"]
    assert result["order_id"].tolist() == expected_order_ids
    assert result["sku"].tolist() == expected_skus
