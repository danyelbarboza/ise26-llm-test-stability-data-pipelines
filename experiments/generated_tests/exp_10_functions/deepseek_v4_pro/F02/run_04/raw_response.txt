import pandas as pd
import pytest
from ise26.targets import deduplicate_events


def test_deduplicate_keeps_most_recent_timestamp_per_id():
    """Basic deduplication: keep row with largest valid timestamp per id."""
    df = pd.DataFrame({
        "event_id": [1, 1, 2],
        "updated_at": ["2023-01-01", "2023-02-01", "2023-03-01"],
        "value": [10, 20, 30]
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # For id=1, the most recent is 2023-02-01 (value=20). Id=2 only one row.
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "updated_at": ["2023-02-01", "2023-03-01"],
        "value": [20, 30]
    })
    pd.testing.assert_frame_equal(result, expected, check_exact=True)


def test_tie_break_keeps_last_original_order():
    """When timestamps tie for same id, keep the row that appears later in original order."""
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2023-01-01", "2023-01-01", "2023-01-01"],
        "value": [10, 20, 30]
    })
    # Order: row 0, row 1, row 2. Last original row is index 2 -> value=30
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-01"],
        "value": [30]
    })
    pd.testing.assert_frame_equal(result, expected, check_exact=True)


def test_null_identifiers_preserved():
    """Rows with null event_id should be kept even if they appear to be duplicates."""
    df = pd.DataFrame({
        "event_id": [None, None, 1, 1],
        "updated_at": ["2023-01-01", "2023-02-01", "2023-01-01", "2023-02-01"],
        "value": [10, 20, 30, 40]
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # All null-id rows retained, identified id=1 deduplicated, keeping last (value=40)
    # Original order: row0 (None,10), row1 (None,20), row2 (id=1,30), row3 (id=1,40)
    # Identified: id=1, keep last -> row3. Unidentified: rows 0,1.
    # Sorted by original order: row0, row1, row3.
    expected = pd.DataFrame({
        "event_id": [None, None, 1.0],
        "updated_at": ["2023-01-01", "2023-02-01", "2023-02-01"],
        "value": [10, 20, 40]
    })
    # event_id becomes float because original had NaN for None? Actually None becomes NaN,
    # and after processing it may remain NaN (float). We compare with NaN.
    pd.testing.assert_frame_equal(result, expected, check_dtype=False, check_exact=False)


def test_invalid_timestamps_treated_as_older():
    """Invalid timestamps are coerced to NaT and treated as older than valid timestamps."""
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["invalid", "2023-01-01", "2023-02-01"],
        "value": [10, 20, 30]
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Sorted ascending: NaT (invalid) first, then 2023-01-01, then 2023-02-01.
    # Last is 2023-02-01, so row with value=30 kept.
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-02-01"],
        "value": [30]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_all_invalid_timestamps_tie_break():
    """When all timestamps for an id are invalid, tie-break keeps last original row."""
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["bad", "also_bad", "not_a_date"],
        "value": [10, 20, 30]
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # All parsed to NaT, then sort stable keeps original order ascending,
    # last original is row index 2 -> value=30.
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["not_a_date"],
        "value": [30]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_mixed_valid_and_invalid_timestamps():
    """Combination of valid and invalid timestamps; keep the most recent valid when present."""
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": ["2023-01-01", "bad_date", "2023-02-01"],
        "value": [10, 20, 30]
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # Parsed: 2023-01-01 (idx0), NaT (idx1), 2023-02-01 (idx2)
    # Sorted: NaT first, then 2023-01-01, then 2023-02-01. Last is idx2 -> value=30.
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-02-01"],
        "value": [30]
    })
    pd.testing.assert_frame_equal(result, expected)


def test_no_duplicates_returns_same_rows_in_order():
    """If no duplicates exist, the returned DataFrame should match original order exactly."""
    df = pd.DataFrame({
        "event_id": [1, 2, 3],
        "updated_at": ["2023-01-01", "2023-02-01", "2023-03-01"],
        "val": ["a", "b", "c"]
    })
    result = deduplicate_events(df)
    pd.testing.assert_frame_equal(result.reset_index(drop=True), df.reset_index(drop=True))


def test_all_identified_duplicates_removed_correct_order():
    """Multiple ids with duplicates, verify order is original positions of kept rows."""
    df = pd.DataFrame({
        "event_id": [1, 2, 1, 2],
        "updated_at": ["2023-01-01", "2023-02-01", "2023-02-01", "2023-02-02"],
        "v": [10, 20, 30, 40]
    })
    # For id=1: rows 0 (2023-01-01) and 2 (2023-02-01). Keep row 2 (value=30).
    # For id=2: rows 1 (2023-02-01) and 3 (2023-02-02). Keep row 3 (value=40).
    # Original order of kept rows: row1? Wait, row1 is for id=2 but it's removed; the kept ones are row2 and row3.
    # In original order, row2 comes before row3. So result should be row2 then row3.
    expected = pd.DataFrame({
        "event_id": [1, 2],
        "updated_at": ["2023-02-01", "2023-02-02"],
        "v": [30, 40]
    })
    result = deduplicate_events(df)
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_input_not_mutated():
    """The original DataFrame should remain unchanged after calling the function."""
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-01", "2023-02-01"],
        "x": [5, 6]
    })
    original = df.copy()
    _ = deduplicate_events(df)
    pd.testing.assert_frame_equal(df, original)


def test_custom_id_and_timestamp_columns():
    """Verify that custom column names work."""
    df = pd.DataFrame({
        "my_id": [10, 10, 20],
        "ts": ["2023-01-01", "2023-01-02", "2023-01-01"],
        "val": [100, 200, 300]
    })
    result = deduplicate_events(df, id_col="my_id", timestamp_col="ts")
    expected = pd.DataFrame({
        "my_id": [10, 20],
        "ts": ["2023-01-02", "2023-01-01"],
        "val": [200, 300]
    })
    pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)


def test_empty_dataframe():
    """Return empty DataFrame unchanged."""
    df = pd.DataFrame(columns=["event_id", "updated_at", "value"])
    result = deduplicate_events(df)
    pd.testing.assert_frame_equal(result.reset_index(drop=True), df)


def test_missing_columns_raises_keyerror():
    """If specified column does not exist, the function should raise KeyError."""
    df = pd.DataFrame({"event_id": [1], "updated_at": ["2023-01-01"]})
    with pytest.raises(KeyError):
        deduplicate_events(df, id_col="missing_col")
