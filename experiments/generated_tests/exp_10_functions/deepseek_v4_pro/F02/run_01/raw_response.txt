import pytest
import pandas as pd
from ise26.targets import deduplicate_events


def test_basic_dedup():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": pd.to_datetime(["2023-01-02", "2023-01-01"]),
        "value": ["a", "b"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": pd.to_datetime(["2023-01-02"]),
        "value": ["a"],
    })
    pd.testing.assert_frame_equal(result, expected)


def test_multiple_duplicates():
    df = pd.DataFrame({
        "event_id": [10, 10, 10],
        "updated_at": pd.to_datetime(["2022-05-01", "2022-06-01", "2022-04-01"]),
        "info": ["x", "y", "z"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # most recent is 2022-06-01, row index 1
    expected = pd.DataFrame({
        "event_id": [10],
        "updated_at": pd.to_datetime(["2022-06-01"]),
        "info": ["y"],
    })
    pd.testing.assert_frame_equal(result, expected)


def test_null_identifiers_preserved():
    df = pd.DataFrame({

def test_mixed_ids():
    df = pd.DataFrame({
        "event_id": [1, None, 1, 2, None],
        "updated_at": pd.to_datetime([
            "2023-01-01", "2023-03-01", "2023-01-02", "2023-04-01", "2023-02-01"
        ]),
        "data": ["a", "b", "c", "d", "e"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # For id=1: rows 0 and 2, keep most recent (row 2, timest 2023-01-02, data 'c')
    # id=2: only row 3 kept
    # nulls rows 1 and 4 both kept
    expected = pd.DataFrame({
        "event_id": [1.0, None, 2.0, None],
        "updated_at": pd.to_datetime(["2023-01-02", "2023-03-01", "2023-04-01", "2023-02-01"]),
        "data": ["c", "b", "d", "e"],
    })
    # order of rows preserved? The function sorts by original order after concat,
    # so rows should be in original: row 2 (data c), row 1 (b), row 3 (d), row 4 (e)
    expected = expected.iloc[[0, 1, 2, 3]].reset_index(drop=True)
    actual_order = [2, 1, 3, 4]  # original indices
    expected = pd.DataFrame({
        "event_id": [1.0, None, 2.0, None],
        "updated_at": pd.to_datetime(["2023-01-02", "2023-03-01", "2023-04-01", "2023-02-01"]),
        "data": ["c", "b", "d", "e"],
    }).reset_index(drop=True)
    pd.testing.assert_frame_equal(result, expected)


def test_invalid_timestamps_treated_as_oldest():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": [
            "not-a-date",
            "2023-01-01",
            None,
        ],
        "value": ["a", "b", "c"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # valid timestamps: row 1 (2023-01-01) is most recent, row 0 and 2 become NaT (older)
    # Keep last in original order among oldest? But the sort puts NaT first (na_position='first'),
    # and among NaT, ascending _original_order, then last overall keeps the one with max _original_order
    # among all NaT rows. So we must examine.
    # rows: index0 "not-a-date" -> NaT, index1 "2023-01-01" valid, index2 None -> NaT.
    # Sorted by id, _parsed_timestamp (NaT first, then 2023-01-01), _original_order asc:
    #   id=1, NaT, orig_order 0 -> first
    #   id=1, NaT, orig_order 2 -> second
    #   id=1, 2023-01-01, orig_order 1 -> third
    # drop_duplicates keep='last' for id=1 keeps third row, which has orig_order 1, value 'b'.
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-01"],
        "value": ["b"],
    })
    pd.testing.assert_frame_equal(result, expected)


def test_tie_timestamps_same_id():
    df = pd.DataFrame({
        "event_id": [5, 5, 5],
        "updated_at": pd.to_datetime(["2023-01-01", "2023-01-01", "2023-01-01"]),
        "info": ["first", "second", "third"],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # tie: keep last row in original order -> info="third"
    expected = pd.DataFrame({
        "event_id": [5],
        "updated_at": pd.to_datetime(["2023-01-01"]),
        "info": ["third"],
    })
    pd.testing.assert_frame_equal(result, expected)


def test_all_invalid_timestamps():
    df = pd.DataFrame({
        "event_id": [42, 42, 42],
        "updated_at": ["???", None, "bad"],
        "data": [1, 2, 3],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # all timestamps become NaT, sorted with na_position='first', then by _original_order asc
    # drop_duplicates keep='last' picks max original order: row idx 2 with data=3
    expected = pd.DataFrame({
        "event_id": [42],
        "updated_at": ["bad"],
        "data": [3],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_empty_dataframe():

    })

def test_reset_index_after_dedup():
    df = pd.DataFrame({
        "event_id": [2, 1, 2, 1],
        "updated_at": pd.to_datetime(["2023-02-01", "2023-02-02", "2023-02-03", "2023-01-01"]),
        "x": [0, 1, 2, 3],
    })
    result = deduplicate_events(df)
    # expected rows: for id=1: row1 (2023-02-02, x=1) vs row3 (2023-01-01, x=3) -> keep row1
    # id=2: row0 (2023-02-01, x=0) vs row2 (2023-02-03, x=2) -> keep row2
    # order original: row2 (id2), row1 (id1)
    expected = pd.DataFrame({
        "event_id": [2, 1],
        "updated_at": pd.to_datetime(["2023-02-03", "2023-02-02"]),
        "x": [2, 1],
    })
    pd.testing.assert_frame_equal(result, expected)
    assert result.index.tolist() == [0, 1]


def test_handling_mixed_data_types_in_timestamp():
    df = pd.DataFrame({
        "event_id": [1, 1],
        "updated_at": ["2023-01-01", pd.Timestamp("2023-01-02")],
        "val": [0, 1],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # should parse correctly, keep row with 2023-01-02
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": ["2023-01-02"],
        "val": [1],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_duplicates_keep_only_last_original_order_when_all_nat():
    df = pd.DataFrame({
        "event_id": [1, 1, 1],
        "updated_at": [None, None, None],
        "val": [10, 20, 30],
    })
    result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
    # all NaT, keep last original row -> index 2 with val=30
    expected = pd.DataFrame({
        "event_id": [1],
        "updated_at": [None],
        "val": [30],
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)
