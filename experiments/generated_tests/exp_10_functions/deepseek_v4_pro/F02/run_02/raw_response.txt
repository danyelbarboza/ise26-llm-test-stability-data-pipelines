import pytest
import pandas as pd
from ise26.targets import deduplicate_events


class TestDeduplicateEvents:
    def test_keep_most_recent_valid_timestamp(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 2],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-01"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df)
        expected_ids = [1, 2]
        expected_values = [20, 30]
        assert list(result["event_id"]) == expected_ids
        assert list(result["value"]) == expected_values

    def test_tie_break_keep_last_row_original_order(self):
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2023-01-01", "2023-01-01"],
            "value": [10, 20]
        })
        result = deduplicate_events(df)
        assert len(result) == 1
        assert result.iloc[0]["event_id"] == 1
        assert result.iloc[0]["value"] == 20

    def test_invalid_timestamp_treated_as_older(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["invalid", "2023-01-02", "2023-01-01"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df)
        assert len(result) == 1
        assert result.iloc[0]["value"] == 20

    def test_null_id_rows_preserved(self):
        df = pd.DataFrame({
            "event_id": [None, 1, None, 1],
            "updated_at": ["2023-01-01", "2023-01-01", "2023-01-02", "2023-01-03"],
            "value": [10, 20, 30, 40]
        })
        result = deduplicate_events(df)
        assert list(result["value"]) == [10, 30, 40]

    def test_input_dataframe_not_mutated(self):
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "value": [10, 20]
        })
        original_df = df.copy()
        _ = deduplicate_events(df)
        pd.testing.assert_frame_equal(df, original_df)

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["event_id", "updated_at"])
        result = deduplicate_events(df)
        assert result.empty
        assert list(result.columns) == ["event_id", "updated_at"]

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "id": [1, 1, 2],
            "time": ["2023-01-02", "2023-01-01", "2023-01-01"],
            "val": [10, 20, 30]
        })
        result = deduplicate_events(df, id_col="id", timestamp_col="time")
        assert list(result["id"]) == [1, 2]
        assert list(result["val"]) == [10, 30]

    def test_all_null_ids_preserved(self):
        df = pd.DataFrame({
            "event_id": [None, None, None],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
            "value": [1, 2, 3]
        })
        result = deduplicate_events(df)
        assert len(result) == 3
        assert list(result["value"]) == [1, 2, 3]

    def test_all_invalid_timestamps_with_same_id(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["bad", "bad2", "bad3"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df)
        assert len(result) == 1
        assert result.iloc[0]["value"] == 30

    def test_mixed_valid_and_invalid_with_duplicates(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1, 1],
            "updated_at": ["2023-01-01", "invalid", "2023-01-01", "2023-01-02"],
            "value": [10, 20, 30, 40]
        })
        result = deduplicate_events(df)
        assert len(result) == 1
        assert result.iloc[0]["value"] == 40

    def test_index_reset(self):
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "value": [10, 20]
        }, index=[10, 20])
        result = deduplicate_events(df)
        assert list(result.index) == [0]
