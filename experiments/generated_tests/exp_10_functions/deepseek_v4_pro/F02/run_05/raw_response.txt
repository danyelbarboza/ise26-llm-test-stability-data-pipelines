import pandas as pd
import pytest
from ise26.targets import deduplicate_events


class TestDeduplicateEvents:
    """Tests for deduplicate_events function."""

    def test_basic_deduplication_keeps_most_recent(self):
        """Keep the row with the most recent timestamp per event id."""
        df = pd.DataFrame({
            "event_id": [1, 1, 2, 2],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-02-10", "2023-02-09"],
            "value": [10, 20, 30, 40]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # Expected: for id=1 keep row with 2023-01-02, for id=2 keep row with 2023-02-10
        expected_ids = [1, 2]
        expected_values = [20, 30]
        assert result["event_id"].tolist() == expected_ids
        assert result["value"].tolist() == expected_values
        # Original order preserved (row 1 first, then row 2)
        assert result.index.tolist() == [0, 1]

    def test_timestamp_ties_last_row_wins(self):
        """When timestamps tie, keep the last row in original order."""
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-01", "2023-01-01"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 1
        assert result.iloc[0]["value"] == 30  # last row kept

    def test_invalid_timestamps_treated_as_older(self):
        """Rows with unparseable timestamps are considered older than valid ones."""
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["invalid", "2023-01-02", "2023-01-01"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 1
        # The valid timestamp 2023-01-02 is most recent, should be kept
        assert result.iloc[0]["value"] == 20

    def test_all_invalid_timestamps_keep_last_row(self):
        """When all timestamps for an id are invalid, keep the last row in order."""
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["bad", "also_bad", "still_bad"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 1
        assert result.iloc[0]["value"] == 30

    def test_null_identifiers_preserved(self):
        """Rows with null id are kept as-is, even if they appear multiple times."""
        df = pd.DataFrame({
            "event_id": [None, None, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-01", "2023-01-02"],
            "value": [10, 20, 30, 40]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # Two null id rows should survive, plus one row for id=1 (the most recent)
        assert len(result) == 3
        # Check null rows are both present
        null_rows = result[result["event_id"].isna()]
        assert len(null_rows) == 2
        assert null_rows["value"].tolist() == [10, 20]
        # The identified row should be the most recent
        identified = result[result["event_id"] == 1].iloc[0]
        assert identified["value"] == 40

    def test_original_dataframe_not_mutated(self):
        """The input DataFrame should remain unchanged."""
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "value": [10, 20]
        })
        df_copy = df.copy()
        _ = deduplicate_events(df)
        pd.testing.assert_frame_equal(df, df_copy)

    def test_column_order_preserved(self):
        """Output columns should match input column order (excluding temp columns)."""
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "value": [10, 20]
        })
        result = deduplicate_events(df)
        assert list(result.columns) == list(df.columns)

    def test_empty_dataframe(self):
        """Empty input returns empty result with same structure."""
        df = pd.DataFrame(columns=["event_id", "updated_at", "value"])
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert result.empty
        assert list(result.columns) == list(df.columns)

    def test_only_null_ids(self):
        """All ids null keeps every row."""
        df = pd.DataFrame({
            "event_id": [None, None, None],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 3
        assert result["value"].tolist() == [10, 20, 30]

    def test_mixed_valid_invalid_timestamps_with_null(self):
        """Complex case: null ids, valid and invalid timestamps, duplicate ids."""
        df = pd.DataFrame({
            "event_id": [1, 1, 2, None, 2, 2],
            "updated_at": ["invalid", "2023-01-02", "2023-03-03", "2023-01-01",
                           "bad_date", "2023-03-01"],
            "value": [10, 20, 30, 40, 50, 60]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        # Null id row preserved
        assert result["event_id"].isna().sum() == 1
        assert result[result["event_id"].isna()]["value"].iloc[0] == 40
        # For id=1: row with invalid timestamp is older than valid 2023-01-02 -> keep row with value 20
        id1 = result[result["event_id"] == 1].iloc[0]
        assert id1["value"] == 20
        # For id=2: rows: value 30 (2023-03-03 - most recent valid), 50 (invalid, older), 60 (2023-03-01, valid but earlier). Keep 30.
        id2 = result[result["event_id"] == 2].iloc[0]
        assert id2["value"] == 30
        # Order preserved: null row (position 3) after id1,2 rows? Input order: id=1(pos0), id=1(pos1), id=2(pos2), null(pos3), id=2(pos4), id=2(pos5). After dedup: id1 (pos1 value20) would be kept, id2 (pos2 value30) kept, null (pos3) kept. Original order: pos1 (20), pos2 (30), pos3 (40). So output should be values: 20, 30, 40.
        assert result["value"].tolist() == [20, 30, 40]

    def test_custom_column_names(self):
        """Function works with custom id and timestamp column names."""
        df = pd.DataFrame({
            "uid": [1, 1, 2],
            "ts": ["2023-01-01", "2023-01-02", "2023-01-01"],
            "data": ["a", "b", "c"]
        })
        result = deduplicate_events(df, id_col="uid", timestamp_col="ts")
        assert len(result) == 2
        # uid=1 keeps the later timestamp (value "b")
        assert set(result[result["uid"] == 1]["data"]) == {"b"}
        # uid=2 stays as is
        assert set(result[result["uid"] == 2]["data"]) == {"c"}

    def test_output_index_is_reset(self):
        """Output should have a default RangeIndex."""
        df = pd.DataFrame({
            "event_id": [5, 5],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "value": [100, 200]
        })
        result = deduplicate_events(df)
        assert isinstance(result.index, pd.RangeIndex)
        assert result.index.start == 0
        assert result.index.stop == len(result)
        assert result.index.step == 1

    def test_timestamp_sorting_with_na_position(self):
        """Verify that NaT timestamps are placed before any valid timestamp."""
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": [pd.NaT, "2023-01-01"],
            "value": [10, 20]
        })
        # NaT row should be considered older; keep row with valid timestamp
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        assert len(result) == 1
        assert result.iloc[0]["value"] == 20

    def test_function_does_not_alter_global_state(self):
        """Quick check that calling function does not change global pandas options inadvertently."""
        import pandas as pd
        _ = deduplicate_events(pd.DataFrame({"event_id": [1], "updated_at": ["2023"]}))
        # No assertion needed, just run without error

    def test_duplicates_with_different_na_values(self):
        """Various null representations (None, np.nan) are treated consistently."""
        df = pd.DataFrame({
            "event_id": [None, pd.NA, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-01", "2023-01-03"],
            "value": [10, 20, 30, 40]
        })
        result = deduplicate_events(df, id_col="event_id", timestamp_col="updated_at")
        null_count = result["event_id"].isna().sum()
        assert null_count == 2
        # id=1 should keep the most recent (value 40)
        assert result[result["event_id"] == 1]["value"].iloc[0] == 40
