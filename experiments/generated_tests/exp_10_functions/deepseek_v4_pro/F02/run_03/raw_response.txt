import pandas as pd
import pytest
from ise26.targets import deduplicate_events


class TestDeduplicateEvents:
    def test_keeps_most_recent_timestamp(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 2],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
            "value": [10, 20, 30]
        })
        result = deduplicate_events(df)
        expected = pd.DataFrame({
            "event_id": [1, 2],
            "updated_at": ["2023-01-02", "2023-01-03"],
            "value": [20, 30]
        })
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_null_identifiers_are_preserved(self):
        df = pd.DataFrame({
            "event_id": [None, None, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04"],
            "value": [1, 2, 3, 4]
        })
        result = deduplicate_events(df)
        # Null ids are kept as-is, identified rows deduplicated (keep most recent)
        expected = pd.DataFrame({
            "event_id": [None, None, 1],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-04"],
            "value": [1, 2, 4]
        })
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_invalid_timestamps_treated_as_older(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": [None, "2023-01-01", "2022-01-01"],
            "value": [1, 2, 3]
        })
        result = deduplicate_events(df)
        # None converts to NaT; NaT is older than any valid date, so "2023-01-01" is kept
        expected = pd.DataFrame({
            "event_id": [1],
            "updated_at": ["2023-01-01"],
            "value": [2]
        })
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_timestamp_tie_keeps_last_original_order(self):
        df = pd.DataFrame({
            "event_id": [1, 1, 1],
            "updated_at": ["2023-01-01", "2023-01-01", "2023-01-01"],
            "value": [1, 2, 3]
        })
        result = deduplicate_events(df)
        # All timestamps equal; keep the row that appeared last (index 2)
        expected = pd.DataFrame({
            "event_id": [1],
            "updated_at": ["2023-01-01"],
            "value": [3]
        })
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_all_nat_timestamp_keeps_last_original_order(self):
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": [None, None],
            "value": [10, 20]
        })
        result = deduplicate_events(df)
        # Both timestamps are NaT; keep the row with larger original order (last row)
        expected = pd.DataFrame({
            "event_id": [1],
            "updated_at": [None],
            "value": [20]
        })
        # updated_at column will have NaT after parsing, but None is preserved? 
        # The function uses pd.to_datetime with errors='coerce', so None -> NaT.
        # When returning, the column type may be datetime64[ns], NaT is displayed as NaT.
        # Let's adjust expected to have NaT for matching.
        expected["updated_at"] = pd.to_datetime(expected["updated_at"])
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_preserves_original_order_when_keeping_rows(self):
        # Original order: 0 (id=1 old), 1 (id=2), 2 (id=1 newest), 3 (id=3)
        df = pd.DataFrame({
            "event_id": [1, 2, 1, 3],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03", "2023-01-04"],
            "value": [10, 20, 30, 40]
        })
        result = deduplicate_events(df)
        # For id=1, the newest is row 2. So kept rows: row 2 (orig order 2) and row 1 (orig order 1), row 3 (orig order 3)
        # Sorted by original_order: row1, row2, row3 -> columns id=2, id=1, id=3
        expected = pd.DataFrame({
            "event_id": [2, 1, 3],
            "updated_at": ["2023-01-02", "2023-01-03", "2023-01-04"],
            "value": [20, 30, 40]
        })
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_input_dataframe_not_mutated(self):
        df = pd.DataFrame({
            "event_id": [1, 1],
            "updated_at": ["2023-01-01", "2023-01-02"],
            "value": [10, 20]
        })
        df_original = df.copy()
        _ = deduplicate_events(df)
        pd.testing.assert_frame_equal(df, df_original)

    def test_custom_column_names(self):
        df = pd.DataFrame({
            "custom_id": [1, 1, 2],
            "ts": ["2023-01-01", "2023-01-02", "2023-01-03"],
            "x": [1, 2, 3]
        })
        result = deduplicate_events(df, id_col="custom_id", timestamp_col="ts")
        expected = pd.DataFrame({
            "custom_id": [1, 2],
            "ts": ["2023-01-02", "2023-01-03"],
            "x": [2, 3]
        })
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["event_id", "updated_at", "value"])
        result = deduplicate_events(df)
        expected = df.copy()
        pd.testing.assert_frame_equal(result, expected)

    def test_single_row(self):
        df = pd.DataFrame({
            "event_id": [1],
            "updated_at": ["2023-01-01"],
            "value": [10]
        })
        result = deduplicate_events(df)
        expected = df.copy()
        pd.testing.assert_frame_equal(result.reset_index(drop=True), expected)

    def test_all_null_identifiers(self):
        df = pd.DataFrame({
            "event_id": [None, None, None],
            "updated_at": ["2023-01-01", "2023-01-02", "2023-01-03"],
            "value": [1, 2, 3]
        })
        result = deduplicate_events(df)
        # All rows preserved
        pd.testing.assert_frame_equal(result.reset_index(drop=True), df.reset_index(drop=True))
