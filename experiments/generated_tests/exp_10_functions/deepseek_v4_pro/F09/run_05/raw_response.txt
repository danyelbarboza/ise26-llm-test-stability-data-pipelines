import pytest
import pandas as pd
import numpy as np
from pandas.testing import assert_frame_equal, assert_series_equal
from ise26.targets import cap_outliers_iqr


def test_caps_outliers_correctly():
    df = pd.DataFrame({"amount": [1, 2, 3, 100]})
    result = cap_outliers_iqr(df)
    expected_capped = pd.Series([1.0, 2.0, 3.0, 65.5], name="amount_capped", dtype="Float64")
    assert "amount_capped" in result.columns
    assert_series_equal(result["amount_capped"], expected_capped)
    assert_series_equal(result["amount"], df["amount"])


def test_no_outliers_all_within_bounds():
    df = pd.DataFrame({"amount": [10, 20, 30, 40, 50]})
    result = cap_outliers_iqr(df)
    expected = pd.Series([10.0, 20.0, 30.0, 40.0, 50.0], name="amount_capped", dtype="Float64")
    assert_series_equal(result["amount_capped"], expected)


def test_missing_values_remain_na():
    df = pd.DataFrame({"amount": [5.0, np.nan, 15.0]})
    result = cap_outliers_iqr(df)
    expected = pd.Series([5.0, pd.NA, 15.0], name="amount_capped", dtype="Float64")
    assert_series_equal(result["amount_capped"], expected)


def test_non_numeric_values_become_na():
    df = pd.DataFrame({"amount": ["a", 10, "b", 20]})
    result = cap_outliers_iqr(df)
    expected = pd.Series([pd.NA, 10.0, pd.NA, 20.0], name="amount_capped", dtype="Float64")
    assert_series_equal(result["amount_capped"], expected)


def test_all_invalid_or_missing_returns_all_na():
    df = pd.DataFrame({"amount": [np.nan, np.nan, "x"]})
    result = cap_outliers_iqr(df)
    expected = pd.Series([pd.NA] * 3, name="amount_capped", dtype="Float64")
    assert_series_equal(result["amount_capped"], expected)


def test_single_value_no_capping():
    df = pd.DataFrame({"amount": [42]})
    result = cap_outliers_iqr(df)
    expected = pd.Series([42.0], name="amount_capped", dtype="Float64")
    assert_series_equal(result["amount_capped"], expected)


def test_two_distinct_values_capping():
    df = pd.DataFrame({"amount": [1, 100]})
    result = cap_outliers_iqr(df)
    # IQR=99, lower=-48.5, upper=148.5 -> both within
    assert list(result["amount_capped"]) == [1.0, 100.0]


def test_duplicates_preserved():
    df = pd.DataFrame({"amount": [1, 100, 1, 100]})
    result = cap_outliers_iqr(df)
    # same as original because all within bounds (IQR=99, LB=-48.5, UB=148.5)
    expected = [1.0, 100.0, 1.0, 100.0]
    assert list(result["amount_capped"]) == expected


def test_custom_column_names():
    df = pd.DataFrame({"val": [1, 2, 3, 200]})
    result = cap_outliers_iqr(df, value_col="val", output_col="val_capped")
    assert "val_capped" in result.columns
    assert "val" in result.columns
    # For [1,2,3,200], Q1=1.75, Q3=3.25? Actually with linear:
    # 4 values: Q1=1.75, Q3=151.25? Wait recalc for 1,2,3,200:
    # sorted: 1,2,3,200. q1=1 + 0.25*(2-1)=1.25? Let's recompute: quantile linear interpolation:
    # For 4 values, rank-based interpolation: position = (n-1)*p + 1?
    # Pandas uses linear interpolation between data points in sorted order with fractional index.
    # 25%: index = 0.25*(4-1) = 0.75 -> between 0 and 1 -> value = 1 + 0.75*(2-1) = 1.75.
    # 75%: index = 0.75*3 = 2.25 -> between index 2 (3) and 3 (200) -> value = 3 + 0.25*(200-3)=3+49.25=52.25.
    # IQR=52.25-1.75=50.5. LB=1.75-75.75=-74.0, UB=52.25+75.75=128.0. So 200 capped to 128.0.
    expected = [1.0, 2.0, 3.0, 128.0]
    actual = list(result["val_capped"])
    assert actual == pytest.approx(expected)


def test_original_dataframe_unchanged():
    df = pd.DataFrame({"amount": [10, 20, 30]})
    original = df.copy()
    _ = cap_outliers_iqr(df)
    assert_frame_equal(df, original)


def test_result_is_new_object():
    df = pd.DataFrame({"amount": [1, 2, 3]})
    result = cap_outliers_iqr(df)
    assert result is not df


def test_output_dtype_float64():
    df = pd.DataFrame({"amount": [1, 2, 3]})
    result = cap_outliers_iqr(df)
    assert result["amount_capped"].dtype.name == "Float64"


def test_existing_output_column_overwritten():
    df = pd.DataFrame({"amount": [1, 2, 3], "amount_capped": [9, 9, 9]})
    result = cap_outliers_iqr(df)
    assert result["amount_capped"].iloc[0] == 1.0
    assert result["amount_capped"].dtype.name == "Float64"


def test_extreme_values_capped_to_bounds():
    df = pd.DataFrame({"amount": [0, 10, 20, 30, 200]})
    result = cap_outliers_iqr(df)
    # For [0,10,20,30,200], Q1=10, Q3=30, IQR=20, LB=-20, UB=60. So 200 capped to 60.
    capped = result["amount_capped"]
    assert capped.min() == -20.0  # 0 not below lower bound
    assert capped.max() == 60.0


def test_all_values_equal_iqr_zero():
    df = pd.DataFrame({"amount": [5, 5, 5]})
    result = cap_outliers_iqr(df)
    expected = [5.0, 5.0, 5.0]
    assert list(result["amount_capped"]) == expected
