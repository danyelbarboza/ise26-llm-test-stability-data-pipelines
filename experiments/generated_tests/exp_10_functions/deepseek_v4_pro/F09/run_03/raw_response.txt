import pandas as pd
import numpy as np
import pytest
from ise26.targets import cap_outliers_iqr


def test_basic_capping():
    """Values beyond IQR limits are capped to the bounds."""
    df = pd.DataFrame({"amount": [1, 2, 3, 4, 100]})
    result = cap_outliers_iqr(df, value_col="amount", output_col="amount_capped")
    expected = [1, 2, 3, 4, 7]  # Q1=2, Q3=4, IQR=2 -> bounds [-1, 7]
    assert list(result["amount_capped"]) == expected
    assert list(result["amount"]) == df["amount"].tolist()


def test_no_mutation_input():
    """Original DataFrame is not modified."""
    df = pd.DataFrame({"amount": [1, 2, 3, 4, 100]})
    df_before = df.copy(deep=True)
    _ = cap_outliers_iqr(df, value_col="amount", output_col="amount_capped")
    pd.testing.assert_frame_equal(df, df_before)


def test_missing_values_remain_missing():
    """NaN or None in input remain as NA in output column."""
    df = pd.DataFrame({"amount": [1, np.nan, 3, None, 100]})
    result = cap_outliers_iqr(df, value_col="amount", output_col="capped")
    # first check that missing positions are NA
    assert pd.isna(result["capped"].iloc[1])
    assert pd.isna(result["capped"].iloc[3])
    # non‑missing are capped correctly
    assert result["capped"].iloc[0] == 1.0
    assert result["capped"].iloc[2] == 3.0
    assert result["capped"].iloc[4] == 7.0


def test_all_missing_values():
    """When no valid numeric values exist, output column is all NA."""
    df = pd.DataFrame({"amount": [np.nan, np.nan, None]})
    result = cap_outliers_iqr(df)
    assert result["amount_capped"].isna().all()
    assert result["amount_capped"].dtype == "Float64"


def test_constant_column():
    """Constant data: no capping occurs, values stay identical."""
    df = pd.DataFrame({"amount": [5, 5, 5]})
    result = cap_outliers_iqr(df)
    assert list(result["amount_capped"]) == [5.0, 5.0, 5.0]


def test_negative_values():
    """Both lower and upper bounds applied when data spans negative and positive."""
    df = pd.DataFrame({"amount": [-20, -1, 0, 1, 30]})
    result = cap_outliers_iqr(df, output_col="capped")
    # Q1=-1, Q3=1, IQR=2, lower=-4, upper=4
    expected = [-4.0, -1.0, 0.0, 1.0, 4.0]
    assert list(result["capped"]) == expected


def test_custom_column_names():
    """Works with any valid column names."""
    df = pd.DataFrame({"x": [10, 20, 200]})
    result = cap_outliers_iqr(df, value_col="x", output_col="x_out")
    # Q1=10, Q3=20, IQR=10 -> bounds [-5, 35], 200 becomes 35
    assert list(result["x_out"]) == [10, 20, 35]
    assert "x" in result.columns
    assert "x_out" in result.columns


def test_output_is_float64():
    """Capped column is always Float64 dtype."""
    df = pd.DataFrame({"amount": [10, 20, 30]})
    result = cap_outliers_iqr(df)
    assert result["amount_capped"].dtype.name == "Float64"


def test_preserves_index():
    """Non‑default index is preserved in output."""
    index = pd.Index([5, 10, 15])
    df = pd.DataFrame({"amount": [1, 100, 3]}, index=index)
    result = cap_outliers_iqr(df)
    pd.testing.assert_index_equal(result.index, index)
    # check values at positions
    assert result.loc[10, "amount_capped"] == 7  # outlier capped


def test_duplicates_handled_individually():
    """Duplicate rows each get capped independently."""
    df = pd.DataFrame({"amount": [100, 100, 5]})
    result = cap_outliers_iqr(df)
    # Q1=5, Q3=100, IQR=95 -> bounds [-137.5, 242.5], so nothing capped
    # Actually with only these three values: sorted: 5,100,100 -> Q1=5, Q3=100.
    # IQR=95, lower=5-142.5= -137.5, upper=100+142.5=242.5. No cap.
    assert list(result["amount_capped"]) == [100, 100, 5]
    # Try with obvious outlier
    df2 = pd.DataFrame({"amount": [1, 1, 1, 1, 100]})
    res2 = cap_outliers_iqr(df2)
    # Q1=1, Q3=1, IQR=0 -> bounds [1,1]. Values above 1 get capped to 1.
    assert list(res2["amount_capped"]) == [1, 1, 1, 1, 1]

    assert df["amount"].dtype == "int64"

    assert result["amount"].dtype == "int64"


def test_empty_dataframe():

    assert list(result["amount_capped"]) == []
    assert result.shape == (0, 2)  # original amount + added capped

def test_iqr_zero_with_variation():
    """When IQR=0 but data varies, all values not equal to the constant get capped."""
    # majority = 10, a few outliers
    data = [10, 10, 10, 10, 12, 8, 10]
    df = pd.DataFrame({"amount": data})
    result = cap_outliers_iqr(df)
    # Q1=10, Q3=10, IQR=0, bounds=[10,10]
    expected = [10.0, 10.0, 10.0, 10.0, 10.0, 10.0, 10.0]
    assert list(result["amount_capped"]) == expected
