import pandas as pd
import pytest
from ise26.targets import cap_outliers_iqr


class TestCapOutliersIqr:
    def test_basic_capping(self):
        df = pd.DataFrame({"amount": [10, 12, 13, 15, 100, -5]})
        result = cap_outliers_iqr(df, value_col="amount", output_col="amount_capped")
        expected_q1, expected_q3 = 11.0, 16.25
        expected_iqr = expected_q3 - expected_q1
        expected_lower = expected_q1 - 1.5 * expected_iqr
        expected_upper = expected_q3 + 1.5 * expected_iqr
        expected_capped = pd.Series(
            [10, 12, 13, 15, expected_upper, expected_lower],
            name="amount_capped",
            dtype="Float64",
        )
        pd.testing.assert_series_equal(result["amount_capped"], expected_capped, check_names=False)
        assert list(result.columns) == ["amount", "amount_capped"]
        # original column unchanged
        pd.testing.assert_series_equal(result["amount"], df["amount"], check_names=False)

    def test_no_outliers(self):
        df = pd.DataFrame({"value": [1, 2, 3, 4, 5]})
        result = cap_outliers_iqr(df, "value", "capped")
        pd.testing.assert_series_equal(result["capped"], pd.Series([1, 2, 3, 4, 5], dtype="Float64", name="capped"))

    def test_high_outliers_only(self):
        df = pd.DataFrame({"x": [1, 2, 3, 100]})
        result = cap_outliers_iqr(df, "x", "x_capped")
        # Q1=1.75, Q3=26.75, IQR=25, lower=-35.75, upper=64.25
        expected = pd.Series([1, 2, 3, 64.25], dtype="Float64", name="x_capped")
        pd.testing.assert_series_equal(result["x_capped"], expected)

    def test_low_outliers_only(self):
        df = pd.DataFrame({"x": [-100, 5, 6, 7]})
        result = cap_outliers_iqr(df, "x", "x_capped")
        # Q1=-41.75, Q3=6.25, IQR=48, lower=-113.75, upper=78.25
        expected = pd.Series([-100, 5, 6, 7], dtype="Float64", name="x_capped")  # -100 > -113.75, no cap needed
        # Actually -100 is above lower bound, so no change
        pd.testing.assert_series_equal(result["x_capped"], expected)
        # Let's test with a value lower than bound
        df2 = pd.DataFrame({"x": [-200, 5, 6, 7]})
        result2 = cap_outliers_iqr(df2, "x", "x_capped")
        # lower = -113.75, so -200 capped at -113.75
        expected2 = pd.Series([-113.75, 5, 6, 7], dtype="Float64", name="x_capped")
        pd.testing.assert_series_equal(result2["x_capped"], expected2)

    def test_missing_values_stay_missing(self):
        df = pd.DataFrame({"amount": [10, None, 15, pd.NA, 100, float("nan")]})
        result = cap_outliers_iqr(df, "amount", "amount_capped")
        # all missing entries should be pd.NA in output
        assert pd.isna(result.loc[1, "amount_capped"])
        assert pd.isna(result.loc[3, "amount_capped"])
        assert pd.isna(result.loc[5, "amount_capped"])
        # non-missing values should be capped normally
        assert result.loc[0, "amount_capped"] == 10.0
        assert result.loc[2, "amount_capped"] == 15.0
        # 100 is outlier
        q1 = df["amount"].quantile(0.25)
        q3 = df["amount"].quantile(0.75)
        iqr = q3 - q1
        assert result.loc[4, "amount_capped"] == q3 + 1.5 * iqr

    def test_invalid_strings_coerced_to_missing(self):
        df = pd.DataFrame({"amount": [10, "invalid", 15], "extra": [1, 2, 3]})
        result = cap_outliers_iqr(df, "amount", "amount_capped")
        assert pd.isna(result.loc[1, "amount_capped"])
        assert result.loc[0, "amount_capped"] == 10.0
        assert result.loc[2, "amount_capped"] == 15.0
        # Original column untouched
        assert result.loc[1, "amount"] == "invalid"

    def test_all_valid_values_empty(self):
        df = pd.DataFrame({"amount": ["a", "b", None, pd.NA]})
        result = cap_outliers_iqr(df, "amount", "capped")
        assert result["capped"].isna().all()
        assert result["capped"].dtype == "Float64"

    def test_all_identical_values(self):
        df = pd.DataFrame({"amount": [5, 5, 5, 5]})
        result = cap_outliers_iqr(df, "amount", "capped")
        expected = pd.Series([5.0, 5.0, 5.0, 5.0], dtype="Float64")
        pd.testing.assert_series_equal(result["capped"], expected, check_names=False)

    def test_input_not_mutated(self):
        original_df = pd.DataFrame({"amount": [10, 100]})
        original_copy = original_df.copy()
        _ = cap_outliers_iqr(original_df)
        pd.testing.assert_frame_equal(original_df, original_copy)

    def test_column_addition_and_order(self):
        df = pd.DataFrame({"amount": [1, 2], "other": [3, 4]})
        result = cap_outliers_iqr(df, "amount", "new_col")
        assert list(result.columns) == ["amount", "other", "new_col"]
        assert "new_col" in result.columns
        assert "amount" in result.columns

    def test_large_values(self):
        df = pd.DataFrame({"val": [1e10, 2e10, 3e10, 1e12]})
        result = cap_outliers_iqr(df, "val", "capped")
        q1 = df["val"].quantile(0.25)
        q3 = df["val"].quantile(0.75)
        iqr = q3 - q1
        upper = q3 + 1.5 * iqr
        assert result["capped"].iloc[-1] == float(upper)
        # ensure floating dtype
        assert result["capped"].dtype == "Float64"

    def test_custom_output_col_name(self):
        df = pd.DataFrame({"amount": [1, 2, 3]})
        result = cap_outliers_iqr(df, output_col="my_capped")
        assert "my_capped" in result.columns

    def test_duplicate_values(self):
        df = pd.DataFrame({"amount": [10, 10, 20, 20, 100]})
        result = cap_outliers_iqr(df)
        expected_upper = 20 + 1.5 * (20 - 10)  # Q3=20, Q1=10, IQR=10, upper=35
        assert all(result.loc[df["amount"] == 100, "amount_capped"] == expected_upper)
        # lower bound 10 - 15 = -5, none below
        assert list(result.loc[df["amount"] == 10, "amount_capped"]) == [10.0, 10.0]
