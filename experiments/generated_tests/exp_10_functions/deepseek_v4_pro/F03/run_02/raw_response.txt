import pytest
import pandas as pd
from ise26.targets import calculate_monthly_revenue


class TestCalculateMonthlyRevenue:
    def test_basic_aggregation(self):
        df = pd.DataFrame(
            {
                "order_date": ["2023-01-15", "2023-01-20", "2023-02-10"],
                "amount": [100.0, 200.0, 50.0],
                "status": ["completed", "completed", "completed"],
            }
        )
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame(
            {"month": ["2023-01", "2023-02"], "revenue": [300.0, 50.0]}
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_ignore_canceled_case_and_spaces(self):
        # Test English spellings and Portuguese, with mixed case and surrounding spaces
        df = pd.DataFrame(
            {
                "order_date": ["2023-03-01", "2023-03-02", "2023-03-03", "2023-03-04"],
                "amount": [10, 20, 30, 40],
                "status": [
                    "  Cancelled  ",
                    "CANCELED",
                    "cancelado",
                    "active",
                ],
            }
        )
        result = calculate_monthly_revenue(df)
        # Only the "active" row should count
        expected = pd.DataFrame({"month": ["2023-03"], "revenue": [40]})
        pd.testing.assert_frame_equal(result, expected, check_dtype=False)

    def test_ignore_invalid_dates(self):
        df = pd.DataFrame(
            {
                "order_date": [
                    "2023-04-10",
                    "not-a-date",
                    None,
                    "",
                    "2023-05-05",
                ],
                "amount": [1, 2, 3, 4, 5],
                "status": ["ok", "ok", "ok", "ok", "ok"],
            }
        )
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame(
            {"month": ["2023-04", "2023-05"], "revenue": [1, 5]}
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_treat_invalid_amounts_as_zero(self):
        df = pd.DataFrame(
            {
                "order_date": ["2023-06-01", "2023-06-02", "2023-06-03"],
                "amount": [10, "abc", None],
                "status": ["ok", "ok", "ok"],
            }
        )
        result = calculate_monthly_revenue(df)
        # "abc" and None become 0, so total is 10
        expected = pd.DataFrame({"month": ["2023-06"], "revenue": [10.0]})
        pd.testing.assert_frame_equal(result, expected, check_dtype=False)

    def test_missing_amounts_default_to_zero(self):
        # Check explicit NaN in amount treated as 0
        df = pd.DataFrame(
            {
                "order_date": ["2023-07-01", "2023-07-02"],
                "amount": [None, 15],
                "status": ["ok", "ok"],
            }
        )
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame({"month": ["2023-07"], "revenue": [15.0]})
        pd.testing.assert_frame_equal(result, expected, check_dtype=False)

    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["order_date", "amount", "status"])
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame(columns=["month", "revenue"])
        # Ensure the output has the expected column structure
        pd.testing.assert_frame_equal(result, expected)

    def test_all_canceled_or_invalid(self):
        df = pd.DataFrame(
            {
                "order_date": ["2023-08-01", "2023-08-02", "not-a-date"],
                "amount": [100, 200, 300],
                "status": ["canceled", "Cancelled ", "ok"],
            }
        )
        result = calculate_monthly_revenue(df)
        # All rows are filtered out: two canceled, one invalid date
        expected = pd.DataFrame(columns=["month", "revenue"])
        pd.testing.assert_frame_equal(result, expected)

    def test_empty_result_structure(self):
        # When no rows survive, result should be empty with correct columns
        df = pd.DataFrame(
            {"order_date": ["2023-09-01"], "amount": [5], "status": ["cancelado"]}
        )
        result = calculate_monthly_revenue(df)
        assert list(result.columns) == ["month", "revenue"]
        assert len(result) == 0

    def test_mixed_statuses_and_amounts(self):
        df = pd.DataFrame(
            {
                "order_date": [
                    "2023-10-01",
                    "2023-10-11",
                    "2023-10-21",
                    "2023-11-01",
                    "2023-11-11",
                ],
                "amount": [10, None, "bad", 40, 50],
                "status": [
                    "ok",
                    "cancelled",
                    "ok",
                    "cancelado",
                    "ok",
                ],
            }
        )
        result = calculate_monthly_revenue(df)
        # Oct: row1 10, row2 canceled ignored, row3 0 (bad -> 0) => total 10
        # Nov: row4 canceled ignored, row5 50 => total 50
        expected = pd.DataFrame(
            {"month": ["2023-10", "2023-11"], "revenue": [10.0, 50.0]}
        )
        pd.testing.assert_frame_equal(result, expected, check_dtype=False)

    def test_output_sorted_by_month(self):
        # Months are given out of order; result must be sorted
        df = pd.DataFrame(
            {
                "order_date": ["2023-12-01", "2023-01-01", "2023-06-01"],
                "amount": [1, 2, 3],
                "status": ["ok", "ok", "ok"],
            }
        )
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame(
            {"month": ["2023-01", "2023-06", "2023-12"], "revenue": [2, 3, 1]}
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_custom_column_names(self):
        df = pd.DataFrame(
            {
                "dt": ["2024-01-01", "2024-02-01"],
                "val": [10, 20],
                "state": ["OK", "OK"],
            }
        )
        result = calculate_monthly_revenue(df, date_col="dt", amount_col="val", status_col="state")
        expected = pd.DataFrame(
            {"month": ["2024-01", "2024-02"], "revenue": [10, 20]}
        )
        pd.testing.assert_frame_equal(result, expected)

    def test_original_dataframe_unchanged(self):
        # Ensure the function does not mutate the original DataFrame
        df = pd.DataFrame(
            {
                "order_date": ["2023-01-05"],
                "amount": [100],
                "status": [" shipped "],
            }
        )
        original = df.copy()
        calculate_monthly_revenue(df)
        pd.testing.assert_frame_equal(df, original)

    def test_missing_column_raises_keyerror(self):
        df = pd.DataFrame({"order_date": ["2023-01-01"], "amount": [1]})
        with pytest.raises(KeyError):
            calculate_monthly_revenue(df)

    def test_single_valid_row(self):
        df = pd.DataFrame(
            {
                "order_date": ["2023-05-05"],
                "amount": [42],
                "status": ["active"],
            }
        )
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame({"month": ["2023-05"], "revenue": [42]})
        pd.testing.assert_frame_equal(result, expected)

    def test_duplicate_months_summed(self):
        df = pd.DataFrame(
            {
                "order_date": ["2023-11-01", "2023-11-10", "2023-12-01"],
                "amount": [5, 15, 2],
                "status": ["ok", "ok", "ok"],
            }
        )
        result = calculate_monthly_revenue(df)
        expected = pd.DataFrame(
            {"month": ["2023-11", "2023-12"], "revenue": [20, 2]}
        )
        pd.testing.assert_frame_equal(result, expected)
