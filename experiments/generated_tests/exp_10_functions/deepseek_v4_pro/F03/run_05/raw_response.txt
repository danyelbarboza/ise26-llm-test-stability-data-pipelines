import pandas as pd
import pytest
from ise26.targets import calculate_monthly_revenue

@pytest.fixture
def sample_df():
    return pd.DataFrame({
        "order_date": ["2023-01-15", "2023-01-20", "2023-02-10", "2023-02-28"],
        "amount": [100, 200, 150, 50],
        "status": ["confirmed", "confirmed", "shipped", "delivered"]
    })

@pytest.fixture
def df_with_canceled():
    return pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-02-01"],
        "amount": [10, 20, 30],
        "status": ["cancelled", " Canceled ", "CANCELADO"]
    })

@pytest.fixture
def df_with_invalid_dates():
    return pd.DataFrame({
        "order_date": ["2023-01-01", "not-a-date", "2023-02-01", None],
        "amount": [100, 200, 300, 400],
        "status": ["ok", "ok", "ok", "ok"]
    })

@pytest.fixture
def df_with_invalid_amounts():
    return pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-02-01"],
        "amount": [50, "bad", None],
        "status": ["ok", "ok", "ok"]
    })

def test_basic_aggregation(sample_df):
    result = calculate_monthly_revenue(sample_df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [300, 200]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_canceled_statuses_are_excluded(df_with_canceled):
    result = calculate_monthly_revenue(df_with_canceled)
    # All orders are canceled, no revenue
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]

def test_canceled_normalization_with_spaces_and_case(df_with_canceled):
    result = calculate_monthly_revenue(df_with_canceled)
    assert result.empty

def test_invalid_dates_ignored(df_with_invalid_dates):
    result = calculate_monthly_revenue(df_with_invalid_dates)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [100, 300]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_missing_amounts_treated_as_zero(df_with_invalid_amounts):
    result = calculate_monthly_revenue(df_with_invalid_amounts)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [50, 0]
    })
    pd.testing.assert_frame_equal(result, expected)

def test_sort_order():
    df = pd.DataFrame({
        "order_date": ["2023-03-01", "2023-01-01", "2023-02-01"],
        "amount": [1, 2, 3],
        "status": ["ok", "ok", "ok"]
    })
    result = calculate_monthly_revenue(df)
    months = result["month"].tolist()
    assert months == ["2023-01", "2023-02", "2023-03"]

def test_empty_dataframe():
    df = pd.DataFrame({"order_date": [], "amount": [], "status": []})
    result = calculate_monthly_revenue(df)
    assert result.empty
    assert list(result.columns) == ["month", "revenue"]

def test_all_invalid_dates():
    df = pd.DataFrame({
        "order_date": ["xyz", "abc", None],
        "amount": [100, 200, 300],
        "status": ["ok", "ok", "ok"]
    })
    result = calculate_monthly_revenue(df)
    assert result.empty

def test_all_canceled_mixed_with_valid():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02", "2023-01-03"],
        "amount": [10, 20, 30],
        "status": ["cancelled", "ok", "Cancelado"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2023-01"], "revenue": [20]})
    pd.testing.assert_frame_equal(result, expected)

def test_columns_exist():
    df = pd.DataFrame({
        "order_date": ["2023-01-01"],
        "amount": [100],
        "status": ["ok"]
    })
    result = calculate_monthly_revenue(df)
    assert list(result.columns) == ["month", "revenue"]

def test_original_dataframe_not_modified():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-02-01"],
        "amount": [100, 200],
        "status": [" ok ", "CANCELED"]
    })
    original = df.copy()
    _ = calculate_monthly_revenue(df)
    pd.testing.assert_frame_equal(df, original)

def test_duplicate_rows_summed():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-01"],
        "amount": [100, 200],
        "status": ["ok", "ok"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({"month": ["2023-01"], "revenue": [300]})
    pd.testing.assert_frame_equal(result, expected)

def test_custom_column_names():
    df = pd.DataFrame({
        "dt": ["2023-01-05"],
        "value": [75],
        "state": ["confirmed"]
    })
    result = calculate_monthly_revenue(df, date_col="dt", amount_col="value", status_col="state")
    expected = pd.DataFrame({"month": ["2023-01"], "revenue": [75]})
    pd.testing.assert_frame_equal(result, expected)

def test_amount_with_currency_symbols_treated_as_invalid():
    df = pd.DataFrame({
        "order_date": ["2023-01-01"],
        "amount": ["$100"],
        "status": ["ok"]
    })
    result = calculate_monthly_revenue(df)
    # pd.to_numeric("$100", errors='coerce') is NaN -> fillna(0)
    expected = pd.DataFrame({"month": ["2023-01"], "revenue": [0.0]})
    pd.testing.assert_frame_equal(result, expected)

def test_leading_trailing_spaces_in_status_ignored():
    df = pd.DataFrame({
        "order_date": ["2023-01-01"],
        "amount": [50],
        "status": ["  cancelled  "]
    })
    result = calculate_monthly_revenue(df)
    assert result.empty

def test_mixed_valid_canceled_and_invalid_dates():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "bad_date", "2023-02-02", "2023-03-03"],
        "amount": [10, 20, 30, 40],
        "status": ["ok", "cancelled", "cancelado", "ok"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-03"],
        "revenue": [10, 40]
    })
    pd.testing.assert_frame_equal(result, expected)
