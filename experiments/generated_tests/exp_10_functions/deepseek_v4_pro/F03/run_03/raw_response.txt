from ise26.targets import calculate_monthly_revenue
import pytest
import pandas as pd
import numpy as np

def test_basic_aggregation():
    """Sum revenue per month for valid non-canceled orders."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-15", "2023-02-01", "2023-02-10"],
        "amount": [100, 200, 150, 50],
        "status": ["active", "active", "pending", "shipped"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [300, 200]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_invalid_dates_ignored():
    """Rows with unparseable dates are excluded from aggregation."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "not-a-date", "2023-02-01", "garbage"],
        "amount": [100, 50, 200, 75],
        "status": ["active", "active", "active", "active"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [100, 200]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_canceled_statuses_ignored():
    """Rows with canceled/cancelled/cancelado (case-insensitive, trimmed) are excluded."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-15", "2023-01-25", "2023-02-01",
                       "2023-02-10", "2023-02-20"],
        "amount": [100, 200, 300, 150, 250, 50],
        "status": ["cancelled", "active", "Canceled", "cancelado",
                   "CANCELED", "active"]
    })
    result = calculate_monthly_revenue(df)
    # Only non-canceled: row 1 (index 1) amount 200 in Jan, row 5 amount 50 in Feb
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [200, 50]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_mixed_case_and_spaces_status():
    """Status normalization handles leading/trailing spaces and mixed case."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-15", "2023-01-25"],
        "amount": [100, 200, 300],
        "status": ["  Cancelled ", "cancelado  ", "   CANCELED   "]
    })
    result = calculate_monthly_revenue(df)
    # All are canceled, result should be empty
    assert len(result) == 0
    assert list(result.columns) == ["month", "revenue"]


def test_invalid_amounts_treated_as_zero():
    """Non-numeric amounts become 0 and are included in revenue for valid orders."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-15", "2023-02-01"],
        "amount": [100, "abc", None],
        "status": ["active", "active", "active"]
    })
    result = calculate_monthly_revenue(df)
    # Jan: 100 + 0 = 100, Feb: 0
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [100.0, 0.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_missing_amount_keeps_row_zero_revenue():
    """NaN amount is treated as 0 and the row is retained if status valid."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-15"],
        "amount": [100, np.nan],
        "status": ["active", "active"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [100.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_all_rows_filtered_out_empty_result():
    """If every row is invalid date or canceled, result is empty DataFrame with correct columns."""
    df = pd.DataFrame({
        "order_date": ["bad", "2023-01-01", "bad"],
        "amount": [10, 20, 30],
        "status": ["active", "cancelled", "active"]
    })
    result = calculate_monthly_revenue(df)
    assert len(result) == 0
    assert list(result.columns) == ["month", "revenue"]


def test_result_sorted_by_month():
    """Output rows are sorted by month ascending as string."""
    df = pd.DataFrame({
        "order_date": ["2022-12-31", "2023-01-01", "2023-02-01",
                       "2023-01-15"],
        "amount": [1, 2, 3, 4],
        "status": ["active"] * 4
    })
    result = calculate_monthly_revenue(df)
    months = result["month"].tolist()
    assert months == ["2022-12", "2023-01", "2023-02"]


def test_duplicate_months_aggregated():
    """Multiple orders in the same month are summed."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-10", "2023-01-15"],
        "amount": [100, 200, 300],
        "status": ["active", "active", "active"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [600]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_custom_column_names():
    """Function works with non-default column names."""
    df = pd.DataFrame({
        "dt": ["2023-01-05", "2023-02-10"],
        "value": [7.5, 2.5],
        "state": ["active", "active"]
    })
    result = calculate_monthly_revenue(df, date_col="dt", amount_col="value", status_col="state")
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [7.5, 2.5]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_empty_dataframe():
    """Passing an empty DataFrame returns an empty result with correct columns."""
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    assert len(result) == 0
    assert list(result.columns) == ["month", "revenue"]


def test_status_nan_kept():
    """Null status is not treated as canceled; row is included."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05"],
        "amount": [42],
        "status": [np.nan]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [42.0]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_amount_types_float_and_int():
    """Amounts can be int or float."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "2023-01-10"],
        "amount": [100, 150.5],
        "status": ["active", "active"]
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [250.5]
    })
    pd.testing.assert_frame_equal(result, expected, check_dtype=False)


def test_original_dataframe_not_modified():
    """The input DataFrame should remain unchanged."""
    df = pd.DataFrame({
        "order_date": ["2023-01-05", "invalid", "2023-02-01"],
        "amount": [100, 200, 300],
        "status": ["active", "cancelled", "active"]
    })
    df_copy = df.copy()
    calculate_monthly_revenue(df)
    pd.testing.assert_frame_equal(df, df_copy)
