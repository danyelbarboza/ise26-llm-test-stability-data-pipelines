from ise26.targets import calculate_monthly_revenue
import pandas as pd
import pytest

def test_basic_revenue():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-15", "2023-02-01", "2023-02-20"],
        "amount": [100, 200, 50, 300],
        "status": ["completed", "shipped", "cancelled", "processing"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-02"],
        "revenue": [300, 300],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_invalid_dates_ignored():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "bad_date", "2023-01-15", "not_a_date"],
        "amount": [100, 999, 200, 888],
        "status": ["completed"] * 4,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [300],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_missing_amounts_treated_as_zero():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02"],
        "amount": [None, float("nan")],
        "status": ["completed", "completed"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [0.0],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_invalid_amount_strings_become_zero():
    df = pd.DataFrame({
        "order_date": ["2023-01-01", "2023-01-02"],
        "amount": ["abc", "def"],
        "status": ["completed"] * 2,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01"],
        "revenue": [0.0],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_cancel_status_variations():
    # English: cancelled, canceled; Portuguese: cancelado; with spaces and mixed case
    df = pd.DataFrame({
        "order_date": ["2023-03-01"] * 5,
        "amount": [100] * 5,
        "status": [
            "cancelled",
            "  CANCELED ",
            "Cancelado",
            "  cancelado",
            "completed",
        ],
    })
    result = calculate_monthly_revenue(df)
    # Only the "completed" row should remain
    expected = pd.DataFrame({
        "month": ["2023-03"],
        "revenue": [100],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_all_canceled_returns_empty():
    df = pd.DataFrame({
        "order_date": ["2023-04-01", "2023-04-15"],
        "amount": [50, 70],
        "status": ["cancelled", "cancelado"],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(columns=["month", "revenue"]).astype({"revenue": "float64"})
    pd.testing.assert_frame_equal(result, expected)

def test_all_invalid_dates_returns_empty():
    df = pd.DataFrame({
        "order_date": ["xyz", "abc"],
        "amount": [50, 70],
        "status": ["completed"] * 2,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(columns=["month", "revenue"]).astype({"revenue": "float64"})
    pd.testing.assert_frame_equal(result, expected)

def test_empty_dataframe():
    df = pd.DataFrame(columns=["order_date", "amount", "status"])
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame(columns=["month", "revenue"]).astype({"revenue": "float64"})
    pd.testing.assert_frame_equal(result, expected)

def test_sorting_by_month():
    # months out of order, cross year boundary
    df = pd.DataFrame({
        "order_date": ["2024-01-05", "2023-12-20", "2023-01-10", "2023-12-31"],
        "amount": [1, 2, 3, 4],
        "status": ["completed"] * 4,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-01", "2023-12", "2024-01"],
        "revenue": [3, 6, 1],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_custom_column_names():
    df = pd.DataFrame({
        "dt": ["2023-05-01", "2023-05-15"],
        "val": [10, 20],
        "state": ["completed", "canceled"],
    })
    result = calculate_monthly_revenue(df, date_col="dt", amount_col="val", status_col="state")
    expected = pd.DataFrame({
        "month": ["2023-05"],
        "revenue": [10],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_negative_amounts_included():
    # negative amounts should be summed, not ignored
    df = pd.DataFrame({
        "order_date": ["2023-06-01", "2023-06-02"],
        "amount": [100, -20],
        "status": ["completed"] * 2,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-06"],
        "revenue": [80],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_whitespace_only_status_not_cancel():
    # whitespace-only status is normalized to empty string, which is not in canceled set
    df = pd.DataFrame({
        "order_date": ["2023-07-01"],
        "amount": [50],
        "status": ["   "],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-07"],
        "revenue": [50],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_month_format_is_yyyy_mm():
    # even if date carries time or day, month extraction must be YYYY-MM
    df = pd.DataFrame({
        "order_date": ["2023-08-15 12:34:56", "2023-08-31"],
        "amount": [1, 2],
        "status": ["completed"] * 2,
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-08"],
        "revenue": [3],
    })
    pd.testing.assert_frame_equal(result, expected)

def test_status_mixed_normalization():
    # adjacent non-cancelled status with extra spaces and uppercase
    df = pd.DataFrame({
        "order_date": ["2023-09-01", "2023-09-02", "2023-09-03"],
        "amount": [10, 20, 30],
        "status": ["  COMPLETED", "completed ", "  cancelled "],
    })
    result = calculate_monthly_revenue(df)
    expected = pd.DataFrame({
        "month": ["2023-09"],
        "revenue": [30],
    })
    pd.testing.assert_frame_equal(result, expected)
