from ise26.targets import validate_schema

import pytest
import pandas as pd


def test_valid_schema_all_correct():
    df = pd.DataFrame(
        {
            "col_int": [1, 2],
            "col_float": [1.0, 2.0],
            "col_number": [3, 4.5],
            "col_string": ["a", "b"],
            "col_datetime": pd.to_datetime(["2021-01-01", "2021-06-06"]),
            "col_bool": [True, False],
        }
    )
    schema = {
        "col_int": "int",
        "col_float": "float",
        "col_number": "number",
        "col_string": "string",
        "col_datetime": "datetime",
        "col_bool": "bool",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_columns():
    df = pd.DataFrame({"col_a": [1, 2]})
    schema = {"col_a": "int", "col_b": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["col_b"]
    assert result["type_errors"] == []


def test_type_mismatch():
    df = pd.DataFrame({"col_a": ["a", "b"]})
    schema = {"col_a": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "col_a"
    assert err["expected_type"] == "int"
    assert err["actual_type"] != "int"  # Should be something like string/object
    assert result["missing_columns"] == []


def test_extra_columns_allowed():
    df = pd.DataFrame({"col_a": [1, 2], "col_b": [3, 4]})
    schema = {"col_a": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_supported_type_labels_normalized():
    df = pd.DataFrame(
        {
            "c1": [1, 2],
            "c2": [1.0, 2.0],
            "c3": [3, 4.5],
            "c4": ["a", "b"],
            "c5": pd.to_datetime(["2021-01-01", "2021-06-06"]),
            "c6": [True, False],
        }
    )
    schema = {
        "c1": "  Int  ",
        "c2": "FLOAT",
        "c3": "Number",
        "c4": "STRING",
        "c5": "Datetime",
        "c6": "Bool",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True


def test_unsupported_expected_type_raises_valueerror():
    df = pd.DataFrame({"col_a": [1]})
    schema = {"col_a": "unknown"}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_empty_schema():
    df = pd.DataFrame({"col_a": [1, 2]})
    schema = {}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_empty_df_no_columns_schema_has_columns():
    df = pd.DataFrame()
    schema = {"col_a": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert result["missing_columns"] == ["col_a"]
    assert result["type_errors"] == []


def test_empty_df_with_matching_columns_still_valid():
    # Empty DataFrame but columns exist with correct dtypes
    df = pd.DataFrame(columns=["col_a", "col_b"])
    df["col_a"] = df["col_a"].astype("int64")
    df["col_b"] = df["col_b"].astype("object")
    schema = {"col_a": "int", "col_b": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is True


def test_mixed_missing_and_type_error():
    df = pd.DataFrame({"col_a": ["a"], "col_b": [1]})
    schema = {"col_a": "int", "col_b": "float", "col_c": "string"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert "col_c" in result["missing_columns"]
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "col_a"
    assert err["expected_type"] == "int"
