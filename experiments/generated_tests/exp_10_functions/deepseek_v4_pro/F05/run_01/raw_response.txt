from ise26.targets import validate_schema

import pandas as pd
import pytest


def test_valid_schema_all_types():
    df = pd.DataFrame(
        {
            "int_col": pd.Series([1, 2, 3], dtype="int64"),
            "float_col": pd.Series([1.0, 2.0, 3.0], dtype="float64"),
            "number_col": pd.Series([1.5, 2.5, 3.5], dtype="float64"),
            "str_col": pd.Series(["a", "b", "c"], dtype="object"),
            "datetime_col": pd.to_datetime(["2020-01-01", "2020-01-02", "2020-01-03"]),
            "bool_col": pd.Series([True, False, True], dtype="bool"),
        }
    )
    schema = {
        "int_col": "int",
        "float_col": "float",
        "number_col": "float",
        "str_col": "string",
        "datetime_col": "datetime",
        "bool_col": "bool",
    }
    result = validate_schema(df, schema)
    assert result["valid"] is True
    assert result["missing_columns"] == []
    assert result["type_errors"] == []


def test_missing_column():
    df = pd.DataFrame({"a": [1, 2], "b": ["x", "y"]})

def test_type_mismatch_int_vs_float():
    df = pd.DataFrame({"a": [1.0, 2.0]})
    schema = {"a": "int"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "a"
    assert err["expected_type"] == "int"
    assert err["actual_type"] == "float"

def test_normalization_of_type_label():
    df = pd.DataFrame({"col": [1, 2]})
    schema = {"col": "  InT "}
    result = validate_schema(df, schema)
    assert result["valid"] is True
    schema2 = {"col": "FLOAT"}
    result2 = validate_schema(df, schema2)
    assert result2["valid"] is False  # int != float
    # with correct float data
    df_float = pd.DataFrame({"col": [1.0, 2.0]})
    result3 = validate_schema(df_float, {"col": "  float "})
    assert result3["valid"] is True


def test_unsupported_type_raises_value_error():
    df = pd.DataFrame({"col": [1, 2]})
    schema = {"col": "unknown"}
    with pytest.raises(ValueError):
        validate_schema(df, schema)


def test_unsupported_type_after_normalization_still_raises():
    # normalization may strip and lower, but invalid type remains invalid.
    df = pd.DataFrame({"col": [1, 2]})
    schema = {"col": "  unknown "}
    with pytest.raises(ValueError):
        validate_schema(df, schema)

def test_multiple_errors_missing_and_type():
    df = pd.DataFrame({"x": [1.0], "y": ["text"]})
    schema = {"x": "int", "y": "string", "z": "datetime"}
    result = validate_schema(df, schema)
    assert result["valid"] is False
    assert set(result["missing_columns"]) == {"z"}
    assert len(result["type_errors"]) == 1
    err = result["type_errors"][0]
    assert err["column"] == "x"
    assert err["expected_type"] == "int"
    assert err["actual_type"] == "float"
