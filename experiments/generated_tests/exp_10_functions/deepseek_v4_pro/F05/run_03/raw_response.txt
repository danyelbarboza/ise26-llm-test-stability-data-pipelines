import pytest
import pandas as pd
from ise26.targets import validate_schema


def test_all_columns_match_types():
    df = pd.DataFrame({
        'int_col': [1, 2],
        'float_col': [1.0, 2.5],
        'string_col': ['a', 'b'],
        'datetime_col': pd.to_datetime(['2021-01-01', '2021-06-15']),
        'bool_col': [True, False],
        'number_int': [10, 20],
        'number_float': [1.5, 2.5],
    })
    schema = {
        'int_col': 'int',
        'float_col': 'float',
        'string_col': 'string',
        'datetime_col': 'datetime',
        'bool_col': 'bool',
        'number_int': 'number',
        'number_float': 'number',
    }
    result = validate_schema(df, schema)
    assert result['valid'] is True
    assert result['missing_columns'] == []
    assert result['type_errors'] == []

def test_type_mismatch_int_expected_float_actual():
    df = pd.DataFrame({'a': [1.0, 2.5]})  # float column
    result = validate_schema(df, {'a': 'int'})
    assert result['valid'] is False
    assert len(result['type_errors']) == 1
    err = result['type_errors'][0]
    assert err['column'] == 'a'
    assert err['expected_type'] == 'int'
    assert err['actual_type'] == 'float'

def test_missing_and_mismatch_together():
    df = pd.DataFrame({'a': [1, 2], 'b': [3.0, 4.5]})  # missing c
    schema = {'a': 'int', 'b': 'int', 'c': 'string'}
    result = validate_schema(df, schema)
    assert result['valid'] is False
    assert result['missing_columns'] == ['c']
    assert len(result['type_errors']) == 1
    err = result['type_errors'][0]
    assert err['column'] == 'b'
    assert err['expected_type'] == 'int'
    assert err['actual_type'] == 'float'

def test_nulls_do_not_affect_type_inference():
    df = pd.DataFrame({'a': [1, None, 2]})  # int with NaN, dtype float64?
    # If int with None, pandas converts to float. But int column with None becomes float64
    # So we need to construct an int column that stays int despite None? Use pd.Series with pd.Int64Dtype()
    # To be safe, we'll use nullable integer type.
    df = pd.DataFrame({'a': pd.array([1, None, 3], dtype='Int64')})
    result = validate_schema(df, {'a': 'int'})
    assert result['valid'] is True
    result2 = validate_schema(df, {'a': 'float'})
    assert result2['valid'] is False  # it's int, not float
