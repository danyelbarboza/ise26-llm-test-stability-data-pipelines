from ise26.targets import standardize_currency_values

import pandas as pd
import numpy as np
import pytest


def test_english_format_thousands_comma_decimal_dot():
    df = pd.DataFrame({"raw": ["1,234.56", "10,000.00", "0.99"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = pd.array([1234.56, 10000.0, 0.99], dtype="Float64")
    pd.testing.assert_extension_array_equal(
        result["amount"].array, expected, check_dtype=True
    )
    # original column unchanged
    assert "raw" in result.columns
    assert list(result["raw"]) == ["1,234.56", "10,000.00", "0.99"]

    df = pd.DataFrame({"raw": [None, np.nan, "1,234.56", pd.NA]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = pd.array([pd.NA, pd.NA, 1234.56, pd.NA], dtype="Float64")

def test_preserves_original_dataframe():
    df_original = pd.DataFrame({"raw": ["1,000.00", "2.500,75"]})
    result = standardize_currency_values(df_original, value_col="raw", output_col="amount")

    # original DataFrame must not be modified
    assert df_original.columns.tolist() == ["raw"]
    assert "amount" not in df_original.columns
    assert df_original["raw"].tolist() == ["1,000.00", "2.500,75"]

    # result is a separate object
    assert id(result) != id(df_original)


def test_output_column_is_nullable_float():
    df = pd.DataFrame({"raw": ["1,234.56", "invalid", "0"]})
    result = standardize_currency_values(df)
    assert result["amount"].dtype == "Float64"


def test_custom_column_names():
    df = pd.DataFrame({"valor_texto": ["1.234,56", "999,99"]})
    result = standardize_currency_values(df, value_col="valor_texto", output_col="valor_numerico")
    assert "valor_numerico" in result.columns
    assert "valor_texto" in result.columns
    expected = pd.array([1234.56, 999.99], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["valor_numerico"].array, expected)


def test_empty_dataframe():
    df = pd.DataFrame({"raw": [], "extra": []})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.empty
    assert list(result.columns) == ["raw", "extra", "amount"]
    assert result["amount"].dtype == "Float64"


def test_with_duplicates_in_values():

    pd.testing.assert_extension_array_equal(result["amount"].array, expected)

def test_whitespace_and_surrounding_text():
    # Assume that symbols and spaces are stripped by the internal parser
    df = pd.DataFrame({"raw": ["  $ 1,234.56 ", "R$ 1.234,56"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = pd.array([1234.56, 1234.56], dtype="Float64")
    pd.testing.assert_extension_array_equal(result["amount"].array, expected)


def test_float_output_precision():
    # Check that float64 precision is maintained (e.g., many decimals)
    df = pd.DataFrame({"raw": ["0.123456789"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result["amount"].iloc[0] == 0.123456789
