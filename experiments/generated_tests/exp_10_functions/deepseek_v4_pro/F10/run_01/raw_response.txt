import pytest
import pandas as pd
from ise26.targets import standardize_currency_values


def test_english_format_with_thousands():
    df = pd.DataFrame({"raw": ["1,234.56", "12,345.67", "0.99"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = pd.Series([1234.56, 12345.67, 0.99], name="amount", dtype="Float64")
    # Compare with a small tolerance because Float64 might not store exact floats
    pd.testing.assert_series_equal(result["amount"], expected, check_names=False)

def test_negative_values():
    df = pd.DataFrame({"raw": ["-1,234.56", "-12.345,67", "-0.99", "(100.00)"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    # Expected behavior for negative formats: assume function handles minus sign.
    # Parentheses may not be supported – they likely become NA. We'll test that they are NA.
    expected = pd.Series([-1234.56, -12345.67, -0.99, pd.NA], name="amount", dtype="Float64")
    pd.testing.assert_series_equal(result["amount"], expected, check_names=False)


def test_invalid_strings_become_na():
    df = pd.DataFrame({"raw": ["abc", "12.34.56", "1,234,56", "12,34.56", "no number"]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result["amount"].isna().all()
    # Ensure all are pd.NA (Float64)
    assert result["amount"].dtype == "Float64"

def test_empty_strings():

def test_mixed_valid_and_invalid():
    df = pd.DataFrame({"raw": ["1.000,00", "invalid", "3,000.50", None]})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = pd.Series([1000.0, pd.NA, 3000.5, pd.NA], name="amount", dtype="Float64")
    pd.testing.assert_series_equal(result["amount"], expected, check_names=False)


def test_original_raw_column_unchanged():
    original_raw = ["1.234,56", "abc", "1,234.56"]
    df = pd.DataFrame({"raw": original_raw})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    # raw column in result is a copy, but original df's "raw" is untouched
    assert list(result["raw"]) == original_raw
    assert "amount" not in df.columns
    # Original df unchanged
    assert list(df["raw"]) == original_raw


def test_output_dtype_is_float64():

    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result["amount"].dtype == "Float64"


def test_custom_column_names():
    df = pd.DataFrame({"valor": ["1.000,00", "2,000.00"]})
    result = standardize_currency_values(df, value_col="valor", output_col="valor_clean")
    assert "valor_clean" in result.columns
    expected = pd.Series([1000.0, 2000.0], name="valor_clean", dtype="Float64")
    pd.testing.assert_series_equal(result["valor_clean"], expected, check_names=False)


def test_duplicate_values():

    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    expected = pd.Series([1234.56, 1234.56, pd.NA], name="amount", dtype="Float64")
    pd.testing.assert_series_equal(result["amount"], expected, check_names=False)


def test_empty_dataframe():
    df = pd.DataFrame({"raw": []})
    result = standardize_currency_values(df, value_col="raw", output_col="amount")
    assert result.empty
    assert "amount" in result.columns
    assert result["amount"].dtype == "Float64"
