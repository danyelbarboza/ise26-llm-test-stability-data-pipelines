import pandas as pd
import numpy as np
import pytest
from ise26.targets import standardize_currency_values


def test_brazilian_format():
    df = pd.DataFrame({"amount_raw": ["R$ 1.234,56", "R$ 9.876,54", "R$ 0,50"]})
    result = standardize_currency_values(df)
    
    assert "amount_raw" in result.columns
    assert "amount" in result.columns
    assert result["amount"].dtype.name == "Float64"
    
    expected = [1234.56, 9876.54, 0.50]
    for val, exp in zip(result["amount"], expected):
        assert val == exp


def test_english_format():
    df = pd.DataFrame({"amount_raw": ["$1,234.56", "9,876.54", "0.50"]})
    result = standardize_currency_values(df)
    expected = [1234.56, 9876.54, 0.50]
    for val, exp in zip(result["amount"], expected):
        assert val == exp


def test_mixed_separators_default_parsing():
    # Ambiguous: assume both Brazilian and English work
    df = pd.DataFrame({"amount_raw": ["1.234,56", "1,234.56"]})
    result = standardize_currency_values(df)
    # Both should be interpreted as 1234.56
    expected = [1234.56, 1234.56]
    for val, exp in zip(result["amount"], expected):
        assert abs(val - exp) < 0.001


def test_negative_values():
    df = pd.DataFrame({"amount_raw": ["-R$ 1.234,56", "-1,234.56", "-0,99"]})
    result = standardize_currency_values(df)
    expected = [-1234.56, -1234.56, -0.99]
    for val, exp in zip(result["amount"], expected):
        assert abs(val - exp) < 0.001


def test_already_numeric_string():
    df = pd.DataFrame({"amount_raw": ["1234.56", "1234,56", "0", "1000"]})
    result = standardize_currency_values(df)
    expected = [1234.56, 1234.56, 0.0, 1000.0]
    for val, exp in zip(result["amount"], expected):
        assert val == exp


def test_invalid_text_returns_na():
    df = pd.DataFrame({"amount_raw": ["invalid", "NA", "abc", "---"]})
    result = standardize_currency_values(df)
    for val in result["amount"]:
        assert pd.isna(val)


def test_empty_string_becomes_na():
    df = pd.DataFrame({"amount_raw": ["", "   ", "\t"]})
    result = standardize_currency_values(df)
    for val in result["amount"]:
        assert pd.isna(val)


def test_none_and_nan_input():
    df = pd.DataFrame({"amount_raw": [None, np.nan, pd.NA, "R$ 10,00"]})
    result = standardize_currency_values(df)
    assert pd.isna(result["amount"].iloc[0])
    assert pd.isna(result["amount"].iloc[1])
    assert pd.isna(result["amount"].iloc[2])
    assert result["amount"].iloc[3] == 10.0


def test_original_column_unchanged():
    original = ["R$ 1,23", "invalid", "$4.56"]
    df = pd.DataFrame({"amount_raw": original.copy()})
    result = standardize_currency_values(df)
    assert list(result["amount_raw"]) == original
    # Ensure raw column dtype is unchanged (object)
    assert result["amount_raw"].dtype == object


def test_custom_column_names():
    df = pd.DataFrame({"raw": ["R$ 1.000,00", "500,33"]})
    result = standardize_currency_values(df, value_col="raw", output_col="clean")
    
    assert "raw" in result.columns
    assert "clean" in result.columns
    assert "amount" not in result.columns
    assert result["clean"].iloc[0] == 1000.00
    assert result["clean"].iloc[1] == 500.33


def test_output_column_overwrites_existing():
    df = pd.DataFrame({"amount_raw": ["10,99"], "amount": ["old"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 10.99


def test_empty_dataframe():
    df = pd.DataFrame({"amount_raw": []}, dtype=str)
    result = standardize_currency_values(df)
    assert len(result) == 0
    assert "amount" in result.columns
    assert result["amount"].dtype.name == "Float64"


def test_preserves_index():
    df = pd.DataFrame({"amount_raw": ["1,23", "45.67"]}, index=[9, 42])
    result = standardize_currency_values(df)
    assert list(result.index) == [9, 42]
    assert result.loc[9, "amount"] == 1.23
    assert result.loc[42, "amount"] == 45.67


def test_duplicate_index():
    df = pd.DataFrame({"amount_raw": ["10,00", "20.00"]}, index=[1, 1])
    result = standardize_currency_values(df)
    assert len(result) == 2
    assert result.iloc[0]["amount"] == 10.0
    assert result.iloc[1]["amount"] == 20.0


def test_currency_symbols_stripped():
    symbols = ["¥1,000", "€ 2.500,50", "£3,000.99", "R$ 500"]
    df = pd.DataFrame({"amount_raw": symbols})
    result = standardize_currency_values(df)
    expected = [1000.0, 2500.50, 3000.99, 500.0]
    for val, exp in zip(result["amount"], expected):
        assert val == exp


def test_thousands_separator_handling():
    df = pd.DataFrame({"amount_raw": ["1.000.000,99", "1,000,000.99"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 1000000.99
    assert result["amount"].iloc[1] == 1000000.99


def test_only_comma_decimal():
    df = pd.DataFrame({"amount_raw": ["1234,56"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 1234.56


def test_only_dot_decimal():
    df = pd.DataFrame({"amount_raw": ["1234.56"]})
    result = standardize_currency_values(df)
    assert result["amount"].iloc[0] == 1234.56
