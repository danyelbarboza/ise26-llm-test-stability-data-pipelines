import pandas as pd
import numpy as np
import pytest
from ise26.targets import standardize_currency_values


# ---------------------------------------------------------------------------
# Helpers for assertions
# ---------------------------------------------------------------------------
def assert_output_column(df_out, output_col, expected_values):
    """Check that the output column matches a list of expected values,
    correctly handling pd.NA."""
    for i, exp in enumerate(expected_values):
        # pd.isna works for both np.nan and pd.NA
        if pd.isna(exp):
            assert pd.isna(df_out.loc[i, output_col]), (
                f"Row {i}: expected NA but got {df_out.loc[i, output_col]}"
            )
        else:
            assert df_out.loc[i, output_col] == exp, (
                f"Row {i}: expected {exp} but got {df_out.loc[i, output_col]}"
            )


def assert_original_unchanged(df_before, df_after, value_col):
    """Check that the original value column is unchanged."""
    pd.testing.assert_series_equal(
        df_before[value_col],
        df_after[value_col],
        check_names=False,
        check_dtype=False,
    )


# ---------------------------------------------------------------------------
# Tests
# ---------------------------------------------------------------------------
class TestStandardizeCurrencyValues:
    def test_default_column_names(self):
        df = pd.DataFrame({"amount_raw": ["1000", "2000,50", "invalid_currency"]})
        result = standardize_currency_values(df)

        assert "amount_raw" in result.columns
        assert "amount" in result.columns
        assert result["amount"].dtype == "Float64"

        expected = [1000.0, 2000.50, pd.NA]
        assert_output_column(result, "amount", expected)
        assert_original_unchanged(df, result, "amount_raw")

    def test_custom_column_names(self):
        df = pd.DataFrame({"raw": ["1.000,00", "abc"]})
        result = standardize_currency_values(df, value_col="raw", output_col="final")

        assert "raw" in result.columns
        assert "final" in result.columns
        assert result["final"].dtype == "Float64"

        expected = [1000.0, pd.NA]
        assert_output_column(result, "final", expected)
        assert_original_unchanged(df, result, "raw")

    def test_original_dataframe_immutable(self):
        df = pd.DataFrame({"amount_raw": ["100", "200"]})
        original = df.copy()
        _ = standardize_currency_values(df)
        pd.testing.assert_frame_equal(df, original)

    def test_brazilian_format(self):
        df = pd.DataFrame({"amount_raw": [
            "1.234,56",
            "1.000.000,00",
            "0,99",
            "-1.234,56",
        ]})
        result = standardize_currency_values(df)

        expected = [1234.56, 1000000.0, 0.99, -1234.56]
        assert_output_column(result, "amount", expected)

    def test_english_format(self):
        df = pd.DataFrame({"amount_raw": [
            "1,234.56",
            "1,000,000.00",
            "0.99",
            "-1,234.56",
        ]})
        result = standardize_currency_values(df)

        expected = [1234.56, 1000000.0, 0.99, -1234.56]
        assert_output_column(result, "amount", expected)

    def test_plain_number_string(self):
        df = pd.DataFrame({"amount_raw": ["1000", "0", "-25"]})
        result = standardize_currency_values(df)
        expected = [1000.0, 0.0, -25.0]
        assert_output_column(result, "amount", expected)

    def test_leading_trailing_spaces_are_stripped(self):
        df = pd.DataFrame({"amount_raw": [
            "  1000,00  ",
            "\t1.234,56\n",
            "   100",
        ]})
        result = standardize_currency_values(df)
        expected = [1000.0, 1234.56, 100.0]
        assert_output_column(result, "amount", expected)

    def test_invalid_strings_return_na(self):
        df = pd.DataFrame({"amount_raw": [
            "não é dinheiro",
            "",
            "12.34.567",
            "abc",
        ]})
        result = standardize_currency_values(df)
        expected = [pd.NA, pd.NA, pd.NA, pd.NA]
        assert_output_column(result, "amount", expected)

    def test_missing_native_nan_input(self):
        df = pd.DataFrame({"amount_raw": [np.nan, "100,00", None]})
        result = standardize_currency_values(df)
        expected = [pd.NA, 100.0, pd.NA]
        assert_output_column(result, "amount", expected)

    def test_empty_dataframe(self):
        df = pd.DataFrame({"amount_raw": []}, dtype=str)
        result = standardize_currency_values(df)
        assert result.empty
        assert "amount" in result.columns
        assert result["amount"].dtype == "Float64"

    def test_duplicate_rows(self):
        df = pd.DataFrame({"amount_raw": ["1.000,00", "1.000,00", "2,000.50"]})
        result = standardize_currency_values(df)
        expected = [1000.0, 1000.0, 2000.50]
        assert_output_column(result, "amount", expected)

    def test_missing_value_column_raises_error(self):
        df = pd.DataFrame({"other": [1, 2, 3]})
        with pytest.raises(KeyError):
            standardize_currency_values(df)

    def test_output_dtype_is_nullable_float(self):
        df = pd.DataFrame({"amount_raw": ["100", "bad"]})
        result = standardize_currency_values(df)
        assert result["amount"].dtype == "Float64"

    def test_no_side_effects_on_original_df_columns(self):
        df = pd.DataFrame({"amount_raw": ["1000,00"], "other": [42]})
        original_cols = set(df.columns)
        _ = standardize_currency_values(df)
        assert set(df.columns) == original_cols

    def test_works_with_single_row(self):
        df = pd.DataFrame({"amount_raw": ["1.234,56"]})
        result = standardize_currency_values(df)
        assert_output_column(result, "amount", [1234.56])
