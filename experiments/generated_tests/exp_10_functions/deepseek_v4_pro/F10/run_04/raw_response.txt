import numpy as np
import pandas as pd
import pytest
from ise26.targets import standardize_currency_values


def test_original_dataframe_unchanged():
    df = pd.DataFrame({"amount_raw": ["1,000.00", "invalid"]})
    df_original = df.copy()
    _ = standardize_currency_values(df)
    pd.testing.assert_frame_equal(df, df_original)


def test_raw_column_unchanged():
    df = pd.DataFrame({"amount_raw": ["1,000.00", "2.000,50"]})
    result = standardize_currency_values(df)
    assert list(result["amount_raw"]) == list(df["amount_raw"])


def test_output_column_exists_and_dtype():
    df = pd.DataFrame({"amount_raw": ["10.00"]})
    result = standardize_currency_values(df)
    assert "amount" in result.columns
    assert result["amount"].dtype == "Float64"


def test_result_index_preserved():
    df = pd.DataFrame({"amount_raw": ["1.000,00", "2.000,00"]}, index=[10, 20])
    result = standardize_currency_values(df)
    assert result.index.tolist() == [10, 20]


def test_new_dataframe_object():
    df = pd.DataFrame({"amount_raw": ["10.00"]})
    result = standardize_currency_values(df)
    assert result is not df

    df = pd.DataFrame({"amount_raw": ["1.234,56"]})

def test_with_brazilian_currency_symbol_space():
    df = pd.DataFrame({"amount_raw": ["R$ 1.234,56"]})
    result = standardize_currency_values(df)
    # parser may or may not strip symbols; if not, it becomes NA.
    # This test documents actual behavior – adjust if expected.
    val = result.loc[0, "amount"]
    # We'll check that it's either the numeric value or NA.
    assert (val == pytest.approx(1234.56)) or pd.isna(val)

def test_empty_string_returns_na():
    df = pd.DataFrame({"amount_raw": [""]})
    result = standardize_currency_values(df)

def test_none_value_returns_na():

def test_mixed_valid_and_invalid():
    df = pd.DataFrame({"amount_raw": ["1.000,00", "??", "2,000.50", ""]})
    result = standardize_currency_values(df)
    expected = [1000.0, pd.NA, 2000.5, pd.NA]
    for i, exp in enumerate(expected):
        val = result.loc[i, "amount"]
        if pd.isna(exp):
            assert pd.isna(val)
        else:
            assert val == pytest.approx(exp)


def test_custom_value_column():
    df = pd.DataFrame({"raw": ["1.000,00"]})
    result = standardize_currency_values(df, value_col="raw")
    assert result.loc[0, "amount"] == pytest.approx(1000.0)


def test_custom_output_column():
    df = pd.DataFrame({"amount_raw": ["1.000,00"]})
    result = standardize_currency_values(df, output_col="clean")
    assert "clean" in result.columns
    assert result.loc[0, "clean"] == pytest.approx(1000.0)


def test_overwrite_existing_output_column():
    df = pd.DataFrame({"amount_raw": ["1.000,00"], "amount": ["old"]})
    result = standardize_currency_values(df, output_col="amount")
    assert result.loc[0, "amount"] == pytest.approx(1000.0)
