import pytest
import pandas as pd
from datetime import datetime
from ise26.targets import classify_payment_status


class TestClassifyPaymentStatus:
    """Test suite for classify_payment_status function."""

    # ------------------------------------------------------------------
    # Invalid reference_date
    # ------------------------------------------------------------------
    def test_invalid_reference_date_raises_valueerror(self):
        """An invalid reference_date must raise ValueError."""
        df = pd.DataFrame({
            "due_date": ["2023-01-01"],
            "paid_date": [pd.NaT],
            "amount": [100.0],
        })
        with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
            classify_payment_status(df, reference_date="not-a-date")

    # Better: one row with bad due_date.
    def test_invalid_due_date_alone_makes_invalid(self):
        df = pd.DataFrame({"due_date": ["bad"], "paid_date": [pd.NaT], "amount": [100]})
        ref = "2023-01-01"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "invalid"

    # ------------------------------------------------------------------
    # Invalid rows: amount
    # ------------------------------------------------------------------
    @pytest.mark.parametrize("bad_amount", [None, float("nan"), "", "abc", -5, 0])
    def test_invalid_amount_marks_invalid(self, bad_amount):
        """Amounts that are NaN, non-numeric, or <= 0 are invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": [pd.NaT],
            "amount": [bad_amount],
        })
        ref = "2023-01-01"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "invalid"

    # ------------------------------------------------------------------
    # Invalid rows: provided but unparseable paid_date
    # ------------------------------------------------------------------
    def test_provided_invalid_paid_date_marks_invalid(self):
        """If a paid_date is given but cannot be parsed, it's invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-03-01"],
            "paid_date": ["not-a-date"],
            "amount": [75.0],
        })
        ref = "2023-02-01"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "invalid"

    def test_empty_string_paid_date_is_missing_not_invalid(self):
        """Empty/whitespace paid_date is treated as missing, not invalid."""
        df = pd.DataFrame({
            "due_date": ["2023-03-01"],
            "paid_date": ["   "],  # whitespace only
            "amount": [75.0],
        })
        ref = "2023-02-01"
        result = classify_payment_status(df, ref)
        # due_date (2023-03-01) > ref (2023-02-01) -> overdue
        assert result["payment_status"].iloc[0] == "overdue"

    def test_none_paid_date_is_missing_not_invalid(self):
        """None in paid_date is treated as missing."""
        df = pd.DataFrame({
            "due_date": ["2023-03-01"],
            "paid_date": [None],
            "amount": [75.0],
        })
        ref = "2023-02-01"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "overdue"

    # ------------------------------------------------------------------
    # Valid payments: paid_on_time and paid_late
    # ------------------------------------------------------------------
    def test_paid_on_time(self):
        """Payment before due date classifies as paid_on_time."""
        df = pd.DataFrame({
            "due_date": ["2023-05-15"],
            "paid_date": ["2023-05-10"],
            "amount": [200.0],
        })
        ref = "2023-05-01"  # any ref doesn't matter if payment exists
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_paid_on_exact_due_date(self):
        """Payment on the exact due date is paid_on_time."""
        df = pd.DataFrame({
            "due_date": ["2023-05-15"],
            "paid_date": ["2023-05-15"],
            "amount": [200.0],
        })
        ref = "2023-05-01"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "paid_on_time"

    def test_paid_late(self):
        """Payment after due date classifies as paid_late."""
        df = pd.DataFrame({
            "due_date": ["2023-04-01"],
            "paid_date": ["2023-04-02"],
            "amount": [50.0],
        })
        ref = "2023-03-30"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "paid_late"

    # ------------------------------------------------------------------
    # Unpaid rows: overdue vs pending
    # ------------------------------------------------------------------
    def test_overdue_when_reference_past_due(self):
        """Unpaid row with reference_date > due_date classifies as overdue."""
        df = pd.DataFrame({
            "due_date": ["2023-01-10"],
            "paid_date": [pd.NaT],
            "amount": [100.0],
        })
        ref = "2023-01-15"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "overdue"

    def test_pending_when_reference_before_or_on_due(self):
        """Unpaid row with reference_date <= due_date classifies as pending."""
        df = pd.DataFrame({
            "due_date": ["2023-01-15"],
            "paid_date": [pd.NaT],
            "amount": [100.0],
        })
        ref = "2023-01-15"
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "pending"

    # ------------------------------------------------------------------
    # Mixed rows
    # ------------------------------------------------------------------
    def test_multiple_rows_with_all_statuses(self):
        df = pd.DataFrame({
            "due_date": [
                "2023-01-10",  # overdue
                "2023-01-20",  # pending
                "2023-02-01",  # paid_on_time
                "2023-02-10",  # paid_late
                "2023-03-01",  # invalid (bad due_date)
            ],
            "paid_date": [
                pd.NaT,
                pd.NaT,
                "2023-01-20",
                "2023-02-15",
                "2023-03-05",
            ],
            "amount": [50, 50, 50, 50, 50],
            "due_date_alt": [
                "2023-01-10",
                "2023-01-20",
                "2023-02-01",
                "2023-02-10",
                "not-a-date",
            ],
        })
        # Need to align: use the column "due_date_alt" for invalid row
        # Actually easier: separate. We'll do all columns same due_date col.
        df["due_date"] = [
            "2023-01-10",
            "2023-01-20",
            "2023-02-01",
            "2023-02-10",
            "bad-date",   # invalid
        ]
        # For paid_on_time row: paid_date = 2023-02-01? Actually due_date=2023-02-01, paid_date=2023-02-01 -> paid_on_time. Good.
        # For paid_late row: due=2023-02-10, paid=2023-02-15 -> late.
        # For invalid row: due date invalid -> invalid, paid_date not considered.
        # Unpaid first two: due=2023-01-10 (< ref 2023-01-15) -> overdue; due=2023-01-20 (> ref) -> pending.
        ref = "2023-01-15"
        result = classify_payment_status(df, ref)
        expected = ["overdue", "pending", "paid_on_time", "paid_late", "invalid"]
        assert result["payment_status"].tolist() == expected

    # ------------------------------------------------------------------
    # Edge: empty DataFrame
    # ------------------------------------------------------------------
    def test_empty_dataframe(self):
        df = pd.DataFrame(columns=["due_date", "paid_date", "amount"])
        ref = "2023-01-01"
        result = classify_payment_status(df, ref)
        assert "payment_status" in result.columns
        assert len(result) == 0

    # ------------------------------------------------------------------
    # Output column name configurable
    # ------------------------------------------------------------------
    def test_custom_output_column(self):
        df = pd.DataFrame({
            "due_date": ["2023-03-01"],
            "paid_date": ["2023-03-01"],
            "amount": [10.0],
        })
        ref = "2023-01-01"
        result = classify_payment_status(df, ref, output_col="classification")
        assert "classification" in result.columns
        assert result["classification"].iloc[0] == "paid_on_time"

    # ------------------------------------------------------------------
    # Immutability: original DataFrame unchanged
    # ------------------------------------------------------------------
    def test_original_dataframe_not_modified(self):
        df = pd.DataFrame({
            "due_date": ["2023-04-01"],
            "paid_date": ["2023-04-02"],
            "amount": [99.9],
        })
        original = df.copy()
        ref = "2023-03-30"
        _ = classify_payment_status(df, ref)
        assert df.equals(original)  # column set is same, values unchanged

    # ------------------------------------------------------------------
    # Behaviour with datetime objects instead of strings
    # ------------------------------------------------------------------
    def test_with_datetime_objects(self):
        df = pd.DataFrame({
            "due_date": [pd.Timestamp("2023-05-05")],
            "paid_date": [pd.Timestamp("2023-05-06")],
            "amount": [100],
        })
        ref = pd.Timestamp("2023-05-01")
        result = classify_payment_status(df, ref)
        assert result["payment_status"].iloc[0] == "paid_late"

    # ------------------------------------------------------------------
    # Missing columns (should raise KeyError)
    # ------------------------------------------------------------------
    def test_missing_required_column_raises(self):
        df = pd.DataFrame({"due_date": ["2023-01-01"], "amount": [10]})
        ref = "2023-01-01"
        with pytest.raises(KeyError):
            classify_payment_status(df, ref, paid_date_col="paid_date")
