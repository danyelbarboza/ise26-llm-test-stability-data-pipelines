import pytest
import pandas as pd
from ise26.targets import classify_payment_status

def test_valid_reference_date_str():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100]
    })
    result = classify_payment_status(df, "2023-01-20")
    assert result["payment_status"].iloc[0] == "paid_on_time"

def test_valid_reference_date_timestamp():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": [None],
        "amount": [100]
    })
    result = classify_payment_status(df, pd.Timestamp("2023-01-20"))
    assert result["payment_status"].iloc[0] == "overdue"

def test_invalid_reference_date():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100]
    })
    with pytest.raises(ValueError):

def test_mixed_scenarios():
    df = pd.DataFrame({
        "due_date": ["2023-01-15", "2023-01-16", "2023-01-17", "2023-01-18", "2023-01-19"],
        "paid_date": [None, "2023-01-16", "2023-01-20", "bad_date", "2023-01-15"],
        "amount": [100, 100, -100, 100, 100]
    })
    result = classify_payment_status(df, "2023-01-20")
    expected = [
        "overdue",       # unpaid, reference > due_date
        "paid_on_time",  # paid == due
        "invalid",       # negative amount
        "invalid",       # invalid paid date
        "paid_on_time"
    ]
    assert list(result["payment_status"]) == expected

def test_custom_output_column():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100]
    })
    result = classify_payment_status(df, "2023-01-20", output_col="status")
    assert "status" in result.columns
    assert result["status"].iloc[0] == "paid_on_time"

def test_preserves_original_columns():
    df = pd.DataFrame({
        "due_date": ["2023-01-15"],
        "paid_date": ["2023-01-10"],
        "amount": [100],
        "extra": [42]
    })
    result = classify_payment_status(df, "2023-01-20")
    assert "extra" in result.columns
    assert result["extra"].iloc[0] == 42

def test_duplicate_rows():
    df = pd.DataFrame({
        "due_date": ["2023-01-15", "2023-01-15"],
        "paid_date": ["2023-01-10", "2023-01-16"],
        "amount": [100, 200]
    })
    result = classify_payment_status(df, "2023-01-20")
    expected = ["paid_on_time", "paid_late"]
    assert list(result["payment_status"]) == expected

def test_all_invalid():
    df = pd.DataFrame({
        "due_date": [None, "bad", "2023-01-15"],
        "paid_date": ["2023-01-10", "2023-01-11", "bad_paid"],
        "amount": [None, 0, 100]
    })
    result = classify_payment_status(df, "2023-01-20")
    assert all(status == "invalid" for status in result["payment_status"])
