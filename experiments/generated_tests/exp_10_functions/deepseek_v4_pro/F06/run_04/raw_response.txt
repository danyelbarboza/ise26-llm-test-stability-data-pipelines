import pytest
import pandas as pd
from ise26.targets import classify_payment_status


def reference():
    return pd.Timestamp("2025-03-15")


def df_with_defaults():
    """Return a minimal valid DataFrame with default column names."""
    return pd.DataFrame(
        {
            "due_date": [pd.Timestamp("2025-03-10")],
            "paid_date": [pd.Timestamp("2025-03-09")],
            "amount": [150.0],
        }
    )


# ---------------------------------------------------------------------------
# Positive cases
# ---------------------------------------------------------------------------

def test_paid_on_time():
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-10", "2025-03-12"],
            "paid_date": ["2025-03-10", "2025-03-11"],
            "amount": [10, 20],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    expected = pd.Series(["paid_on_time", "paid_on_time"], name="payment_status")
    pd.testing.assert_series_equal(
        result["payment_status"], expected, check_names=False
    )


def test_paid_late():
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-10", "2025-03-12"],
            "paid_date": ["2025-03-11", "2025-03-13"],
            "amount": [10, 20],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    expected = pd.Series(["paid_late", "paid_late"], name="payment_status")
    pd.testing.assert_series_equal(
        result["payment_status"], expected, check_names=False
    )


def test_overdue():
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-08", "2025-03-09"],
            "paid_date": [None, None],
            "amount": [5, 5],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    expected = pd.Series(["overdue", "overdue"], name="payment_status")
    pd.testing.assert_series_equal(
        result["payment_status"], expected, check_names=False
    )


def test_pending():
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-16", "2025-03-20"],
            "paid_date": [None, None],
            "amount": [7, 7],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    expected = pd.Series(["pending", "pending"], name="payment_status")
    pd.testing.assert_series_equal(
        result["payment_status"], expected, check_names=False
    )


def test_reference_equal_due_pending():
    df = pd.DataFrame(
        {"due_date": ["2025-03-15"], "paid_date": [None], "amount": [30]}
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "pending"


def test_mixed_statuses():
    df = pd.DataFrame(
        {
            "due_date": [
                "2025-03-05",
                "2025-03-08",
                "2025-03-10",
                "2025-03-12",
                "2025-03-15",
            ],
            "paid_date": [
                "2025-03-05",
                "2025-03-09",
                None,
                None,
                "2025-03-14",
            ],
            "amount": [100, 200, 300, 400, 500],
        }
    )
    ref = "2025-03-10"
    result = classify_payment_status(df, ref)
    expected = pd.Series(
        ["paid_on_time", "paid_late", "overdue", "pending", "paid_on_time"],
        name="payment_status",
    )
    pd.testing.assert_series_equal(
        result["payment_status"], expected, check_names=False
    )


# ---------------------------------------------------------------------------
# Invalid rows
# ---------------------------------------------------------------------------

def test_invalid_amount_missing():
    df = pd.DataFrame(
        {"due_date": ["2025-03-10"], "paid_date": [None], "amount": [None]}
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "invalid"


def test_invalid_amount_non_numeric():
    df = pd.DataFrame(
        {"due_date": ["2025-03-10"], "paid_date": [None], "amount": ["abc"]}
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "invalid"


def test_invalid_amount_zero_or_negative():
    df = pd.DataFrame(
        {"due_date": ["2025-03-10"], "paid_date": [None], "amount": [0, -5]}
    )
    result = classify_payment_status(df, "2025-03-15")
    assert (result["payment_status"].values == ["invalid", "invalid"]).all()


def test_invalid_due_date():
    # unparseable due date
    df = pd.DataFrame(
        {
            "due_date": ["not-a-date", "2025-03-01", "2025-03-02"],
            "paid_date": [
                "2025-03-01",
                "2025-03-01",
                "2025-03-02",
            ],
            "amount": [100, 100, 100],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "invalid"
    # other rows should be valid
    assert result["payment_status"].iloc[1] in ["paid_on_time", "paid_late"]
    assert result["payment_status"].iloc[2] in ["paid_on_time", "paid_late"]


def test_invalid_paid_date():
    # paid_date provided but unparseable -> invalid
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-10", "2025-03-11"],
            "paid_date": ["garbage", "2025-03-11"],
            "amount": [10, 20],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "invalid"
    assert result["payment_status"].iloc[1] == "paid_on_time"


def test_paid_date_empty_string_treated_as_missing():
    # empty string is considered not provided -> unpaid, not invalid
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-10"],
            "paid_date": [""],
            "amount": [10],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    # due date < ref -> overdue
    assert result["payment_status"].iloc[0] == "overdue"


def test_paid_date_whitespace_string_treated_as_missing():
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-10"],
            "paid_date": ["   "],
            "amount": [10],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "overdue"


def test_invalid_paid_and_due_combination():
    df = pd.DataFrame(
        {
            "due_date": ["2025-03-10", "bad-date"],
            "paid_date": ["not-a-date", None],
            "amount": [5, 5],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    assert result["payment_status"].iloc[0] == "invalid"
    assert result["payment_status"].iloc[1] == "invalid"


# ---------------------------------------------------------------------------
# Reference date validation
# ---------------------------------------------------------------------------

def test_invalid_reference_date_raises():
    df = df_with_defaults()
    with pytest.raises(ValueError, match="reference_date must be a valid"):
        classify_payment_status(df, "not-a-date")


def test_reference_date_passed_as_timestamp():
    df = pd.DataFrame(
        {"due_date": ["2025-03-10"], "paid_date": [None], "amount": [50]}
    )
    result = classify_payment_status(df, pd.Timestamp("2025-03-15"))
    assert result["payment_status"].iloc[0] == "overdue"


# ---------------------------------------------------------------------------
# Output column handling
# ---------------------------------------------------------------------------

def test_default_output_column():
    df = df_with_defaults()
    result = classify_payment_status(df, "2025-03-15")
    assert "payment_status" in result.columns


def test_custom_output_column():
    df = df_with_defaults()
    result = classify_payment_status(df, "2025-03-15", output_col="my_status")
    assert "my_status" in result.columns
    assert "payment_status" not in result.columns
    assert result["my_status"].iloc[0] == "paid_on_time"


# ---------------------------------------------------------------------------
# Immutability of original DataFrame
# ---------------------------------------------------------------------------

def test_original_df_not_modified():
    df = df_with_defaults().copy()
    original_columns = df.columns.tolist()
    original_values = df.copy()
    classify_payment_status(df, "2025-03-15")
    # original should be unchanged
    assert df.columns.tolist() == original_columns
    pd.testing.assert_frame_equal(df, original_values)


# ---------------------------------------------------------------------------
# Edge case: all rows invalid
# ---------------------------------------------------------------------------

def test_all_rows_invalid():
    df = pd.DataFrame(
        {
            "due_date": ["bad", "bad2"],
            "paid_date": [None, "bad3"],
            "amount": [None, -1],
        }
    )
    result = classify_payment_status(df, "2025-03-15")
    assert (result["payment_status"] == "invalid").all()


# ---------------------------------------------------------------------------
# Custom column names
# ---------------------------------------------------------------------------

def test_custom_column_names():
    df = pd.DataFrame(
        {
            "when_due": ["2025-03-10"],
            "when_paid": [None],
            "value": [5],
        }
    )
    result = classify_payment_status(
        df,
        "2025-03-15",
        due_date_col="when_due",
        paid_date_col="when_paid",
        amount_col="value",
        output_col="status",
    )
    assert result["status"].iloc[0] == "overdue"
