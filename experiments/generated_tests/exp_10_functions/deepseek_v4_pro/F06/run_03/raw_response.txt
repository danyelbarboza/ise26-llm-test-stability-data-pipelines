import pandas as pd
import numpy as np
import pytest
from ise26.targets import classify_payment_status


def test_default_columns_paid_on_time():
    df = pd.DataFrame({
        "due_date": ["2024-01-10", "2024-01-15"],
        "paid_date": ["2024-01-10", "2024-01-14"],
        "amount": [200.0, 150.0],
    })
    reference_date = "2024-01-20"
    result = classify_payment_status(df, reference_date)
    expected = ["paid_on_time", "paid_on_time"]
    assert "payment_status" in result.columns
    assert result["payment_status"].tolist() == expected

def test_paid_date_empty_string_treated_as_unpaid():
    df = pd.DataFrame({
        "due_date": ["2024-01-10"],
        "paid_date": [""],
        "amount": [100.0],
    })
    reference_date = "2024-01-20"
    result = classify_payment_status(df, reference_date)
    # Empty string is not considered provided → acts as unpaid → overdue because ref > due
    assert result["payment_status"].iloc[0] == "overdue"

def test_custom_column_names():
    df = pd.DataFrame({
        "vencimento": ["2024-02-01"],
        "pagamento": ["2024-02-01"],
        "valor": [50.0],
    })
    reference_date = "2024-02-10"
    result = classify_payment_status(
        df,
        reference_date,
        due_date_col="vencimento",
        paid_date_col="pagamento",
        amount_col="valor",
        output_col="status",
    )
    assert "status" in result.columns
    assert result["status"].iloc[0] == "paid_on_time"


def test_reference_date_valid_types():
    df = pd.DataFrame({
        "due_date": ["2024-01-10"],
        "paid_date": [None],
        "amount": [100.0],
    })
    # reference_date as Timestamp
    result = classify_payment_status(df, pd.Timestamp("2024-01-15"))
    assert result["payment_status"].iloc[0] == "overdue"


def test_reference_date_invalid_raises_valueerror():
    df = pd.DataFrame({
        "due_date": ["2024-01-10"],
        "paid_date": [None],
        "amount": [100.0],
    })
    with pytest.raises(ValueError, match="reference_date must be a valid date-like value"):
        classify_payment_status(df, "not_a_date")


def test_does_not_mutate_original_df():
    original = pd.DataFrame({
        "due_date": ["2024-01-10"],
        "paid_date": ["2024-01-10"],
        "amount": [100.0],
    })
    reference_date = "2024-01-20"
    df_copy = original.copy()
    result = classify_payment_status(original, reference_date)
    pd.testing.assert_frame_equal(original, df_copy)
    assert "payment_status" not in original.columns


def test_mixed_scenarios():
    df = pd.DataFrame({
        "due_date": [
            "2024-01-10",   # valid, paid on time
            "2024-01-11",   # valid, paid late
            "2024-01-12",   # unpaid, overdue (ref > due)
            "2024-02-01",   # unpaid, pending (due > ref)
            "invalid_due",  # invalid due date
            "2024-01-10",   # invalid amount (zero)
        ],
        "paid_date": [
            "2024-01-10",
            "2024-01-13",
            None,
            None,
            "2024-01-10",
            "2024-01-10",
        ],
        "amount": [
            200.0,
            150.0,
            100.0,
            100.0,
            50.0,
            0.0,
        ],
    })
    reference_date = "2024-01-15"
    result = classify_payment_status(df, reference_date)
    expected = [
        "paid_on_time",
        "paid_late",
        "overdue",
        "pending",
        "invalid",
        "invalid",
    ]
    assert result["payment_status"].tolist() == expected
